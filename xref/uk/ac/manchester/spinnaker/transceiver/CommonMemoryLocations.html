17  package uk.ac.manchester.spinnaker.transceiver;
18  
19  import static uk.ac.manchester.spinnaker.messages.Constants.CPU_INFO_OFFSET;
20  import static uk.ac.manchester.spinnaker.messages.Constants.ROUTER_REGISTER_P2P_ADDRESS;
21  import static uk.ac.manchester.spinnaker.messages.Constants.SYSTEM_VARIABLE_BASE_ADDRESS;
22  
23  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
24  
25  /**
26   * Common memory locations. Note that many locations really come in both
27   * buffered and unbuffered varieties, but this class doesn't hold all standard
28   * locations and versions:
29   * <p>
30   * <blockquote> Buffered SDRAM means that writes go through a write buffer.
31   * Unbuffered means that they go directly to SDRAM. Reads are unaffected in
32   * general. If you are writing lots of data, it is unlikely to matter much since
33   * the write buffer is limited in size. Here you probably want to use Unbuffered
34   * anyway, as it will then block until the write is definitely done. Using
35   * Buffered writing means that the write may or may not have happened at the
36   * time of the response. </blockquote>
37   *
38   * @author Donal Fellows
39   */
40  public abstract class CommonMemoryLocations {
41  	private CommonMemoryLocations() {
42  	}
43  
44  	/**
45  	 * Start of unbuffered access to SDRAM. Writes will block until they have
46  	 * completed.
47  	 */
48  	public static final MemoryLocation UNBUFFERED_SDRAM_START =
49  			new MemoryLocation(0x60000000);
50  
51  	/** Location of routing table data in transit. Reserved by SCAMP. */
52  	public static final MemoryLocation ROUTING_TABLE_DATA =
53  			new MemoryLocation(0x67800000);
54  
55  	/**
56  	 * Where executables are written to prior to launching them. Reserved by
57  	 * SCAMP.
58  	 */
59  	public static final MemoryLocation EXECUTABLE_ADDRESS =
60  			new MemoryLocation(0x67800000);
61  
62  	/**
63  	 * Start of buffered access to SDRAM. Writes will finish rapidly, but data
64  	 * may take some cycles to appear in memory.
65  	 * <p>
66  	 * It doesn't matter too much when working from host; the time to do the
67  	 * network communications is rather longer than the normal buffering time
68  	 * for SDRAM.
69  	 */
70  	public static final MemoryLocation BUFFERED_SDRAM_START =
71  			new MemoryLocation(0x70000000);
72  
73  	/** Location of the bank of memory-mapped router registers. Buffered. */
74  	public static final MemoryLocation ROUTER_BASE =
75  			new MemoryLocation(0xe1000000);
76  
77  	/** Location of the memory-mapped router control register (r0). */
78  	public static final MemoryLocation ROUTER_CONTROL =
79  			new MemoryLocation(0xe1000000);
80  
81  	/** Location of the memory-mapped router error register (r5). */
82  	public static final MemoryLocation ROUTER_ERROR =
83  			new MemoryLocation(0xe1000014);
84  
85  	/**
86  	 * Where to write router diagnostic counters control data to (r11).
87  	 * Unbuffered.
88  	 */
89  	public static final MemoryLocation ROUTER_DIAGNOSTIC_COUNTER =
90  			new MemoryLocation(0xf100002c);
91  
92  	/** Location of the memory-mapped router filtering registers (rFN). */
93  	public static final MemoryLocation ROUTER_FILTERS =
94  			new MemoryLocation(0xe1000200);
95  
96  	/** Location of the memory-mapped router diagnostics registers (rCN). */
97  	public static final MemoryLocation ROUTER_DIAGNOSTICS =
98  			new MemoryLocation(0xe1000300);
99  
100 	/** The base address of a router's P2P routing table, 3 bits per route. */
101 	public static final MemoryLocation ROUTER_P2P =
102 			new MemoryLocation(ROUTER_REGISTER_P2P_ADDRESS);
103 
104 	/** Where the CPU information structure is located. Buffered system RAM. */
105 	public static final MemoryLocation CPU_INFO =
106 			new MemoryLocation(CPU_INFO_OFFSET);
107 
108 	/** Where the system variables are located. Unbuffered system RAM. */
109 	public static final MemoryLocation SYS_VARS =
110 			new MemoryLocation(SYSTEM_VARIABLE_BASE_ADDRESS);
111 }
