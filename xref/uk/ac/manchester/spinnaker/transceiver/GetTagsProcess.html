17  package uk.ac.manchester.spinnaker.transceiver;
18  
19  import static java.util.stream.IntStream.range;
20  
21  import java.io.IOException;
22  import java.net.InetAddress;
23  import java.util.Collection;
24  import java.util.Map;
25  import java.util.TreeMap;
26  
27  import uk.ac.manchester.spinnaker.connections.ConnectionSelector;
28  import uk.ac.manchester.spinnaker.connections.SCPConnection;
29  import uk.ac.manchester.spinnaker.machine.tags.IPTag;
30  import uk.ac.manchester.spinnaker.machine.tags.ReverseIPTag;
31  import uk.ac.manchester.spinnaker.machine.tags.Tag;
32  import uk.ac.manchester.spinnaker.messages.scp.IPTagGet;
33  import uk.ac.manchester.spinnaker.messages.scp.IPTagGetInfo;
34  
35  /** Gets IP tags and reverse IP tags. */
36  class GetTagsProcess extends MultiConnectionProcess<SCPConnection> {
37  	/**
38  	 * @param connectionSelector
39  	 *            How to select how to communicate.
40  	 * @param retryTracker
41  	 *            Object used to track how many retries were used in an
42  	 *            operation. May be {@code null} if no suck tracking is
43  	 *            required.
44  	 */
45  	GetTagsProcess(ConnectionSelector<SCPConnection> connectionSelector,
46  			RetryTracker retryTracker) {
47  		super(connectionSelector, retryTracker);
48  	}
49  
50  	/**
51  	 * Get the set of tags that are associated with a connection.
52  	 *
53  	 * @param connection
54  	 *            The connection that the tags are associated with.
55  	 * @return The allocated tags in ID order. Unallocated tags are absent.
56  	 * @throws IOException
57  	 *             If anything goes wrong with networking.
58  	 * @throws ProcessException
59  	 *             If SpiNNaker rejects a message.
60  	 */
61  	Collection<Tag> getTags(SCPConnection connection)
62  			throws IOException, ProcessException {
63  		var tags = new TreeMap<Integer, Tag>();
64  		for (var tag : range(0, getTagCount(connection)).toArray()) {
65  			sendRequest(new IPTagGet(connection.getChip(), tag), response -> {
66  				if (response.isInUse()) {
67  					tags.put(tag, createTag(connection.getRemoteIPAddress(),
68  							tag, response));
69  				}
70  			});
71  		}
72  		finish();
73  		checkForError();
74  		return tags.values();
75  	}
76  
77  	private int getTagCount(SCPConnection connection)
78  			throws IOException, ProcessException {
79  		var tagInfo = synchronousCall(new IPTagGetInfo(connection.getChip()));
80  		return tagInfo.poolSize + tagInfo.fixedSize;
81  	}
82  
83  	private static Tag createTag(InetAddress host, int tag,
84  			IPTagGet.Response res) {
85  		if (res.isReverse()) {
86  			return new ReverseIPTag(host, tag, res.rxPort, res.spinCore,
87  					res.spinPort);
88  		} else {
89  			return new IPTag(host, res.sdpHeader.getSource().asChipLocation(),
90  					tag, res.ipAddress, res.port, res.isStrippingSDP());
91  		}
92  	}
93  
94  	/**
95  	 * Get the usage of each of the (active) tags associated with a connection.
96  	 *
97  	 * @param connection
98  	 *            The connection that the tags are associated with.
99  	 * @return A map from each active tag to the number of packets sent through
100 	 *         that tag.
101 	 * @throws IOException
102 	 *             If anything goes wrong with networking.
103 	 * @throws ProcessException
104 	 *             If SpiNNaker rejects a message.
105 	 */
106 	Map<Tag, Integer> getTagUsage(SCPConnection connection)
107 			throws IOException, ProcessException {
108 		var tagUsages = new TreeMap<Tag, Integer>();
109 		for (var tag : range(0, getTagCount(connection)).toArray()) {
110 			sendRequest(new IPTagGet(connection.getChip(), tag), response -> {
111 				if (response.isInUse()) {
112 					tagUsages.put(createTag(connection.getRemoteIPAddress(),
113 							tag, response), response.count);
114 				}
115 			});
116 		}
117 		finish();
118 		checkForError();
119 		return tagUsages;
120 	}
121 }
