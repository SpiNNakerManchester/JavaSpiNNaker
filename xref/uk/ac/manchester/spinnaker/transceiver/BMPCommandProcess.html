49  /**
50   * A process for handling communicating with the BMP.
51   * <p>
52   * Does not inherit from {@link Process} for ugly type reasons.
53   *
54   * @param <R>
55   *            The type of the response; implicit in the type of the request.
56   * @author Donal Fellows
57   * @see uk.ac.manchester.spinnaker.connections.SCPRequestPipeline
58   *      SCPRequestPipeline
59   */
60  class BMPCommandProcess<R extends BMPResponse> {
61  	private static final Logger log = getLogger(BMPCommandProcess.class);
62  	/** How long to wait for a BMP to respond. */
63  	private static final int DEFAULT_TIMEOUT =
64  			(int) (MSEC_PER_SEC * BMP_TIMEOUT);
65  
66  	private static final String TIMEOUT_TOKEN = "BMP timed out";
67  	private static final int BMP_RETRIES = 3;
68  
69  	private final ConnectionSelector<BMPConnection> connectionSelector;
70  	private final int timeout;
71  	private final RetryTracker retryTracker;
72  	private BMPRequest<?> errorRequest;
73  	private Throwable exception;
74  
75  	/**
76  	 * @param connectionSelector
77  	 *            How to select how to communicate.
78  	 * @param retryTracker
79  	 *            Object used to track how many retries were used in an
80  	 *            operation. May be {@code null} if no suck tracking is
81  	 *            required.
82  	 */
83  	BMPCommandProcess(ConnectionSelector<BMPConnection> connectionSelector,
84  			RetryTracker retryTracker) {
85  		this(connectionSelector, DEFAULT_TIMEOUT, retryTracker);
86  	}
87  
88  	/**
89  	 * @param connectionSelector
90  	 *            How to select how to communicate.
91  	 * @param timeout
92  	 *            The timeout on the connection, in milliseconds.
93  	 * @param retryTracker
94  	 *            Object used to track how many retries were used in an
95  	 *            operation. May be {@code null} if no suck tracking is
96  	 *            required.
97  	 */
98  	BMPCommandProcess(ConnectionSelector<BMPConnection> connectionSelector,
99  			int timeout, RetryTracker retryTracker) {
100 		this.timeout = timeout;
101 		this.connectionSelector = connectionSelector;
102 		this.retryTracker = retryTracker;
103 	}

105 	/**
106 	 * Do a synchronous call of a BMP operation, sending the given message and
107 	 * completely processing the interaction before returning its response.
108 	 *
109 	 * @param request
110 	 *            The request to send
111 	 * @return The successful response to the request
112 	 * @throws IOException
113 	 *             If the communications fail
114 	 * @throws ProcessException
115 	 *             If the other side responds with a failure code
116 	 */
117 	R execute(BMPRequest<R> request) throws IOException, ProcessException {
118 		ValueHolder<R> holder = new ValueHolder<>();
119 		/*
120 		 * If no pipeline built yet, build one on the connection selected for
121 		 * it.
122 		 */
123 		RequestPipeline requestPipeline = new RequestPipeline(
124 				connectionSelector.getNextConnection(request));
125 		requestPipeline.sendRequest(request, holder::setValue);
126 		requestPipeline.finish();
127 		if (exception != null) {
128 			throw new ProcessException(errorRequest.sdpHeader.getDestination(),
129 					exception);
130 		}
131 		return holder.getValue();
132 	}
133 
134 	/**
135 	 * Allows a set of BMP requests to be grouped together in a communication
136 	 * across a number of channels for a given connection.
137 	 * <p>
138 	 * This class implements SCP windowing, first suggested by Andrew Mundy.
139 	 * This extends the idea by having both send and receive windows. These are
140 	 * represented by the <i>numChannels</i> and the
141 	 * <i>intermediateChannelWaits</i> parameters respectively. This seems to
142 	 * help with the timeout issue; when a timeout is received, all requests for
143 	 * which a reply has not been received can also timeout.
144 	 *
145 	 * @author Andrew Mundy
146 	 * @author Andrew Rowley
147 	 * @author Donal Fellows
148 	 */
149 	private final class RequestPipeline {
150 		/** The connection over which the communication is to take place. */
151 		private BMPConnection connection;
152 		/** A dictionary of sequence number &rarr; requests in progress. */
153 		private final Map<Integer, Request> requests =
154 				synchronizedMap(new HashMap<>());
155 
156 		/** Per message record. */
157 		private final class Request {
158 			/** request in progress. */
159 			private final BMPRequest<R> request;
160 			/** payload of request in progress. */
161 			private final ByteBuffer requestData;
162 			/** callback function for response. */
163 			private final Consumer<R> callback;
164 			/** retry reason. */
165 			private final List<String> retryReason = new ArrayList<>();
166 			/** number of retries for the packet. */
167 			private int retries = BMP_RETRIES;
168 
169 			private Request(BMPRequest<R> request, Consumer<R> callback) {
170 				this.request = request;
171 				this.requestData = request.getMessageData(connection.getChip());
172 				this.callback = callback;
173 			}
174 
175 			private void send() throws IOException {
176 				connection.send(requestData.asReadOnlyBuffer());
177 			}
178 
179 			private void resend(String reason) throws IOException {
180 				retries--;
181 				retryReason.add(reason);
182 				if (retryTracker != null) {
183 					retryTracker.retryNeeded();
184 				}
185 				send();
186 			}
187 
188 			private boolean hasRetries() {
189 				return retries > 0;
190 			}
191 
192 			private boolean allTimeoutFailures() {
193 				return retryReason.stream().allMatch(TIMEOUT_TOKEN::equals);
194 			}
195 
196 			private void parseReceivedResponse(SCPResultMessage msg)
197 					throws Exception {
198 				R response = msg.parsePayload(request);
199 				if (callback != null) {
200 					callback.accept(response);
201 				}
202 			}
203 
204 			private HasCoreLocation dest() {
205 				return request.sdpHeader.getDestination();
206 			}
207 		}
208 
209 		/**
210 		 * Create a request handling pipeline.
211 		 *
212 		 * @param connection
213 		 *            The connection over which the communication is to take
214 		 *            place.
215 		 */
216 		private RequestPipeline(BMPConnection connection) {
217 			this.connection = connection;
218 		}
219 
220 		/**
221 		 * Add a BMP request to the set to be sent.
222 		 *
223 		 * @param request
224 		 *            The BMP request to be sent
225 		 * @param callback
226 		 *            A callback function to call when the response has been
227 		 *            received; takes an SCPResponse as a parameter, or a
228 		 *            {@code null} if the response doesn't need to be processed.
229 		 * @throws IOException
230 		 *             If things go really wrong.
231 		 */
232 		private void sendRequest(BMPRequest<R> request, Consumer<R> callback)
233 				throws IOException {
234 			// Get the next sequence to be used and store it in the header
235 			int sequence = request.scpRequestHeader
236 					.issueSequenceNumber(requests.keySet());
237 
238 			// Send the request, keeping track of how many are sent
239 			Request req = new Request(request, callback);
240 			if (requests.put(sequence, req) != null) {
241 				throw new RuntimeException(
242 						"duplicate sequence number catastrophe");
243 			}
244 			req.send();
245 		}
246 
247 		/**
248 		 * Indicate the end of the packets to be sent. This must be called to
249 		 * ensure that all responses are received and handled.
250 		 *
251 		 * @throws IOException
252 		 *             If anything goes wrong with communications.
253 		 */
254 		private void finish() throws IOException {
255 			// While there are still more packets in progress than some
256 			// threshold
257 			while (!requests.isEmpty()) {
258 				try {
259 					// Receive the next response
260 					retrieve();
261 				} catch (SocketTimeoutException e) {
262 					handleReceiveTimeout();
263 				}
264 			}
265 		}
266 
267 		private void retrieve() throws IOException {
268 			// Receive the next response
269 			SCPResultMessage msg = connection.receiveSCPResponse(timeout);
270 			Request req = msg.pickRequest(requests);
271 			if (req == null) {
272 				// Only process responses which have matching requests
273 				log.info("discarding message with unknown sequence number: {}",
274 						msg.getSequenceNumber());
275 				return;
276 			}
277 
278 			// If the response can be retried, retry it
279 			try {
280 				if (msg.isRetriable()) {
281 					sleep(RETRY_DELAY_MS);
282 					resend(req, msg.getResult());
283 				} else {
284 					// No retry is possible. Try constructing the result
285 					req.parseReceivedResponse(msg);
286 					// Remove the sequence from the outstanding responses
287 					msg.removeRequest(requests);
288 				}
289 			} catch (Exception e) {
290 				errorRequest = req.request;
291 				exception = e;
292 				msg.removeRequest(requests);
293 			}
294 		}
295 
296 		private void handleReceiveTimeout() {
297 			// If there is a timeout, all packets remaining are resent
298 			BitSet toRemove = new BitSet(SEQUENCE_LENGTH);
299 			for (int seq : new ArrayList<>(requests.keySet())) {
300 				Request req = requests.get(seq);
301 				if (req == null) {
302 					// Shouldn't happen, but if it does we should nuke it.
303 					toRemove.set(seq);
304 					continue;
305 				}
306 
307 				try {
308 					resend(req, TIMEOUT_TOKEN);
309 				} catch (Exception e) {
310 					errorRequest = req.request;
311 					exception = e;
312 					toRemove.set(seq);
313 				}
314 			}
315 
316 			toRemove.stream().forEach(requests::remove);
317 		}
318 
319 		private void resend(Request req, Object reason) throws IOException {
320 			if (!req.hasRetries()) {
321 				// Report timeouts as timeout exception
322 				if (req.allTimeoutFailures()) {
323 					throw new SendTimedOutException(
324 							req.request.scpRequestHeader, timeout);
325 				}
326 
327 				// Report any other exception
328 				throw new SendFailedException(req.request.scpRequestHeader,
329 						req.dest(), req.retryReason);
330 			}
331 
332 			// If the request can be retried, retry it
333 			req.resend(reason.toString());
334 		}
335 	}

337 	/**
338 	 * Indicates that message sending timed out.
339 	 */
340 	static final class SendTimedOutException extends SocketTimeoutException {
341 		private static final long serialVersionUID = 1660563278795501381L;
342 
343 		SendTimedOutException(SCPRequestHeader hdr, int timeout) {
344 			super(format("Operation %s timed out after %f seconds", hdr.command,
345 					timeout / (double) MSEC_PER_SEC));
346 		}
347 	}
348 
349 	/**
350 	 * Indicates that message sending failed for various reasons.
351 	 */
352 	static final class SendFailedException extends IOException {
353 		private static final long serialVersionUID = -7806549580351626377L;
354 
355 		SendFailedException(SCPRequestHeader hdr, HasCoreLocation core,
356 				List<String> retryReason) {
357 			super(format(
358 					"Errors sending request %s to %d,%d,%d over %d retries: %s",
359 					hdr.command, core.getX(), core.getY(), core.getP(),
360 					BMP_RETRIES, retryReason));
361 		}
