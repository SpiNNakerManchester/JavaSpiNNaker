17  package uk.ac.manchester.spinnaker.alloc.model;
18  
19  import static com.fasterxml.jackson.annotation.JsonFormat.Shape.ARRAY;
20  
21  import com.fasterxml.jackson.annotation.JsonFormat;
22  import com.google.errorprone.annotations.Immutable;
23  
24  /**
25   * Describes a link that is disabled.
26   *
27   * @author Donal Fellows
28   */
29  @Immutable
30  @JsonFormat(shape = ARRAY)
31  public final class DownLink {
32  	/**
33  	 * Describes one end of a link that is disabled.
34  	 *
35  	 * @author Donal Fellows
36  	 */
37  	@Immutable
38  	public static final class End {
39  		private End(BoardCoords board, Direction direction) {
40  			this.board = board;
41  			this.direction = direction;
42  		}
43  
44  		/**
45  		 * On what board is this end of the link.
46  		 */
47  		public final BoardCoords board;
48  
49  		/**
50  		 * In which direction does this end of the link go?
51  		 */
52  		public final Direction direction;
53  	}
54  
55  	/**
56  	 * Create a down link description.
57  	 *
58  	 * @param board1
59  	 *            Which board is one end of the link.
60  	 * @param dir1
61  	 *            In which direction off of {@code board1} is the link.
62  	 * @param board2
63  	 *            Which board is one end of the link.
64  	 * @param dir2
65  	 *            In which direction off of {@code board2} is the link.
66  	 */
67  	public DownLink(BoardCoords board1, Direction dir1, BoardCoords board2,
68  			Direction dir2) {
69  		end1 = new End(board1, dir1);
70  		end2 = new End(board2, dir2);
71  	}
72  
73  	/** One end of the down link. */
74  	public final DownLink.End end1;
75  
76  	/** The other end of the down link. */
77  	public final DownLink.End end2;
78  }
