157 	@Override
158 	protected final JobMachineInfo getJobMachineInfo(int jobId)
159 			throws TaskException {
160 		var job = getJob(jobId);
161 		job.access(host());
162 		var machine = job.getMachine()
163 				.orElseThrow(() -> new TaskException("boards not allocated"));
164 		int w, h;
165 		if (machine.getDepth() == 1) {
166 			// Special case: single board alloc
167 			w = SIZE_X_OF_ONE_BOARD;
168 			h = SIZE_Y_OF_ONE_BOARD;
169 		} else {
170 			// Everything else is in terms of triads; careful of wraparounds
171 			var full = machine.getMachine();
172 			int linearTriadSize = job.getWidth().orElse(0);
173 			w = linearTriadSize * TRIAD_WIDTH;
174 			if (linearTriadSize != full.getWidth()
175 					|| !full.isHorizonallyWrapped()) {
176 				w += HALF_SIZE;
177 			}
178 			linearTriadSize = job.getHeight().orElse(0);
179 			h = linearTriadSize * TRIAD_HEIGHT;
180 			if (linearTriadSize != full.getHeight()
181 					|| !full.isVerticallyWrapped()) {
182 				h += HALF_SIZE;
183 			}
184 		}
185 		return new JobMachineInfo(w, h, machine.getConnections().stream()
186 				.map(ci -> new Connection(ci.getChip(), ci.getHostname()))
187 				.collect(toList()), machine.getMachine().getName(),
188 				machine.getBoards());
189 	}

191 	@Override
192 	protected final JobState getJobState(int jobId) throws TaskException {
193 		var job = getJob(jobId);
194 		job.access(host());
195 		var js = new JobState.Builder();
196 		js.setKeepalive(timestamp(job.getKeepaliveTimestamp()));
197 		js.setKeepalivehost(job.getKeepaliveHost().orElse(""));
198 		js.setPower(job.getMachine().map(m -> {
199 			try {
200 				return m.getPower() == ON;
201 			} catch (DataAccessException e) {
202 				log.warn("problem getting job power state", e);
203 				return false;
204 			}
205 		}).orElse(false));
206 		js.setReason(job.getReason().orElse(""));
207 		js.setStartTime(timestamp(job.getStartTime()));
208 		js.setState(state(job));
209 		return js.build();
210 	}

217 	/**
218 	 * Parse a value to get a keepalive duration.
219 	 *
220 	 * @param keepalive
221 	 *            The number to parse. May be {@code null} to get a default.
222 	 * @return The duration. Never {@code null}.
223 	 */
224 	private Duration parseKeepalive(Number keepalive) {
225 		if (isNull(keepalive)) {
226 			return mainProps.getCompat().getDefaultKeepalive();
227 		}
228 		var d = Duration.ofSeconds(keepalive.longValue());
229 		if (!(keepalive instanceof Double || keepalive instanceof Float)) {
230 			return d;
231 		}
232 		double fractionalPart = keepalive.doubleValue() - keepalive.longValue();
233 		return d.plusNanos((long) (fractionalPart * NSEC_PER_SEC));
234 	}

236 	/**
237 	 * Parse a value to get a list of machine tags to match.
238 	 *
239 	 * @param src
240 	 *            The value to parse.
241 	 * @param mayForceDefault
242 	 *            Whether we want to force a default value into the tags.
243 	 * @return The list of tags. Never {@code null}.
244 	 */
245 	private static List<String> tags(Object src, boolean mayForceDefault) {
246 		var vals = new ArrayList<String>();
247 		if (src instanceof List) {
248 			for (var o : (List<?>) src) {
249 				vals.add(Objects.toString(o));
250 			}
251 		} else if (src instanceof String) {
252 			vals.add((String) src);
253 		}
254 		if (vals.isEmpty() && mayForceDefault) {
255 			return List.of("default");
256 		}
257 		return vals;
258 	}

281 	private static String getOwner(Map<String, Object> kwargs)
282 			throws TaskException {
283 		var owner = Objects.toString(kwargs.get("owner"), "").strip();
284 		if (owner.isBlank()) {
285 			throw new TaskException(
286 					"invalid owner identifier; must be non-empty");
287 		}
288 		if (!isAsciiPrintable(owner)) {
289 			throw new TaskException(
290 					"invalid owner identifier; must be printable ASCII");
291 		}
292 		return owner;
293 	}

295 	private Optional<Integer> createJob(SpallocAPI.CreateDescriptor create,
296 			Map<String, Object> kwargs, byte[] cmd) throws TaskException {
297 		var owner = getOwner(kwargs);
298 		var keepalive = parseKeepalive((Number) kwargs.get("keepalive"));
299 		var machineName = (String) kwargs.get("machine");
300 		var ts = tags(kwargs.get("tags"), isNull(machineName));
301 		var result = permit.authorize(() -> spalloc.createJob(permit.name,
302 				groupName, create, machineName, ts, keepalive, cmd));
303 		result.ifPresent(
304 				j -> log.info(
305 						"made compatibility-mode job {} "
306 								+ "on behalf of claimed user {}",
307 						j.getId(), owner));
308 		return result.map(Job::getId);
309 	}

343 		private JobDescription[] listJobs(V1TaskImpl task) {
344 			// Messy; hits the database many times
345 			return spalloc.getJobs(false, LOTS, 0).jobs().stream()
346 					.map(job -> buildJobDescription(task,
347 							// NB: convert partial job description to full
348 							spalloc.getJob(task.permit, job.getId())
349 									.orElseThrow(IllegalStateException::new)))
350 					.collect(collectToArray(JobDescription[]::new));
351 		}
352 
353 		private static JobDescription buildJobDescription(V1TaskImpl task,
354 				Job job) {
355 			var jd = new JobDescription.Builder();
356 			jd.setJobID(job.getId());
357 			jd.setOwner("");
358 			jd.setKeepAlive(timestamp(job.getKeepaliveTimestamp()));
359 			jd.setKeepAliveHost(job.getKeepaliveHost().orElse(""));
360 			jd.setReason(job.getReason().orElse(""));
361 			jd.setStartTime(timestamp(job.getStartTime()));
362 			jd.setState(state(job));
363 			job.getOriginalRequest().map(task::parseCommand).ifPresent(cmd -> {
364 				// In order to get here, this must be safe
365 				// Validation was when job was created
366 				@SuppressWarnings({ "unchecked", "rawtypes" })
367 				List<Integer> args = (List) cmd.getArgs();
368 				jd.setArgs(args);
369 				jd.setKwargs(cmd.getKwargs());
370 				// Override shrouded owner from above
371 				jd.setOwner(cmd.getKwargs().get("owner").toString());
372 			});
373 			job.getMachine().ifPresent(sm -> {
374 				jd.setMachine(sm.getMachine().getName());
375 				jd.setBoards(sm.getBoards());
376 				jd.setPower(sm.getPower() == ON);
377 			});
378 			return jd.build();
379 		}

388 		private static Machine buildMachineDescription(SpallocAPI.Machine m) {
389 			return new Machine(m.getName(), List.copyOf(m.getTags()),

392 							.collect(toList()),

408 	@Override
409 	protected final void notifyJob(Integer jobId, boolean wantNotify)
410 			throws TaskException {
411 		if (nonNull(jobId)) {
412 			var job = getJob(jobId);
413 			job.access(host());
414 		}
415 		manageNotifier(jobNotifiers, jobId, wantNotify, () -> {
416 			var actual = permit.authorize(() -> {
417 				spalloc.getJobs(false, LOTS, 0).waitForChange(
418 						mainProps.getCompat().getNotifyWaitTime());
419 				return spalloc.getJobs(false, LOTS, 0).ids();
420 			});
421 			if (nonNull(jobId)) {
422 				actual.retainAll(List.of(jobId));
423 			}
424 			writeJobNotification(actual);
425 		});
426 	}

428 	@Override
429 	protected final void notifyMachine(String machine, boolean wantNotify)
430 			throws TaskException {
431 		if (nonNull(machine)) {
432 			// Validate
433 			getMachine(machine);
434 		}
435 		manageNotifier(machNotifiers, machine, wantNotify, () -> {
436 			epochs.getMachineEpoch()
437 					.waitForChange(mainProps.getCompat().getNotifyWaitTime());
438 			var actual = new ArrayList<>(permit
439 					.authorize(() -> spalloc.getMachines(false)).keySet());
440 			if (nonNull(machine)) {
441 				actual.retainAll(List.of(machine));
442 			}
443 			writeMachineNotification(actual);
444 		});
445 	}

459 			Integer p, String description) {
460 		HasChipLocation locus;
461 		if (nonNull(x) && nonNull(y)) {
462 			if (nonNull(p)) {
463 				locus = new CoreLocation(x, y, p);
464 			} else {
465 				locus = new ChipLocation(x, y);
466 			}
467 		} else {
468 			locus = null;
469 		}
470 		permit.authorize(() -> {
471 			spalloc.reportProblem(address, locus, description, permit);
472 			return this; // Ignored value
473 		});

539 	private <T> void manageNotifier(Map<T, Future<Void>> notifiers, T key,
540 			boolean wantNotify, Notifier notifier) {
541 		if (wantNotify) {
542 			if (!notifiers.containsKey(key)) {
543 				notifiers.put(key,
544 						getExecutor().submit(Notifier.toCallable(notifier)));
545 			}
546 		} else {
547 			var n = notifiers.remove(key);
548 			if (nonNull(n)) {
549 				n.cancel(true);
550 			}
551 		}
