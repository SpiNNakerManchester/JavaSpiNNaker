169 	@Override
170 	protected final JobState getJobState(int jobId) throws TaskException {
171 		var job = getJob(jobId);
172 		job.access(host());
173 		var js = new JobState.Builder();
174 		js.setKeepalive(timestamp(job.getKeepaliveTimestamp()));
175 		js.setKeepalivehost(job.getKeepaliveHost().orElse(""));
176 		js.setPower(job.getMachine().map(m -> {
177 			try {
178 				return m.getPower() == ON;
179 			} catch (DataAccessException e) {
180 				log.warn("problem getting job power state", e);
181 				return false;
182 			}
183 		}).orElse(false));
184 		js.setReason(job.getReason().orElse(""));
185 		js.setStartTime(timestamp(job.getStartTime()));
186 		js.setState(state(job));
187 		return js.build();
188 	}

195 	/**
196 	 * Parse a value to get a keepalive duration.
197 	 *
198 	 * @param keepalive
199 	 *            The number to parse. May be {@code null} to get a default.
200 	 * @return The duration. Never {@code null}.
201 	 */
202 	private Duration parseKeepalive(Number keepalive) {
203 		if (isNull(keepalive)) {
204 			return mainProps.getCompat().getDefaultKeepalive();
205 		}
206 		var d = Duration.ofSeconds(keepalive.longValue());
207 		if (!(keepalive instanceof Double || keepalive instanceof Float)) {
208 			return d;
209 		}
210 		double fractionalPart = keepalive.doubleValue() - keepalive.longValue();
211 		return d.plusNanos((long) (fractionalPart * NSEC_PER_SEC));
212 	}

214 	/**
215 	 * Parse a value to get a list of machine tags to match.
216 	 *
217 	 * @param src
218 	 *            The value to parse.
219 	 * @param mayForceDefault
220 	 *            Whether we want to force a default value into the tags.
221 	 * @return The list of tags. Never {@code null}.
222 	 */
223 	private static List<String> tags(Object src, boolean mayForceDefault) {
224 		var vals = new ArrayList<String>();
225 		if (src instanceof List) {
226 			for (var o : (List<?>) src) {
227 				vals.add(Objects.toString(o));
228 			}
229 		} else if (src instanceof String) {
230 			vals.add((String) src);
231 		}
232 		if (vals.isEmpty() && mayForceDefault) {
233 			return List.of("default");
234 		}
235 		return vals;
236 	}

259 	private static String getOwner(Map<String, Object> kwargs)
260 			throws TaskException {
261 		var owner = Objects.toString(kwargs.get("owner"), "").strip();
262 		if (owner.isBlank()) {
263 			throw new TaskException(
264 					"invalid owner identifier; must be non-empty");
265 		}
266 		if (!isAsciiPrintable(owner)) {
267 			throw new TaskException(
268 					"invalid owner identifier; must be printable ASCII");
269 		}
270 		return owner;
271 	}

273 	private Optional<Integer> createJob(SpallocAPI.CreateDescriptor create,
274 			Map<String, Object> kwargs, byte[] cmd) throws TaskException {
275 		var owner = getOwner(kwargs);
276 		var keepalive = parseKeepalive((Number) kwargs.get("keepalive"));
277 		var machineName = (String) kwargs.get("machine");
278 		var ts = tags(kwargs.get("tags"), isNull(machineName));
279 		var result = permit.authorize(() -> spalloc.createJob(permit.name,
280 				groupName, create, machineName, ts, keepalive, cmd));
281 		result.ifPresent(
282 				j -> log.info(
283 						"made compatibility-mode job {} "
284 								+ "on behalf of claimed user {}",
285 						j.getId(), owner));
286 		return result.map(Job::getId);
287 	}

321 		private JobDescription[] listJobs(V1TaskImpl task) {
322 			// Messy; hits the database many times
323 			return spalloc.getJobs(false, LOTS, 0).jobs().stream()
324 					.map(job -> buildJobDescription(task,
325 							// NB: convert partial job description to full
326 							spalloc.getJob(task.permit, job.getId())
327 									.orElseThrow(IllegalStateException::new)))
328 					.collect(collectToArray(JobDescription[]::new));
329 		}
330 
331 		private static JobDescription buildJobDescription(V1TaskImpl task,
332 				Job job) {
333 			var jd = new JobDescription.Builder();
334 			jd.setJobID(job.getId());
335 			jd.setOwner("");
336 			jd.setKeepAlive(timestamp(job.getKeepaliveTimestamp()));
337 			jd.setKeepAliveHost(job.getKeepaliveHost().orElse(""));
338 			jd.setReason(job.getReason().orElse(""));
339 			jd.setStartTime(timestamp(job.getStartTime()));
340 			jd.setState(state(job));
341 			job.getOriginalRequest().map(task::parseCommand).ifPresent(cmd -> {
342 				// In order to get here, this must be safe
343 				// Validation was when job was created
344 				@SuppressWarnings({ "unchecked", "rawtypes" })
345 				List<Integer> args = (List) cmd.getArgs();
346 				jd.setArgs(args);
347 				jd.setKwargs(cmd.getKwargs());
348 				// Override shrouded owner from above
349 				jd.setOwner(cmd.getKwargs().get("owner").toString());
350 			});
351 			job.getMachine().ifPresent(sm -> {
352 				jd.setMachine(sm.getMachine().getName());
353 				jd.setBoards(sm.getBoards());
354 				jd.setPower(sm.getPower() == ON);
355 			});
356 			return jd.build();
357 		}

366 		private static Machine buildMachineDescription(SpallocAPI.Machine m) {

386 	@Override
387 	protected final void notifyJob(Integer jobId, boolean wantNotify)
388 			throws TaskException {
389 		if (nonNull(jobId)) {
390 			var job = getJob(jobId);
391 			job.access(host());
392 		}
393 		manageNotifier(jobNotifiers, jobId, wantNotify, () -> {
394 			var actual = permit.authorize(() -> {
395 				spalloc.getJobs(false, LOTS, 0).waitForChange(
396 						mainProps.getCompat().getNotifyWaitTime());
397 				return spalloc.getJobs(false, LOTS, 0).ids();
398 			});
399 			if (nonNull(jobId)) {
400 				actual.retainAll(List.of(jobId));
401 			}
402 			writeJobNotification(actual);
403 		});
404 	}

406 	@Override
407 	protected final void notifyMachine(String machine, boolean wantNotify)
408 			throws TaskException {
409 		if (nonNull(machine)) {
410 			// Validate
411 			getMachine(machine);
412 		}
413 		manageNotifier(machNotifiers, machine, wantNotify, () -> {
414 			epochs.getMachineEpoch()
415 					.waitForChange(mainProps.getCompat().getNotifyWaitTime());
416 			var actual = new ArrayList<>(permit
417 					.authorize(() -> spalloc.getMachines(false)).keySet());
418 			if (nonNull(machine)) {
419 				actual.retainAll(List.of(machine));
420 			}
421 			writeMachineNotification(actual);
422 		});
423 	}

517 	private <T> void manageNotifier(Map<T, Future<Void>> notifiers, T key,
518 			boolean wantNotify, Notifier notifier) {
519 		if (wantNotify) {
520 			if (!notifiers.containsKey(key)) {
521 				notifiers.put(key,
522 						getExecutor().submit(Notifier.toCallable(notifier)));
523 			}
524 		} else {
525 			var n = notifiers.remove(key);
526 			if (nonNull(n)) {
527 				n.cancel(true);
528 			}
529 		}
