41  /**
42   * Manages user quotas.
43   *
44   * @author Donal Fellows
45   */
46  @Service
47  public class QuotaManager extends DatabaseAwareBean {
48  	private static final Logger log = getLogger(QuotaManager.class);
49  
50  	@Autowired
51  	private ServiceMasterControl control;
52  
53  	/**
54  	 * Can the user (in a specific group) create another job at this point? If
55  	 * not, they're currently out of resources.
56  	 *
57  	 * @param groupId
58  	 *            What group will the job be accounted against.
59  	 * @return True if they can make a job. False if they can't.
60  	 */
61  	public boolean mayCreateJob(int groupId) {
62  		try (var sql = new CreateCheckSQL()) {
63  			return sql.transactionRead(() -> sql.mayCreateJob(groupId));
64  		}
65  	}
66  
67  	private class CreateCheckSQL extends AbstractSQL {
68  		// TODO These should be combined (but one is an aggregate so...)
69  		private final Query getQuota = conn.query(GET_GROUP_QUOTA);
70  
71  		private final Query getCurrentUsage = conn.query(GET_CURRENT_USAGE);
72  
73  		@Override
74  		public void close() {
75  			getQuota.close();
76  			getCurrentUsage.close();
77  			super.close();
78  		}
79  
80  		private boolean mayCreateJob(int groupId) {
81  			return getQuota.call1(result -> {
82  				var quota = result.getInteger("quota");
83  				if (isNull(quota)) {
84  					return true;
85  				}
86  				// Quota is defined; check if current usage exceeds it
87  				int usage = getCurrentUsage.call1(
88  						integer("current_usage"), groupId).orElse(0);
89  				// If board-seconds are left, we're good to go
90  				return (quota > usage);
91  			}, groupId).orElse(true);
92  		}
93  	}
94  
95  	/**
96  	 * Has the execution of a job remained within its group's resource
97  	 * allocation at this point?
98  	 *
99  	 * @param jobId
100 	 *            What job is consuming resources?
101 	 * @return True if the job can continue to run. False if it can't.
102 	 */
103 	public boolean mayLetJobContinue(int jobId) {
104 		try (var sql = new ContinueCheckSQL()) {
105 			return sql.transactionRead(() -> sql.mayLetJobContinue(jobId));
106 		}
107 	}
108 
109 	/**
110 	 * Has the execution of a job exceeded its group's resource allocation at
111 	 * this point?
112 	 *
113 	 * @param jobId
114 	 *            What job is consuming resources?
115 	 * @return False if the job should be killed. True otherwise.
116 	 */
117 	public boolean shouldKillJob(int jobId) {
118 		return !mayLetJobContinue(jobId);
119 	}
120 
121 	private class ContinueCheckSQL extends AbstractSQL {
122 		private final Query getUsageAndQuota =
123 				conn.query(GET_JOB_USAGE_AND_QUOTA);
124 
125 		@Override
126 		public void close() {
127 			getUsageAndQuota.close();
128 			super.close();
129 		}
130 
131 		private boolean mayLetJobContinue(int jobId) {
132 			return getUsageAndQuota.call1(
133 					// If we have an entry, check if usage <= quota
134 					row -> row.getInt("quota_used") <= row.getInt("quota"),
135 					jobId)
136 
137 					// Otherwise, we'll just allow it
138 					.orElse(true);
139 		}
140 	}
141 
142 	/**
143 	 * Adjust a group's quota.
144 	 *
145 	 * @param groupId
146 	 *            Which group's quota to change
147 	 * @param delta
148 	 *            Amount to change by, in board-seconds
149 	 * @return Information about what group's quota was adjusted and what it has
150 	 *         become.
151 	 */
152 	public Optional<AdjustedQuota> addQuota(int groupId, int delta) {
153 		try (var sql = new AdjustQuotaSQL()) {
154 			return sql.transaction(() -> sql.adjustQuota(groupId, delta));
155 		}
156 	}
157 
158 	/**
159 	 * Describes the result of the {@link QuotaManager#addQuota(int,int)}
160 	 * operation.
161 	 *
162 	 * @author Donal Fellows
163 	 */
164 	public static final class AdjustedQuota {
165 		private final String name;
166 
167 		private final Long quota;
168 
169 		private AdjustedQuota(Row row) {
170 			this.name = row.getString("group_name");
171 			this.quota = row.getLong("quota");
172 		}
173 
174 		/** @return The name of the group. */
175 		public String getName() {
176 			return name;
177 		}
178 
179 		/** @return The new quota of the group. */
180 		public Long getQuota() {
181 			return quota;
182 		}
183 	}
184 
185 	private class AdjustQuotaSQL extends AbstractSQL {
186 		private final Update adjustQuota = conn.update(ADJUST_QUOTA);
187 
188 		private final Query getQuota = conn.query(GET_GROUP_QUOTA);
189 
190 		@Override
191 		public void close() {
192 			adjustQuota.close();
193 			super.close();
194 		}
195 
196 		private Optional<AdjustedQuota> adjustQuota(int groupId, int delta) {
197 			if (adjustQuota.call(delta, groupId) == 0) {
198 				return Optional.empty();
199 			}
200 			return getQuota.call1(AdjustedQuota::new, groupId);
201 		}
202 	}
203 
204 	/**
205 	 * Consolidates usage from finished jobs onto quotas. Runs hourly.
206 	 */
207 	@Scheduled(cron = "#{quotaProperties.consolidationSchedule}")
208 	public void consolidateQuotas() {
209 		if (control.isPaused()) {
210 			return;
211 		}
212 		// Split off for testability
213 		try (var c = getConnection()) {
214 			doConsolidate(c);
215 		} catch (DataAccessException e) {
216 			if (isBusy(e)) {
217 				log.info("database is busy; "
218 						+ "will try job quota consolidation processing later");
219 				return;
220 			}
221 			throw e;
222 		}
223 	}
224 
225 	private void doConsolidate(Connection c) {
226 		try (var sql = new ConsolidateSQL(c)) {
227 			sql.transaction(sql::consolidate);
228 		}
229 	}
230 
231 	private class ConsolidateSQL extends AbstractSQL {
232 		private final Query getConsoldationTargets =
233 				conn.query(GET_CONSOLIDATION_TARGETS);
234 
235 		private final Update decrementQuota = conn.update(DECREMENT_QUOTA);
236 
237 		private final Update markConsolidated = conn.update(MARK_CONSOLIDATED);
238 
239 		ConsolidateSQL(Connection c) {
240 			super(c);
241 		}
242 
243 		@Override
244 		public void close() {
245 			getConsoldationTargets.close();
246 			decrementQuota.close();
247 			markConsolidated.close();
248 			super.close();
249 		}
250 
251 		// Result is arbitrary and ignored
252 		private Void consolidate() {
253 			for (var target : getConsoldationTargets.call(Target::new)) {
254 				decrementQuota.call(target.quotaUsed, target.groupId);
255 				markConsolidated.call(target.jobId);
256 			}
257 			return null;
258 		}
259 
260 		private class Target {
261 			Object quotaUsed;
262 
263 			int groupId;
264 
265 			int jobId;
266 
267 			Target(Row row) {
268 				quotaUsed = row.getObject("quota_used");
269 				groupId = row.getInt("group_id");
270 				jobId = row.getInt("job_id");
271 			}
272 		}
273 	}
