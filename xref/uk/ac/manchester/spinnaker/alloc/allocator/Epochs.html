43  /**
44   * Manages waiting for values.
45   *
46   * @author Donal Fellows
47   * @author Andrew Rowley
48   */
49  @Service
50  public class Epochs {
51  	private static final Logger log = getLogger(Epochs.class);
52  
53  	/** How long to wait between cleaning of the maps. */
54  	private static final Duration CLEANING_SCHEDULE = Duration.ofSeconds(30);
55  
56  	@Autowired
57  	private TaskScheduler scheduler;
58  
59  	/**
60  	 * Mapping from job ID to set of epoch handles waiting on it for a state
61  	 * change.
62  	 */
63  	private final EpochMap jobs = new EpochMap();
64  
65  	/**
66  	 * Mapping from machine ID to set of epoch handles waiting on it for a state
67  	 * change.
68  	 */
69  	private final EpochMap machines = new EpochMap();
70  
71  	/**
72  	 * Mapping from board ID to set of epoch handles waiting on it for blacklist
73  	 * handling.
74  	 */
75  	private final EpochMap blacklists = new EpochMap();
76  
77  	@PostConstruct
78  	private void init() {
79  		scheduler.scheduleAtFixedRate(this::checkEmpties, CLEANING_SCHEDULE);
80  	}
81  
82  	/**
83  	 * The maps contain weak reference maps, where Epochs are removed when they
84  	 * are no longer referenced, but the map will still contain the empty weak
85  	 * reference map unless removed. This tasklet handles that cleanup.
86  	 */
87  	private void checkEmpties() {
88  		if (jobs.checkEmptyValues()) {
89  			log.debug("Job map now contains jobs {}", jobs.getIds());
90  		}
91  		if (machines.checkEmptyValues()) {
92  			log.debug("Machine map now contains machines {}",
93  					machines.getIds());
94  		}
95  		if (blacklists.checkEmptyValues()) {
96  			log.debug("Blacklist map now contains boards {}",
97  					blacklists.getIds());
98  		}
99  	}

192 	/**
193 	 * A waitable epoch checkpoint.
194 	 *
195 	 * @author Donal Fellows
196 	 * @author Andrew Rowley
197 	 */
198 	public final class Epoch {
199 		private final EpochMap map;
200 
201 		private final List<Integer> ids;
202 
203 		private final Set<Integer> changed = new HashSet<>();
204 
205 		private Epoch(EpochMap map, int id) {
206 			this.map = map;
207 			this.ids = List.of(id);
208 			map.addAll(this, ids);
209 		}
210 
211 		private Epoch(EpochMap map, List<Integer> ids) {
212 			if (ids.isEmpty()) {
213 				log.warn("empty ID list; will never wake");
214 			}
215 			this.map = map;
216 			this.ids = List.copyOf(ids);
217 			map.addAll(this, ids);
218 		}
219 
220 		synchronized void updateChanged(int id) {
221 			log.debug("Change to {}, id {}", this, id);
222 			changed.add(id);
223 			notifyAll();
224 		}
225 
226 		/**
227 		 * Wait, for up to {@code timeout}, for a change.
228 		 *
229 		 * @param timeout
230 		 *            The time to wait, in milliseconds.
231 		 * @return Whether the item has changed or not.
232 		 * @throws InterruptedException
233 		 *             If the wait is interrupted.
234 		 */
235 		public boolean waitForChange(Duration timeout)
236 				throws InterruptedException {
237 			return !getChanged(timeout).isEmpty();
238 		}
239 
240 		/**
241 		 * Get the set of changed items.
242 		 *
243 		 * @param timeout
244 		 *            The timeout to wait for one item to change.
245 		 * @return The changed items.
246 		 * @throws InterruptedException
247 		 *             If the wait is interrupted.
248 		 */
249 		public Collection<Integer> getChanged(Duration timeout)
250 				throws InterruptedException {
251 			synchronized (this) {
252 				log.debug("Waiting for change to {}", this);
253 				wait(timeout.toMillis());
254 				log.debug("After wait, changed: {}", changed);
255 				return Set.copyOf(changed);
256 			}
257 		}
258 
259 		/**
260 		 * Check if this epoch is the current one.
261 		 *
262 		 * @return Whether this is a valid epoch.
263 		 */
264 		public boolean isValid() {
265 			return map.containsAnyKey(ids);
266 		}
267 	}
268 }

270 /**
271  * A weak mapping from an ID to the epoch handles that care about it. Handles
272  * will be removed when they get garbage collected.
273  *
274  * @author Donal Fellows
275  */
276 class EpochMap {
277 	/** The value in {@link #map} leaves. Shared. Unimportant. */
278 	private static final Object OBJ = new Object();
279 
280 	/** A map from integers to weak sets of epochs. */
281 	@GuardedBy("this")
282 	private final Map<Integer, Map<Epochs.Epoch, Object>> map = new HashMap<>();
283 
284 	synchronized boolean checkEmptyValues() {
285 		return map.entrySet().removeIf(entry -> entry.getValue().isEmpty());
286 	}
287 
288 	void changed(int id) {
289 		var items = getSet(id);
290 		if (nonNull(items)) {
291 			synchronized (items) {
292 				for (var item : items) {
293 					item.updateChanged(id);
294 				}
295 			}
296 		}
297 	}
298 
299 	/**
300 	 * Take the set of epochs for a particular ID.
301 	 *
302 	 * @param id
303 	 *            The key into the map.
304 	 * @return The removed set of epochs. Empty if the key is absent.
305 	 */
306 	@UsedInJavadocOnly({
307 		BiConsumer.class, ConcurrentModificationException.class
308 	})
309 	private synchronized Set<Epochs.Epoch> getSet(Integer id) {
310 		var weakmap = map.get(id);
311 		if (weakmap == null) {
312 			return null;
313 		}
314 		// Copy the set here while still synchronized to avoid
315 		// ConcurrentModificationException when updated elsewhere.
316 		return Set.copyOf(weakmap.keySet());
317 	}
318 
319 	synchronized void addAll(Epochs.Epoch epoch, List<Integer> ids) {
320 		for (var id : ids) {
321 			map.computeIfAbsent(id, key -> new WeakHashMap<>()).put(epoch, OBJ);
322 		}
323 	}
324 
325 	@SuppressWarnings("GuardedBy")
326 	synchronized boolean containsAnyKey(Collection<Integer> ids) {
327 		return ids.stream().allMatch(map::containsKey);
328 	}
329 
330 	synchronized Collection<Integer> getIds() {
331 		return map.keySet();
332 	}
333 }
