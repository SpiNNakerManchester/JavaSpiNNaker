151 	/**
152 	 * Describe the enable state of a board.
153 	 *
154 	 * @param machineName
155 	 *            The name of the machine; required
156 	 * @param x
157 	 *            The X coordinate, used to identify a board by triad coords
158 	 * @param y
159 	 *            The Y coordinate, used to identify a board by triad coords
160 	 * @param z
161 	 *            The Z coordinate, used to identify a board by triad coords
162 	 * @param c
163 	 *            The cabinet number, used to identify a board physically
164 	 * @param f
165 	 *            The frame number, used to identify a board physically
166 	 * @param b
167 	 *            The board number, used to identify a board physically
168 	 * @param address
169 	 *            The IP address of the board (dotted quad)
170 	 * @return Whether the board is enabled
171 	 * @throws RequestFailedException
172 	 *             If bad query combinations are supplied
173 	 */
174 	@GET
175 	@Produces(TEXT_PLAIN)
176 	@Path(BOARD)
177 	default boolean getBoardState(
178 			@NotBlank(message = "machine name is required")
179 			@QueryParam("machine") String machineName,
180 			@QueryParam("x") Integer x, @QueryParam("y") Integer y,
181 			@QueryParam("z") Integer z, @QueryParam("cabinet") Integer c,
182 			@QueryParam("frame") Integer f, @QueryParam("board") Integer b,
183 			@QueryParam("address") String address) {
184 		if (nonNull(x) && nonNull(y) && nonNull(z)) {
185 			return getBoardStateXYZ(machineName, x, y, z);
186 		}
187 		if (nonNull(c) && nonNull(f) && nonNull(b)) {
188 			return getBoardStateCFB(machineName, c, f, b);
189 		}
190 		if (nonNull(address)) {
191 			return getBoardStateAddress(machineName, address);
192 		}
193 		throw new RequestFailedException(BAD_REQUEST,
194 				"You need to supply coordinates (as "
195 						+ "x=&y=&z=, c=&f=&b= or address=)");
196 	}

241 	/**
242 	 * Enable or disable a board.
243 	 *
244 	 * @param machineName
245 	 *            The name of the machine
246 	 * @param x
247 	 *            The X coordinate, used to identify a board by triad coords
248 	 * @param y
249 	 *            The Y coordinate, used to identify a board by triad coords
250 	 * @param z
251 	 *            The Z coordinate, used to identify a board by triad coords
252 	 * @param c
253 	 *            The cabinet number, used to identify a board physically
254 	 * @param f
255 	 *            The frame number, used to identify a board physically
256 	 * @param b
257 	 *            The board number, used to identify a board physically
258 	 * @param address
259 	 *            The IP address of the board (dotted quad)
260 	 * @param enabled
261 	 *            Whether the board should be set to the enabled state
262 	 * @return Whether the board is enabled
263 	 * @throws RequestFailedException
264 	 *             If bad query combinations are supplied
265 	 */
266 	@PUT
267 	@Consumes(TEXT_PLAIN)
268 	@Produces(TEXT_PLAIN)
269 	@Path(BOARD)
270 	default boolean setBoardState(
271 			@NotBlank(message = "machine name is required")
272 			@QueryParam("machine") String machineName,
273 			@QueryParam("x") Integer x, @QueryParam("y") Integer y,
274 			@QueryParam("z") Integer z, @QueryParam("cabinet") Integer c,
275 			@QueryParam("frame") Integer f, @QueryParam("board") Integer b,
276 			@QueryParam("address") String address, boolean enabled) {
277 		if (nonNull(x) && nonNull(y) && nonNull(z)) {
278 			return setBoardStateXYZ(machineName, x, y, z, enabled);
279 		}
280 		if (nonNull(c) && nonNull(f) && nonNull(b)) {
281 			return setBoardStateCFB(machineName, c, f, b, enabled);
282 		}
283 		if (nonNull(address)) {
284 			return setBoardStateAddress(machineName, address, enabled);
285 		}
286 		throw new RequestFailedException(BAD_REQUEST,
287 				"You need to supply coordinates (as "
288 						+ "x=&y=&z=, c=&f=&b= or address=)");
289 	}
