16  package uk.ac.manchester.spinnaker.alloc.security;
17  
18  import static java.util.Objects.isNull;
19  import static org.springframework.beans.factory.config.BeanDefinition.ROLE_INFRASTRUCTURE;
20  
21  import java.util.Collection;
22  import java.util.Map;
23  import java.util.Optional;
24  import java.util.stream.Stream;
25  
26  import org.springframework.context.annotation.Role;
27  import org.springframework.expression.EvaluationContext;
28  import org.springframework.expression.Expression;
29  import org.springframework.security.access.expression.method.DefaultMethodSecurityExpressionHandler;
30  import org.springframework.security.access.prepost.PostFilter;
31  import org.springframework.stereotype.Component;
32  
33  import uk.ac.manchester.spinnaker.utils.UsedInJavadocOnly;
34  
35  /**
36   * Because Spring otherwise can't apply {@link PostFilter} to {@link Optional}.
37   *
38   * @author Donal Fellows
39   * @see <a href="https://stackoverflow.com/q/66107075/301832">Stack Overflow</a>
40   */
41  @Component
42  @Role(ROLE_INFRASTRUCTURE)
43  @UsedInJavadocOnly(PostFilter.class)
44  class AnyTypeMethodSecurityExpressionHandler
45  		extends DefaultMethodSecurityExpressionHandler {
46  	@Override
47  	public Object filter(Object target, Expression expr,
48  			EvaluationContext ctx) {
49  		if (isNull(target)) {
50  			// We can handle this case here!
51  			return null;
52  		}
53  		if (target instanceof Collection || target.getClass().isArray()
54  				|| target instanceof Map || target instanceof Stream) {
55  			return super.filter(target, expr, ctx);
56  		}
57  		if (target instanceof Optional) {
58  			return filterOptional((Optional<?>) target, expr, ctx);
59  		} else {
60  			return filterOptional(Optional.of(target), expr, ctx).orElse(null);
61  		}
62  	}
63  
64  	@SuppressWarnings("unchecked")
65  	private <T> Optional<T> filterOptional(Optional<T> target, Expression expr,
66  			EvaluationContext ctx) {
67  		return ((Stream<T>) super.filter(target.stream(), expr, ctx))
68  				.findFirst();
69  	}
70  }
