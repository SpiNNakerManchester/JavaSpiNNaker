124 	private Optional<UserRecord> makeInitUser(String username) {
125 		if (!authProps.isAddDummyUser()) {
126 			return Optional.empty();
127 		}
128 		var pass = DUMMY_PASSWORD;
129 		boolean poorPassword = true;
130 		if (authProps.isDummyRandomPass()) {
131 			pass = passServices.generatePassword();
132 			poorPassword = false;
133 		}
134 		if (!createUser(username, pass, ADMIN)) {
135 			// User creation failed; probably already exists, which is OK
136 			return Optional.empty();
137 		}
138 		if (authProps.isDummyRandomPass()) {
139 			log.info("admin user {} has password: {}", username, pass);
140 		}
141 		if (poorPassword) {
142 			log.warn("user {} has default password!", username);
143 		}
144 		return Optional.of(userController.getUser(username, null)
145 				.orElseThrow(() -> new SetupException(
146 						"default user was created, yet wasn't found!")));
147 	}

149 	private Optional<GroupRecord> makeInitGroup(String groupname) {
150 		if (groupname.isBlank()) {
151 			// No system group name, so ignore group setup
152 			return Optional.empty();
153 		}
154 		var template = new GroupRecord();
155 		template.setGroupName(groupname);
156 		// What should the default quota be here? Using no-quota-at-all for now
157 		template.setQuota(null);
158 		try {
159 			return userController.createGroup(template, INTERNAL).map(g -> {
160 				log.info("system group '{}' created", groupname);
161 				return g;
162 			});
163 		} catch (DataIntegrityViolationException e) {
164 			if (e.getMessage().contains("A UNIQUE constraint failed")) {
165 				// Already exists; no big deal
166 				return Optional.empty();
167 			} else if (e.getMessage().contains("Duplicate entry")) {
168 				return Optional.empty();
169 			}
170 			throw e;
171 		}
172 	}

199 	@Override
200 	@PreAuthorize(IS_ADMIN)
201 	public boolean createUser(String username, String password,
202 			TrustLevel trustLevel) {
203 		var name = username.strip();
204 		if (name.isBlank()) {
205 			// Won't touch the DB if the username is empty
206 			throw new UsernameNotFoundException("empty user name?");
207 		}
208 		var encPass = passServices.encodePassword(password);
209 		try (var sql = new AuthQueries()) {
210 			return sql.transaction(
211 					() -> sql.createUser(username, encPass, trustLevel, null)
212 							.isPresent());
213 		}

216 	/** Marks an authentication that we've already taken a decision about. */
217 	private interface AlreadyDoneMarker extends Authentication {
218 	}
219 
220 	@Override
221 	public Authentication authenticate(Authentication auth)
222 			throws AuthenticationException {
223 		if (isNull(auth)) {
224 			return null;
225 		}
226 		log.debug("requesting auth decision about {}", auth);
227 		if (auth instanceof AlreadyDoneMarker) {
228 			// It's ours; don't repeat the auth
229 			return auth;
230 		}
231 
232 		try {
233 			if (auth instanceof UsernamePasswordAuthenticationToken) {
234 				return authenticateDirect(
235 						(UsernamePasswordAuthenticationToken) auth);
236 			} else if (auth instanceof OAuth2AuthenticationToken) {
237 				/*
238 				 * Technically, at this point we're already authenticated as
239 				 * we've checked that the token from Keycloak is valid. We still
240 				 * have to take an authorization decision though.
241 				 */
242 				var user = ((OAuth2AuthenticationToken) auth).getPrincipal();
243 				return authorizeOpenId(
244 						authProps.getOpenid().getUsernamePrefix()
245 								+ user.getAttribute(PREFERRED_USERNAME),
246 						user.getAttribute(SUB), new OriginatingCredential(user),
247 						auth.getAuthorities());
248 			} else if (auth instanceof BearerTokenAuthentication) {
249 				/*
250 				 * Technically, at this point we're already authenticated as
251 				 * we've checked that the token from Keycloak is valid. We still
252 				 * have to take an authorization decision though.
253 				 */
254 				var bearerAuth = (BearerTokenAuthentication) auth;
255 				var token = bearerAuth.getToken();
256 				return authorizeOpenId(
257 						authProps.getOpenid().getUsernamePrefix()
258 								+ bearerAuth.getTokenAttributes().get(
259 										PREFERRED_USERNAME),
260 						bearerAuth.getName(), new OriginatingCredential(token),
261 						auth.getAuthorities());
262 			} else {
263 				return null;
264 			}
265 		} catch (DataAccessException e) {
266 			throw new InternalAuthenticationServiceException(
267 					"database access problem", e);
268 		}
269 	}
270 
271 	@Override
272 	public Authentication updateAuthentication(HttpServletRequest req,
273 			SecurityContext ctx) {
274 		var current = ctx.getAuthentication();
275 		if (nonNull(current)) {
276 			if (supports(current.getClass())) {
277 				var updated = authenticate(current);
278 				if (nonNull(updated) && updated != current) {
279 					log.debug("filter updated security from {} to {}", current,
280 							updated);
281 					ctx.setAuthentication(updated);
282 					return updated;
283 				}
284 			} else if (!isUnsupportedAuthTokenClass(current.getClass())) {
285 				log.warn("unexpected authentication type {} (token: {})"
286 						+ "on request {}",
287 						current.getClass(), current, req.getRequestURI());
288 			}
289 		}
290 		return null;
291 	}

295 		UsernamePasswordAuthenticationToken.class,
296 		OAuth2AuthenticationToken.class, BearerTokenAuthentication.class,

300 	/** The classes that we <em>know</em> we don't ever want to handle. */
301 	private static final Class<?>[] UNSUPPORTED_AUTH_TOKEN_CLASSES = {
302 		AnonymousAuthenticationToken.class, RememberMeAuthenticationToken.class,
303 		RunAsUserToken.class, TestingAuthenticationToken.class
304 	};
305 
306 	@Override
307 	public boolean supports(Class<?> cls) {
308 		for (var c : SUPPORTED_AUTH_TOKEN_CLASSES) {
309 			if (c.isAssignableFrom(cls)) {
310 				return true;
311 			}
312 		}
313 		if (!isUnsupportedAuthTokenClass(cls)) {
314 			log.warn("asked about supporting {}", cls);
315 		}
316 		return false;
317 	}

319 	/**
320 	 * Check if we're talking about a class that we know we don't ever want to
321 	 * handle.
322 	 *
323 	 * @param cls
324 	 *            The class to check.
325 	 * @return {@code true} if we never want to take a full auth decision about
326 	 *         it.
327 	 */
328 	static boolean isUnsupportedAuthTokenClass(Class<?> cls) {
329 		for (var c : UNSUPPORTED_AUTH_TOKEN_CLASSES) {
330 			if (c.isAssignableFrom(cls)) {
331 				return true;
332 			}
333 		}
334 		return false;
335 	}

349 	 * Do authorization mapping for users coming in with a direct username and
350 	 * password.
351 	 *
352 	 * @param auth
353 	 *            The credentials they presented
354 	 * @return The remapped auth token, or {@code null} if we are throwing
355 	 *         things out at this stage.
356 	 * @throws AuthenticationException
357 	 *             If something is badly wrong.
358 	 */
359 	private Authentication authenticateDirect(
360 			UsernamePasswordAuthenticationToken auth)
361 			throws AuthenticationException {
362 		log.info("authenticating Local Login {}", auth.toString());
363 		// We ALWAYS trim the username; extraneous whitespace is bogus
364 		var name = auth.getName().strip();
365 		if (name.isBlank()) {
366 			// Won't touch the DB if the username is empty
367 			throw new UsernameNotFoundException("empty user name?");
368 		}
369 		var password = auth.getCredentials().toString();
370 		var authorities = new ArrayList<GrantedAuthority>();
371 
372 		try (var queries = new AuthQueries()) {
373 			if (!authLocalAgainstDB(name, password, authorities, queries)) {
374 				return null;
375 			}
376 		}
377 		return new PerformedUsernamePasswordAuthenticationToken(name, password,
378 				authorities);
379 	}

381 	/**
382 	 * Do authorization mapping for users coming in via OpenID Connect from
383 	 * HBP/EBRAINS.
384 	 *
385 	 * @param name
386 	 *            The name of the user <em>with standard namespacing prefix
387 	 *            added</em>.
388 	 * @param subject
389 	 *            The OpenID subject.
390 	 * @param credential
391 	 *            The reason we believe the user should be allowed to use the
392 	 *            service.
393 	 * @param authorities
394 	 *            The claimed authorities derived from the HBP OpenID service.
395 	 *            Interesting because they include claims about organisation and
396 	 *            collabratory membership.
397 	 * @return The remapped auth token, or {@code null} if we are throwing
398 	 *         things out at this stage.
399 	 * @throws AuthenticationException
400 	 *             If something is badly wrong.
401 	 */
402 	private Authentication authorizeOpenId(String name, String subject,
403 			OriginatingCredential credential,
404 			Collection<? extends GrantedAuthority> authorities)
405 			throws AuthenticationException {
406 		log.debug("authenticating OpenID {}", credential);
407 		if (isNull(name)
408 				|| name.equals(authProps.getOpenid().getUsernamePrefix())) {
409 			// No actual name there?
410 			throw new UsernameNotFoundException("empty user name?");
411 		}
412 		try (var queries = new AuthQueries()) {
413 			if (!queries.transaction(() -> authOpenIDAgainstDB(name, subject,
414 					authorities, queries))) {
415 				return null;
416 			}
417 		} catch (RuntimeException e) {
418 			log.warn("serious problem when processing login for OpenID user {}",
419 					name, e);
420 			return null;
421 		}
422 		// Users from OpenID always have the same permissions
423 		return new OpenIDDerivedAuthenticationToken(name, credential);
424 	}

426 	/** Holds either a {@link OAuth2User} or a {@link Jwt}. */
427 	private static final class OriginatingCredential {
428 		private final OAuth2User user;
429 
430 		private final OAuth2AccessToken token;
431 
432 		OriginatingCredential(OAuth2User user) {
433 			this.user = requireNonNull(user);
434 			this.token = null;
435 		}
436 
437 		OriginatingCredential(OAuth2AccessToken token) {
438 			this.user = null;
439 			this.token = requireNonNull(token);
440 		}
441 
442 		@Override
443 		public String toString() {
444 			if (nonNull(user)) {
445 				return user.toString();
446 			} else {
447 				return token.toString();
448 			}
449 		}
450 	}

610 	/**
611 	 * Collection of queries for setting the group memberships of a user. These
612 	 * are queries that would be part of {@link AuthQueries} except for the
613 	 * dance of handling temporary database tables.
614 	 * <p>
615 	 * Use this class by calling {@link #define(String,GroupType)} several
616 	 * times, and then calling {@link #apply(int)} once. Closing this resource
617 	 * removes temporary storage; it does not support nested use.
618 	 *
619 	 * @author Donal Fellows
620 	 */
621 	private final class GroupSynch extends AbstractSQL {
622 		private final Update insert;
623 
624 		private final Update add;
625 
626 		private final Update remove;
627 
628 		GroupSynch(AuthQueries sql) {
629 			super(sql.getConnection());
630 			try (var make = conn.update(GROUP_SYNC_MAKE_TEMP_TABLE)) {
631 				make.call();
632 			} catch (BadSqlGrammarException e) {
633 				if (!e.getMessage().contains("already exists")) {
634 					throw e;
635 				}
636 				log.warn("Group temp table already exists ("
637 						+ e.getMessage() + ")");
638 			}
639 			insert = conn.update(GROUP_SYNC_INSERT_TEMP_ROW);
640 			add = conn.update(GROUP_SYNC_ADD_GROUPS);
641 			remove = conn.update(GROUP_SYNC_REMOVE_GROUPS);
642 		}
643 
644 		@Override
645 		public void close() {
646 			remove.close();
647 			add.close();
648 			insert.close();
649 			/*
650 			 * Seems we can only drop a temporary table if there are no (other)
651 			 * open statements that refer to it *AND* there is no transaction
652 			 * open that used it. Or is that no connection? Either way,
653 			 * SQLITE_LOCKED is not a desired failure state!
654 			 *
655 			 * https://sqlite.org/forum/forumpost/433d2fdb07fc8f13 says some of
656 			 * the constraints, but not all; the need for the transaction to be
657 			 * closed comes from elsewhere in that thread.
658 			 *
659 			 * Fortunately, we can just delete the contents of the temporary
660 			 * table instead, and that's just as good *and can be done in the
661 			 * transaction*.
662 			 */
663 			try (var drop = conn.update(GROUP_SYNC_DROP_TEMP_TABLE)) {
664 				drop.call();
665 			}
666 			super.close();
667 		}

732 			Pattern.compile("^collab-(.*)-(admin|editor|viewer)$");

734 	/**
735 	 * Convert a list of claimed collabs into authorities.
736 	 *
737 	 * @param claim
738 	 *            The (sub-)claim of collabs.
739 	 * @param results
740 	 *            Where to add the generated authorities.
741 	 * @return Whether we processed a claim at all. (The claim could be empty;
742 	 *         that's OK.)
743 	 */
744 	private boolean collabToAuthority(List<String> claim,
745 			Collection<GrantedAuthority> results) {
746 		if (isNull(claim)) {
747 			return false;
748 		}
749 		var seen = new HashSet<>();
750 		for (var collab : claim) {
751 			var reduced = COLLAB_MATCHER.matcher(collab).replaceFirst("$1");
752 			if (!seen.contains(reduced)) {
753 				results.add(new CollabratoryAuthority(reduced));
754 				seen.add(reduced);
755 			}
756 		}
757 		return true;
758 	}

787 	 * the promised types are all inferred and the <em>actual</em> types
788 	 * guaranteed by the way claims are encoded are more that we are taking an
789 	 * {@link Object} because {@link ClaimAccessor#getClaim(String)} is merely
790 	 * pinky-swearing that objects are of the right type, not enforcing it.

792 	 * @param rolesClaim

795 	 *         looks like a list of strings.

798 			getTeamsFromClaim(Map<String, List<String>> rolesClaim) {
799 		// Messy; all the implicit types and hidden casts!
800 		try {
801 			var teamsClaim = rolesClaim.get("team");
802 			if (!teamsClaim.isEmpty()) {
803 				/*
804 				 * Check invokes a String instance method on the first item of
805 				 * the list; if this works, we're valid. It's pitched into
806 				 * log.trace so things don't complain, but we never expect to
807 				 * enable this.
808 				 */
809 				log.trace("team claim first item length",
810 						teamsClaim.get(0).length());

814 				| IndexOutOfBoundsException e) {
815 			log.debug("failed to convert claim", e);

866 	 * Check if a user can log in, and determine what permissions they have.
867 	 *
868 	 * @param username
869 	 *            The username.
870 	 * @param password
871 	 *            The <em>unencrypted</em> password.
872 	 * @param authorities
873 	 *            Filled out with permissions.
874 	 * @param queries
875 	 *            How to access the database.
876 	 * @return Whether the user is known. If {@code false}, the database cannot
877 	 *         authenticate the user and some other auth method should be tried.
878 	 * @throws UsernameNotFoundException
879 	 *             If no account exists.
880 	 * @throws DisabledException
881 	 *             If the account is administratively disabled.
882 	 * @throws LockedException
883 	 *             If the account is temporarily locked.
884 	 * @throws BadCredentialsException
885 	 *             If the password is invalid.
886 	 */
887 	private boolean authLocalAgainstDB(String username, String password,
888 			List<GrantedAuthority> authorities, AuthQueries queries) {
889 		return ifElse(
890 				queries.transaction(() -> lookUpUserDetails(username, queries)),
891 				details -> {
892 					checkPassword(username, password, details, queries);
893 					// Succeeded; finalize into external form
894 					return queries.transaction(() -> {
895 						queries.noteLoginSuccessForUser(details.userId);
896 						// Convert tiered trust level to grant form
897 						details.trustLevel.getGrants()
898 								.forEach(authorities::add);
899 						return true;
900 					});
901 				}, () -> false);
902 	}

919 	/**
920 	 * Look up a local user. Does all checks <em>except</em> the password check;
921 	 * that's slow (because bcrypt) so we do it outside the transaction. This
922 	 * needs to be done in a writable (exclusive) transaction.
923 	 *
924 	 * @param username
925 	 *            The user name we're looking up.
926 	 * @param queries
927 	 *            How to access the database.
928 	 * @return Whether we know the user, and if so, what do we know about them.
929 	 * @throws AuthenticationException
930 	 *             If the user is known and definitely can't log in for some
931 	 *             reason.
932 	 */
933 	private static Optional<LocalAuthResult> lookUpUserDetails(String username,
934 			AuthQueries queries) {
935 		return queries.getUserBlocked.call1(row -> {
936 			int userId = row.getInt("user_id");
937 			if (row.getBoolean("disabled")) {
938 				log.info("login failure for {}: account is disabled", username);
939 				throw new DisabledException("account is disabled");
940 			}
941 			try {
942 				if (row.getBoolean("locked")) {
943 					// Note that this extends the lock!
944 					throw new LockedException("account is locked");
945 				}
946 				var authInfo = queries.userAuthorities.call1(AuthInfo::new,
947 						userId).get();
948 				var encPass = authInfo.encryptedPassword;
949 				if (isNull(encPass)) {
950 					/*
951 					 * We know this user, but they can't use this authentication
952 					 * method. They'll probably have to use OpenID.
953 					 */
954 					return null;
955 				}
956 				var trust = authInfo.trustLevel;
957 				log.info("User trust level is " + trust);
958 				return new LocalAuthResult(userId, trust, encPass);
959 			} catch (AuthenticationException e) {
960 				queries.noteLoginFailureForUser(userId, username);
961 				log.info("login failure for {}", username, e);
962 				throw e;
963 			}
964 		}, username);
965 	}

967 	/**
968 	 * This is slow (100ms!) so we make sure we aren't holding a transaction
969 	 * open while this step is ongoing.
970 	 *
971 	 * @param username
972 	 *            The username (now validated as existing).
973 	 * @param password
974 	 *            The user-provided password that we're checking.
975 	 * @param queries
976 	 *            How to access the DB.
977 	 * @param details
978 	 *            The results of looking up the user
979 	 */
980 	private void checkPassword(String username, String password,
981 			LocalAuthResult details, AuthQueries queries) {
982 		if (!passServices.matchPassword(password, details.passInfo)) {
983 			queries.transaction(() -> {
984 				queries.noteLoginFailureForUser(details.userId, username);
985 				log.info("login failure for {}: bad password", username);
986 				throw new BadCredentialsException("bad password");
987 			});
988 		}

991 	/**
992 	 * Check if an OpenID user can use the service. A transaction <em>must</em>
993 	 * be held open when calling this.
994 	 *
995 	 * @param username
996 	 *            The username, already obtained and verified.
997 	 * @param subject
998 	 *            The real OpenID {@code sub} claim, if available/known.
999 	 * @param authorities
1000 	 *            The claimed authorities derived from the HBP OpenID service.
1001 	 *            Interesting because they include claims about organisation and
1002 	 *            collabratory membership.
1003 	 * @param queries
1004 	 *            How to access the database.
1005 	 * @return Whether the user is allowed to use the service. If {@code false},
1006 	 *         the database cannot authorize the user and some other auth method
1007 	 *         should be tried.
1008 	 * @throws DisabledException
1009 	 *             If the account is administratively disabled.
1010 	 * @throws LockedException
1011 	 *             If the account is temporarily locked.
1012 	 */
1013 	private boolean authOpenIDAgainstDB(String username, String subject,
1014 			Collection<? extends GrantedAuthority> authorities,
1015 			AuthQueries queries) {
1016 		var collabs = new ArrayList<String>();
1017 		var orgs = new ArrayList<String>();
1018 		authorities.forEach(ga -> inflateGroup(ga, collabs, orgs, queries));
1019 		boolean ok = queries.getUserBlocked.call1(userInfo -> {
1020 			log.info("Found user " + username + " in database");
1021 			int userId = userInfo.getInt("user_id");
1022 			synchExternalGroups(username, userId, orgs, collabs, queries);
1023 			if (userInfo.getBoolean("disabled")) {
1024 				log.info("user {} has a disabled account", username);
1025 				throw new DisabledException("account is disabled");
1026 			}
1027 			try {
1028 				if (userInfo.getBoolean("locked")) {
1029 					// Note that this extends the lock!
1030 					throw new LockedException("account is locked");
1031 				}
1032 				var authInfo = queries.userAuthorities.call1(AuthInfo::new,
1033 						userId).get();
1034 				if (nonNull(authInfo.encryptedPassword)) {
1035 					/*
1036 					 * We know this user, but they can't use this authentication
1037 					 * method. They'll probably have to use username+password.
1038 					 */
1039 					return false;
1040 				}
1041 				checkSubject(username, subject,	authInfo.openIdSubject);
1042 				queries.noteLoginSuccessForUser(userId, subject);
1043 				log.info("login success for {}", username);
1044 				return true;
1045 			} catch (AuthenticationException e) {
1046 				queries.noteLoginFailureForUser(userId, username);
1047 				log.info("login failure for {}", username, e);
1048 				throw e;
1049 			}
1050 		}, username).get();
1051 
1052 		if (ok) {
1053 			return true;
1054 		}
1055 
1056 		/*
1057 		 * No such user; need to inflate one now. If we successfully make
1058 		 * the user, they're also immediately authorised as they're
1059 		 * definitely not disabled or locked.
1060 		 */
1061 		return ifElse(queries.createUser(username, null, USER, subject),
1062 				id -> {
1063 					synchExternalGroups(username, id, orgs, collabs,
1064 							queries);
1065 					return true;
1066 				}, () -> {
1067 					// Can't note a failure; no record to note it in!
1068 					log.warn("failed to make user {}", username);
1069 					return false;
1070 				});
1071 	}

1074 	 * Check if the subject matches and issue a warning if there's a potential
1075 	 * problem.
1076 	 *
1077 	 * @param username
1078 	 *            The username (for logging).
1079 	 * @param subject
1080 	 *            The new user subject ID.
1081 	 * @param oldSubject
1082 	 *            The old user subject ID.
1083 	 */
1084 	private void checkSubject(String username, String subject,

1087 			log.warn("null subject for {}", username);
1088 		} else if (subject.isBlank()) {
1089 			log.warn("empty subject for {}", username);
1090 		}

1092 			/*
1093 			 * We know this user from before; double check that they're the same

1095 			 */
1096 			if (!requireNonNull(subject).equals(oldSubject)) {
1097 				log.warn("user {} subject changed from {} to {}", username,
1098 						oldSubject, subject);
1099 			}
1100 		}

1104 			List<String> orgs, AuthQueries queries) {
1105 		if (ga instanceof CollabratoryAuthority) {
1106 			var collab = (CollabratoryAuthority) ga;
1107 			var collab1 = collab.getCollabratory();
1108 			if (queries.createGroup(collab1, COLLABRATORY,

1114 			var org = (OrganisationAuthority) ga;
1115 			var org1 = org.getOrganisation();

1117 					quotaProps.getDefaultOrgQuota())) {
1118 				log.info("created organisation '{}'", org1);
1119 			}
1120 			orgs.add(org.getOrganisation());

1125 	 * Synchronise the lists of different types of groups for a user. Messy
1126 	 * because we want to do minimal inserts and deletes.
1127 	 *
1128 	 * @param username
1129 	 *            Which user (for logging only)
1130 	 * @param userId
1131 	 *            Which user (for DB access only)
1132 	 * @param orgs
1133 	 *            The list of organisation names
1134 	 * @param collabs
1135 	 *            The list of collab names
1136 	 * @param queries
1137 	 *            How to touch the DB
1138 	 */
1139 	private void synchExternalGroups(String username, int userId,
1140 			List<String> orgs, List<String> collabs, AuthQueries queries) {
1141 		if (orgs.isEmpty() && collabs.isEmpty()) {
1142 			log.warn(
1143 					"no organisations or collabratories for user {}; "
1144 							+ "not synching in case this is a glitch "
1145 							+ "(if this is the first time they've logged in, "
1146 							+ "creating a job will fail in this session)",
1147 					username);
1148 			return;
1149 		}
1150 		try (var synch = new GroupSynch(queries)) {
1151 			for (var org : orgs) {
1152 				synch.define(org, ORGANISATION);
1153 			}
1154 			for (var collab : collabs) {
1155 				synch.define(collab, COLLABRATORY);
1156 			}
1157 			synch.apply(userId);
1158 		} catch (RuntimeException e) {
1159 			log.warn("problem when synchronizing group memberships for {}",
1160 					username, e);
1161 			throw e;
1162 		}
