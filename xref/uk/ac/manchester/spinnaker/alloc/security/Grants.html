17  package uk.ac.manchester.spinnaker.alloc.security;
18  
19  /**
20   * The strings that describe the roles that may be granted.
21   *
22   * @author Donal Fellows
23   */
24  public interface Grants {
25  	/**
26  	 * The authority used to grant a user permission to create jobs, manipulate
27  	 * them, and read their details. Note that many features are locked to the
28  	 * owner of the job or admins. Users should also have {@link #GRANT_READER}.
29  	 */
30  	String GRANT_USER = "ROLE_USER";
31  
32  	/**
33  	 * The authority used to grant a user permission to get general machine
34  	 * information and summaries of jobs. Without this, only the service root
35  	 * (and the parts required for logging in) will be visible.
36  	 */
37  	String GRANT_READER = "ROLE_READER";
38  
39  	/**
40  	 * The authority used to grant a user permission to use administrative
41  	 * actions. Admins should also have {@link #GRANT_USER} and
42  	 * {@link #GRANT_READER}.
43  	 */
44  	String GRANT_ADMIN = "ROLE_ADMIN";
45  }
