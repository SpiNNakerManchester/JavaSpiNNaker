64   * <p>
65   * <strong>Note:</strong> role expressions ({@link #IS_USER} and
66   * {@link #IS_ADMIN}) must be applied (with {@code @}{@link PreAuthorize}) to

112 	/**
113 	 * Builds a custom trust manager to plug into the Java runtime. This is so
114 	 * that we can access resources managed by Keycloak, which is necessary
115 	 * because Java doesn't trust its certificate by default (for messy
116 	 * reasons).
117 	 *
118 	 * @param props
119 	 *            Configuration properties
120 	 * @return the custom trust manager, <em>already injected</em>
121 	 * @throws IOException
122 	 *             If the trust store can't be loaded because of I/O
123 	 * @throws GeneralSecurityException
124 	 *             If there is a security problem with the trust store
125 	 * @see <a href="https://stackoverflow.com/a/24561444/301832">Stack
126 	 *      Overflow</a>
127 	 */
128 	@Bean
129 	@Role(ROLE_SUPPORT)
130 	static X509TrustManager customTrustManager(AuthProperties props)
131 			throws IOException, GeneralSecurityException {
132 		OpenIDProperties p = props.getOpenid();
133 		X509TrustManager tm = trustManager(loadTrustStore(p));
134 		customTm = tm;
135 		log.info("set trust store from {}", p.getTruststorePath().getURI());
136 		return tm;
137 	}

170 	private String oidcPath(String suffix) {
171 		return urlMaker.systemUrl("perform_oidc/" + suffix);
172 	}

174 	/**
175 	 * Set up access control policies where they're not done by method security.
176 	 * The {@code /info} part reveals admin details; you need {@code ROLE_ADMIN}
177 	 * to view it. Everything to do with logging in <strong>must not</strong>
178 	 * require being logged in. For anything else, as long as you are
179 	 * authenticated we're happy. <em>Some</em> calls have additional
180 	 * requirements; those are annotated with {@link PreAuthorize @PreAuthorize}
181 	 * and a suitable auth expression.
182 	 *
183 	 * @param http
184 	 *            Where the configuration is applied to.
185 	 * @throws Exception
186 	 *             If anything goes wrong with setting up.
187 	 */
188 	private void defineAccessPolicy(HttpSecurity http) throws Exception {
189 		http.authorizeRequests()
190 				// General metadata pages require ADMIN access
191 				.antMatchers(urlMaker.serviceUrl("info*"),
192 						urlMaker.serviceUrl("info/**"))
193 				.hasRole("ADMIN")
194 				// Login process and static resources are available to all
195 				.antMatchers(urlMaker.systemUrl("login*"),
196 						urlMaker.systemUrl("perform_*"), oidcPath("**"),
197 						urlMaker.systemUrl("error"),
198 						urlMaker.systemUrl("resources/*"))
199 				.permitAll()
200 				// Everything else requires post-login
201 				.anyRequest().authenticated();
202 	}

204 	/**
205 	 * How we handle the mechanics of login with the REST API.
206 	 *
207 	 * @param http
208 	 *            Where the configuration is applied to.
209 	 * @throws Exception
210 	 *             If anything goes wrong with setting up. Not expected.
211 	 */
212 	private void defineAPILoginRules(HttpSecurity http) throws Exception {
213 		if (properties.isBasic()) {
214 			http.httpBasic().authenticationEntryPoint(authenticationEntryPoint);
215 		}
216 		if (properties.getOpenid().isEnable()) {
217 			http.oauth2ResourceServer()
218 					.authenticationEntryPoint(authenticationEntryPoint).jwt()
219 					.jwtAuthenticationConverter(authConverter());
220 		}
221 	}
222 
223 	/**
224 	 * How we handle the mechanics of login within the web UI.
225 	 *
226 	 * @param http
227 	 *            Where the configuration is applied to.
228 	 * @throws Exception
229 	 *             If anything goes wrong with setting up. Not expected.
230 	 */
231 	private void defineWebUILoginRules(HttpSecurity http) throws Exception {
232 		String loginUrl = urlMaker.systemUrl("login.html");
233 		String rootPage = urlMaker.systemUrl("");
234 		if (properties.getOpenid().isEnable()) {
235 			/*
236 			 * We're both, so we can have logins AND tokens. The logins are for
237 			 * using the HTML UI, and the tokens are for using from SpiNNaker
238 			 * tools (especially within the collabratory and the Jupyter
239 			 * notebook).
240 			 */
241 			http.oauth2Login().loginPage(loginUrl)
242 					.loginProcessingUrl(oidcPath("login/code/*"))
243 					.authorizationEndpoint().baseUri(oidcPath("auth")).and()
244 					.defaultSuccessUrl(rootPage, true)
245 					.failureUrl(loginUrl + "?error=true").userInfoEndpoint()
246 					.userAuthoritiesMapper(userAuthoritiesMapper());
247 			http.oauth2Client();
248 		}
249 		if (properties.isLocalForm()) {
250 			http.formLogin().loginPage(loginUrl)
251 					.loginProcessingUrl(urlMaker.systemUrl("perform_login"))
252 					.defaultSuccessUrl(rootPage, true)
253 					.failureUrl(loginUrl + "?error=true")
254 					.failureHandler(authenticationFailureHandler);
255 		}
256 	}
257 
258 	/**
259 	 * Logging out is common code between the UI and the API, but pretty
260 	 * pointless for Basic Auth as browsers will just log straight back in
261 	 * again. Still, it is meaningful (it invalidates the session).
262 	 *
263 	 * @param http
264 	 *            Where the configuration is applied to.
265 	 * @throws Exception
266 	 *             If anything goes wrong with setting up. Not expected.
267 	 */
268 	private void defineLogoutRules(HttpSecurity http) throws Exception {
269 		String loginUrl = urlMaker.systemUrl("login.html");
270 		http.logout().logoutUrl(urlMaker.systemUrl("perform_logout"))
271 				.addLogoutHandler((req, resp, auth) -> clearToken(req))
272 				.deleteCookies(SESSION_COOKIE).invalidateHttpSession(true)
273 				.logoutSuccessUrl(loginUrl);
274 	}

276 	/**
277 	 * Define our main security controls.
278 	 *
279 	 * @param http
280 	 *            Used to build the filter chain.
281 	 * @return The filter chain that implements the controls.
282 	 * @throws Exception
283 	 *             If anything goes wrong with setting up. Not expected.
284 	 */
285 	@Bean
286 	@Role(ROLE_SUPPORT)
287 	public SecurityFilterChain securityFilter(HttpSecurity http)
288 			throws Exception {
289 		defineAccessPolicy(http);
290 		defineAPILoginRules(http);
291 		defineWebUILoginRules(http);
292 		defineLogoutRules(http);
293 		http.addFilterAfter(authApplicationFilter,
294 				BasicAuthenticationFilter.class);
295 		return http.build();
296 	}

298 	/**
299 	 * @return A converter that handles the initial extraction of collabratories
300 	 *         and organisations from the info we have available with a bearer
301 	 *         token.
302 	 * @see LocalAuthProviderImpl#mapAuthorities(Jwt, Collection)
303 	 */
304 	@Bean("hbp.collab-and-org.bearer-converter.shim")
305 	@Role(ROLE_SUPPORT)
306 	Converter<Jwt, ? extends AbstractAuthenticationToken> authConverter() {
307 		JwtGrantedAuthoritiesConverter jgac =
308 				new JwtGrantedAuthoritiesConverter();
309 		return jwt -> {
310 			Collection<GrantedAuthority> mappedAuthorities = jgac.convert(jwt);
311 			// ASSUME that this is a modifiable collection
312 			localAuthProvider.mapAuthorities(jwt, mappedAuthorities);
313 			return new JwtAuthenticationToken(jwt, mappedAuthorities);
314 		};
315 	}
316 
317 	/**
318 	 * @return A converter that handles the initial extraction of collabratories
319 	 *         and organisations from the info we have available when a user
320 	 *         logs in explicitly in the web UI.
321 	 * @see LocalAuthProviderImpl#mapAuthorities(OidcUserAuthority, Collection)
322 	 */
323 	@Bean("hbp.collab-and-org.user-converter.shim")
324 	@Role(ROLE_SUPPORT)
325 	GrantedAuthoritiesMapper userAuthoritiesMapper() {
326 		SimpleAuthorityMapper sam = new SimpleAuthorityMapper();
327 		return authorities -> {
328 			Collection<GrantedAuthority> mappedAuthorities =
329 					sam.mapAuthorities(authorities);
330 			authorities.forEach(authority -> {
331 				/*
332 				 * Check for OidcUserAuthority because Spring Security 5.2
333 				 * returns each scope as a GrantedAuthority, which we don't care
334 				 * about.
335 				 */
336 				if (authority instanceof OidcUserAuthority) {
337 					localAuthProvider.mapAuthorities(
338 							(OidcUserAuthority) authority, mappedAuthorities);
339 				}
340 				mappedAuthorities.add(authority);
341 			});
342 			return mappedAuthorities;
343 		};
344 	}

346 	@Bean
347 	@Role(ROLE_SUPPORT)
348 	LogoutHandler logoutHandler() {
349 		SecurityContextLogoutHandler sclh = new SecurityContextLogoutHandler();
350 		sclh.setClearAuthentication(true);
351 		sclh.setInvalidateHttpSession(true);
352 		return sclh;
