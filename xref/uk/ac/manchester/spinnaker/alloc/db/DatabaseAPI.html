34  /**
35   * The primary interface to the database. It conceptually represents an SQLite
36   * database that can have connections opened on it.
37   *
38   * @author Donal Fellows
39   */
40  public interface DatabaseAPI {
41  	/**
42  	 * Get a connection. This connection is thread-bound and pooled; it <em>must
43  	 * not</em> be passed to other threads. They should get their own
44  	 * connections instead. The connection has auto-commit disabled; use the
45  	 * {@link Connection#transaction(DatabaseEngine.TransactedWithResult)
46  	 * transaction()} method instead.
47  	 * <p>
48  	 * Note that if an in-memory database is used (see
49  	 * {@link #getInMemoryDB()}), that DB can <em>only</em> be accessed from the
50  	 * connection returned from this method; the next call to this method
51  	 * (whether from the current thread or another one) will get an independent
52  	 * database. Such in-memory databases are not subject to thread-bound
53  	 * cleanup actions; they're simply deleted from memory when no longer used
54  	 * (but the connection should be {@code close()}d after use for efficiency
55  	 * nonetheless).
56  	 * <p>
57  	 * This would be marked with {@link MustBeClosed} except that causes a mess
58  	 * elsewhere.
59  	 *
60  	 * @return A configured initialised connection to the database.
61  	 */
62  	@MustBeClosed
63  	Connection getConnection();
64  
65  	/**
66  	 * Create an engine interface for an in-memory database. This is intended
67  	 * mainly for testing purposes. Note that various coupled automatic services
68  	 * are disabled, in particular connections are not closed automatically.
69  	 *
70  	 * @return The in-memory database interface.
71  	 */
72  	DatabaseAPI getInMemoryDB();

74  	/**
75  	 * A connection manager and transaction runner. If the {@code operation}
76  	 * completes normally (and this isn't a nested use), the transaction
77  	 * commits. If an exception is thrown, the transaction is rolled back. The
78  	 * connection is closed up in any case.
79  	 *
80  	 * @param lockForWriting
81  	 *            Whether to lock for writing. Multiple read locks can be held
82  	 *            at once, but only one write lock. Locks <em>cannot</em> be
83  	 *            upgraded (because that causes deadlocks).
84  	 * @param operation
85  	 *            The operation to run
86  	 * @throws RuntimeException
87  	 *             If something goes wrong with the database access or the
88  	 *             contained code.
89  	 */
90  	void executeVoid(boolean lockForWriting, Connected operation);
91  
92  	/**
93  	 * A connection manager and transaction runner. If the {@code operation}
94  	 * completes normally (and this isn't a nested use), the transaction
95  	 * commits. If an exception is thrown, the transaction is rolled back. The
96  	 * connection is closed up in any case. This uses a write lock.
97  	 *
98  	 * @param operation
99  	 *            The operation to run
100 	 * @throws RuntimeException
101 	 *             If something goes wrong with the database access or the
102 	 *             contained code.
103 	 */
104 	default void executeVoid(Connected operation) {
105 		executeVoid(true, operation);
106 	}

108 	/**
109 	 * A connection manager and transaction runner. If the {@code operation}
110 	 * completes normally (and this isn't a nested use), the transaction
111 	 * commits. If an exception is thrown, the transaction is rolled back. The
112 	 * connection is closed up in any case.
113 	 *
114 	 * @param <T>
115 	 *            The type of the result of {@code operation}
116 	 * @param lockForWriting
117 	 *            Whether to lock for writing. Multiple read locks can be held
118 	 *            at once, but only one write lock. Locks <em>cannot</em> be
119 	 *            upgraded (because that causes deadlocks).
120 	 * @param operation
121 	 *            The operation to run
122 	 * @return the value returned by {@code operation}
123 	 * @throws RuntimeException
124 	 *             If something other than database access goes wrong with the
125 	 *             contained code.
126 	 */
127 	<T> T execute(boolean lockForWriting, ConnectedWithResult<T> operation);
128 
129 	/**
130 	 * A connection manager and transaction runner. If the {@code operation}
131 	 * completes normally (and this isn't a nested use), the transaction
132 	 * commits. If an exception is thrown, the transaction is rolled back. The
133 	 * connection is closed up in any case. This uses a write lock.
134 	 *
135 	 * @param <T>
136 	 *            The type of the result of {@code operation}
137 	 * @param operation
138 	 *            The operation to run
139 	 * @return the value returned by {@code operation}
140 	 * @throws RuntimeException
141 	 *             If something other than database access goes wrong with the
142 	 *             contained code.
143 	 */
144 	default <T> T execute(ConnectedWithResult<T> operation) {
145 		return execute(true, operation);
146 	}

168 	/**
169 	 * Connections made by the database engine bean. Its methods do not throw
170 	 * checked exceptions. The connection is thread-bound, and will be cleaned
171 	 * up correctly when the thread exits (ideal for thread pools).
172 	 */
173 	interface Connection extends AutoCloseable {
174 		/**
175 		 * Closes this connection and releases any resources. The actual
176 		 * underlying connection may remain open if the connection pool wishes
177 		 * to maintain it, but this handle should not be retained by the caller
178 		 * after this point.
179 		 *
180 		 * @see java.sql.Connection#close()
181 		 */
182 		@Override
183 		void close();
184 
185 		/**
186 		 * Undoes all changes made in the current transaction and releases any
187 		 * database locks currently held by this connection.
188 		 * <p>
189 		 * This method should be used only when in a transaction; it is only
190 		 * required when the transaction is to be rolled back without throwing
191 		 * an exception, as the normal behaviour of the internal transaction
192 		 * manager is to roll the transaction back when an exception leaves the
193 		 * code inside the transaction boundary.
194 		 *
195 		 * @see java.sql.Connection#rollback()
196 		 */
197 		void rollback();
198 
199 		/**
200 		 * Retrieves whether this connection is in read-only mode.
201 		 *
202 		 * @return {@code true} if this connection is read-only; {@code false}
203 		 *         otherwise.
204 		 * @see java.sql.Connection#isReadOnly()
205 		 */
206 		boolean isReadOnly();
207 
208 		/**
209 		 * Whether the historical data DB is available. If it isn't, you can't
210 		 * move any data to longer-term storage, but ordinary operations should
211 		 * be fine.
212 		 *
213 		 * @return Whether the historical data DB is available.
214 		 */
215 		boolean isHistoricalDBAvailable();
216 
217 		/**
218 		 * A nestable transaction runner. If the {@code operation} completes
219 		 * normally (and this isn't a nested use), the transaction commits. If
220 		 * an exception is thrown, the transaction is rolled back.
221 		 *
222 		 * @param lockForWriting
223 		 *            Whether to lock for writing. Multiple read locks can be
224 		 *            held at once, but only one write lock. Locks
225 		 *            <em>cannot</em> be upgraded (because that causes
226 		 *            deadlocks).
227 		 * @param operation
228 		 *            The operation to run
229 		 * @see #transaction(DatabaseEngine.TransactedWithResult)
230 		 */
231 		void transaction(boolean lockForWriting, Transacted operation);
232 
233 		/**
234 		 * A nestable transaction runner. If the {@code operation} completes
235 		 * normally (and this isn't a nested use), the transaction commits. If
236 		 * an exception is thrown, the transaction is rolled back. This uses a
237 		 * write lock.
238 		 *
239 		 * @param operation
240 		 *            The operation to run
241 		 * @see #transaction(DatabaseEngine.TransactedWithResult)
242 		 */
243 		void transaction(Transacted operation);
244 
245 		/**
246 		 * A nestable transaction runner. If the {@code operation} completes
247 		 * normally (and this isn't a nested use), the transaction commits. If
248 		 * an exception is thrown, the transaction is rolled back. This uses a
249 		 * write lock.
250 		 *
251 		 * @param <T>
252 		 *            The type of the result of {@code operation}
253 		 * @param operation
254 		 *            The operation to run
255 		 * @return the value returned by {@code operation}
256 		 * @see #transaction(DatabaseEngine.Transacted)
257 		 */
258 		<T> T transaction(TransactedWithResult<T> operation);
259 
260 		/**
261 		 * A nestable transaction runner. If the {@code operation} completes
262 		 * normally (and this isn't a nested use), the transaction commits. If
263 		 * an exception is thrown, the transaction is rolled back.
264 		 *
265 		 * @param <T>
266 		 *            The type of the result of {@code operation}
267 		 * @param lockForWriting
268 		 *            Whether to lock for writing. Multiple read locks can be
269 		 *            held at once, but only one write lock. Locks
270 		 *            <em>cannot</em> be upgraded (because that causes
271 		 *            deadlocks).
272 		 * @param operation
273 		 *            The operation to run
274 		 * @return the value returned by {@code operation}
275 		 * @see #transaction(DatabaseEngine.Transacted)
276 		 */
277 		<T> T transaction(boolean lockForWriting,
278 				TransactedWithResult<T> operation);

280 		// @formatter:off
281 		/**
282 		 * Create a new query. Usage pattern:
283 		 * <pre>
284 		 * try (var q = conn.query(SQL_SELECT)) {
285 		 *     for (var row : u.call(argument1, argument2)) {
286 		 *         // Do something with the row
287 		 *     }
288 		 * }
289 		 * </pre>
290 		 * or:
291 		 * <pre>
292 		 * try (var q = conn.query(SQL_SELECT)) {
293 		 *     u.call(argument1, argument2).forEach(row -&gt; {
294 		 *         // Do something with the row
295 		 *     });
296 		 * }
297 		 * </pre>
298 		 *
299 		 * @param sql
300 		 *            The SQL of the query.
301 		 * @return The query object.
302 		 * @see #query(Resource)
303 		 * @see #update(String)
304 		 * @see SQLQueries
305 		 */
306 		// @formatter:on
307 		Query query(@CompileTimeConstant String sql);
308 
309 		// @formatter:off
310 		/**
311 		 * Create a new query. Usage pattern:
312 		 * <pre>
313 		 * try (var q = conn.query(SQL_SELECT)) {
314 		 *     for (var row : u.call(argument1, argument2)) {
315 		 *         // Do something with the row
316 		 *     }
317 		 * }
318 		 * </pre>
319 		 * or:
320 		 * <pre>
321 		 * try (var q = conn.query(SQL_SELECT)) {
322 		 *     u.call(argument1, argument2).forEach(row -&gt; {
323 		 *         // Do something with the row
324 		 *     });
325 		 * }
326 		 * </pre>
327 		 *
328 		 * @param sql
329 		 *            The SQL of the query.
330 		 * @param lockType
331 		 *            Whether we expect to have a write lock. This is vital
332 		 * @return The query object.
333 		 * @see #query(Resource)
334 		 * @see #update(String)
335 		 * @see SQLQueries
336 		 */
337 		// @formatter:on
338 		Query query(@CompileTimeConstant String sql, boolean lockType);
339 
340 		// @formatter:off
341 		/**
342 		 * Create a new query.
343 		 * <pre>
344 		 * try (var q = conn.query(sqlSelectResource)) {
345 		 *     for (var row : u.call(argument1, argument2)) {
346 		 *         // Do something with the row
347 		 *     }
348 		 * }
349 		 * </pre>
350 		 * or:
351 		 * <pre>
352 		 * try (var q = conn.query(sqlSelectResource)) {
353 		 *     u.call(argument1, argument2).forEach(row -&gt; {
354 		 *         // Do something with the row
355 		 *     });
356 		 * }
357 		 * </pre>
358 		 *
359 		 * @param sqlResource
360 		 *            Reference to the SQL of the query.
361 		 * @return The query object.
362 		 * @see #query(String)
363 		 * @see #update(Resource)
364 		 * @see SQLQueries
365 		 */
366 		// @formatter:on
367 		Query query(Resource sqlResource);
368 
369 		// @formatter:off
370 		/**
371 		 * Create a new query.
372 		 * <pre>
373 		 * try (var q = conn.query(sqlSelectResource)) {
374 		 *     for (var row : u.call(argument1, argument2)) {
375 		 *         // Do something with the row
376 		 *     }
377 		 * }
378 		 * </pre>
379 		 * or:
380 		 * <pre>
381 		 * try (var q = conn.query(sqlSelectResource)) {
382 		 *     u.call(argument1, argument2).forEach(row -&gt; {
383 		 *         // Do something with the row
384 		 *     });
385 		 * }
386 		 * </pre>
387 		 *
388 		 * @param sqlResource
389 		 *            Reference to the SQL of the query.
390 		 * @param lockType
391 		 *            Whether we expect to have a write lock. This is vital
392 		 *            only when the query is an {@code UPDATE RETURNING}.
393 		 * @return The query object.
394 		 * @see #query(String)
395 		 * @see #update(Resource)
396 		 * @see SQLQueries
397 		 */
398 		// @formatter:on
399 		Query query(Resource sqlResource, boolean lockType);
400 
401 		// @formatter:off
402 		/**
403 		 * Create a new update. Usage pattern:
404 		 * <pre>
405 		 * try (var u = conn.update(SQL_UPDATE)) {
406 		 *     int numRows = u.call(argument1, argument2);
407 		 * }
408 		 * </pre>
409 		 * or:
410 		 * <pre>
411 		 * try (var u = conn.update(SQL_INSERT)) {
412 		 *     for (var key : u.keys(argument1, argument2)) {
413 		 *         // Do something with the key
414 		 *     }
415 		 * }
416 		 * </pre>
417 		 * or even:
418 		 * <pre>
419 		 * try (var u = conn.update(SQL_INSERT)) {
420 		 *     u.key(argument1, argument2).ifPresent(key -&gt; {
421 		 *         // Do something with the key
422 		 *     });
423 		 * }
424 		 * </pre>
425 		 * <p>
426 		 * <strong>Note:</strong> If you use a {@code RETURNING} clause then
427 		 * you should use a {@link Query} with the {@code lockType} set to
428 		 * {@code true}.
429 		 *
430 		 * @param sql
431 		 *            The SQL of the update.
432 		 * @return The update object.
433 		 * @see #update(Resource)
434 		 * @see #query(String)
435 		 * @see SQLQueries
436 		 */
437 		// @formatter:on
438 		Update update(@CompileTimeConstant String sql);

442 		 * Create a new update.
443 		 * <pre>
444 		 * try (var u = conn.update(sqlUpdateResource)) {
445 		 *     int numRows = u.call(argument1, argument2);
446 		 * }
447 		 * </pre>
448 		 * or:
449 		 * <pre>

480 	/**
481 	 * Some code that may be run within a transaction.
482 	 */
483 	@FunctionalInterface
484 	interface Transacted {
485 		/**
486 		 * The operation to run.
487 		 */
488 		void act();
489 	}
490 
491 	/**
492 	 * Some code that may be run within a transaction that returns a result.
493 	 *
494 	 * @param <T>
495 	 *            The type of the result of the code.
496 	 */
497 	@FunctionalInterface
498 	interface TransactedWithResult<T> {
499 		/**
500 		 * The operation to run.
501 		 *
502 		 * @return The result of the operation.
503 		 */
504 		T act();
505 	}
506 
507 	/**
508 	 * Some code that may be run within a transaction and which will be given a
509 	 * new connection for the duration.
510 	 */
511 	@FunctionalInterface
512 	interface Connected {
513 		/**
514 		 * The operation to run.
515 		 *
516 		 * @param connection
517 		 *            The newly-created connection. Do not save beyond the scope
518 		 *            of this action.
519 		 */
520 		void act(Connection connection);
521 	}
522 
523 	/**
524 	 * Some code that may be run within a transaction that returns a result and
525 	 * which will be given a new connection for the duration.
526 	 *
527 	 * @param <T>
528 	 *            The type of the result of the code.
529 	 */
530 	@FunctionalInterface
531 	interface ConnectedWithResult<T> {
532 		/**
533 		 * The operation to run.
534 		 *
535 		 * @param connection
536 		 *            The newly-created connection. Do not save beyond the scope
537 		 *            of this action.
538 		 * @return The result of the operation.
539 		 */
540 		T act(Connection connection);
541 	}

543 	/**
544 	 * Common shared API between {@link Query} and {@link Update}.
545 	 *
546 	 * @author Donal Fellows
547 	 */
548 	interface StatementCommon extends AutoCloseable {
549 		/**
550 		 * Get the number of arguments expected when calling this statement.
551 		 *
552 		 * @return The number of arguments. Types are arbitrary (because SQLite)
553 		 */
554 		int getNumArguments();
555 
556 		/**
557 		 * Get the set of names of columns produced when calling this statement.
558 		 *
559 		 * @return A set of names. The order is the order in the SQL producing
560 		 *         the result set, but this should normally be insignificant.
561 		 */
562 		Set<String> getRowColumnNames();
563 
564 		/**
565 		 * Close this statement. This never throws a checked exception.
566 		 */
567 		@Override
568 		void close();
569 
570 		/**
571 		 * Get the query plan explanation. Note that DML and DDL <em>may</em>
572 		 * have empty query plans; that's up to the DB.
573 		 *
574 		 * @return A list of lines that describe the query plan.
575 		 * @see <a href="https://www.sqlite.org/eqp.html">SQLite
576 		 *      documentation</a>
577 		 */
578 		List<String> explainQueryPlan();
579 	}

581 	/**
582 	 * Wrapping a prepared query to be more suitable for Java 8 onwards.
583 	 *
584 	 * @author Donal Fellows
585 	 */
586 	interface Query extends StatementCommon {
587 		/**
588 		 * Run the query on the given arguments.
589 		 *
590 		 * @param arguments
591 		 *            Positional argument to the query
592 		 * @return The results, wrapped as a one-shot iterable. The
593 		 *         {@linkplain Row rows} in the iterable <em>must not</em> be
594 		 *         retained by callers; they may share state and might not
595 		 *         outlive the iteration.
596 		 */
597 		MappableIterable<Row> call(Object... arguments);
598 
599 		/**
600 		 * Run the query on the given arguments. The query must be one that only
601 		 * produces a single row result.
602 		 *
603 		 * @param arguments
604 		 *            Positional argument to the query
605 		 * @return The single row with the results, or empty if there is no such
606 		 *         row.
607 		 * @see SingleRowResult
608 		 */
609 		Optional<Row> call1(Object... arguments);
610 	}

612 	/**
613 	 * Wrapping a prepared update to be more suitable for Java 8 onwards.

615 	 * @author Donal Fellows

619 		/**
620 		 * Run the update on the given arguments.
621 		 *
622 		 * @param arguments
623 		 *            Positional argument to the query
624 		 * @return The number of rows updated
625 		 */
626 		int call(Object... arguments);
627 
628 		/**
629 		 * Run the update on the given arguments.
630 		 *
631 		 * @param arguments
632 		 *            Positional arguments to the query
633 		 * @return The integer primary keys generated by the update.
634 		 * @see GeneratesID
635 		 */
636 		MappableIterable<Integer> keys(Object... arguments);
637 
638 		/**
639 		 * Run the update on the given arguments. This is expected to generate a
640 		 * single integer primary key (common with {@code INSERT}).
641 		 *
642 		 * @param arguments
643 		 *            Positional arguments to the query
644 		 * @return The integer primary key generated by the update.
645 		 * @see GeneratesID
646 		 */
647 		Optional<Integer> key(Object... arguments);
648 	}
