17  package uk.ac.manchester.spinnaker.alloc.db;
18  
19  /**
20   * Flags in SQLite that the driver doesn't name for us.
21   */
22  public interface SQLiteFlags {
23  	/**
24  	 * Flag direct from SQLite. Used to mark a function as ineligible for use
25  	 * other than by written directly in an SQL query.
26  	 *
27  	 * <blockquote>The {@code SQLITE_DIRECTONLY} flag means that the function
28  	 * may only be invoked from top-level SQL, and cannot be used in
29  	 * <em>VIEW</em>s or <em>TRIGGER</em>s nor in schema structures such as
30  	 * <em>CHECK</em> constraints, <em>DEFAULT</em> clauses, expression indexes,
31  	 * partial indexes, or generated columns. The {@code SQLITE_DIRECTONLY}
32  	 * flags is a security feature which is recommended for all
33  	 * application-defined SQL functions, and especially for functions that have
34  	 * side-effects or that could potentially leak sensitive information.
35  	 * </blockquote>
36  	 * <p>
37  	 * Note that the password-related functions we used to install were
38  	 * definitely examples of functions that are only usable directly. (They
39  	 * were removed for performance reasons only.)
40  	 *
41  	 * @see <a href=
42  	 *      "https://www.sqlite.org/c3ref/c_deterministic.html">SQLite</a>
43  	 */
44  	int SQLITE_DIRECTONLY = 0x000080000;
45  
46  	/**
47  	 * Flag direct from SQLite. Used to mark a function as "safe".
48  	 *
49  	 * <blockquote>The {@code SQLITE_INNOCUOUS} flag means that the function is
50  	 * unlikely to cause problems even if misused. An innocuous function should
51  	 * have no side effects and should not depend on any values other than its
52  	 * input parameters. The {@code abs()} function is an example of an
53  	 * innocuous function. The {@code load_extension()} SQL function is not
54  	 * innocuous because of its side effects.
55  	 * <p>
56  	 * {@code SQLITE_INNOCUOUS} is similar to
57  	 * {@link org.sqlite.Function#FLAG_DETERMINISTIC SQLITE_DETERMINISTIC}, but
58  	 * is not exactly the same. The {@code random()} function is an example of a
59  	 * function that is innocuous but not deterministic.
60  	 * <p>
61  	 * Some heightened security settings ({@code SQLITE_DBCONFIG_TRUSTED_SCHEMA}
62  	 * and {@code PRAGMA trusted_schema=OFF}) disable the use of SQL functions
63  	 * inside views and triggers and in schema structures such as <em>CHECK</em>
64  	 * constraints, <em>DEFAULT</em> clauses, expression indexes, partial
65  	 * indexes, and generated columns unless the function is tagged with
66  	 * {@code SQLITE_INNOCUOUS}. Most built-in functions are innocuous.
67  	 * Developers are advised to avoid using the {@code SQLITE_INNOCUOUS} flag
68  	 * for application-defined functions unless the function has been carefully
69  	 * audited and found to be free of potentially security-adverse side-effects
70  	 * and information-leaks. </blockquote>
71  	 * <p>
72  	 * Note that this engine marks non-innocuous functions as
73  	 * {@link #SQLITE_DIRECTONLY}; this is slightly over-eager, but likely
74  	 * correct.
75  	 *
76  	 * @see <a href=
77  	 *      "https://www.sqlite.org/c3ref/c_deterministic.html">SQLite</a>
78  	 */
79  	int SQLITE_INNOCUOUS = 0x000200000;
80  }
