58  /**
59   * Manufactures instances of {@link MachineAPI} and {@link JobAPI}. This
60   * indirection allows Spring to insert any necessary security interceptors on
61   * methods.
62   * <p>
63   * Do not call the {@link #machine(Machine,UriInfo) machine()} and
64   * {@link #job(Job,String,Permit,UriInfo) job()} methods directly. Use the
65   * factories.
66   */
67  @Configuration
68  @Role(ROLE_SUPPORT)
69  class SpallocServiceAPIImplBuilder extends BackgroundSupport {
70  	private static final Logger log =
71  			getLogger(SpallocServiceAPIImplBuilder.class);
72  
73  	@Autowired
74  	private SpallocAPI core;
75  
76  	@Autowired
77  	private JsonMapper mapper;
78  
79  	@Autowired
80  	private SpallocProperties props;
81  
82  	@Value("${spring.mvc.servlet.path}")
83  	private String servletPath;
84  
85  	/**
86  	 * Make a machine access interface.
87  	 *
88  	 * @param machine
89  	 *            The machine object to wrap
90  	 * @param ui
91  	 *            How the API was accessed
92  	 * @return A machine access interface. Will be wrapped by Spring with
93  	 *         security support (if needed).
94  	 */
95  	@Bean
96  	@Prototype
97  	@Role(ROLE_APPLICATION)
98  	public MachineAPI machine(Machine machine, UriInfo ui) {
99  		return new MachineAPI() {
100 			@Override
101 			public void describeMachine(boolean wait, AsyncResponse response) {
102 				if (wait) {
103 					bgAction(response, () -> {
104 						log.debug("starting wait for change of machine");
105 						machine.waitForChange(props.getWait());
106 						/*
107 						 * Assume that machines don't change often enough for us
108 						 * to care about whether they vanish; therefore handle
109 						 * still valid after wait
110 						 */
111 						return new MachineResponse(machine, ui);
112 					});
113 				} else {
114 					fgAction(response, () -> new MachineResponse(machine, ui));
115 				}
116 			}
117 
118 			private WhereIsResponse
119 					makeResponse(Optional<BoardLocation> boardLocation) {
120 				return new WhereIsResponse(
121 						boardLocation.orElseThrow(
122 								() -> new NotFound("failed to locate board")),
123 						ui);
124 			}
125 
126 			@Override
127 			public WhereIsResponse whereIsLogicalPosition(int x, int y, int z) {
128 				return makeResponse(machine
129 						.getBoardByLogicalCoords(new TriadCoords(x, y, z)));
130 			}
131 
132 			@Override
133 			public WhereIsResponse whereIsPhysicalPosition(int cabinet,
134 					int frame, int board) {
135 				return makeResponse(machine.getBoardByPhysicalCoords(
136 						new PhysicalCoords(cabinet, frame, board)));
137 			}
138 
139 			@Override
140 			public WhereIsResponse whereIsMachineChipLocation(int x, int y) {
141 				return makeResponse(
142 						machine.getBoardByChip(new ChipLocation(x, y)));
143 			}
144 
145 			@Override
146 			public WhereIsResponse whereIsIPAddress(String address) {
147 				return makeResponse(machine.getBoardByIPAddress(address));
148 			}
149 		};
150 	}
151 
152 	/**
153 	 * Make a job access interface.
154 	 *
155 	 * @param j
156 	 *            The job object to wrap
157 	 * @param caller
158 	 *            What host made the call? Used in keepalive tracking.
159 	 * @param permit
160 	 *            The security permissions pertinent to the job
161 	 * @param ui
162 	 *            How the API was accessed
163 	 * @return A machine access interface. Will be wrapped by Spring with
164 	 *         security support (if needed).
165 	 */
166 	@Bean
167 	@Prototype
168 	@Role(ROLE_APPLICATION)
169 	public JobAPI job(Job j, String caller, Permit permit, UriInfo ui) {
170 		var sp = props.getProxy().isEnable() ? servletPath : null;
171 		return new JobAPI() {
172 			@Override
173 			public String keepAlive(String reqBody) {
174 				log.debug("keeping job {} alive: {}", j.getId(), caller);
175 				j.access(caller);
176 				return "ok";
177 			}
178 
179 			@Override
180 			public void getState(boolean wait, AsyncResponse response) {
181 				if (wait) {
182 					bgAction(response, permit, () -> {
183 						log.debug("starting wait for change of job");
184 						j.waitForChange(props.getWait());
185 						// Refresh the handle
186 						var nj = core.getJob(permit, j.getId())
187 								.orElseThrow(() -> new ItsGone("no such job"));
188 						return new JobStateResponse(nj, ui, mapper, sp);
189 					});
190 				} else {
191 					fgAction(response, () -> new JobStateResponse(j, ui, mapper,
192 							sp));
193 				}
194 			}
195 
196 			@Override
197 			public Response deleteJob(String reason) {
198 				if (isNull(reason)) {
199 					reason = "unspecified";
200 				}
201 				j.destroy(reason);
202 				return noContent().build();
203 			}
204 
205 			private SubMachine allocation() {
206 				return j.getMachine().orElseThrow(
207 						// No machine allocated yet
208 						EmptyResponse::new);
209 			}
210 
211 			@Override
212 			public SubMachineResponse getMachine() {
213 				j.access(caller);
214 				return new SubMachineResponse(allocation(), ui);
215 			}
216 
217 			@Override
218 			public MachinePower getMachinePower() {
219 				j.access(caller);
220 				return new MachinePower(allocation().getPower());
221 			}
222 
223 			@Override
224 			public void setMachinePower(MachinePower reqBody,
225 					AsyncResponse response) {
226 				// Async because it involves getting a write lock
227 				if (isNull(reqBody)) {
228 					throw new BadArgs("bad power description");
229 				}
230 				bgAction(response, permit, () -> {
231 					j.access(caller);
232 					allocation().setPower(reqBody.getPower());
233 					return accepted().build();
234 				});
235 			}
236 
237 			@Override
238 			public WhereIsResponse getJobChipLocation(int x, int y) {
239 				j.access(caller);
240 				var loc = j.whereIs(x, y).orElseThrow(EmptyResponse::new);
241 				return new WhereIsResponse(loc, ui);
242 			}
243 
244 			@Override
245 			public void reportBoardIssue(IssueReportRequest reqBody,
246 					AsyncResponse response) {
247 				// Async because it involves getting a write lock
248 				if (isNull(reqBody)) {
249 					throw new BadArgs("bad issue description");
250 				}
251 				bgAction(response, () -> new IssueReportResponse(
252 						j.reportIssue(reqBody, permit)));
253 			}
254 		};
255 	}
