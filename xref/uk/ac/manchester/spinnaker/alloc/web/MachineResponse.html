16  package uk.ac.manchester.spinnaker.alloc.web;
17  
18  import static com.fasterxml.jackson.annotation.JsonInclude.Include.NON_NULL;
19  import static java.util.Objects.nonNull;
20  import static uk.ac.manchester.spinnaker.alloc.web.WebServiceComponentNames.MACH_BOARD_BY_ADDRESS;
21  import static uk.ac.manchester.spinnaker.alloc.web.WebServiceComponentNames.MACH_BOARD_BY_CHIP;
22  import static uk.ac.manchester.spinnaker.alloc.web.WebServiceComponentNames.MACH_BOARD_BY_LOGICAL;
23  import static uk.ac.manchester.spinnaker.alloc.web.WebServiceComponentNames.MACH_BOARD_BY_PHYSICAL;
24  
25  import java.net.URI;
26  import java.util.List;
27  
28  import javax.ws.rs.core.UriInfo;
29  
30  import com.fasterxml.jackson.annotation.JsonInclude;
31  
32  import uk.ac.manchester.spinnaker.alloc.allocator.SpallocAPI;
33  import uk.ac.manchester.spinnaker.alloc.model.BoardCoords;
34  import uk.ac.manchester.spinnaker.alloc.model.DownLink;
35  
36  /**
37   * The description of an individual machine.
38   *
39   * @author Donal Fellows
40   */
41  public final class MachineResponse {
42  	/** The name of the machine. */
43  	public final String name;
44  
45  	/** The tags of the machine. */
46  	public final List<String> tags;
47  
48  	/** The width of the machine, in triads. */
49  	public final int width;
50  
51  	/** The height of the machine, in triads. */
52  	public final int height;
53  
54  	/** The boards of the machine marked as down. */
55  	public final List<BoardCoords> downBoards;
56  
57  	/** The links of the machine marked as down. */
58  	public final List<DownLink> downLinks;
59  
60  	/** Where to look up a board by physical coordinates. */
61  	@JsonInclude(NON_NULL)
62  	public final URI lookupByPhysicalBoard;
63  
64  	/** Where to look up a board by logical coordinates. */
65  	@JsonInclude(NON_NULL)
66  	public final URI lookupByLogicalBoard;
67  
68  	/** Where to look up a board by global chip address. */
69  	@JsonInclude(NON_NULL)
70  	public final URI lookupByChip;
71  
72  	/** Where to look up a board by its ethernet IP address. */
73  	@JsonInclude(NON_NULL)
74  	public final URI lookupByAddress;
75  
76  	/**
77  	 * Create a machine response.
78  	 *
79  	 * @param machine
80  	 *            The machine being described.
81  	 * @param ui
82  	 *            How to manufacture URIs. May be {@code null}.
83  	 */
84  	public MachineResponse(SpallocAPI.Machine machine, UriInfo ui) {
85  		name = machine.getName();
86  		tags = List.copyOf(machine.getTags());
87  		width = machine.getWidth();
88  		height = machine.getHeight();
89  		downBoards = machine.getDeadBoards();
90  		downLinks = machine.getDownLinks();
91  
92  		if (nonNull(ui)) {
93  			var b = ui.getAbsolutePathBuilder().path("{resource}");
94  			lookupByPhysicalBoard = b.build(MACH_BOARD_BY_PHYSICAL);
95  			lookupByLogicalBoard = b.build(MACH_BOARD_BY_LOGICAL);
96  			lookupByChip = b.build(MACH_BOARD_BY_CHIP);
97  			lookupByAddress = b.build(MACH_BOARD_BY_ADDRESS);
98  		} else {
99  			lookupByPhysicalBoard = null;
100 			lookupByLogicalBoard = null;
101 			lookupByChip = null;
102 			lookupByAddress = null;
103 		}
104 	}
105 }
