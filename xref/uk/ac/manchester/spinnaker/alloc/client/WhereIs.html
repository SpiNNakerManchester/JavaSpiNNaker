16  package uk.ac.manchester.spinnaker.alloc.client;
17  
18  import java.net.URI;
19  
20  import com.fasterxml.jackson.annotation.JsonAlias;
21  import com.fasterxml.jackson.annotation.JsonIgnore;
22  import com.fasterxml.jackson.annotation.JsonProperty;
23  import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
24  import com.fasterxml.jackson.databind.annotation.JsonPOJOBuilder;
25  
26  import uk.ac.manchester.spinnaker.alloc.client.SpallocClient.Machine;
27  import uk.ac.manchester.spinnaker.machine.ChipLocation;
28  import uk.ac.manchester.spinnaker.machine.board.PhysicalCoords;
29  import uk.ac.manchester.spinnaker.machine.board.TriadCoords;
30  
31  /** A description of where a board is and what it is doing. */
32  @JsonDeserialize(builder = WhereIs.Builder.class)
33  public final class WhereIs {
34  	@JsonProperty("job-id")
35  	private final Integer jobId;
36  
37  	@JsonProperty("job-ref")
38  	private final URI jobRef;
39  
40  	@JsonProperty("job-chip")
41  	private final ChipLocation jobChip;
42  
43  	@JsonProperty("chip")
44  	private final ChipLocation chip;
45  
46  	@JsonIgnore
47  	private transient Machine machineHandle;
48  
49  	@JsonProperty("machine")
50  	private final String machineName;
51  
52  	@JsonProperty("machine-ref")
53  	private URI machineRef;
54  
55  	@JsonProperty("board-chip")
56  	private final ChipLocation boardChip;
57  
58  	@JsonProperty("logical-board-coordinates")
59  	private final TriadCoords logicalCoords;
60  
61  	@JsonProperty("physical-board-coordinates")
62  	private final PhysicalCoords physicalCoords;
63  
64  	private WhereIs(Integer jobId, URI jobRef, ChipLocation jobChip,
65  			ChipLocation chip, String machineName, URI machineRef,
66  			ChipLocation boardChip, TriadCoords logicalCoords,
67  			PhysicalCoords physicalCoords) {
68  		this.jobId = jobId;
69  		this.jobRef = jobRef;
70  		this.jobChip = jobChip;
71  		this.chip = chip;
72  		this.machineName = machineName;
73  		this.machineRef = machineRef;
74  		this.boardChip = boardChip;
75  		this.logicalCoords = logicalCoords;
76  		this.physicalCoords = physicalCoords;
77  	}
78  
79  	/**
80  	 * @return The ID of the job allocated to the board with the chip.
81  	 *         {@code null} if none.
82  	 */
83  	public Integer getJobId() {
84  		return jobId;
85  	}
86  
87  	/**
88  	 * @return The location of more information about the job allocated to the
89  	 *         board with the chip. {@code null} if none.
90  	 */
91  	public URI getJobRef() {
92  		return jobRef;
93  	}
94  
95  	/**
96  	 * @return The global location of the chip at the root of the job
97  	 *         allocation, if one exists.
98  	 */
99  	public ChipLocation getJobChip() {
100 		return jobChip;
101 	}
102 
103 	/** @return The global location of the chip. */
104 	public ChipLocation getChip() {
105 		return chip;
106 	}
107 
108 	/** @return Information about the machine containing the chip. */
109 	public Machine getMachineHandle() {
110 		return machineHandle;
111 	}
112 
113 	void setMachineHandle(Machine machineHandle) {
114 		this.machineHandle = machineHandle;
115 	}
116 
117 	/** @return The name of the machine containing the chip. */
118 	public String getMachineName() {
119 		return machineName;
120 	}
121 
122 	/**
123 	 * @return The location of more information about the machine containing the
124 	 *         chip.
125 	 */
126 	URI getMachineRef() {
127 		return machineRef;
128 	}
129 
130 	void clearMachineRef() {
131 		machineRef = null;
132 	}
133 
134 	/**
135 	 * @return The global location of the chip at the root of the board
136 	 *         containing the chip being described.
137 	 */
138 	public ChipLocation getBoardChip() {
139 		return boardChip;
140 	}
141 
142 	/** @return The logical coordinates for the board containing the chip. */
143 	public TriadCoords getLogicalCoords() {
144 		return logicalCoords;
145 	}
146 
147 	/** @return The physical coordinates for the board containing the chip. */
148 	public PhysicalCoords getPhysicalCoords() {
149 		return physicalCoords;
150 	}
151 
152 	@JsonPOJOBuilder(withPrefix = "set")
153 	static class Builder {
154 		private Integer jobId;
155 
156 		private URI jobRef;
157 
158 		private ChipLocation jobChip;
159 
160 		private ChipLocation chip;
161 
162 		private String machineName;
163 
164 		private URI machineRef;
165 
166 		private ChipLocation boardChip;
167 
168 		private TriadCoords logicalCoords;
169 
170 		private PhysicalCoords physicalCoords;
171 
172 		void setJobId(Integer jobId) {
173 			this.jobId = jobId;
174 		}
175 
176 		void setJobRef(URI jobRef) {
177 			this.jobRef = jobRef;
178 		}
179 
180 		void setJobChip(ChipLocation jobChip) {
181 			this.jobChip = jobChip;
182 		}
183 
184 		void setChip(ChipLocation chip) {
185 			this.chip = chip;
186 		}
187 
188 		@JsonProperty("machine")
189 		@JsonAlias("machine-name")
190 		void setMachineName(String machineName) {
191 			this.machineName = machineName;
192 		}
193 
194 		void setMachineRef(URI machineRef) {
195 			this.machineRef = machineRef;
196 		}
197 
198 		void setBoardChip(ChipLocation boardChip) {
199 			this.boardChip = boardChip;
200 		}
201 
202 		@JsonProperty("logical-board-coordinates")
203 		void setLogicalCoords(TriadCoords logicalCoords) {
204 			this.logicalCoords = logicalCoords;
205 		}
206 
207 		@JsonProperty("physical-board-coordinates")
208 		void setPhysicalCoords(PhysicalCoords physicalCoords) {
209 			this.physicalCoords = physicalCoords;
210 		}
211 
212 		WhereIs build() {
213 			return new WhereIs(jobId, jobRef, jobChip, chip, machineName,
214 					machineRef, boardChip, logicalCoords, physicalCoords);
215 		}
216 	}
217 }
