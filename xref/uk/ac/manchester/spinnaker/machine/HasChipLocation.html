17  package uk.ac.manchester.spinnaker.machine;
18  
19  /**
20   * The interface supported by any object that is associated with a chip.
21   *
22   * @author Donal Fellows
23   */
24  public interface HasChipLocation {
25      /**
26       * @return The X coordinate of the chip.
27       */
28      int getX();
29  
30      /**
31       * @return The X coordinate of the chip.
32       */
33      int getY();
34  
35      /**
36       * Check if two locations are colocated at the chip level. This does
37       * <i>not</i> imply that the two are equal.
38       *
39       * @param other
40       *            The other location to compare to.
41       * @return If the two locations have the same X and Y coordinates.
42       */
43      default boolean onSameChipAs(HasChipLocation other) {
44          return (getX() == other.getX()) && (getY() == other.getY());
45      }
46  
47      /**
48       * Get the core of the chip that will be running SC&amp;MP. This is always
49       * core 0 of the chip, as the core that runs SC&amp;MP always maps itself to
50       * be virtual core ID 0.
51       *
52       * @return The location of the SC&amp;MP core.
53       */
54      default HasCoreLocation getScampCore() {
55          // SCAMP always runs on core 0 or we can't talk to the chip at all
56          return new CoreLocation(getX(), getY(), 0);
57      }
58  
59      /**
60       * Converts (if required) this to a simple X, Y tuple.
61       *
62       * @return A ChipLocation representation of the X and Y tuple
63       */
64      default ChipLocation asChipLocation() {
65          return new ChipLocation(getX(), getY());
66      }
67  }
