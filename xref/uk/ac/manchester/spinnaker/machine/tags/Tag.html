17  package uk.ac.manchester.spinnaker.machine.tags;
18  
19  import java.net.InetAddress;
20  
21  /** Common properties of SpiNNaker IP tags and reverse IP tags. */
22  public abstract class Tag implements Comparable<Tag> {
23      /** The board address associated with this tagID. */
24      private final InetAddress boardAddress;
25      /** The tagID ID associated with this tagID. */
26      private final int tagID;
27      /** The port number associated with this tagID. */
28      private Integer port;
29  
30      /**
31       * Create a tag.
32       *
33       * @param boardAddress
34       *            The address of the board where the tag is.
35       * @param tagID
36       *            The ID of the tag (0-7?)
37       * @param port
38       *            The port of the tag.
39       */
40      Tag(InetAddress boardAddress, int tagID, Integer port) {
41          this.boardAddress = boardAddress;
42          this.tagID = tagID;
43          this.port = port;
44      }
45  
46      /** @return The board address of the tag. */
47      public InetAddress getBoardAddress() {
48          return boardAddress;
49      }
50  
51      /**
52       * @return The ID of the tag. Note that this is the only property used for
53       *         determining order.
54       */
55      public int getTag() {
56          return tagID;
57      }
58  
59      /**
60       * @return The port of the tagID, or {@code null} if there isn't one yet.
61       */
62      public Integer getPort() {
63          return port;
64      }
65  
66      /**
67       * Set the port.
68       *
69       * @param port
70       *            the port to set
71       * @throws IllegalStateException
72       *             If the port has already been set.
73       */
74      public void setPort(int port) {
75          if (this.port != null && this.port != port) {
76              throw new IllegalStateException("port cannot be changed to " + port
77                      + " once set to " + this.port);
78          }
79          this.port = port;
80      }
81  
82      // Subclasses *must* create an equality test and hash code
83      /** {@inheritDoc} */
84      @Override
85      public abstract boolean equals(Object other);
86  
87      /** {@inheritDoc} */
88      @Override
89      public abstract int hashCode();
90  
91      /**
92       * Partial equality test between two tags. Only compares on fields defined
93       * in the {@link Tag} class. Used to make implementing a full equality test
94       * simpler.
95       *
96       * @param otherTag
97       *            The other tag to compare to.
98       * @return Whether the two tags are partially equal.
99       */
100     protected final boolean partialEquals(Tag otherTag) {
101         return tagID == otherTag.tagID
102                 && boardAddress.equals(otherTag.boardAddress)
103                 && (port == null ? otherTag.port == null
104                         : (otherTag.port != null
105                                 && port.equals(otherTag.port)));
106     }
107 
108     private static final int MAGIC1 = 43;
109     private static final int MAGIC2 = 17;
110 
111     /**
112      * Partial hash code of a tag. Used to make implementing a full hash
113      * simpler.
114      *
115      * @return The hash of this class's fields.
116      */
117     protected final int partialHashCode() {
118         int h = (tagID * MAGIC1) ^ boardAddress.hashCode();
119         if (port != null) {
120             h ^= port * MAGIC2;
121         }
122         return h;
123     }
124 
125     /**
126      * Compare this tag to another one for ordering <strong>Note:</strong> this
127      * class has a natural ordering that is inconsistent with equals; ordering
128      * is by the tag ID only (this is fine for tags obtained from the same chip
129      * at the same time, which is the primary use case).
130      * <p>
131      * {@inheritDoc}
132      */
133     @Override
134     public int compareTo(Tag o) {
135         return getTag() - o.getTag();
136     }
137 }
