16  package uk.ac.manchester.spinnaker.machine;
17  
18  /**
19   * A direction from one SpiNNaker chip to another.
20   *
21   * @author Christian-B
22   */
23  public enum Direction {
24  	/** Direction 0 typically towards a location x + 1, y. */
25  	EAST(0, +1, 0, "east"),
26  	/** Direction 1 typically towards a location x + 1, y +1. */
27  	NORTHEAST(1, +1, +1, "north_east"),
28  	/** Direction 2 typically towards a location x, y +1. */
29  	NORTH(2, 0, +1, "north"),
30  	/** Direction 3 typically towards a location x - 1, y. */
31  	WEST(3, -1, 0, "west"),
32  	/** Direction 4 typically towards a location x -1, y -1. */
33  	SOUTHWEST(4, -1, -1, "south_west"),
34  	/** Direction 5 typically towards a location x, y -1. */
35  	SOUTH(5, 0, -1, "south");
36  
37  	private static final Direction[] BY_ID = {
38  		EAST, NORTHEAST, NORTH, WEST, SOUTHWEST, SOUTH
39  	};
40  
41  	/** The ID of this direction when it is expressed as an {@code int}. */
42  	public final int id;
43  
44  	/**
45  	 * The typical change to X when moving in this direction.
46  	 */
47  	public final int xChange;
48  
49  	/**
50  	 * The typical change to Y when moving in this direction.
51  	 */
52  	public final int yChange;
53  
54  	/**
55  	 * The string representation, for example used in JSON.
56  	 */
57  	public final String label;
58  
59  	/**
60  	 * @param id
61  	 *            ID of this direction.
62  	 * @param xChange
63  	 *            Typical change to X if moving in this direction.
64  	 * @param yChange
65  	 *            Typical change to Y if moving in this direction.
66  	 */
67  	Direction(int id, int xChange, int yChange, String label) {
68  		this.id = id;
69  		this.xChange = xChange;
70  		this.yChange = yChange;
71  		this.label = label;
72  	}
73  
74  	/**
75  	 * Obtains the inverse direction.
76  	 *
77  	 * @return The inverse direction.
78  	 */
79  	Direction inverse() {
80  		switch (this) {
81  		case EAST:
82  			return WEST;
83  		case NORTHEAST:
84  			return SOUTHWEST;
85  		case NORTH:
86  			return SOUTH;
87  		case WEST:
88  			return EAST;
89  		case SOUTHWEST:
90  			return NORTHEAST;
91  		default:
92  			return NORTH;
93  		}
94  	}
95  
96  	/**
97  	 * The Direction with this ID when expressed as an {@code int}.
98  	 *
99  	 * @param id
100 	 *            ID of this Direction
101 	 * @return Direction with this ID
102 	 * @throws ArrayIndexOutOfBoundsException
103 	 *             if the ID is not correct
104 	 */
105 	public static Direction byId(int id) throws ArrayIndexOutOfBoundsException {
106 		if (id < 0 || id >= BY_ID.length) {
107 			throw new ArrayIndexOutOfBoundsException("direction ID " + id
108 					+ " not in range 0 to " + (BY_ID.length - 1));
109 		}
110 		return BY_ID[id];
111 	}
112 
113 	/**
114 	 * The Direction with this label
115 	 * <p>
116 	 * The current implementation assumes the labels are lower-case and words
117 	 * are separated by underscore (e.g., {@code north_east}).
118 	 *
119 	 * @param label
120 	 *            Label of a direction
121 	 * @return A valid direction
122 	 * @throws IllegalArgumentException
123 	 *             If no direction is found.
124 	 */
125 	public static Direction byLabel(String label) {
126 		for (var direction : Direction.values()) {
127 			if (direction.label.equals(label)) {
128 				return direction;
129 			}
130 		}
131 		throw new IllegalArgumentException(
132 				"No direction found for \"" + label + "\"");
133 	}
134 }
