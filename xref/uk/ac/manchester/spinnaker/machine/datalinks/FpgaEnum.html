200 	/**
201 	 * Constructor for the enum.
202 	 *
203 	 * @param x
204 	 *            X coordinate of the source Chip.
205 	 * @param y
206 	 *            Y coordinate of the source Chip.
207 	 * @param direction
208 	 *            Direction of the link as it comes out of the source Chip.
209 	 * @param fpgaId
210 	 *            ID of the FPGA device.
211 	 * @param id
212 	 *            ID of the link on this FPGA device.
213 	 */
214 	FpgaEnum(int x, int y, Direction direction, FpgaId fpgaId, int id) {
215 		this.x = x;
216 		this.y = y;
217 		this.direction = direction;
218 		this.fpgaId = fpgaId;
219 		this.id = id;
220 	}
221 
222 	@Override
223 	public int getX() {
224 		return x;
225 	}
226 
227 	@Override
228 	public int getY() {
229 		return y;
230 	}
231 
232 	@Override
233 	public ChipLocation asChipLocation() {
234 		return new ChipLocation(x, y);
235 	}
236 
237 	/**
238 	 * Find the enum by chip coordinates and the direction.
239 	 *
240 	 * @param x
241 	 *            X coordinate of chip.
242 	 * @param y
243 	 *            X coordinate of chip.
244 	 * @param direction
245 	 *            Direction of the link covered by the FPGA.
246 	 * @return The corresponding enum value.
247 	 * @throws IllegalArgumentException
248 	 *             If the arguments don't describe a supported FPGA link.
249 	 */
250 	public static FpgaEnum findId(int x, int y, Direction direction) {
251 		for (FpgaEnum linkId : FpgaEnum.values()) {
252 			if (linkId.x == x && linkId.y == y
253 					&& linkId.direction == direction) {
254 				return linkId;
255 			}
256 		}
257 		throw new IllegalArgumentException("No FPGA link found for y: " + x
258 				+ " x: " + y + " direction: " + direction);
259 	}
260 
261 	/**
262 	 * Find the enum by FPGA and link IDs.
263 	 *
264 	 * @param fpgaId
265 	 *            ID of the FPGA device.
266 	 * @param id
267 	 *            ID of the link on that device.
268 	 * @return The corresponding enum value.
269 	 * @throws IllegalArgumentException
270 	 *             If the arguments don't describe a supported FPGA link.
271 	 */
272 	public static FpgaEnum findId(FpgaId fpgaId, int id) {
273 		for (FpgaEnum linkId : FpgaEnum.values()) {
274 			if (linkId.fpgaId == fpgaId && linkId.id == id) {
275 				return linkId;
276 			}
277 		}
278 		throw new IllegalArgumentException(
279 				"No FPGA link found for fpgaId: " + fpgaId + " id: " + id);
280 	}
