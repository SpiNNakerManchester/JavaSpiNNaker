17  package uk.ac.manchester.spinnaker.machine;
18  
19  import static java.lang.Integer.compareUnsigned;
20  import static java.lang.String.format;
21  
22  /**
23   * A location in SpiNNaker or BMP memory. Does not say which address space this
24   * is in.
25   *
26   * @author Donal Fellows
27   */
28  public final class MemoryLocation implements Comparable<MemoryLocation> {
29  	/** Number of bytes in a SpiNNaker (ARM) word. */
30  	private static final int WORD_SIZE = 4;
31  
32  	/** The zero memory location. Often means "no actual address". */
33  	public static final MemoryLocation NULL = new MemoryLocation(0);
34  
35  	/** The actual location. */
36  	public final int address;
37  
38  	/** @param address The actual location. */
39  	public MemoryLocation(int address) {
40  		this.address = address;
41  	}
42  
43  	/** @param address The actual location. */
44  	public MemoryLocation(long address) {
45  		this.address = convert(address);
46  	}
47  
48  	/**
49  	 * Add an offset to this location to get a new memory location.
50  	 *
51  	 * @param offset
52  	 *            The offset to add.
53  	 * @return The new memory location.
54  	 */
55  	public MemoryLocation add(int offset) {
56  		return new MemoryLocation(address + offset);
57  	}
58  
59  	/**
60  	 * Get the difference between this location and another.
61  	 *
62  	 * @param other
63  	 *            The other location.
64  	 * @return This location's address minus the other location's address.
65  	 */
66  	public int diff(MemoryLocation other) {
67  		return address - other.address;
68  	}
69  
70  	/** @return Whether this location is really {@linkplain #NULL null}. */
71  	public boolean isNull() {
72  		return address == 0;
73  	}
74  
75  	/**
76  	 * How many bytes is this location's address above a word-aligned address?
77  	 *
78  	 * @return The number of bytes offset.
79  	 */
80  	public int subWordAlignment() {
81  		return address % WORD_SIZE;
82  	}
83  
84  	/** @return Whether this is a word-aligned location. */
85  	public boolean isAligned() {
86  		return subWordAlignment() == 0;
87  	}
88  
89  	@Override
90  	public int hashCode() {
91  		return address;
92  	}
93  
94  	@Override
95  	public boolean equals(Object other) {
96  		if (!(other instanceof MemoryLocation)) {
97  			return false;
98  		}
99  		MemoryLocation o = (MemoryLocation) other;
100 		return o.address == address;
101 	}
102 
103 	@Override
104 	public int compareTo(MemoryLocation other) {
105 		return compareUnsigned(address, other.address);
106 	}
107 
108 	/**
109 	 * Test if this location is less than another location.
110 	 *
111 	 * @param other
112 	 *            The other location.
113 	 * @return True if this location's address comes before.
114 	 */
115 	public boolean lessThan(MemoryLocation other) {
116 		return compareTo(other) < 0;
117 	}
118 
119 	/**
120 	 * Test if this location is greater than another location.
121 	 *
122 	 * @param other
123 	 *            The other location.
124 	 * @return True if this location's address comes after.
125 	 */
126 	public boolean greaterThan(MemoryLocation other) {
127 		return compareTo(other) > 0;
128 	}
129 
130 	@Override
131 	public String toString() {
132 		// Leading '0x' then exactly 8 hexadecimal characters; 10 chars total
133 		return format("%#010x", address);
134 	}
135 
136 	/** Maximum legal SpiNNaker address. It's a 32-bit machine. */
137 	private static final long MAX_ADDR = 0xFFFFFFFFL;
138 
139 	/**
140 	 * Convert an address to a 32-bit integer.
141 	 *
142 	 * @param baseAddress
143 	 *            The address as a long.
144 	 * @return the address as an int.
145 	 * @throws IllegalArgumentException
146 	 *             if the value is outside the unsigned 32-bit integer range.
147 	 */
148 	private static int convert(long baseAddress) {
149 		if (baseAddress < 0 || baseAddress > MAX_ADDR) {
150 			throw new IllegalArgumentException(
151 					"address must be in 32-bit unsigned integer range");
152 		}
153 		return (int) baseAddress;
154 	}
155 }
