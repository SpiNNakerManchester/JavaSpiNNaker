17  package uk.ac.manchester.spinnaker.utils;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.HashMap;
22  import java.util.function.Supplier;
23  
24  /**
25   * A map that will extend itself with new values (on get) when the key is
26   * otherwise absent from the map.
27   *
28   * @author Donal Fellows
29   *
30   * @param <K>
31   *            The type of keys.
32   * @param <V>
33   *            The type of values.
34   */
35  public class DefaultMap<K, V> extends HashMap<K, V> {
36  	private static final long serialVersionUID = -3805864660424802906L;
37  
38  	private final boolean direct;
39  
40  	private final V defValue;
41  
42  	private final Supplier<? extends V> defFactory;
43  
44  	private final KeyAwareFactory<? super K, ? extends V> advFactory;
45  
46  	/**
47  	 * Create a new map.
48  	 *
49  	 * @param <DV>
50  	 *            The type of the default value.
51  	 * @param defaultValue
52  	 *            The default value to use in the map. This should be an
53  	 *            immutable value as it can be potentially inserted for many
54  	 *            keys.
55  	 */
56  	public <DV extends V> DefaultMap(DV defaultValue) {
57  		direct = true;
58  		defValue = requireNonNull(defaultValue);
59  		defFactory = null;
60  		advFactory = null;
61  	}
62  
63  	/**
64  	 * Create a new map.
65       * <p>
66       * The defaultFactory can be a method or an Object.
67       * If V is mutable it is HIGHLY recommended to pass in a method and not an
68       *     Object.
69       * For example use ArrayList::new and not new ArrayList() otherwise the
70       *     Single will be used every time and values added after one get will
71       *     be in the default for the next get.
72       *
73  	 * @param defaultFactory
74  	 *         A method to create a new value/ an Object to insert in the map.
75   	 */
76  	public DefaultMap(Supplier<? extends V> defaultFactory) {
77  		direct = false;
78  		defValue = null;
79  		defFactory = requireNonNull(defaultFactory);
80  		advFactory = null;
81  	}
82  
83  	/**
84  	 * Create a new map.
85  	 *
86  	 * @param dummy
87  	 *            Just something to make this constructor distinct.
88  	 * @param defaultFactory
89  	 *            Can be an Object of a Class that implement KeyAwareFactory.
90       *            Can be a method expressed as a lambda.
91  	 */
92  	private DefaultMap(Marker dummy,
93  			KeyAwareFactory<? super K, ? extends V> defaultFactory) {
94  		direct = false;
95  		defValue = null;
96  		defFactory = null;
97  		advFactory = requireNonNull(defaultFactory);
98  	}
99  
100 	/**
101 	 * A marker used only to differentiate a constructor.
102 	 */
103 	private static class Marker {
104 		// Nothing interesting
105 		static final Marker INSTANCE = new Marker();
106 	}
107 
108 	/**
109 	 * Create a new map that manufactures new elements that are aware of their
110 	 * key from the beginning. This is done through this method because
111 	 * otherwise it clashes with the more common case of the unaware factory.
112 	 * <p>
113      * The Factory can be a lambda method to create a me value based on the key.
114      * <br>For example:
115      * <p>
116      * {@code DefaultMap.newAdvancedDefaultMap(i -> i*2);}
117      * <p>
118      * The Factory can also be a Object of a class that implements the
119      * KeyAwareFactory interface.
120      *
121 	 * @param <K>
122 	 *            The type of keys.
123 	 * @param <V>
124 	 *            The type of values.
125 	 * @param keyAwareFactory
126      *            Method or Object to create the default values.
127 	 *
128 	 * @return The new default map.
129 	 */
130 	public static <K, V> DefaultMap<K, V> newAdvancedDefaultMap(
131 			KeyAwareFactory<? super K, ? extends V> keyAwareFactory) {
132 		return new DefaultMap<>(Marker.INSTANCE, keyAwareFactory);
133 	}
134 
135 	private V defaultFactory(K key) {
136 		if (direct) {
137 			return defValue;
138 		}
139 		if (defFactory != null) {
140 			return defFactory.get();
141 		}
142 		return advFactory.createValue(key);
143 	}
144 
145 	/**
146 	 * Gets a value from the dictionary, inserting a newly manufactured value if
147 	 * the key has no mapping.
148 	 */
149 	@SuppressWarnings("unchecked")
150 	@Override
151 	public V get(Object key) {
152 		V value = super.get(key);
153 		if (value == null) {
154 			value = defaultFactory((K) key);
155 			put((K) key, value);
156 		}
157 		return value;
158 	}
159 
160 	/**
161 	 * An advanced factory that has access to the key when it is creating the
162 	 * value to associate with it.
163 	 *
164 	 * @author Donal Fellows
165 	 * @param <K>
166 	 *            The type of keys.
167 	 * @param <V>
168 	 *            The type of values.
169 	 */
170 	@FunctionalInterface
171 	public interface KeyAwareFactory<K, V> {
172 		/**
173 		 * Create a new value for the {@linkplain DefaultMap default map}.
174 		 *
175 		 * @param key
176 		 *            The key that will be used to store the value in the map.
177 		 * @return the value to store
178 		 */
179 		V createValue(K key);
180 	}
181 }
