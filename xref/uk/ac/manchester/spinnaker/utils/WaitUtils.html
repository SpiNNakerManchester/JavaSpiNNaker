17  package uk.ac.manchester.spinnaker.utils;
18  
19  import static java.lang.System.nanoTime;
20  import static java.lang.Thread.interrupted;
21  import static java.util.concurrent.locks.LockSupport.parkNanos;
22  
23  import java.util.concurrent.locks.LockSupport;
24  
25  /**
26   * Utilities for waiting very short periods of time.
27   *
28   * @author Donal Fellows
29   */
30  public abstract class WaitUtils {
31  	@UsedInJavadocOnly(LockSupport.class)
32  	private WaitUtils() {
33  	}
34  
35  	/**
36  	 * Wait until the given time has passed. May decide to not wait at all.
37  	 *
38  	 * @param nanoTimestamp
39  	 *            The first time at which the code may return. If in the past,
40  	 *            returns immediately.
41  	 * @return Whether the wait stopped because the thread was interrupted. The
42  	 *         caller <em>must</em> decide what to do in that case.
43  	 * @see System#nanoTime()
44  	 * @see LockSupport#parkNanos(long)
45  	 * @see <a href="https://stackoverflow.com/q/35875117/301832">Stack
46  	 *      Overflow</a>
47  	 */
48  	public static boolean waitUntil(long nanoTimestamp) {
49  		// Critical: this is static so JVM can inline this code!
50  		while (true) {
51  			long dt = nanoTimestamp - nanoTime();
52  			if (dt <= 0) {
53  				return interrupted();
54  			} else if (interrupted()) {
55  				return true;
56  			}
57  			parkNanos(dt);
58  		}
59  	}
60  }
