40  /**
41   * A context class that can disable dropping of packets on the SpiNNaker on-chip
42   * network. <em>Use very carefully indeed!</em> A network that can't drop
43   * packets is a network that can deadlock. It is not believed safe to use this
44   * class on more than one board at a time, and it is definitely not safe to do
45   * anything else than data transfers while the context is set.
46   *
47   * @author Donal Fellows
48   * @author Alan Stokes
49   */
50  public class NoDropPacketContext implements AutoCloseable {
51  	private static final Logger log = getLogger(NoDropPacketContext.class);
52  	private ReinjectionStatus lastStatus;
53  	private final CoreSubsets monitorCores;
54  	private final Transceiver txrx;
55  	private final ChipLocation firstChip;
56  	private final CoreSubsets gatherers;
57

59  	 * Standard short timeout for emergency routing.
60  	 */

63  	private static final RouterTimeoutdel/RouterTimeout.html#RouterTimeout">RouterTimeout TEMP_TIMEOUT = new RouterTimeout(15, 4);
64  	private static final RouterTimeoutdel/RouterTimeout.html#RouterTimeout">RouterTimeout ZERO_TIMEOUT = new RouterTimeout(0, 0);

66  	/**
67  	 * Create a no-drop-packets context. This can manage multiple boards at
68  	 * once, but it is <em>recommended</em> that only a single board be handled
69  	 * by a context.
70  	 *
71  	 * @param txrx
72  	 *            The transceiver to use for talking to SpiNNaker.
73  	 * @param monitorCores
74  	 *            The extra monitor cores on the SpiNNaker system that control
75  	 *            the routers. These must be on the same board as one of the
76  	 *            gatherers; this is not checked.
77  	 * @param gatherers
78  	 *            The gatherer cores on the SpiNNaker system that supports the
79  	 *            multicast router control API.
80  	 * @throws IOException
81  	 *             If communications fail.
82  	 * @throws ProcessException
83  	 *             If SCAMP or an extra monitor rejects a message.
84  	 */
85  	public NoDropPacketContext(Transceiver txrx, CoreSubsets monitorCores,
86  			CoreSubsets gatherers) throws IOException, ProcessException {
87  		this.txrx = txrx;
88  		this.monitorCores = monitorCores;
89  		// Store the last reinjection status for resetting
90  		// NOTE: This assumes the status is the same on all cores
91  		CoreLocation firstCore = monitorCores.iterator().next();
92  		firstChip = firstCore.asChipLocation();
93  		lastStatus = txrx.getReinjectionStatus(firstCore);
94  		this.gatherers = gatherers;
95  		log.info(
96  				"switching board at {} ({} monitor cores) to "
97  						+ "non-drop mode (saved status: {})",
98  				firstChip, monitorCores.size(), lastStatus);
99  		try {
100 			// Set to not inject dropped packets
101 			txrx.setReinjection(monitorCores, false);
102 			// Clear any outstanding packets from reinjection
103 			txrx.clearReinjectionQueues(gatherers);
104 			// Set time outs
105 			txrx.setReinjectionEmergencyTimeout(gatherers, SHORT_TIMEOUT);
106 			txrx.setReinjectionTimeout(gatherers, LONG_TIMEOUT);
107 		} catch (IOException | ProcessException e) {
108 			log.error("failed to switch board at {} to non-drop mode",
109 					firstChip, e);
110 			throw e;
111 		}
112 	}

185 	private static CoreSubsets convertToCoreSubset(
186 			List<? extends HasCoreLocation> coreLocationList) {
187 		CoreSubsets cores = new CoreSubsets();
188 		for (HasCoreLocation coreLocation : coreLocationList) {
189 			cores.addCore(coreLocation.asCoreLocation());
190 		}
191 		return cores;
192 	}
193 
194 	private static CoreSubsets convertToCoreSubset(Gather gather) {
195 		CoreSubsets cores = new CoreSubsets();
196 		cores.addCore(gather.asCoreLocation());
197 		return cores;
198 	}
199 
200 	private static CoreSubsets convertToCoreSubset(
201 			Stream<? extends HasCoreLocation> coreLocations) {
202 		CoreSubsets cores = new CoreSubsets();
203 		coreLocations.forEach(loc -> cores.addCore(loc.asCoreLocation()));
204 		return cores;
205 	}

207 	/**
208 	 * Restore the SpiNNaker board (or boards, for the brave) to its normal
209 	 * operating mode.
210 	 *
211 	 * @throws IOException
212 	 *             If communications fail.
213 	 * @throws ProcessException
214 	 *             If SCAMP or an extra monitor rejects a message.
215 	 */
216 	@Override
217 	public void close() throws IOException, ProcessException {
218 		log.info("switching board at {} to standard mode", firstChip);
219 		quietlySetTemporaryTimeouts();
220 
221 		try {
222 			// Do the real reset
223 			txrx.setReinjectionTimeout(gatherers, lastStatus);
224 			txrx.setReinjectionEmergencyTimeout(gatherers, lastStatus);
225 			txrx.setReinjection(monitorCores, lastStatus);
226 			log.debug("switched board at {} to standard mode", firstChip);
227 			return;
228 		} catch (Exception e) {
229 			log.error("error resetting router timeouts", e);
230 		}
231 		try {
232 			log.error("checking to see of the cores are OK...");
233 			Map<CoreLocation, CPUInfo> errorCores =
234 					txrx.getCoresNotInState(monitorCores, RUNNING);
235 			if (!errorCores.isEmpty()) {
236 				log.error("cores in an unexpected state: {}", errorCores);
237 			}
238 		} catch (Exception e) {
239 			log.error("couldn't get core state", e);
240 		}
241 	}
242 
243 	/**
244 	 * This sets some temporary timeouts so that we can use SDP more safely. We
245 	 * hope. Failures are ignored; if they happen, failures when setting the
246 	 * real values are also likely and we'll get error messages then.
247 	 */
248 	private void quietlySetTemporaryTimeouts() {
249 		try {
250 			txrx.setReinjectionTimeout(gatherers, TEMP_TIMEOUT);
251 		} catch (Exception ignored) {
252 		}
253 		try {
254 			txrx.setReinjectionEmergencyTimeout(gatherers, ZERO_TIMEOUT);
255 		} catch (Exception ignored) {
256 		}
257 	}
