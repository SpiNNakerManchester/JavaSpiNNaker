39  /**
40   * A context class that can disable dropping of packets on the SpiNNaker on-chip
41   * network. <em>Use very carefully indeed!</em> A network that can't drop
42   * packets is a network that can deadlock. It is not believed safe to use this
43   * class on more than one board at a time, and it is definitely not safe to do
44   * anything else than data transfers while the context is set.
45   *
46   * @author Donal Fellows
47   * @author Alan Stokes
48   */
49  public class NoDropPacketContext implements AutoCloseable {
50  	private static final Logger log = getLogger(NoDropPacketContext.class);
51  
52  	private ReinjectionStatus lastStatus;
53  
54  	private final CoreSubsets monitorCores;
55  
56  	private final TransceiverInterface txrx;
57  
58  	private final ChipLocation firstChip;
59  
60  	private final CoreSubsets gatherers;
61  
62  	/**
63  	 * Standard short timeout for emergency routing.
64  	 */
65  	private static final RouterTimeout SHORT_TIMEOUT = new RouterTimeout(1, 1);
66  
67  	private static final RouterTimeout LONG_TIMEOUT = new RouterTimeout(14, 14);
68  
69  	private static final RouterTimeout TEMP_TIMEOUT = new RouterTimeout(15, 4);
70  
71  	private static final RouterTimeout ZERO_TIMEOUT = new RouterTimeout(0, 0);
72  
73  	/**
74  	 * Create a no-drop-packets context. This can manage multiple boards at
75  	 * once, but it is <em>recommended</em> that only a single board be handled
76  	 * by a context.
77  	 *
78  	 * @param txrx
79  	 *            The transceiver to use for talking to SpiNNaker.
80  	 * @param monitorCores
81  	 *            The extra monitor cores on the SpiNNaker system that control
82  	 *            the routers. These must be on the same board as one of the
83  	 *            gatherers; this is not checked.
84  	 * @param gatherers
85  	 *            The gatherer cores on the SpiNNaker system that supports the
86  	 *            multicast router control API.
87  	 * @throws IOException
88  	 *             If communications fail.
89  	 * @throws ProcessException
90  	 *             If SCAMP or an extra monitor rejects a message.
91  	 * @throws InterruptedException
92  	 *             If communications are interrupted.
93  	 */
94  	@MustBeClosed
95  	public NoDropPacketContext(TransceiverInterface txrx,
96  			CoreSubsets monitorCores, CoreSubsets gatherers)
97  			throws IOException, ProcessException, InterruptedException {
98  		this.txrx = txrx;
99  		this.monitorCores = monitorCores;
100 		// Store the last reinjection status for resetting
101 		// NOTE: This assumes the status is the same on all cores
102 		var firstCore = monitorCores.first().orElseThrow();
103 		firstChip = firstCore.asChipLocation();
104 		lastStatus = txrx.getReinjectionStatus(firstCore);
105 		this.gatherers = gatherers;
106 		log.info(
107 				"switching board at {} ({} monitor cores) to "
108 						+ "non-drop mode (saved status: {})",
109 				firstChip, monitorCores.size(), lastStatus);
110 		try {
111 			// Set to not inject dropped packets
112 			txrx.setReinjection(monitorCores, false);
113 			// Clear any outstanding packets from reinjection
114 			txrx.clearReinjectionQueues(gatherers);
115 			// Set time outs
116 			txrx.setReinjectionEmergencyTimeout(gatherers, SHORT_TIMEOUT);
117 			txrx.setReinjectionTimeout(gatherers, LONG_TIMEOUT);
118 		} catch (IOException | ProcessException e) {
119 			log.error("failed to switch board at {} to non-drop mode",
120 					firstChip, e);
121 			throw e;
122 		}
123 	}

129 	 *            The transceiver to use for talking to SpiNNaker.

133 	 *            this is not checked.
134 	 * @param gatherer
135 	 *            The gatherer for this context and linked to these extra
136 	 *            monitor cores.
137 	 * @throws IOException
138 	 *             If communications fail.
139 	 * @throws ProcessException
140 	 *             If SCAMP or an extra monitor rejects a message.
141 	 * @throws InterruptedException
142 	 *             If communications are interrupted.

146 			CoreSubsets monitorCoreLocations, Gather gatherer)
147 			throws IOException, ProcessException, InterruptedException {
148 		this(txrx, monitorCoreLocations, convertToCoreSubset(gatherer));

207 	private static CoreSubsets convertToCoreSubset(
208 			List<? extends HasCoreLocation> coreLocationList) {
209 		var cores = new CoreSubsets();
210 		for (var coreLocation : coreLocationList) {
211 			cores.addCore(coreLocation.asCoreLocation());
212 		}
213 		return cores;
214 	}
215 
216 	private static CoreSubsets convertToCoreSubset(Gather gather) {
217 		var cores = new CoreSubsets();
218 		cores.addCore(gather.asCoreLocation());
219 		return cores;
220 	}
221 
222 	private static CoreSubsets convertToCoreSubset(
223 			Stream<? extends HasCoreLocation> coreLocations) {
224 		var cores = new CoreSubsets();
225 		coreLocations.forEach(loc -> cores.addCore(loc.asCoreLocation()));
226 		return cores;
227 	}

229 	/**
230 	 * Restore the SpiNNaker board (or boards, for the brave) to its normal
231 	 * operating mode.
232 	 *
233 	 * @throws IOException
234 	 *             If communications fail.
235 	 * @throws ProcessException
236 	 *             If SCAMP or an extra monitor rejects a message.
237 	 * @throws InterruptedException
238 	 *             If communications are interrupted.
239 	 */
240 	@Override
241 	public void close()
242 			throws IOException, ProcessException, InterruptedException {
243 		log.info("switching board at {} to standard mode", firstChip);
244 		quietlySetTemporaryTimeouts();
245 
246 		try {
247 			// Do the real reset
248 			txrx.setReinjectionTimeout(gatherers, lastStatus);
249 			txrx.setReinjectionEmergencyTimeout(gatherers, lastStatus);
250 			txrx.setReinjection(monitorCores, lastStatus);
251 			log.debug("switched board at {} to standard mode", firstChip);
252 			return;
253 		} catch (IOException | ProcessException | InterruptedException
254 				| RuntimeException e) {
255 			log.error("error resetting router timeouts", e);
256 			throw e;
257 		} catch (Exception e) {
258 			log.error("error resetting router timeouts", e);
259 		}
260 		try {
261 			log.error("checking to see of the cores are OK...");
262 			var errorCores = txrx.getCoresNotInState(monitorCores, RUNNING);
263 			if (!errorCores.isEmpty()) {
264 				log.error("cores in an unexpected state: {}", errorCores);
265 			}
266 		} catch (Exception e) {
267 			log.error("couldn't get core state", e);
268 		}
269 	}
270 
271 	/**
272 	 * This sets some temporary timeouts so that we can use SDP more safely. We
273 	 * hope. Failures are ignored; if they happen, failures when setting the
274 	 * real values are also likely and we'll get error messages then.
275 	 */
276 	private void quietlySetTemporaryTimeouts() {
277 		try {
278 			txrx.setReinjectionTimeout(gatherers, TEMP_TIMEOUT);
279 		} catch (Exception e) {
280 			log.debug("failed to reset reinjection timeout", e);
281 		}
282 		try {
283 			txrx.setReinjectionEmergencyTimeout(gatherers, ZERO_TIMEOUT);
284 		} catch (Exception e) {
285 			log.debug("failed to reset emergency reinjection timeout", e);
286 		}
287 	}
