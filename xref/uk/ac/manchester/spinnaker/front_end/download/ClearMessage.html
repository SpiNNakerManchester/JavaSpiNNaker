16  package uk.ac.manchester.spinnaker.front_end.download;
17  
18  import static java.nio.ByteBuffer.allocate;
19  import static java.nio.ByteOrder.LITTLE_ENDIAN;
20  import static uk.ac.manchester.spinnaker.front_end.download.GatherProtocolMessage.ID.CLEAR_TRANSMISSIONS;
21  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
22  import static uk.ac.manchester.spinnaker.messages.sdp.SDPPort.EXTRA_MONITOR_CORE_DATA_SPEED_UP;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
27  
28  /**
29   * A message used to request fast data transfer from SpiNNaker to Host.
30   *
31   * @author Alan Stokes
32   */
33  
34  public final class ClearMessage extends GatherProtocolMessage {
35  	private static final int NUM_WORDS = 2;
36  
37  	/**
38  	 * Create a message used to request fast data transfer from SpiNNaker to
39  	 * Host.
40  	 *
41  	 * @param destination
42  	 *            Where to send the message
43  	 * @param transactionId
44  	 *            the transaction id needed
45  	 * @return The created message.
46  	 */
47  	static ClearMessage create(HasCoreLocation destination, int transactionId) {
48  		var payload = allocate(NUM_WORDS * WORD_SIZE).order(LITTLE_ENDIAN);
49  		payload.putInt(CLEAR_TRANSMISSIONS.value);
50  		payload.putInt(transactionId);
51  		payload.flip();
52  		return new ClearMessage(destination, payload);
53  	}
54  
55  	private ClearMessage(HasCoreLocation destination, ByteBuffer payload) {
56  		super(destination, EXTRA_MONITOR_CORE_DATA_SPEED_UP, payload);
57  	}
58  }
