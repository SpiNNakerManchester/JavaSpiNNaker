17  package uk.ac.manchester.spinnaker.front_end.download;
18  
19  import static java.nio.ByteBuffer.allocate;
20  import static java.nio.ByteOrder.LITTLE_ENDIAN;
21  import static uk.ac.manchester.spinnaker.front_end.download.GatherProtocolMessage.ID.START_SENDING_DATA;
22  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
23  import static uk.ac.manchester.spinnaker.messages.sdp.SDPPort.EXTRA_MONITOR_CORE_DATA_SPEED_UP;
24  
25  import java.nio.ByteBuffer;
26  
27  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
28  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
29  
30  /**
31   * A message used to request fast data transfer from SpiNNaker to Host.
32   */
33  public final class StartSendingMessage extends GatherProtocolMessage {
34  	private static final int NUM_WORDS = 4;
35  
36  	/**
37  	 * Create a message used to request fast data transfer from SpiNNaker to
38  	 * Host.
39  	 *
40  	 * @param destination
41  	 *            Where to send the message
42  	 * @param address
43  	 *            Where to start reading from
44  	 * @param length
45  	 *            How many bytes to read
46  	 * @param transactionId
47  	 *            the transaction ID needed
48  	 * @return The created message.
49  	 */
50  	static StartSendingMessage create(HasCoreLocation destination,
51  			MemoryLocation address, int length, int transactionId) {
52  		var payload = allocate(NUM_WORDS * WORD_SIZE).order(LITTLE_ENDIAN);
53  		payload.putInt(START_SENDING_DATA.value);
54  		payload.putInt(transactionId);
55  		payload.putInt(address.address);
56  		payload.putInt(length);
57  		payload.flip();
58  		return new StartSendingMessage(destination, payload);
59  	}
60  
61  	private StartSendingMessage(HasCoreLocation destination,
62  			ByteBuffer payload) {
63  		super(destination, EXTRA_MONITOR_CORE_DATA_SPEED_UP, payload);
64  	}
65  }
