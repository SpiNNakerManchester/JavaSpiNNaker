31  import java.util.Map;
32  import java.util.concurrent.ExecutorService;

35

39  import uk.ac.manchester.spinnaker.front_end.download.request.Placement;
40  import uk.ac.manchester.spinnaker.machine.Machine;
41  import uk.ac.manchester.spinnaker.storage.BufferManagerStorage;
42  import uk.ac.manchester.spinnaker.storage.BufferManagerStorage.Region;

47  /**
48   * A data gatherer that pulls the data from a recording region. Internally, this
49   * accepts requests to store data (after it has been retrieved from SpiNNaker)
50   * in parallel and passes them to a worker thread so that only that thread needs
51   * to hold a write transaction open.
52   *
53   * @author Donal Fellows
54   */
55  public class RecordingRegionDataGatherer extends DataGatherer {
56  	/**
57  	 * How long a termination delay has to be to be worth reporting, in
58  	 * milliseconds.
59  	 */
60  	private static final int TERMINATION_REPORT_THRESHOLD = 250;
61  
62  	private static final Logger log =
63  			getLogger(RecordingRegionDataGatherer.class);
64  
65  	private final TransceiverInterface txrx;
66  
67  	private final BufferManagerStorage database;
68  
69  	@GuardedBy("itself")
70  	private final Map<Placement, List<RecordingRegion>> recordingRegions =
71  			new HashMap<>();
72  
73  	private final ExecutorService dbWorker = newSingleThreadExecutor();
74  
75  	private int numWrites = 0;
76  
77  	/**
78  	 * Create a data gatherer.
79  	 *
80  	 * @param transceiver
81  	 *            How to talk to the machine.
82  	 * @param machine
83  	 *            The description of the machine talked to.
84  	 * @param database
85  	 *            Where to put the retrieved data.
86  	 * @throws ProcessException
87  	 *             If we can't discover the machine details due to SpiNNaker
88  	 *             rejecting messages
89  	 * @throws IOException
90  	 *             If we can't discover the machine details due to I/O problems
91  	 */
92  	@MustBeClosed
93  	public RecordingRegionDataGatherer(TransceiverInterface transceiver,
94  			Machine machine, BufferManagerStorage database)
95  			throws IOException, ProcessException {
96  		super(transceiver, machine);
97  		this.txrx = transceiver;
98  		this.database = database;
99  	}
100 
101 	private List<RecordingRegion> getRegions(Placement placement)
102 			throws IOException, ProcessException {
103 		// Cheap check first
104 		synchronized (recordingRegions) {
105 			var regions = recordingRegions.get(placement);
106 			if (regions != null) {
107 				return regions;
108 			}
109 		}
110 
111 		// Need to go to the machine; don't hold the lock while doing so
112 		var regions = getRecordingRegionDescriptors(txrx, placement);
113 		synchronized (recordingRegions) {
114 			// Put the value in the map if it wasn't already there
115 			return recordingRegions.computeIfAbsent(placement, __ -> regions);
116 		}
117 	}
118 
119 	@Override
120 	protected List<Region> getRegion(Placement placement, int index)
121 			throws IOException, ProcessException {
122 		var region = getRegions(placement).get(index);
123 		log.debug("got region of {} R:{} as {}", placement.asCoreLocation(),
124 				index, region);
125 		if (region.size > 0) {
126 			return List.of(new Region(placement, index, region.data,
127 					(int) region.size));
128 		}
129 		return List.of();
130 	}
131 
132 	@Override
133 	protected void storeData(Region r, ByteBuffer data) {
134 		if (data == null) {
135 			log.warn("failed to download data for {} R:{} from {}:{}", r.core,
136 					r.regionIndex, r.startAddress, r.size);
137 			return;
138 		}
139 		dbWorker.execute(() -> {
140 			log.info("storing region data for {} R:{} from {} as {} bytes",
141 					r.core, r.regionIndex, r.startAddress, data.remaining());
142 			try {
143 				database.appendRecordingContents(r, data);
144 				numWrites++;
145 			} catch (StorageException e) {
146 				log.error("failed to write to database", e);
147 			}
148 		});
149 	}
150 
151 	@Override
152 	public void close() throws InterruptedException {
153 		super.close();
154 		log.info("waiting for database usage to complete");
155 		long start = currentTimeMillis();
156 		dbWorker.shutdown();
157 		// It really shouldn't take a minute to finish
158 		dbWorker.awaitTermination(1, MINUTES);
159 		long end = currentTimeMillis();
160 		log.info("total of {} database writes done", numWrites);
161 		if (end - start > TERMINATION_REPORT_THRESHOLD) {
162 			double diff = (end - start) / (double) MSEC_PER_SEC;
163 			log.info("DB shutdown took {}s", format("%.2f", diff));
164 		}
165 	}
166 }
