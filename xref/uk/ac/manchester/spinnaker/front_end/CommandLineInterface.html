103 	static class BuildPropsLoader implements IModelTransformer {
104 		private Properties loadProps() throws IOException {
105 			var prop = new Properties();
106 			try (var clp = getClass().getResourceAsStream(PROPS)) {
107 				prop.load(clp);
108 			}
109 			return prop;
110 		}
111 
112 		@Override
113 		public CommandSpec transform(CommandSpec commandSpec) {
114 			try {
115 				var prop = loadProps();
116 				var jar = prop.getProperty("jar");
117 				var ver = prop.getProperty("version");
118 
119 				commandSpec.name("java -jar " + jar);
120 				commandSpec.version(jar + " version " + ver);
121 			} catch (IOException e) {
122 				e.printStackTrace();
123 				System.exit(USAGE);
124 			}
125 			return commandSpec;
126 		}
127 	}

134 	/**
135 	 * The main command line interface. Dispatches to other classes based on the
136 	 * first argument, which is a command word.
137 	 *
138 	 * @param args
139 	 *            The command line arguments.
140 	 */
141 	public static void main(String... args) {
142 		var cmd = new CommandLine(new CommandLineInterface());
143 		if (args.length == 0) {
144 			cmd.usage(cmd.getErr());
145 			System.exit(USAGE);
146 		} else {
147 			System.exit(cmd.execute(args));
148 		}
149 	}

170 	/**
171 	 * Run the data specifications in parallel.
172 	 *
173 	 * @param machine
174 	 *            Description of overall machine.
175 	 * @param runFolder
176 	 *            Directory containing per-run information (i.e., the database
177 	 *            that holds the data specifications to execute).
178 	 * @param filterSystemCores
179 	 *            If {@code true}, only run the DSE for application vertices. If
180 	 *            {@code false}, only run the DSE for system vertices. If
181 	 *            {@code null}, run the DSE for all vertices.
182 	 * @throws IOException
183 	 *             If the communications fail.
184 	 * @throws SpinnmanException
185 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
186 	 * @throws StorageException
187 	 *             If the database is in an illegal state.
188 	 * @throws ExecutionException
189 	 *             If there was a problem in the parallel queue.
190 	 * @throws InterruptedException
191 	 *             If the wait for everything to complete is interrupted.
192 	 * @throws DataSpecificationException
193 	 *             If an invalid data specification file is executed.
194 	 * @throws URISyntaxException
195 	 *             If the proxy URI is provided but not valid.
196 	 */
197 	public void runDSEUploadingViaClassicTransfer(Machine machine,
198 			File runFolder, Boolean filterSystemCores)
199 			throws IOException, SpinnmanException, StorageException,
200 			ExecutionException, InterruptedException,
201 			DataSpecificationException, URISyntaxException {
202 		setLoggerDir(runFolder);
203 		var db = getDataSpecDB(runFolder);
204 
205 		try (var dseExec = new HostExecuteDataSpecification(machine, db)) {
206 			if (filterSystemCores == null) {
207 				dseExec.loadAllCores();
208 			} else if (filterSystemCores) {
209 				dseExec.loadApplicationCores();
210 			} else {
211 				dseExec.loadSystemCores();
212 			}
213 		}
214 	}

226 	 * @param reportFolder
227 	 *            Directory containing reports. If {@link Optional#empty()}, no
228 	 *            report will be written.
229 	 * @throws IOException
230 	 *             If the communications fail.
231 	 * @throws SpinnmanException
232 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
233 	 * @throws StorageException
234 	 *             If the database is in an illegal state.
235 	 * @throws ExecutionException
236 	 *             If there was a problem in the parallel queue.
237 	 * @throws InterruptedException
238 	 *             If the wait for everything to complete is interrupted.
239 	 * @throws DataSpecificationException
240 	 *             If an invalid data specification file is executed.
241 	 * @throws URISyntaxException
242 	 *             If a proxy URI is provided but invalid.
243 	 */
244 	@Command(name = "dse_app_mon", description = DSE_MON_DESC)
245 	public void runDSEForAppCoresUploadingViaMonitorStreaming(
246 			@Mixin GatherersParam gatherers, @Mixin MachineParam machine,
247 			@Parameters(description = RUN) File runFolder,
248 			@Parameters(description = REPORT, arity = "0..1") Optional<
249 					File> reportFolder)
250 			throws IOException, SpinnmanException, StorageException,
251 			ExecutionException, InterruptedException,
252 			DataSpecificationException, URISyntaxException {
253 		setLoggerDir(runFolder);
254 		var db = getDataSpecDB(runFolder);
255 
256 		try (var dseExec = new FastExecuteDataSpecification(machine.get(),
257 				gatherers.get(), reportFolder.orElse(null), db)) {
258 			dseExec.loadCores();
259 		}
260 	}

262 	/**
263 	 * Retrieve IOBUFs in parallel.
264 	 *
265 	 * @param machine
266 	 *            Description of overall machine.
267 	 * @param iobuf
268 	 *            Mapping from APLX executable names (full paths) to what cores
269 	 *            are running those executables, and which we will download
270 	 *            IOBUFs for.
271 	 * @param runFolder
272 	 *            Name of directory containing per-run information (i.e., the
273 	 *            database that holds the data specifications to execute).
274 	 * @throws IOException
275 	 *             If the communications fail.
276 	 * @throws SpinnmanException
277 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
278 	 * @throws InterruptedException
279 	 *             If interrupted (not expected).
280 	 * @throws URISyntaxException
281 	 *             If the proxy URI is invalid
282 	 * @throws StorageException
283 	 *             If there is an error reading the database
284 	 */
285 	@Command(name = "iobuf", description = IOBUF_DESC)
286 	public void retrieveIOBUFs(@Mixin MachineParam machine,
287 			@Mixin IobufMapParam iobuf,
288 			@Parameters(description = RUN) File runFolder)
289 			throws IOException, SpinnmanException, InterruptedException,
290 			StorageException, URISyntaxException {
291 		setLoggerDir(runFolder);
292 		var db = getBufferManagerDB(runFolder);
293 		var job = getJob(db);
294 
295 		try (var txrx = getTransceiver(machine.get(), job);
296 				var r = new IobufRetriever(txrx, machine.get(),
297 						PARALLEL_SIZE)) {
298 			var result = r.retrieveIobufContents(iobuf.get(), runFolder);
299 			MAPPER.writeValue(System.out, result);
300 		}
301 	}

343 	 * @param gatherers
344 	 *            List of descriptions of gatherers.
345 	 * @param machine
346 	 *            Description of overall machine.
347 	 * @param runFolder
348 	 *            Directory containing per-run information (i.e., the database
349 	 *            that receives the output).
350 	 * @throws IOException
351 	 *             If the communications fail
352 	 * @throws SpinnmanException
353 	 *             If a BMP is uncontactable or SpiNNaker rejects a message
354 	 * @throws StorageException
355 	 *             If the database is in an illegal state
356 	 * @throws InterruptedException
357 	 *             If things are interrupted while waiting for all the downloads
358 	 *             to be done
359 	 * @throws URISyntaxException
360 	 *             If the URI of the proxy is invalid
361 	 */
362 	@Command(name = "gather", description = GATHER_DESC)
363 	public void downloadRecordingChannelsViaMonitorStreaming(
364 			@Mixin GatherersParam gatherers, @Mixin MachineParam machine,
365 			@Parameters(description = RUN) File runFolder)
366 			throws IOException, SpinnmanException, StorageException,
367 			InterruptedException, URISyntaxException {
368 		setLoggerDir(runFolder);
369 		var db = getBufferManagerDB(runFolder);
370 		var job = getJob(db);
371 
372 		try (var trans = getTransceiver(machine.get(), job);
373 				var r = new RecordingRegionDataGatherer(trans, machine.get(),
374 						db, job)) {
375 			int misses = r.gather(gatherers.get());
376 			getLogger(CommandLineInterface.class).info("total misses: {}",
377 					misses);
378 		}
379 	}
