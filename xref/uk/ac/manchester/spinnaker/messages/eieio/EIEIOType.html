17  package uk.ac.manchester.spinnaker.messages.eieio;
18  
19  import static java.util.Objects.requireNonNull;
20  import static uk.ac.manchester.spinnaker.messages.Constants.NBBY;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  /** Possible types of EIEIO packets. */
26  public enum EIEIOType {
27  	/** Indicates that data is keys which are 16 bits. */
28  	KEY_16_BIT(0, 2, 0),
29  	/** Indicates that data is keys and payloads of 16 bits. */
30  	KEY_PAYLOAD_16_BIT(1, 2, 2),
31  	/** Indicates that data is keys of 32 bits. */
32  	KEY_32_BIT(2, 4, 0),
33  	/** Indicates that data is keys and payloads of 32 bits. */
34  	KEY_PAYLOAD_32_BIT(3, 4, 4);
35  
36  	private static final Map<Integer, EIEIOType> MAP;
37  
38  	static {
39  		MAP = new HashMap<>();
40  		for (EIEIOType v : values()) {
41  			MAP.put(v.value, v);
42  		}
43  	}
44  
45  	private final int value;
46  
47  	/** The number of bytes used by each key element. */
48  	public final int keyBytes;
49  
50  	/** The number of bytes used by each payload element. */
51  	public final int payloadBytes;
52  
53  	/** The maximum value of the key or payload (if there is a payload). */
54  	public final long maxValue;
55  
56  	/**
57  	 * @param value the value
58  	 * @param keyBytes the number of bytes per key
59  	 * @param payloadBytes the number of bytes per value
60  	 */
61  	EIEIOType(int value, int keyBytes, int payloadBytes) {
62  		this.value = value;
63  		this.keyBytes = keyBytes;
64  		this.payloadBytes = payloadBytes;
65  		this.maxValue = (1L << (keyBytes * NBBY)) - 1;
66  	}
67  
68  	/** @return The encoded type. */
69  	public int getValue() {
70  		return value;
71  	}
72  
73  	/**
74  	 * Get the type given its encoded form.
75  	 *
76  	 * @param value
77  	 *            The encoded type.
78  	 * @return The type object
79  	 * @throws IllegalArgumentException
80  	 *             if the encoded type is unrecognised.
81  	 */
82  	public static EIEIOType getByValue(int value) {
83  		return requireNonNull(MAP.get(value), "no such type");
84  	}
85  }
