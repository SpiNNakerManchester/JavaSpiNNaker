47  /** Represents a set of boot messages to be sent to boot the board. */
48  public class BootMessages {
49  	/**
50  	 * The (maximum) number of data words in an individual boot data message.
51  	 */
52  	static final int BOOT_MESSAGE_DATA_WORDS = 256;
53  	private static final int BOOT_MESSAGE_DATA_BYTES =
54  			BOOT_MESSAGE_DATA_WORDS * WORD_SIZE;
55  	private static final int BOOT_IMAGE_MAX_BYTES = 32 * 1024;
56  	private static final int BOOT_STRUCT_REPLACE_OFFSET = 384;
57  	private static final int BOOT_STRUCT_REPLACE_LENGTH = 128;
58  	private static final String BOOT_IMAGE = "scamp.boot";
59  
60  	private final ByteBuffer bootData;
61  	private final int numDataPackets;
62  
63  	private static SystemVariableBootValues initFlags(
64  			SystemVariableBootValues bootVars) {
65  		SystemVariableBootValues specific =
66  				bootVars == null ? new SystemVariableBootValues()
67  						: new SystemVariableBootValues(bootVars);
68  		int currentTime = (int) (System.currentTimeMillis() / MSEC_PER_SEC);
69  		specific.setValue(unix_timestamp, currentTime);
70  		specific.setValue(boot_signature, currentTime);
71  		specific.setValue(is_root_chip, 1);
72  		return specific;
73  	}
74  
75  	private BootMessages(SystemVariableBootValues bootVariables,
76  			Map<SystemVariableDefinition, Object> extraBootValues) {
77  		bootVariables = initFlags(bootVariables);
78  		if (extraBootValues != null) {
79  			for (Entry<SystemVariableDefinition, Object> entry : extraBootValues
80  					.entrySet()) {
81  				bootVariables.setValue(entry.getKey(), entry.getValue());
82  			}
83  		}
84  		bootData = readBootImage(getClass().getResource(BOOT_IMAGE));
85  		injectBootVariableBlock(bootVariables);
86  		numDataPackets =
87  				(int) ceil(bootData.limit() / (float) BOOT_MESSAGE_DATA_BYTES);
88  	}
89  
90  	private void injectBootVariableBlock(
91  			SystemVariableBootValues bootVariables) {
92  		// NB: Endian shenanigans!
93  		ByteBuffer buffer = allocate(BOOT_VARIABLE_SIZE).order(LITTLE_ENDIAN);
94  		bootVariables.addToBuffer(buffer);
95  		buffer.position(0);
96  		for (int i = 0; i < BOOT_STRUCT_REPLACE_LENGTH / WORD_SIZE; i++) {
97  			bootData.putInt(BOOT_STRUCT_REPLACE_OFFSET + i * WORD_SIZE,
98  					buffer.getInt());
99  		}
100 	}
101 
102 	private static ByteBuffer readBootImage(URL bootImage) {
103 		// NB: This data is BIG endian!
104 		ByteBuffer buffer =
105 				allocate(BOOT_IMAGE_MAX_BYTES + WORD_SIZE).order(BIG_ENDIAN);
106 
107 		try (DataInputStream is = new DataInputStream(bootImage.openStream())) {
108 			while (true) {
109 				// NB: Byte-swap the data from the file!
110 				buffer.putInt(reverseBytes(is.readInt()));
111 			}
112 		} catch (EOFException e) {
113 			// EOF signifies end of file; nothing to do here
114 		} catch (IOException e) {
115 			throw new Error("could not load boot image: " + BOOT_IMAGE, e);
116 		}
117 
118 		// Sanity-check the results
119 		if (buffer.position() > BOOT_IMAGE_MAX_BYTES) {
120 			throw new Error(format(
121 					"The boot file is too big at %d bytes"
122 							+ " (only files up to 32KiB are acceptable)",
123 					buffer.position()));
124 		}
125 		if (buffer.position() % WORD_SIZE != 0) {
126 			// This ought to be unreachable...
127 			throw new Error(format(
128 					"The boot file size of %d bytes must be divisible by 4",
129 					buffer.position()));
130 		}
131 		if (buffer.position() < BOOT_STRUCT_REPLACE_OFFSET
132 				+ BOOT_STRUCT_REPLACE_LENGTH) {
133 			throw new Error(format(
134 					"The boot file size of %d bytes is not large enough to"
135 							+ " contain the boot configuration",
136 					buffer.position()));
137 		}
138 
139 		buffer.limit(buffer.position());
140 		return buffer;
141 	}

181 	private BootMessage getBootMessage(int blockID) {
182 		/*
183 		 * Compute the data in the payload; note that this is a pure byte
184 		 * sequence right now so endianness checks are moot.
185 		 */
186 		ByteBuffer buffer = bootData.duplicate();
187 		buffer.position(blockID * BOOT_MESSAGE_DATA_BYTES);
188 		buffer.limit(buffer.position()
189 				+ min(buffer.remaining(), BOOT_MESSAGE_DATA_BYTES));
190 
191 		// Make the message
192 		return new BootDataBlock(blockID, buffer);
193 	}
194 
195 	/** @return a stream of messages to be sent. */
196 	public Stream<BootMessage> getMessages() {
197 		// The bookending control messages
198 		BootMessage start = new StartOfBootMessages(numDataPackets);
199 		BootMessage finish = new EndOfBootMessages();
200 
201 		// Concatenate everything in the right order
202 		Stream<BootMessage> imageMessages =
203 				range(0, numDataPackets).mapToObj(this::getBootMessage);
204 		return concat(Stream.of(start),
205 				concat(imageMessages, Stream.of(finish)));
206 	}
