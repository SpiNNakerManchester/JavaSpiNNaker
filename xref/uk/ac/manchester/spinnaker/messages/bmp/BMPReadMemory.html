16  package uk.ac.manchester.spinnaker.messages.bmp;
17  
18  import static java.nio.ByteOrder.LITTLE_ENDIAN;
19  import static uk.ac.manchester.spinnaker.messages.Constants.UDP_MESSAGE_MAX_SIZE;
20  import static uk.ac.manchester.spinnaker.messages.model.TransferUnit.efficientTransferUnit;
21  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_READ;
22  
23  import java.nio.ByteBuffer;
24  
25  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
26  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
27  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
28  
29  /**
30   * A request to read a region of memory from a BMP. The response payload is
31   * a read-only little-endian {@link ByteBuffer} intended to be read once.
32   * <p>
33   * Calls {@code cmd_read()} in {@code bmp_cmd.c}.
34   */
35  public class BMPReadMemory extends BMPRequest<BMPReadMemory.Response> {
36  	private static int validate(int size) {
37  		if (size < 1 || size > UDP_MESSAGE_MAX_SIZE) {
38  			throw new IllegalArgumentException(
39  					"size must be in range 1 to 256");
40  		}
41  		return size;
42  	}
43  
44  	/**
45  	 * @param board
46  	 *            which board to read the memory of
47  	 * @param address
48  	 *            The positive base address to start the read from
49  	 * @param size
50  	 *            The number of bytes to read, between 1 and 256
51  	 */
52  	public BMPReadMemory(BMPBoard board, MemoryLocation address, int size) {
53  		super(board, CMD_READ, address.address, validate(size),
54  				efficientTransferUnit(address, size).value);
55  	}
56  
57  	@Override
58  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
59  		return new Response(buffer);
60  	}
61  
62  	/**
63  	 * An SCP response to a request to read a region of memory on a chip. Note
64  	 * that it is up to the caller to manage the buffer position of the returned
65  	 * response if it is to be read from multiple times.
66  	 */
67  	protected final class Response
68  			extends BMPRequest.PayloadedResponse<ByteBuffer> {
69  		Response(ByteBuffer buffer) throws UnexpectedResponseCodeException {
70  			super("Read", CMD_READ, buffer);
71  		}
72  
73  		/** @return The data read, in a little-endian read-only buffer. */
74  		@Override
75  		protected ByteBuffer parse(ByteBuffer buffer) {
76  			return buffer.asReadOnlyBuffer().order(LITTLE_ENDIAN);
77  		}
78  	}
79  }
