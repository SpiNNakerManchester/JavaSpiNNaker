16  package uk.ac.manchester.spinnaker.messages.bmp;
17  
18  import static java.util.stream.Collectors.toUnmodifiableList;
19  import static java.util.stream.IntStream.range;
20  import static uk.ac.manchester.spinnaker.messages.bmp.BMPInfo.CAN_STATUS;
21  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_BMP_INFO;
22  
23  import java.nio.ByteBuffer;
24  
25  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
26  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
27  import uk.ac.manchester.spinnaker.utils.MappableIterable;
28  
29  /**
30   * A request for the CAN bus status data from the BMP. The response payload is
31   * an {@linkplain MappableIterable iterable} of enabled {@linkplain BMPBoard
32   * board numbers}.
33   * <p>
34   * Handled in {@code cmd_bmp_info()} (in {@code bmp_cmd.c}) by reading from
35   * {@code can_status}.
36   */
37  public class ReadCANStatus extends BMPRequest<ReadCANStatus.Response> {
38  	private static final int MAX_BOARDS_PER_FRAME = 24;
39  
40  	private static final BMPBoard FRAME_ROOT = new BMPBoard(0);
41  
42  	/** Create a request. */
43  	public ReadCANStatus() {
44  		// The CAN status is shared between all BMPs; it's how they communicate
45  		super(FRAME_ROOT, CMD_BMP_INFO, CAN_STATUS.value);
46  	}
47  
48  	@Override
49  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
50  		return new Response(buffer);
51  	}
52  
53  	/** An SCP response to a request for the CAN status. */
54  	protected final class Response
55  			extends BMPRequest.PayloadedResponse<MappableIterable<BMPBoard>> {
56  		private Response(ByteBuffer buffer)
57  				throws UnexpectedResponseCodeException {
58  			super("Read CAN Status", CMD_BMP_INFO, buffer);
59  		}
60  
61  		/**
62  		 * @return Ordered sequence of board numbers available to be managed by
63  		 *         the BMP.
64  		 */
65  		@Override
66  		protected MappableIterable<BMPBoard> parse(ByteBuffer buffer) {
67  			/*
68  			 * The status data. The byte at {@code x} is zero if the BMP with
69  			 * that index is disabled.
70  			 */
71  			var statusData = new byte[buffer.remaining()];
72  			buffer.get(statusData);
73  			var boards = range(0, MAX_BOARDS_PER_FRAME)
74  					.filter(i -> statusData[i] != 0).mapToObj(BMPBoard::new)
75  					.collect(toUnmodifiableList());
76  			return boards::iterator;
77  		}
78  	}
79  }
