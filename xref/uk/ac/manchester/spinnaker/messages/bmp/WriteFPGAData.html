16  package uk.ac.manchester.spinnaker.messages.bmp;
17  
18  import static uk.ac.manchester.spinnaker.messages.bmp.XilinxCommand.LoadData;
19  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_XILINX;
20  
21  import java.nio.ByteBuffer;
22  
23  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
24  
25  /**
26   * Upload a chunk of FPGA initialisation data. Must have been set up by
27   * {@link InitFPGA}. Upload process will be terminated by {@link ResetFPGA}.
28   * There is no response payload.
29   * <p>
30   * Calls {@code ssp1_copy()} in {@code bmp_ssp.c}.
31   */
32  public class WriteFPGAData extends BMPRequest<BMPRequest.BMPResponse> {
33  	/**
34  	 * @param board
35  	 *            Which board to upload the FPGA data to.
36  	 * @param data
37  	 *            What data to upload.
38  	 */
39  	public WriteFPGAData(BMPBoard board, byte[] data) {
40  		super(board, CMD_XILINX, LoadData.code, data.length, 0,
41  				ByteBuffer.wrap(data));
42  	}
43  
44  	/**
45  	 * @param board
46  	 *            Which board to upload the FPGA data to.
47  	 * @param data
48  	 *            What data to upload. The position and limit of the buffer will
49  	 *            not be updated by this method.
50  	 */
51  	public WriteFPGAData(BMPBoard board, ByteBuffer data) {
52  		super(board, CMD_XILINX, LoadData.code, data.remaining(), 0, data);
53  	}
54  
55  	@Override
56  	public BMPResponse getSCPResponse(ByteBuffer buffer) throws Exception {
57  		return new BMPResponse("Load data into XILINX buffer", CMD_XILINX,
58  				buffer);
59  	}
60  }
