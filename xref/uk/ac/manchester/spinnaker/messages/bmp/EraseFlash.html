17  package uk.ac.manchester.spinnaker.messages.bmp;
18  
19  import static uk.ac.manchester.spinnaker.messages.bmp.WriteFlashBuffer.FLASH_CHUNK_SIZE;
20  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_FLASH_ERASE;
21  
22  import java.nio.ByteBuffer;
23  
24  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
25  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
26  
27  /** A request to erase flash memory on a BMP. */
28  public final class EraseFlash extends BMPRequest<EraseFlash.Response> {
29  	private static final int LOW_BOUNDARY = 65536;
30  
31  	private static final int MEMORY_LIMIT = 524288;
32  
33  	/**
34  	 * @param board
35  	 *            the board with the BMP to write the memory of
36  	 * @param baseAddress
37  	 *            The positive base address to start the erase at
38  	 * @param size
39  	 *            The number of bytes to erase
40  	 * @throws IllegalArgumentException
41  	 *             If the baseAddress or size make no sense
42  	 */
43  	public EraseFlash(BMPBoard board, MemoryLocation baseAddress, int size) {
44  		super(board, CMD_FLASH_ERASE, baseAddress.address,
45  				baseAddress.address + size);
46  		// Check that we've been actually asked to do something sane!
47  		if (size <= 0) {
48  			throw new IllegalArgumentException("no data");
49  		}
50  		int addr = baseAddress.address;
51  		if (addr < 0 || addr + size > MEMORY_LIMIT || addr + size < 0) {
52  			throw new IllegalArgumentException("address not in flash");
53  		}
54  		if ((addr % FLASH_CHUNK_SIZE) != 0) {
55  			throw new IllegalArgumentException("not on 4kB boundary");
56  		}
57  		if (addr < LOW_BOUNDARY && addr + size > LOW_BOUNDARY) {
58  			throw new IllegalArgumentException("crosses flash 4k/32k boundary");
59  		}
60  	}
61  
62  	@Override
63  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
64  		return new Response(buffer);
65  	}
66  
67  	/** The response from a request to erase flash. */
68  	public final class Response extends BMPRequest.BMPResponse {
69  		/** Where the buffer is located. */
70  		public final MemoryLocation address;
71  
72  		private Response(ByteBuffer buffer)
73  				throws UnexpectedResponseCodeException {
74  			super("Erase flash memory", CMD_FLASH_ERASE, buffer);
75  			address = new MemoryLocation(buffer.getInt());
76  		}
77  	}
78  }
