16  package uk.ac.manchester.spinnaker.messages.bmp;
17  
18  import static java.nio.ByteOrder.LITTLE_ENDIAN;
19  import static uk.ac.manchester.spinnaker.messages.Constants.UDP_MESSAGE_MAX_SIZE;
20  import static uk.ac.manchester.spinnaker.messages.bmp.SerialFlashOp.READ;
21  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_BMP_SF;
22  
23  import java.nio.ByteBuffer;
24  
25  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
26  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
27  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
28  
29  /**
30   * A request to read a region of serial flash from a BMP. The response payload
31   * is a read-only little-endian {@link ByteBuffer} intended to be read once.
32   * <p>
33   * Calls {@code sf_read()} in {@code bmp_ssp.c}.
34   */
35  public class ReadSerialFlash extends BMPRequest<ReadSerialFlash.Response> {
36  	private static int validate(int size) {
37  		if (size < 1 || size > UDP_MESSAGE_MAX_SIZE) {
38  			throw new IllegalArgumentException(
39  					"size must be in range 1 to 256");
40  		}
41  		return size;
42  	}
43  
44  	/**
45  	 * @param board
46  	 *            which board to read the flash of
47  	 * @param address
48  	 *            The positive base address to start the read from
49  	 * @param size
50  	 *            The number of bytes to read, between 1 and 256
51  	 */
52  	public ReadSerialFlash(BMPBoard board, MemoryLocation address, int size) {
53  		super(board, CMD_BMP_SF, address.address, validate(size), READ.value);
54  	}
55  
56  	@Override
57  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
58  		return new Response(buffer);
59  	}
60  
61  	/**
62  	 * An SCP response to a request to read a region of memory on a chip. Note
63  	 * that it is up to the caller to manage the buffer position of the returned
64  	 * response if it is to be read from multiple times.
65  	 */
66  	protected final class Response
67  			extends BMPRequest.PayloadedResponse<ByteBuffer> {
68  		Response(ByteBuffer buffer) throws UnexpectedResponseCodeException {
69  			super("Read Serial Flash", CMD_BMP_SF, buffer);
70  		}
71  
72  		/** @return The data read, in a little-endian read-only buffer. */
73  		@Override
74  		protected ByteBuffer parse(ByteBuffer buffer) {
75  			return buffer.asReadOnlyBuffer().order(LITTLE_ENDIAN);
76  		}
77  	}
78  }
