16  package uk.ac.manchester.spinnaker.messages.scp;

18  import static java.lang.Byte.toUnsignedInt;
19  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
20  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE2;
22  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE3;
23  import static uk.ac.manchester.spinnaker.messages.scp.Bits.TOP_BIT;
24  import static uk.ac.manchester.spinnaker.messages.scp.FloodFillConstants.ADD_ID;
25  import static uk.ac.manchester.spinnaker.messages.scp.FloodFillConstants.DELAY;
26  import static uk.ac.manchester.spinnaker.messages.scp.FloodFillConstants.FORWARD_LINKS;
27  import static uk.ac.manchester.spinnaker.messages.scp.FloodFillConstants.INIT_LEVEL;
28  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_NNP;
29  
30  import java.nio.ByteBuffer;
31  
32  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
33  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
34  
35  /**
36   * A request to start a flood fill of data. There is no response payload.
37   * <p>
38   * Calls {@code nn_cmd_ffs()} in {@code scamp-nn.c} on all cores via
39   * {@code ff_nn_send()} in {@code scamp-nn.c}.
40   */
41  public final class FloodFillStart extends SCPRequest<EmptyResponse> {
42  	// See nn_rcv_pkt()
43  	private static final int NN_CMD_FFS = 6;
44  
45  	private static final int NNP_FORWARD_RETRY =
46  			(ADD_ID << TOP_BIT) | (FORWARD_LINKS << BYTE1) | (DELAY << BYTE0);
47  
48  	private static final int NO_CHIP = 0xFFFF;
49  
50  	private static final int LOW_BITS_MASK = 0b00000011;
51  
52  	private static final int HIGH_BITS_MASK = 0b11111100;
53  
54  	/**
55  	 * Flood fill onto all chips.
56  	 *
57  	 * @param nearestNeighbourID
58  	 *            The ID of the packet, between 0 and 127
59  	 * @param numBlocks
60  	 *            The number of blocks of data that will be sent, between 0 and
61  	 *            255
62  	 */
63  	public FloodFillStart(byte nearestNeighbourID, int numBlocks) {
64  		super(BOOT_MONITOR_CORE, CMD_NNP, key(nearestNeighbourID, numBlocks),
65  				NO_CHIP, NNP_FORWARD_RETRY);
66  	}
67  
68  	/**
69  	 * Flood fill on a specific chip.
70  	 *
71  	 * @param nearestNeighbourID
72  	 *            The ID of the packet, between 0 and 127
73  	 * @param numBlocks
74  	 *            The number of blocks of data that will be sent, between 0 and
75  	 *            255
76  	 * @param chip
77  	 *            The chip to load the data on to.
78  	 */
79  	public FloodFillStart(byte nearestNeighbourID, int numBlocks,
80  			HasChipLocation chip) {
81  		super(BOOT_MONITOR_CORE, CMD_NNP, key(nearestNeighbourID, numBlocks),
82  				data(chip), NNP_FORWARD_RETRY);
83  	}
84  
85  	private static int key(byte nearestNeighbourID, int numBlocks) {
86  		if (numBlocks != toUnsignedInt((byte) numBlocks)) {
87  			throw new IllegalArgumentException(
88  					"number of blocks must be representable in 8 bits");
89  		}
90  		return (NN_CMD_FFS << BYTE3)
91  				| (toUnsignedInt(nearestNeighbourID) << BYTE2)
92  				| (numBlocks << BYTE1);
93  	}
94  
95  	private static int data(HasChipLocation chip) {
96  		int mask = 1 << (((chip.getY() & LOW_BITS_MASK) << 2)
97  				+ (chip.getX() & LOW_BITS_MASK));
98  		int region = ((chip.getX() & HIGH_BITS_MASK) << BYTE1)
99  				+ (chip.getY() & HIGH_BITS_MASK);
100 		return (region << BYTE2) + (INIT_LEVEL << BYTE2) + (mask << BYTE0);
101 	}
102 
103 	@Override
104 	public EmptyResponse getSCPResponse(ByteBuffer buffer)
105 			throws UnexpectedResponseCodeException {
106 		return new EmptyResponse("Flood Fill", CMD_NNP, buffer);
107 	}
108 }
