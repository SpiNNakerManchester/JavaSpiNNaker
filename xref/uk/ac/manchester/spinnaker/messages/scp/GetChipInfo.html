17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_INFO;
20  
21  import java.nio.ByteBuffer;
22  
23  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
24  import uk.ac.manchester.spinnaker.messages.model.ChipSummaryInfo;
25  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
26  
27  /** An SCP request to read the chip information from a core. */
28  public class GetChipInfo extends SCPRequest<GetChipInfo.Response> {
29  	private static final int FLAGS = 0x5F;
30  	private static final int SIZE_FLAG = 0x20;
31  
32  	/**
33  	 * @param chip
34  	 *            the chip to read from
35  	 */
36  	public GetChipInfo(HasChipLocation chip) {
37  		this(chip, false);
38  	}
39  
40  	/**
41  	 * @param chip
42  	 *            the chip to read from
43  	 * @param withSize
44  	 *            Whether the size should be included in the response
45  	 */
46  	public GetChipInfo(HasChipLocation chip, boolean withSize) {
47  		super(chip.getScampCore(), CMD_INFO, argument1(withSize));
48  	}
49  
50  	private static int argument1(boolean withSize) {
51  		// Bits 0-4 + bit 6 = all information except size
52  		// Bits 0-6 = all information including size
53  		return FLAGS | (withSize ? SIZE_FLAG : 0);
54  	}
55  
56  	@Override
57  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
58  		return new Response(buffer);
59  	}
60  
61  	/** An SCP response to a request for the version of software running. */
62  	public static final class Response extends CheckOKResponse {
63  		/** The chip information received. */
64  		public final ChipSummaryInfo chipInfo;
65  
66  		private Response(ByteBuffer buffer)
67  				throws UnexpectedResponseCodeException {
68  			super("Version", CMD_INFO, buffer);
69  			this.chipInfo = new ChipSummaryInfo(buffer, sdpHeader.getSource());
70  		}
71  	}
72  }
