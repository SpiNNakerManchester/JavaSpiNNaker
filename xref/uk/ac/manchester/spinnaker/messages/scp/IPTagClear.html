16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.COMMAND_FIELD;
19  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.THREE_BITS_MASK;
20  import static uk.ac.manchester.spinnaker.messages.scp.IPTagCommand.CLR;
21  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_IPTAG;
22  
23  import java.nio.ByteBuffer;
24  
25  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
26  import uk.ac.manchester.spinnaker.machine.tags.TagID;
27  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
28  
29  /**
30   * A request to clear an IP Tag. There is no response payload.
31   * <p>
32   * Handled by {@code cmd_iptag()} in {@code scamp-cmd.c} (or {@code bmp_cmd.c},
33   * if sent to a BMP).
34   */
35  public class IPTagClear extends SCPRequest<EmptyResponse> {
36  	/**
37  	 * @param chip
38  	 *            The chip to clear the tag on.
39  	 * @param tag
40  	 *            The ID of the tag to clear (0..7)
41  	 */
42  	public IPTagClear(HasChipLocation chip, @TagID int tag) {
43  		super(chip.getScampCore(), CMD_IPTAG, argument1(tag));
44  	}
45  
46  	// arg1 = flags[11:8] : timeout : command : dest_port : tag
47  	private static Integer argument1(int tag) {
48  		return (CLR.value << COMMAND_FIELD) | (tag & THREE_BITS_MASK);
49  	}
50  
51  	@Override
52  	public EmptyResponse getSCPResponse(ByteBuffer buffer)
53  			throws UnexpectedResponseCodeException {
54  		return new EmptyResponse("Clear IP Tag", CMD_IPTAG, buffer);
55  	}
56  }
