17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static java.util.Objects.requireNonNull;
20  import static uk.ac.manchester.spinnaker.machine.MachineDefaults.MAX_NUM_CORES;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE3;
22  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_AR;
23  
24  import java.nio.ByteBuffer;
25  import java.util.Collection;
26  
27  import javax.validation.Valid;
28  
29  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
30  import uk.ac.manchester.spinnaker.machine.ValidP;
31  import uk.ac.manchester.spinnaker.messages.model.AppID;
32  import uk.ac.manchester.spinnaker.messages.model.SystemVariableDefinition;
33  import uk.ac.manchester.spinnaker.utils.UsedInJavadocOnly;
34  
35  /**
36   * An SCP request to run an application. The application code must have been
37   * placed at the location
38   * {@link SystemVariableDefinition#system_sdram_base_address} points at. There
39   * is no response payload.
40   * <p>
41   * Calls {@code proc_start_app()} in {@code scamp-app.c}.
42   */
43  @UsedInJavadocOnly(SystemVariableDefinition.class)
44  public class ApplicationRun extends SCPRequest<CheckOKResponse> {
45  	private static final int WAIT_BIT = 18;
46  
47  	/**
48  	 * @param appID
49  	 *            The ID of the application to run
50  	 * @param chip
51  	 *            The coordinates of the chip to run on
52  	 * @param processors
53  	 *            The processors of the chip to run on, between 1 and 17
54  	 */
55  	public ApplicationRun(AppID appID, @Valid HasChipLocation chip,
56  			Collection<@ValidP Integer> processors) {
57  		this(appID, chip, processors, false);
58  	}
59  
60  	/**
61  	 * @param appId
62  	 *            The ID of the application to run
63  	 * @param chip
64  	 *            The coordinates of the chip to run on
65  	 * @param processors
66  	 *            The processors of the chip to run on, between 1 and 17
67  	 * @param wait
68  	 *            True if the processors should enter a "wait" state on starting
69  	 */
70  	public ApplicationRun(AppID appId, @Valid HasChipLocation chip,
71  			Collection<@ValidP Integer> processors, boolean wait) {
72  		super(chip.getScampCore(), CMD_AR, idOpMask(appId, processors, wait));
73  	}
74  
75  	// @formatter:off
76  	/*
77  	 * bits:   [  31-24 |     23-18 |           17-0 ]
78  	 * labels: [ app_id | app_flags | processor_mask ]
79  	 *
80  	 * It is an error to ask for this operation for the SCAMP core.
81  	 * The only app_flag we use is in the lowest bit: WAIT.
82  	 */
83  	// @formatter:on
84  	private static int idOpMask(AppID appId, Collection<Integer> processors,
85  			boolean wait) {
86  		int mask = 0;
87  		for (int p : requireNonNull(processors,
88  				"set of processors on chip must be not null")) {
89  			if (p >= 1 && p < MAX_NUM_CORES) {
90  				mask |= 1 << p;
91  			}
92  		}
93  		mask |= appId.appID << BYTE3;
94  		if (wait) {
95  			mask |= 1 << WAIT_BIT;
96  		}
97  		return mask;
98  	}
99  
100 	@Override
101 	public CheckOKResponse getSCPResponse(ByteBuffer buffer) throws Exception {
102 		return new CheckOKResponse("Run Application", CMD_AR, buffer);
103 	}
104 }
