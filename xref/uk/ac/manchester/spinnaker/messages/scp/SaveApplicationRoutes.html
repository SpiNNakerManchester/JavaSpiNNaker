17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.RouterTableCommand.SAVE_APPLICATION_ROUTES;
20  
21  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
22  
23  /**
24   * An SDP Request to save the currently-installed application multicast router
25   * table. There is no response payload.
26   * <p>
27   * Handled by {@code data_in_save_router()} in {@code extra_monitor_support.c}.
28   */
29  public final class SaveApplicationRoutes extends RouterTableRequest {
30  	/**
31  	 * @param core
32  	 *            The coordinates of the monitor core.
33  	 */
34  	public SaveApplicationRoutes(HasCoreLocation core) {
35  		super(core, SAVE_APPLICATION_ROUTES);
36  	}
37  
38  	@Override
39  	String describe() {
40  		return "Save application multicast routes";
41  	}
42  }
