17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_LINK_WRITE;
20  
21  import java.nio.ByteBuffer;
22  
23  import uk.ac.manchester.spinnaker.machine.Direction;
24  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
25  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
26  
27  /** A request to write memory on a neighbouring chip. */
28  public class WriteLink extends SCPRequest<CheckOKResponse> {
29  	/**
30  	 * @param core
31  	 *            the core to write via
32  	 * @param link
33  	 *            The ID of the link down which to send the write
34  	 * @param baseAddress
35  	 *            The positive base address to start the write at
36  	 * @param data
37  	 *            The data to write (up to 256 bytes); the <i>position</i> of
38  	 *            the buffer must be the point where the data starts.
39  	 */
40  	public WriteLink(HasCoreLocation core, Direction link,
41  			MemoryLocation baseAddress, ByteBuffer data) {
42  		super(core, CMD_LINK_WRITE, baseAddress.address, data.remaining(),
43  				link.id, data);
44  	}
45  
46  	@Override
47  	public CheckOKResponse getSCPResponse(ByteBuffer buffer) throws Exception {
48  		return new CheckOKResponse("Write Memory", CMD_LINK_WRITE, buffer);
49  	}
50  }
