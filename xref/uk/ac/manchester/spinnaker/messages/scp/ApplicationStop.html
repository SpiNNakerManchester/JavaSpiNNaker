18  import static uk.ac.manchester.spinnaker.messages.model.Signal.STOP;
19  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
20  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE2;
22  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE3;
23  import static uk.ac.manchester.spinnaker.messages.scp.Bits.TOP_BIT;
24  import static uk.ac.manchester.spinnaker.messages.scp.Constants.APP_MASK;
25  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_NNP;
26  
27  import java.nio.ByteBuffer;
28  
29  import uk.ac.manchester.spinnaker.messages.model.AppID;
30  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
31  
32  /**
33   * A request to stop an application. There is no response payload.
34   * <p>
35   * This maps to a call to {@code ff_nn_send()} in {@code scamp-nn.c}, which in
36   * turn triggers a call to {@code proc_stop_app()} in {@code scamp-app.c} across
37   * all SCAMP instances.
38   */
39  public final class ApplicationStop extends SCPRequest<EmptyResponse> {
40  	private static final int SHIFT = 28;
41  
42  	private static final int NN_CMD_SIG0 = 0;
43  
44  	private static final int FORWARD = 0x3f;
45  
46  	private static final int RETRY = 0;
47  
48  	private static final int SIG0_APP = 5;
49  
50  	private static final int ADD_ID = 1;
51  
52  	private static final int ALL_LINKS = 0x3f;
53  
54  	private static int key() {
55  		return (NN_CMD_SIG0 << BYTE3) | (FORWARD << BYTE2) | (RETRY << BYTE1);
56  	}
57  
58  	private static int data(AppID appID) {
59  		// Call the SIG0 NN operation, subcode STOP (stop application)
60  		// No masking of the app ID
61  		return (SIG0_APP << SHIFT) | (STOP.value << BYTE2) | (APP_MASK << BYTE1)
62  				| (appID.appID << BYTE0);
63  	}
64  
65  	private static int nnConfig() {
66  		// Add a suitable ID in SCAMP, send message on all links
67  		// Number of repeats is zero, inter-repeat delay is zero
68  		return (ADD_ID << TOP_BIT) | (ALL_LINKS << BYTE1);
69  	}
70  
71  	/**
72  	 * @param appID
73  	 *            The ID of the application
74  	 */
75  	public ApplicationStop(AppID appID) {
76  		super(BOOT_MONITOR_CORE, CMD_NNP, key(), data(appID), nnConfig());
77  	}
78  
79  	@Override
80  	public EmptyResponse getSCPResponse(ByteBuffer buffer)
81  			throws UnexpectedResponseCodeException {
82  		return new EmptyResponse("Send Stop", CMD_NNP, buffer);
83  	}
84  }
