16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static java.util.Objects.requireNonNull;
19  import static uk.ac.manchester.spinnaker.utils.CollectionUtils.makeEnumBackingMap;
20  
21  import java.util.Map;
22  
23  /** The SCP Command codes. */
24  public enum SCPCommand implements CommandCode {
25  	/** Get SCAMP Version. */
26  	CMD_VER(0),
27  	/**
28  	 * Run at PC.
29  	 *
30  	 * @deprecated see {@link #CMD_AS}, which has a superior API
31  	 */
32  	@Deprecated(forRemoval = true)
33  	CMD_RUN(1),
34  	/** Read SDRAM. */
35  	CMD_READ(2),
36  	/** Write SDRAM. */
37  	CMD_WRITE(3),
38  	/**
39  	 * Run via APLX.
40  	 *
41  	 * @deprecated see {@link #CMD_AS}, which has a superior API
42  	 */
43  	@Deprecated(forRemoval = true)
44  	CMD_APLX(4),
45  	/** Fill memory. */
46  	CMD_FILL(5),
47  	/** Remap application core. */
48  	CMD_REMAP(16),
49  	/** Read neighbouring chip's memory. */
50  	CMD_LINK_READ(17),
51  	/** Write neighbouring chip's memory. */
52  	CMD_LINK_WRITE(18),
53  	/** Application core reset. */
54  	CMD_AR(19),
55  	/** Send a broadcast Nearest-Neighbour packet. */
56  	CMD_NNP(20),
57  	/** Send a Signal. */
58  	CMD_SIG(22),
59  	/** Send Flood-Fill Data. */
60  	CMD_FFD(23),
61  	/** Application core APLX start. */
62  	CMD_AS(24),
63  	/** Control the LEDs. */
64  	CMD_LED(25),
65  	/** Set an IP tag. */
66  	CMD_IPTAG(26),
67  	/** Read/write/erase serial ROM. */
68  	CMD_SROM(27),
69  	/** Allocate or Free SDRAM or Routing entries. */
70  	CMD_ALLOC(28),
71  	/** Initialise the router. */
72  	CMD_RTR(29),
73  	/** Dropped Packet Reinjection setup. */
74  	CMD_DPRI(30),
75  	/** Get Chip Summary Information. */
76  	CMD_INFO(31),
77  	/** Get BMP info structures. */
78  	CMD_BMP_INFO(48),
79  	/** Copy working buffer to flash memory. BMP-only operation. */
80  	CMD_FLASH_COPY(49),
81  	/** Erase part of flash memory. BMP-only operation. */
82  	CMD_FLASH_ERASE(50),
83  	/** Write to flash memory. BMP-only operation. */
84  	CMD_FLASH_WRITE(51),
85  	/** Serial flash access. BMP-only operation. */
86  	CMD_BMP_SF(53),
87  	/**
88  	 * EEPROM access. BMP-only operation.
89  	 *
90  	 * @deprecated This message is under-documented. See {@code cmd_ee()} in
91  	 *             {@code bmp_cmd.c} for what it does.
92  	 */
93  	@Deprecated
94  	CMD_BMP_EE(54),
95  	/** BMP-only operation. */
96  	CMD_RESET(55),
97  	/** FPGA control. BMP-only operation. */
98  	CMD_XILINX(56),
99  	/** Turns on or off the machine via BMP. */
100 	CMD_BMP_POWER(57),
101 	/**
102 	 * Access I<sup>2</sup>C bus. BMP-only operation.
103 	 *
104 	 * @deprecated This message is under-documented. See {@code cmd_i2c()} in
105 	 *             {@code bmp_cmd.c} for what it does.
106 	 */
107 	@Deprecated
108 	CMD_BMP_I2C(61),
109 	/**
110 	 * Configure pulse-width modulation hardware. BMP-only operation.
111 	 *
112 	 * @deprecated This message is under-documented. See {@code configure_pwm()}
113 	 *             in {@code bmp_hw.c} for what it does.
114 	 */
115 	@Deprecated
116 	CMD_BMP_PWM(62),
117 	/**
118 	 * Test feature control. BMP-only operation.
119 	 *
120 	 * @deprecated Not implemented by current BMP.
121 	 * @hidden
122 	 */
123 	@Deprecated(forRemoval = true)
124 	CMD_BMP_TEST(63),
125 	/**
126 	 * Tube output. Special.
127 	 *
128 	 * @deprecated For tubotron use only.
129 	 * @hidden
130 	 */
131 	@Deprecated(forRemoval = true)
132 	CMD_TUBE(64);
133 
134 	/**
135 	 * Read FPGA's memory/registers. BMP-only operation.
136 	 * <p>
137 	 * Same as {@link #CMD_LINK_READ} but used differently.
138 	 */
139 	public static final SCPCommand CMD_FPGA_READ = CMD_LINK_READ;
140 
141 	/**
142 	 * Write FPGA's memory/registers. BMP-only operation.
143 	 * <p>
144 	 * Same as {@link #CMD_LINK_WRITE} but used differently.
145 	 */
146 	public static final SCPCommand CMD_FPGA_WRITE = CMD_LINK_WRITE;
147 
148 	/** The SCAMP encoding. */
149 	public final short value;
150 
151 	private static final Map<Short, SCPCommand> MAP =
152 			makeEnumBackingMap(values(), v -> v.value);
153 
154 	SCPCommand(int value) {
155 		this.value = (short) value;
156 	}
157 
158 	/**
159 	 * Convert an encoded value into an enum element.
160 	 *
161 	 * @param value
162 	 *            The value to convert
163 	 * @return The enum element
164 	 */
165 	public static SCPCommand get(short value) {
166 		return requireNonNull(MAP.get(value),
167 				"unrecognised command value: " + value);
168 	}
169 
170 	@Override
171 	public short getValue() {
172 		return value;
173 	}
174 }
