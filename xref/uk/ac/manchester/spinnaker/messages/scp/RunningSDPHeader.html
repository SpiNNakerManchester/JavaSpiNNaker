17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.sdp.SDPHeader.Flag.REPLY_EXPECTED;
20  import static uk.ac.manchester.spinnaker.messages.sdp.SDPHeader.Flag.REPLY_NOT_EXPECTED;
21  import static uk.ac.manchester.spinnaker.messages.sdp.SDPPort.RUNNING_COMMAND_SDP_PORT;
22  
23  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
24  import uk.ac.manchester.spinnaker.messages.sdp.SDPHeader;
25  
26  /**
27   * Variant of SCP header that is used to talk to running cores.
28   *
29   * @author Donal Fellows
30   */
31  class RunningSDPHeader extends SDPHeader {
32  	/**
33  	 * Make a header.
34  	 *
35  	 * @param core
36  	 *            The SpiNNaker core that we want to talk to.
37  	 * @param replyExpected
38  	 *            Whether we expect a reply.
39  	 */
40  	RunningSDPHeader(HasCoreLocation core, boolean replyExpected) {
41  		super(replyExpected ? REPLY_EXPECTED : REPLY_NOT_EXPECTED, core,
42  				RUNNING_COMMAND_SDP_PORT);
43  	}
44  }
