17  package uk.ac.manchester.spinnaker.messages.model;
18  
19  import java.nio.ByteBuffer;
20  
21  /** Enum for data types of system variables. */
22  public enum DataType {
23  	/** The value is one byte long. */
24  	BYTE(1),
25  	/** The value is two bytes long. */
26  	SHORT(2),
27  	/** The value is four bytes long. */
28  	INT(4),
29  	/** The value is eight bytes long. */
30  	LONG(8),
31  	/** The value is an array of bytes. */
32  	BYTE_ARRAY(16);
33  
34  	/** The SCAMP data type descriptor code. */
35  	public final int value;
36  
37  	DataType(int value) {
38  		this.value = value;
39  	}
40  
41  	/**
42  	 * Writes an object described by this data type into the given buffer at the
43  	 * <i>position</i> as a contiguous range of bytes. This assumes that the
44  	 * buffer has been configured to be
45  	 * {@linkplain java.nio.ByteOrder#LITTLE_ENDIAN little-endian} and that its
46  	 * <i>position</i> is at the point where this method should begin writing.
47  	 * Once it has finished, the <i>position</i> should be immediately after the
48  	 * last byte written by this method.
49  	 *
50  	 * @param value
51  	 *            The value to write.
52  	 * @param buffer
53  	 *            The buffer to write into.
54  	 */
55  	void addToBuffer(Object value, ByteBuffer buffer) {
56  		switch (this) {
57  		case BYTE:
58  			buffer.put(((Number) value).byteValue());
59  			return;
60  		case SHORT:
61  			buffer.putShort(((Number) value).shortValue());
62  			return;
63  		case INT:
64  			buffer.putInt(((Number) value).intValue());
65  			return;
66  		case LONG:
67  			buffer.putLong(((Number) value).longValue());
68  			return;
69  		case BYTE_ARRAY:
70  			buffer.put((byte[]) value);
71  			return;
72  		default:
73  			// CHECKSTYLE:OFF
74  			throw new Error("unreachable?");
75  			// CHECKSTYLE:ON
76  		}
77  	}
78  }
