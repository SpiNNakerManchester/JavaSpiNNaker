16  package uk.ac.manchester.spinnaker.messages.model;
17  
18  import static java.lang.Math.min;
19  import static java.util.stream.IntStream.range;
20  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
21  import static uk.ac.manchester.spinnaker.messages.model.P2PTableRoute.NONE;
22  import static uk.ac.manchester.spinnaker.utils.CollectionUtils.copy;
23  
24  import java.nio.ByteBuffer;
25  import java.util.Collection;
26  import java.util.HashMap;
27  import java.util.Map;
28  import java.util.Set;
29  
30  import uk.ac.manchester.spinnaker.machine.ChipLocation;
31  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
32  import uk.ac.manchester.spinnaker.machine.MachineDimensions;
33  import uk.ac.manchester.spinnaker.machine.ValidMachineHeight;
34  import uk.ac.manchester.spinnaker.machine.ValidMachineWidth;
35  
36  /** Represents a P2P routing table read from the machine. */
37  public class P2PTable {
38  	private final Map<ChipLocation, P2PTableRoute> routes;
39  
40  	/** The width of the machine that this table represents. */
41  	@ValidMachineWidth
42  	public final int width;
43  
44  	/** The height of the machine that this table represents. */
45  	@ValidMachineHeight
46  	public final int height;
47  
48  	private static final int ROUTE_CHUNK = 8;
49  
50  	private static final int COLUMN_SIZE = 256;
51  
52  	private static final int ROUTE_BITS = 3;
53  
54  	private static final int ROUTE_MASK = (1 << ROUTE_BITS) - 1;
55  
56  	/** Number of bits per byte. */
57  	private static final int NBBY = 8;
58  
59  	/**
60  	 * Construct a routing table from data.
61  	 *
62  	 * @param dimensions
63  	 *            The size of the machine.
64  	 * @param columnData
65  	 *            The routing data to parse.
66  	 */
67  	public P2PTable(MachineDimensions dimensions,
68  			Collection<ByteBuffer> columnData) {
69  		this.routes = new HashMap<>();
70  		this.width = dimensions.width;
71  		this.height = dimensions.height;
72  		parseColumnData(columnData);
73  	}
74  
75  	private void parseColumnData(Iterable<ByteBuffer> columnData) {
76  		int x = 0;
77  		for (var buffer : columnData) {
78  			var data = buffer.asIntBuffer();
79  			int chipX = x++;
80  			for (int y = 0; y < height; y += ROUTE_CHUNK) {
81  				extractRoutes(chipX, y, data.get());
82  			}
83  		}
84  	}
85  
86  	private void extractRoutes(int chipX, int chipYBase, int word) {
87  		range(0, min(ROUTE_CHUNK, height - chipYBase)).forEach(y -> {
88  			var route =
89  					P2PTableRoute.get((word >> (ROUTE_BITS * y)) & ROUTE_MASK);
90  			if (route != null && route != NONE) {
91  				routes.put(new ChipLocation(chipX, chipYBase + y), route);
92  			}
93  		});
94  	}
95  
96  	/**
97  	 * Get the number of bytes to be read for each column of the table.
98  	 *
99  	 * @param height
100 	 *            The height of the machine
101 	 * @return The number of bytes for the column
102 	 */
103 	public static int getNumColumnBytes(int height) {
104 		return ((height + NBBY - 1) / NBBY) * WORD_SIZE;
105 	}
106 
107 	/**
108 	 * Get the offset of the next column in the table from the P2P base address.
109 	 *
110 	 * @param column
111 	 *            The column to be read
112 	 * @return Where the column is located within the table.
113 	 */
114 	public static int getColumnOffset(int column) {
115 		return ((COLUMN_SIZE * column) / NBBY) * WORD_SIZE;
116 	}
117 
118 	/** @return The coordinates of chips in the table. */
119 	public Set<ChipLocation> getChips() {
120 		return copy(routes.keySet());
121 	}
122 
123 	/**
124 	 * Determines if there is a route in the P2P table to the given chip.
125 	 *
126 	 * @param chip
127 	 *            The coordinates of the chip to look up.
128 	 * @return True if anything in the table routes to the chip.
129 	 */
130 	public boolean isRoute(HasChipLocation chip) {
131 		var r = routes.get(chip.asChipLocation());
132 		return r != null && r != NONE;
133 	}
134 
135 	/**
136 	 * Get the route to follow from this chip to the given chip.
137 	 *
138 	 * @param chip
139 	 *            The coordinates of the chip to find the route to
140 	 * @return Get the route descriptor for a chip.
141 	 */
142 	public P2PTableRoute getRoute(HasChipLocation chip) {
143 		var r = routes.get(chip.asChipLocation());
144 		return r == null ? NONE : r;
145 	}
146 }
