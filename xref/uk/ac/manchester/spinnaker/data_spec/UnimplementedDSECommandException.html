16  package uk.ac.manchester.spinnaker.data_spec;
17  
18  import static java.lang.String.format;
19  
20  /**
21   * An exception which occurs when trying to execute an unimplemented command.
22   */
23  public class UnimplementedDSECommandException
24  		extends UnsupportedOperationException {
25  	private static final long serialVersionUID = -2215740111501788733L;
26  
27  	/**
28  	 * Create an instance.
29  	 *
30  	 * @param index
31  	 *            Where the command was located.
32  	 * @param command
33  	 *            The command that was unimplemented.
34  	 */
35  	UnimplementedDSECommandException(int index, Commands command) {
36  		super(format(
37  				"Command %s (at index %d) in the data specification "
38  						+ "executor has not yet been implemented",
39  				command, index));
40  	}
41  
42  	/**
43  	 * Create an instance.
44  	 *
45  	 * @param index
46  	 *            Where the opcode was located.
47  	 * @param opcode
48  	 *            The opcode that couldn't be converted into a command.
49  	 */
50  	UnimplementedDSECommandException(int index, int opcode) {
51  		super(format("unknown opcocode (%d) at index %d", opcode, index));
52  	}
53  }
