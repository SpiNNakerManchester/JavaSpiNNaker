17  package uk.ac.manchester.spinnaker.data_spec;
18  
19  import static uk.ac.manchester.spinnaker.data_spec.Commands.BREAK;
20  import static uk.ac.manchester.spinnaker.data_spec.Commands.NOP;
21  import static uk.ac.manchester.spinnaker.data_spec.Functions.OPCODE;
22  import static uk.ac.manchester.spinnaker.data_spec.OperationMapper.getOperationImpl;
23  
24  /**
25   * The interface supported by the {@link Functions} class. This makes generating
26   * an {@link Callable} much easier.
27   *
28   * @author Donal Fellows
29   */
30  public interface FunctionAPI {
31  	/**
32  	 * Decode the command, storing the fields in variables in this object.
33  	 *
34  	 * @param cmd
35  	 *            The encoded command
36  	 */
37  	void unpack(int cmd);
38  
39  	/**
40  	 * Get the implementation of an operation.
41  	 *
42  	 * @param cmdOpcode
43  	 *            The <i>encoded</i> opcode value.
44  	 * @param index
45  	 *            The location in the data specification data stream where the
46  	 *            command was read from, for reporting errors to the user.
47  	 * @return The operation implementation. Never {@code null}.
48  	 * @throws UnimplementedDSECommandException
49  	 *             If the opcode used is not implemented.
50  	 */
51  	default Callable getOperation(int cmdOpcode, int index)
52  			throws UnimplementedDSECommandException {
53  		int opcode = OPCODE.getValue(cmdOpcode);
54  		Commands c = Commands.get(opcode);
55  		if (c == null) {
56  			throw new UnimplementedDSECommandException(index, opcode);
57  		}
58  		Callable opImpl = getOperationImpl(this, c);
59  		if (opImpl == null) {
60  			throw new UnimplementedDSECommandException(index, c);
61  		}
62  		return opImpl;
63  	}
64  
65  	/**
66  	 * This command executes no operation.
67  	 */
68  	@Operation(NOP)
69  	default void doNOP() {
70  		// Does nothing
71  	}
72  
73  	/**
74  	 * This command raises an exception to stop the execution of the data
75  	 * specification executor (DSE).
76  	 *
77  	 * @throws ExecuteBreakInstruction
78  	 *             Always
79  	 */
80  	@Operation(BREAK)
81  	default void doBreak() throws ExecuteBreakInstruction {
82  		throw new ExecuteBreakInstruction();
83  	}
84  }
