32  import java.util.WeakHashMap;

34  import org.slf4j.Logger;

36  import com.google.errorprone.annotations.CheckReturnValue;

38  /**
39   * Clever stuff to turn a method annotated with {@link Operation @Operation}
40   * into a {@link Callable}.
41   *
42   * @author Donal Fellows
43   */
44  abstract class OperationMapper {
45  	private static final Logger log = getLogger(OperationMapper.class);
46  
47  	/** Cache of what methods implement operations in a class. */
48  	private static final Map<Class<? extends FunctionAPI>,
49  			Map<Commands, WrappedMethod>> OPS_MAP =
50  					synchronizedMap(new HashMap<>());
51  
52  	/**
53  	 * Cache of callables for a particular operation on a particular executor.
54  	 * This is a <em>weak</em> map; entries are reclaimed when the instance
55  	 * isn't used elsewhere.
56  	 */
57  	private static final Map<FunctionAPI, Map<Commands, Callable>> MAP =
58  			synchronizedMap(new WeakHashMap<>());
59  
60  	private OperationMapper() {
61  	}
62  
63  	/**
64  	 * Given an object that has (or should have) {@link Operation}-annotated
65  	 * methods, and given a {@link Commands} that we want to invoke, get the
66  	 * implementation of how to invoke the operation that is done by calling
67  	 * first {@link FunctionAPI#unpack(int) unpack(...)} and then the annotated
68  	 * method.
69  	 *
70  	 * @param funcs
71  	 *            The object containing the method implementations.
72  	 * @param opcode
73  	 *            The opcode that we want to invoke.
74  	 * @return How to invoke that operation, or {@code null} if that operation
75  	 *         has no registered implementation.
76  	 */
77  	@CheckReturnValue
78  	static Callable getOperationImpl(FunctionAPI funcs, Commands opcode) {
79  		// Note that MAP is using the object identity; this is by design
80  		var map = MAP.computeIfAbsent(requireNonNull(funcs,
81  				"can only look up method implementations of real objects"),
82  				/*
83  				 * Careful! Must only pass in references to funcs as a weak so
84  				 * that it doesn't compromise the weak hash map. But we can use
85  				 * the same weak reference for all the method wrappers that we
86  				 * create.
87  				 */
88  				__ -> manufactureCallables(new WeakReference<>(funcs),
89  						OPS_MAP.computeIfAbsent(funcs.getClass(),
90  								OperationMapper::getOperations)));
91  		return map.get(opcode);
92  	}
93  
94  	/**
95  	 * Manufacture callables for a particular function API instance.
96  	 *
97  	 * @param objref
98  	 *            The reference to the function API instance.
99  	 * @param ops
100 	 *            The parsed wrapped methods in the function API class.
101 	 * @return The callables for the commands.
102 	 */
103 	private static Map<Commands, Callable> manufactureCallables(
104 			WeakReference<FunctionAPI> objref,
105 			Map<Commands, WrappedMethod> ops) {
106 		return ops.entrySet().stream().collect(toMap(Map.Entry::getKey,
107 				e -> cmd -> e.getValue().call(objref, e.getKey(), cmd)));
108 	}
109 
110 	/**
111 	 * Manufacture wrapped methods for a function API class. Methods are only
112 	 * wrapped if they are annotated with {@code @}{@link Operation}.
113 	 *
114 	 * @param cls
115 	 *            The function API class to analyse.
116 	 * @return The wrapped (and annotated) methods for the commands.
117 	 */
118 	private static Map<Commands, WrappedMethod> getOperations(Class<?> cls) {
119 		return stream(cls.getMethods())
120 				.filter(m -> m.isAnnotationPresent(Operation.class))
121 				.collect(toMap(m -> m.getAnnotation(Operation.class).value(),
122 						m -> wrapMethod(cls, m)));
123 	}
124 
125 	/** The types we allow as return values: {@code int} and {@code void}. */
126 	private static final Set<Class<?>> ALLOWED_RETURN_TYPES =
127 			Set.of(Void.TYPE, Integer.TYPE);
128 
129 	/**
130 	 * Check a particular method for validity and wrap it. If there are any
131 	 * arguments, or the method has a return type that isn't {@code void} or
132 	 * {@code int}, that's an error.
133 	 *
134 	 * @param cls
135 	 *            The class, for the exception message.
136 	 * @param m
137 	 *            The (annotated) method to check.
138 	 * @return The checked method, wrapped.
139 	 */
140 	private static WrappedMethod wrapMethod(Class<?> cls, Method m) {
141 		if (m.getParameterCount() != 0
142 				|| !ALLOWED_RETURN_TYPES.contains(m.getReturnType())) {
143 			throw new IllegalArgumentException(
144 					format("bad Operation annotation on method %s of %s",
145 							m.getName(), cls));
146 		}
147 
148 		if (m.getReturnType().equals(Void.TYPE)) {
149 			// Synthesise a return value
150 			return obj -> {
151 				m.invoke(obj);
152 				return 0;
153 			};
154 		} else {
155 			return obj -> (Integer) m.invoke(obj);
156 		}
157 	}
158 
159 	/**
160 	 * A wrapper around {@link Method} of a {@link FunctionAPI} subclass.
161 	 */
162 	@FunctionalInterface
163 	private interface WrappedMethod {
164 		/**
165 		 * How to actually call the wrapped method.
166 		 *
167 		 * @param funcs
168 		 *            The instance to call the method on.
169 		 * @return The integer result.
170 		 * @throws InvocationTargetException
171 		 *             If the method throws
172 		 * @throws IllegalAccessException
173 		 *             If the method can't be accessed.
174 		 */
175 		int operation(FunctionAPI funcs)
176 				throws InvocationTargetException, IllegalAccessException;
177 
178 		/**
179 		 * Ugly stuff to wrap {@link WrappedMethod} as {@link Callable}. The
180 		 * truly nasty part is the handling of exceptions, which have to be
181 		 * unwrapped from the dynamic method calling machinery.
182 		 *
183 		 * @param objref
184 		 *            Reference to target object.
185 		 * @param command
186 		 *            What command this operation is.
187 		 * @param encodedOpcode
188 		 *            The actual operation code from the DSE bytecode.
189 		 * @return The result of the operation
190 		 * @throws DataSpecificationException
191 		 *             If the operation fails.
192 		 */
193 		default int call(WeakReference<FunctionAPI> objref, Commands command,
194 				int encodedOpcode) throws DataSpecificationException {
195 			var funcs = objref.get();
196 			requireNonNull(funcs, "unexpectedly early deallocation");
197 			funcs.unpack(encodedOpcode);
198 			if (log.isDebugEnabled()) {
199 				log.debug("EXEC: {} ({})", command,
200 						format("%08x", encodedOpcode));
201 			}
202 			try {
203 				try {
204 					return operation(funcs);
205 				} catch (InvocationTargetException innerException) {
206 					// Unwrap the inner exception
207 					throw innerException.getTargetException();
208 				}
209 			} catch (RuntimeException | Error
210 					| DataSpecificationException realException) {
211 				// These are the real things that can be thrown
212 				throw realException;
213 			} catch (Throwable badException) {
214 				// Should be unreachable
215 				throw new RuntimeException("bad call", badException);
216 			}
217 		}
218 	}
