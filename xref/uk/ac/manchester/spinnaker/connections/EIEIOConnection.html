17  package uk.ac.manchester.spinnaker.connections;
18  
19  import static uk.ac.manchester.spinnaker.messages.eieio.EIEIOMessageFactory.readCommandMessage;
20  import static uk.ac.manchester.spinnaker.messages.eieio.EIEIOMessageFactory.readDataMessage;
21  import static uk.ac.manchester.spinnaker.transceiver.Utils.newMessageBuffer;
22  
23  import java.io.IOException;
24  import java.net.InetAddress;
25  import java.nio.ByteBuffer;
26  
27  import uk.ac.manchester.spinnaker.connections.model.EIEIOReceiver;
28  import uk.ac.manchester.spinnaker.connections.model.EIEIOSender;
29  import uk.ac.manchester.spinnaker.messages.eieio.EIEIOCommand;
30  import uk.ac.manchester.spinnaker.messages.eieio.EIEIOCommandID;
31  import uk.ac.manchester.spinnaker.messages.eieio.EIEIOCommandMessage;
32  import uk.ac.manchester.spinnaker.messages.eieio.EIEIOHeader;
33  import uk.ac.manchester.spinnaker.messages.eieio.EIEIOMessage;
34  
35  /** A UDP connection for sending and receiving raw EIEIO messages. */
36  public class EIEIOConnection
37  		extends UDPConnection<EIEIOMessage<? extends EIEIOHeader>>
38  		implements EIEIOReceiver, EIEIOSender {
39  	/**
40  	 * Create an EIEIO connection only available for listening, using default
41  	 * local port.
42  	 *
43  	 * @param localHost
44  	 *            The local IP address to bind to. If not specified, it defaults
45  	 *            to binding to all interfaces, unless remoteHost is specified,
46  	 *            in which case binding is done to the IP address that will be
47  	 *            used to send packets.
48  	 * @throws IOException
49  	 *             If there is an error setting up the communication channel
50  	 */
51  	public EIEIOConnection(InetAddress localHost) throws IOException {
52  		super(localHost, null, null, null);
53  	}
54  
55  	/**
56  	 * Create an EIEIO connection only available for listening.
57  	 *
58  	 * @param localHost
59  	 *            The local IP address to bind to. If not specified, it defaults
60  	 *            to binding to all interfaces, unless remoteHost is specified,
61  	 *            in which case binding is done to the IP address that will be
62  	 *            used to send packets.
63  	 * @param localPort
64  	 *            The local port to bind to, 0 or between 1025 and 65535.
65  	 * @throws IOException
66  	 *             If there is an error setting up the communication channel
67  	 */
68  	public EIEIOConnection(InetAddress localHost, Integer localPort)
69  			throws IOException {
70  		super(localHost, localPort, null, null);
71  	}
72  
73  	/**
74  	 * Create an EIEIO connection.
75  	 *
76  	 * @param localHost
77  	 *            The local host to bind to. If not specified, it defaults to
78  	 *            binding to all interfaces, unless remoteHost is specified, in
79  	 *            which case binding is done to the IP address that will be used
80  	 *            to send packets.
81  	 * @param localPort
82  	 *            The local port to bind to, 0 or between 1025 and 65535.
83  	 * @param remoteHost
84  	 *            The remote host to send packets to. If not specified, the
85  	 *            socket will be available for listening only, and will throw
86  	 *            and exception if used for sending.
87  	 * @param remotePort
88  	 *            The remote port to send packets to. If remoteHost is
89  	 *            {@code null}, this is ignored. If remoteHost is specified,
90  	 *            this must also be specified as non-zero for the connection to
91  	 *            allow sending.
92  	 * @throws IOException
93  	 *             If there is an error setting up the communication channel
94  	 */
95  	public EIEIOConnection(InetAddress localHost, Integer localPort,
96  			InetAddress remoteHost, Integer remotePort) throws IOException {
97  		super(localHost, localPort, remoteHost, remotePort);
98  	}
99  
100 	@Override
101 	public void sendEIEIOMessage(EIEIOMessage<?> eieioMessage)
102 			throws IOException {
103 		ByteBuffer b = newMessageBuffer();
104 		eieioMessage.addToBuffer(b);
105 		b.flip();
106 		send(b);
107 	}
108 
109 	/**
110 	 * Send a raw command.
111 	 *
112 	 * @param command
113 	 *            The command to send.
114 	 * @throws IOException
115 	 *             If sending fails.
116 	 */
117 	protected void sendCommand(int command) throws IOException {
118 		sendCommand(EIEIOCommandID.get(command));
119 	}
120 
121 	/**
122 	 * Send a raw command.
123 	 *
124 	 * @param command
125 	 *            The command to send.
126 	 * @throws IOException
127 	 *             If sending fails.
128 	 */
129 	protected void sendCommand(EIEIOCommand command) throws IOException {
130 		sendEIEIOMessage(new EIEIOCommandMessage(command));
131 	}
132 
133 	/**
134 	 * Send a raw command.
135 	 *
136 	 * @param command
137 	 *            The command to send.
138 	 * @param ipAddress
139 	 *            The host to send to.
140 	 * @param port
141 	 *            The port to send to.
142 	 * @throws IOException
143 	 *             If sending fails.
144 	 */
145 	protected void sendCommand(EIEIOCommandID command, InetAddress ipAddress,
146 			int port) throws IOException {
147 		sendEIEIOMessageTo(new EIEIOCommandMessage(command), ipAddress, port);
148 	}
149 
150 	/**
151 	 * Receive a raw command.
152 	 *
153 	 * @return the command ID
154 	 * @throws IOException
155 	 *             If receiving fails.
156 	 */
157 	protected EIEIOCommand receiveCommand() throws IOException {
158 		EIEIOMessage<?> msg = receiveMessage();
159 		if (msg instanceof EIEIOCommandMessage) {
160 			return ((EIEIOCommandMessage) msg).getHeader().command;
161 		}
162 		throw new IOException("unexpected data message");
163 	}
164 
165 	/**
166 	 * Send an EIEIO message to a specific destination.
167 	 *
168 	 * @param eieioMessage
169 	 *            The message to send.
170 	 * @param ipAddress
171 	 *            The host to send to.
172 	 * @param port
173 	 *            The port to send to.
174 	 * @throws IOException
175 	 *             If anything goes wrong in sending.
176 	 */
177 	public void sendEIEIOMessageTo(
178 			EIEIOMessage<? extends EIEIOHeader> eieioMessage,
179 			InetAddress ipAddress, int port) throws IOException {
180 		ByteBuffer b = newMessageBuffer();
181 		eieioMessage.addToBuffer(b);
182 		b.flip();
183 		sendTo(b, ipAddress, port);
184 	}
185 
186 	private static final int MASK = 0xC000;
187 
188 	private static final int FLAG = 0x4000;
189 
190 	@Override
191 	public EIEIOMessage<? extends EIEIOHeader> receiveMessage(Integer timeout)
192 			throws IOException {
193 		ByteBuffer b = receive(timeout);
194 		short header = b.getShort();
195 		if ((header & MASK) == FLAG) {
196 			return readCommandMessage(b);
197 		} else {
198 			return readDataMessage(b);
199 		}
200 	}
201 }
