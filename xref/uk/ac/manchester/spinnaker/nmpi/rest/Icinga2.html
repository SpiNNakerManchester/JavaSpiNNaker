16  package uk.ac.manchester.spinnaker.nmpi.rest;
17  
18  import static com.fasterxml.jackson.databind.PropertyNamingStrategies.SNAKE_CASE;
19  import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
20  
21  import java.util.List;
22  import java.util.Map;
23  
24  import javax.ws.rs.Consumes;
25  import javax.ws.rs.HeaderParam;
26  import javax.ws.rs.POST;
27  import javax.ws.rs.Path;
28  import javax.ws.rs.Produces;
29  
30  import org.apache.cxf.jaxrs.client.JAXRSClientFactory;
31  
32  import com.fasterxml.jackson.databind.ObjectMapper;
33  import com.fasterxml.jackson.jaxrs.json.JacksonJsonProvider;
34  
35  import uk.ac.manchester.spinnaker.nmpi.model.Icinga2CheckResult;
36  
37  /**
38   * Interface to the Icinga2 API for status monitoring.
39   */
40  public interface Icinga2 {
41  
42  	/**
43  	 * Update the status of a service or host.
44  	 *
45  	 * @param authHeader The authorization header value to authenticate with
46  	 * @param result The result of a status check to update with.
47  	 * @return The response from the server as a String.
48  	 */
49  	@Produces(APPLICATION_JSON)
50  	@Consumes(APPLICATION_JSON)
51  	@POST
52  	@Path("/v1/actions/process-check-result")
53  	Map<String, Object> processCheckResult(
54  			@HeaderParam("Authorization") String authHeader,
55  			Icinga2CheckResult result);
56  
57  	/**
58  	 * Get a client for the API.
59  	 * @param url The URL to connect to.
60  	 * @return A proxy of the API.
61  	 */
62  	static Icinga2 createClient(String url) {
63  		ObjectMapper mapper = new ObjectMapper();
64  		mapper.setPropertyNamingStrategy(SNAKE_CASE);
65  		return JAXRSClientFactory.create(url, Icinga2.class,
66  				List.of(new JacksonJsonProvider()));
67  	}
68  }
