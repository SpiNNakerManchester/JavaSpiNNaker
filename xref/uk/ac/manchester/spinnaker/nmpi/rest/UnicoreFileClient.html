16  package uk.ac.manchester.spinnaker.nmpi.rest;
17  
18  import static com.fasterxml.jackson.databind.PropertyNamingStrategies.SNAKE_CASE;
19  
20  import java.io.InputStream;
21  import java.util.List;
22  
23  import javax.ws.rs.Consumes;
24  import javax.ws.rs.HeaderParam;
25  import javax.ws.rs.PUT;
26  import javax.ws.rs.Path;
27  import javax.ws.rs.PathParam;
28  import javax.ws.rs.WebApplicationException;
29  
30  import org.apache.cxf.jaxrs.client.JAXRSClientFactory;
31  
32  import com.fasterxml.jackson.databind.ObjectMapper;
33  import com.fasterxml.jackson.jaxrs.json.JacksonJsonProvider;
34  
35  /**
36   * An interface to the UNICORE storage REST API.
37   */
38  @Path("/storages")
39  public interface UnicoreFileClient {
40  	/**
41  	 * Upload a file.
42  	 *
43  	 * @param authHeader
44  	 *            The authorization header to authenticate with.
45  	 * @param id
46  	 *            The id of the storage on the server.
47  	 * @param filePath
48  	 *            The path at which to store the file (directories are
49  	 *            automatically created).
50  	 * @param input
51  	 *            The input stream containing the file to upload.
52  	 * @throws WebApplicationException
53  	 *             If anything goes wrong.
54  	 */
55  	@PUT
56  	@Path("{id}/files/{filePath}")
57  	@Consumes("application/octet-stream")
58  	void upload(@HeaderParam("Authorization") String authHeader,
59  			@PathParam("id") String id,
60  			@PathParam("filePath") String filePath, InputStream input)
61  			throws WebApplicationException;
62  
63  	/**
64  	 * Get a client for the API.
65  	 *
66  	 * @param url
67  	 *            The URL to connect to.
68  	 * @return A proxy of the API.
69  	 */
70  	static UnicoreFileClient createClient(String url) {
71  		var mapper = new ObjectMapper();
72  		mapper.setPropertyNamingStrategy(SNAKE_CASE);
73  		return JAXRSClientFactory.create(url, UnicoreFileClient.class,
74  				List.of(new JacksonJsonProvider()));
75  	}
76  }
