762 	public Statement getStatement() throws SQLException {
763 		if (s != null) {
764 			return s;
765 		}
766 		validateThread();
767 		var st = r.getStatement();
768 		if (st instanceof CallableStatement) {
769 			return wrap((CallableStatement) st);
770 		} else if (st instanceof PreparedStatement) {
771 			return wrap((PreparedStatement) st);
772 		} else {
773 			return wrap(st);
774 		}
775 	}
