101 	@PostConstruct
102 	private void launchBackground() {
103 		props = properties.getStateControl();
104 		// After a minute, start retrieving board serial numbers
105 		readAllTask =
106 				executor.schedule((Runnable) this::readAllBoardSerialNumbers,
107 						props.getBlacklistTimeout().getSeconds(), SECONDS);
108 		// Why can't I pass a Duration directly there?
109 	}
110 
111 	@PreDestroy
112 	private void stopBackground() {
113 		var t = readAllTask;
114 		if (t != null) {
115 			readAllTask = null;
116 			try {
117 				t.cancel(true);
118 				t.get();
119 			} catch (InterruptedException | CancellationException e) {
120 				log.trace("stopped background loader", e);
121 			} catch (Exception e) {
122 				log.info("failure in background board serial number fetch", e);
123 			}
124 		}
125 	}

450 	 * Retrieve the blacklist for the given board from the board and store it in
451 	 * the database.
452 	 *
453 	 * @param board
454 	 *            The board to get the blacklist of.
455 	 * @return The blacklist that was transferred, if any.
456 	 */
457 	public Optional<Blacklist> pullBlacklist(BoardState board) {
458 		try {
459 			return readBlacklistFromMachine(board).map(bl -> {
460 				blacklistStore.writeBlacklist(board.id, bl);
461 				execute(c -> {
462 					// These must be done in ONE transaction
463 					changed(c, board.id);
464 					synched(c, board.id);
465 					return this; // Unimportant result
466 				});
467 				return bl;
468 			});
469 		} catch (InterruptedException e) {
470 			return Optional.empty();
471 		}
472 	}

539 	private interface InterruptableConsumer<T> {
540 		/**
541 		 * Performs this operation on the given argument.
542 		 *
543 		 * @param t
544 		 *            the input argument
545 		 * @throws InterruptedException
546 		 *             if interrupted
547 		 */
548 		void accept(T t) throws InterruptedException;
549 	}
550 
551 	/**
552 	 * Perform an action for all boards in a machine (or all known), batching
553 	 * them as necessary. If interrupted, will complete the currently processing
554 	 * batch but will not perform further batches.
555 	 *
556 	 * @param machineName
557 	 *            Which machine are we talking about? If {@code null}, all
558 	 *            boards of all machines will be processed.
559 	 * @param action
560 	 *            What are we doing (for log messages)?
561 	 * @param batchSize
562 	 *            How many requests to handle at once in a batch of requests to
563 	 *            the back end engine.
564 	 * @param opGenerator
565 	 *            How to generate an individual operation to perform.
566 	 * @param opResultsHandler
567 	 *            How to process the results of an individual operation.
568 	 */
569 	private void batchReqs(String machineName, String action, int batchSize,
570 			Function<Integer, Op> opGenerator,
571 			InterruptableConsumer<Op> opResultsHandler) {
572 		var boards = executeRead(c -> listAllBoards(c, machineName));
573 		for (var batch : batch(batchSize, boards)) {
574 			/*
575 			 * Theoretically, this could be more efficiently done. Practically,
576 			 * a proper multi-op scheme is really complex, even before
577 			 * considering how to handle failure modes! This isn't a performance
578 			 * sensitive part of the code.
579 			 */
580 			var ops = lmap(batch, opGenerator);
581 			boolean stop = false;
582 			try {
583 				for (var op : ops) {
584 					try {
585 						opResultsHandler.accept(op);
586 					} catch (RuntimeException e) {
587 						log.warn("failed while {}", action, e);
588 					} catch (InterruptedException e) {
589 						log.info("interrupted while {}", action, e);
590 						stop = true;
591 					}
592 				}
593 			} finally {
594 				ops.forEach(Op::close);
595 			}
596 			if (stop) {
597 				// Mark as interrupted
598 				currentThread().interrupt();
599 				break;
600 			}
601 		}
602 	}

604 	/**
605 	 * Retrieve all blacklists, parse them, and store them in the DB's model.
606 	 *
607 	 * @param machineName
608 	 *            Which machine to get the blacklists of.
609 	 */
610 	@SuppressWarnings("MustBeClosed")
611 	public void updateAllBlacklists(@NotBlank String machineName) {
612 		batchReqs(requireNonNull(machineName), "retrieving blacklists",
613 				props.getBlacklistReadBatchSize(),
614 				id -> new Op(CREATE_BLACKLIST_READ, id),
615 				op -> op.getResult(serial("data", Blacklist.class))
616 						.ifPresent(bl -> {
617 							blacklistStore.writeBlacklist(op.boardId, bl);
618 							execute(c -> {
619 								// These must be done in ONE transaction
620 								changed(c, op.boardId);
621 								synched(c, op.boardId);
622 								return this; // Unimportant result
623 							});
624 						}));
625 	}

627 	private static List<Integer> listAllBoards(Connection conn,
628 			String machineName) {
629 		try (var machines = conn.query(GET_NAMED_MACHINE);
630 				var boards = conn.query(GET_ALL_BOARDS);
631 				var all = conn.query(GET_ALL_BOARDS_OF_ALL_MACHINES)) {
632 			if (machineName == null) {
633 				return all.call().map(integer("board_id")).toList();
634 			}
635 			return machines.call1(machineName).map(integer("machine_id")).map(
636 					mid -> boards.call(mid).map(integer("board_id")).toList())
637 					.orElse(List.of());
638 		}
639 	}

658 	 * @param board
659 	 *            Which board to write the blacklist of.
660 	 * @param blacklist

662 	 * @throws DataAccessException
663 	 *             If access to the DB fails.
664 	 */

666 			@Valid Blacklist blacklist) {
667 		blacklistStore.writeBlacklist(board.id, blacklist);
668 		execute(c -> changed(c, board.id)); // Unimportant result
669 	}

753 	/**
754 	 * Manages the transactions used to safely talk with the BMP controller.
755 	 *
756 	 * @author Donal Fellows
757 	 */
758 	private final class Op implements AutoCloseable {
759 		private final int op;
760 
761 		private final Epoch epoch;
762 
763 		private final int boardId;
764 
765 		/**
766 		 * @param operation
767 		 *            The SQL to create the operation to carry out. Must
768 		 *            generate an ID, so presumably is an {@code INSERT}.
769 		 * @param args
770 		 *            Values to bind to parameters in the SQL. The first
771 		 *            parameter <em>must</em> be the board ID! Presumably that
772 		 *            will be the board that the operation refers to.
773 		 */
774 		@MustBeClosed
775 		Op(@CompileTimeConstant final String operation, Object... args) {
776 			boardId = (Integer) args[0];
777 			epoch = epochs.getBlacklistEpoch();
778 			op = execute(conn -> {
779 				try (var readReq = conn.update(operation)) {
780 					return readReq.key(args);
781 				}
782 			}).orElseThrow(() -> new BlacklistException(
783 					"could not create blacklist request"));
784 		}
785 
786 		/**
787 		 * Wait for the result of the request to be ready.
788 		 *
789 		 * @param <T>
790 		 *            The type of the result value.
791 		 * @param retriever
792 		 *            How to convert the row containing the result into the
793 		 *            actual result of the transaction.
794 		 * @return The wrapped result, or empty if the operation times out.
795 		 * @throws InterruptedException
796 		 *             If the thread is interrupted.
797 		 * @throws BlacklistException
798 		 *             If the BMP throws an exception.
799 		 * @throws DataAccessException
800 		 *             If there is a problem accessing the database.
801 		 */
802 		<T> Optional<T> getResult(Function<Row, T> retriever)
803 				throws InterruptedException, BlacklistException,
804 				DataAccessException {
805 			var end = now().plus(props.getBlacklistTimeout());
806 			while (end.isAfter(now())) {
807 				var result = executeRead(conn -> {
808 					try (var getResult =
809 							conn.query(GET_COMPLETED_BLACKLIST_OP)) {
810 						return getResult.call1(op).map(this::throwIfFailed)
811 								.map(retriever);
812 					}
813 				});
814 				if (result.isPresent()) {
815 					return result;
816 				}
817 				epoch.waitForChange(props.getBlacklistPoll());
818 			}
819 			return Optional.empty();
820 		}
821 
822 		/**
823 		 * Wait for the result of the request to be ready, then discard that
824 		 * result. Used instead of {@link #getResult(Function)} when the value
825 		 * of the result is not interesting at all.
826 		 *
827 		 * @throws InterruptedException
828 		 *             If the thread is interrupted.
829 		 * @throws BlacklistException
830 		 *             If the BMP throws an exception.
831 		 * @throws DataAccessException
832 		 *             If there is a problem accessing the database.
833 		 */
834 		void completed() throws DataAccessException, BlacklistException,
835 				InterruptedException {
836 			getResult(__ -> this);
837 		}
838 
839 		/**
840 		 * If a row encodes a failure state, unpack the exception from the row
841 		 * and throw it as a wrapped exception. Otherwise, just pass on the row.
842 		 *
843 		 * @param row
844 		 *            The row to examine.
845 		 * @return The row, which is now guaranteed to not be a failure.
846 		 * @throws BlacklistException
847 		 *             If the row encodes a failure.
848 		 */
849 		private Row throwIfFailed(Row row) throws BlacklistException {
850 			if (row.getBoolean("failed")) {
851 				throw new BlacklistException(
852 						"failed to access hardware blacklist",
853 						row.getSerial("failure", Exception.class));
854 			}
855 			return row;
856 		}
857 
858 		/**
859 		 * Deletes the temporary operation description row in the DB.
860 		 * <p>
861 		 * {@inheritDoc}
862 		 */
863 		@Override
864 		public void close() {
865 			execute(conn -> {
866 				try (var delReq = conn.update(DELETE_BLACKLIST_OP)) {
867 					return delReq.call(op);
868 				}
869 			});
870 		}
871 	}
