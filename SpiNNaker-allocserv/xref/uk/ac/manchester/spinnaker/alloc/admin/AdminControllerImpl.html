145 	private Map<String, Boolean> getMachineNames(boolean allowOutOfService) {
146 		try (var conn = getConnection();
147 				var listMachines = conn.query(LIST_MACHINE_NAMES)) {
148 			return conn.transaction(false,
149 					() -> listMachines.call(allowOutOfService)
150 							.toMap(string("machine_name"), bool("in_service")));
151 		} catch (DataAccessException e) {
152 			log.warn("problem when listing machines", e);
153 			return Map.of();
154 		}
155 	}

168 	 * All models should contain a common set of attributes that describe where
169 	 * the view is rendering and where other parts of the admin interface are.
170 	 * Only call from
171 	 * {@link #addStandardContext(ModelAndView, RedirectAttributes)}.
172 	 *
173 	 * @param model
174 	 *            The base model to add to. This may be the real model or the
175 	 *            flash attributes.
176 	 */
177 	private static void addStandardContextAttrs(Map<String, Object> model) {
178 		var auth = getContext().getAuthentication();
179 		boolean mayChangePassword =
180 				auth instanceof UsernamePasswordAuthenticationToken;
181 
182 		model.put(BASE_URI, fromCurrentRequestUri().toUriString());
183 		model.put(TRUST_LEVELS, TrustLevel.values());
184 		model.put(USERS_URI, uri(admin().listUsers()));
185 		model.put(CREATE_USER_URI, uri(admin().getUserCreationForm()));
186 		model.put(CREATE_GROUP_URI, uri(admin().getGroupCreationForm()));
187 		model.put(GROUPS_URI, uri(admin().listGroups()));
188 		model.put(BOARDS_URI, uri(admin().boards()));
189 		model.put(MACHINE_URI, uri(admin().machineManagement()));
190 		model.put(USER_MAY_CHANGE_PASSWORD, mayChangePassword);
191 	}

194 	 * All models should contain a common set of attributes that describe where
195 	 * the view is rendering and where other parts of the admin interface are.
196 	 *
197 	 * @param mav
198 	 *            The model-and-view.
199 	 * @param attrs
200 	 *            The redirect attributes, or {@code null} if this is not a
201 	 *            redirect.
202 	 * @return The enhanced model-and-view.
203 	 */
204 	private static ModelAndView addStandardContext(ModelAndView mav,
205 			RedirectAttributes attrs) {
206 		addStandardContextAttrs(nonNull(attrs)
207 				// Real implementation of flash attrs is always a ModelMap
208 				? (ModelMap) attrs.getFlashAttributes()
209 				: mav.getModel());
210 		return mav;
211 	}

251 	@ExceptionHandler(BindException.class)
252 	ModelAndView validationError(BindException result) {
253 		if (result.hasGlobalErrors()) {
254 			// I don't believe this is really reachable code
255 			log.debug("binding problem", result);
256 			return errors(errorMessage(result.getGlobalError()));
257 		} else if (result.hasFieldErrors()) {
258 			log.debug("binding problem", result);
259 			return errors(errorMessage(result.getFieldError()));
260 		} else {
261 			// This should definitely be unreachable
262 			log.error("unknown binding error", result);
263 			return errors("unknown error");
264 		}
265 	}

267 	/**
268 	 * Convert thrown exceptions from the DB layer to views so the rest of the
269 	 * code doesn't have to.
270 	 *
271 	 * @param e
272 	 *            A database access exception.
273 	 * @param hm
274 	 *            What method generated the problem? Used to look up what was
275 	 *            supposed to happen.
276 	 * @return The view to render.
277 	 */
278 	@ExceptionHandler(DataAccessException.class)
279 	ModelAndView dbException(DataAccessException e, HandlerMethod hm) {
280 		var a = hm.getMethodAnnotation(Action.class);
281 		if (nonNull(a)) {
282 			log.warn("database access issue when {}", a.value(), e);
283 		} else {
284 			log.warn("database access issue", e);
285 		}
286 		return errors(e);
287 	}

556 	@Action("adding a user to a group")
557 	public ModelAndView addUserToGroup(int id, String user,
558 			RedirectAttributes attrs) {
559 		var g = userManager.getGroup(id, null).orElseThrow(NoGroup::new);
560 		var u = userManager.getUser(user, null).orElseThrow(NoUser::new);
561 		if (userManager.addUserToGroup(u, g).isPresent()) {
562 			log.info("added user {} to group {}", u.getUserName(),
563 					g.getGroupName());
564 			addNotice(attrs, format("added user %s to group %s",
565 					u.getUserName(), g.getGroupName()));
566 		} else {
567 			addNotice(attrs, format("user %s is already a member of group %s",
568 					u.getUserName(), g.getGroupName()));
569 		}
570 		return redirectTo(showGroupInfoUrl(id), attrs);
571 	}

623 	private Optional<BoardState> getBoardState(BoardRecord board) {
624 		if (nonNull(board.getId())) {
625 			return machineController.findId(board.getId());
626 		} else if (board.isTriadCoordPresent()) {
627 			return machineController.findTriad(board.getMachineName(),
628 					new TriadCoords(board.getX(), board.getY(), board.getZ()));
629 		} else if (board.isPhysicalCoordPresent()) {
630 			return machineController.findPhysical(board.getMachineName(),
631 					new PhysicalCoords(board.getCabinet(), board.getFrame(),
632 							board.getBoard()));
633 		} else if (board.isAddressPresent()) {
634 			return machineController.findIP(board.getMachineName(),
635 					board.getIpAddress());
636 		} else {
637 			// unreachable because of validation
638 			return Optional.empty();
639 		}
640 	}

644 	@Action("processing changes to a board's configuration")
645 	public ModelAndView board(BoardRecord board, ModelMap model) {
646 		var bs = getBoardState(board).orElseThrow(NoBoard::new);
647 
648 		inflateBoardRecord(board, bs);
649 
650 		if (board.isEnabledDefined()) {
651 			// We're doing a set
652 			log.info("setting board-allocatable state for board {} to {}", bs,
653 					board.isEnabled());
654 			bs.setState(board.isEnabled());
655 			spalloc.purgeDownCache();
656 		}
657 
658 		board.setEnabled(bs.getState());
659 		addBlacklistData(bs, model);
660 		inflateBoardRecord(board, bs);
661 		addBoard(model, board);
662 		addMachineList(model, getMachineNames(true));
663 		return addStandardContext(BOARD_VIEW.view(model));
664 	}

751 	 * Copy settings from the record out of the DB.
752 	 *
753 	 * @param br
754 	 *            Where the values are copied to. A partial state.
