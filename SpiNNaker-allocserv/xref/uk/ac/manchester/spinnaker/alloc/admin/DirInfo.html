19  import static java.util.Objects.requireNonNull;
20  import static org.slf4j.LoggerFactory.getLogger;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import org.slf4j.Logger;

27  import uk.ac.manchester.spinnaker.alloc.db.DatabaseAPI.Connection;
28  import uk.ac.manchester.spinnaker.alloc.db.SQLQueries;
29  import uk.ac.manchester.spinnaker.alloc.model.Direction;
30  import uk.ac.manchester.spinnaker.machine.board.ValidTriadZ;

32  /**
33   * A mapping that says how to go from one board's coordinates (only the Z
34   * coordinate matters for this) to another when you move in a particular
35   * direction. Comes from the {@code movement_directions} table in the database.
36   * <p>
37   * Consider this board layout (a classic 24 board machine, with wrap-arounds not
38   * shown):
39   * <p>
40   * <img src="doc-files/DirInfo1.png" width="450" alt="24-board layout">
41   * <p>
42   * Bear in mind that 0,1,0 is <em>actually</em> 12 chips vertically and 0 chips
43   * horizontally offset from 0,0,0. (Also, the real boards are slightly offset
44   * from this layout.)
45   *
46   * @author Donal Fellows
47   * @see Direction
48   * @see MachineDefinitionLoader
49   */
50  public final class DirInfo extends SQLQueries {
51  	private static final Logger log = getLogger(DirInfo.class);
52  
53  	private static final Map<Integer, Map<Direction, DirInfo>> MAP =
54  			new HashMap<>();
55  
56  	/**
57  	 * When your Z coordinate is this.
58  	 */
59  	@ValidTriadZ
60  	public final int z;
61  
62  	/** When you are moving in this direction. */
63  	public final Direction dir;
64  
65  	/** Change your X coordinate by this. */
66  	public final int dx;
67  
68  	/** Change your Y coordinate by this. */
69  	public final int dy;
70  
71  	/** Change your Z coordinate by this. */
72  	public final int dz;
73  
74  	private DirInfo(int z, Direction d, int dx, int dy, int dz) {
75  		this.z = z;
76  		this.dir = requireNonNull(d);
77  		this.dx = dx;
78  		this.dy = dy;
79  		this.dz = dz;
80  
81  		MAP.computeIfAbsent(z, __ -> new HashMap<>()).put(d, this);
82  	}
83  
84  	/**
85  	 * Obtain the correct motion information given a starting point and a
86  	 * direction.
87  	 *
88  	 * @param z
89  	 *            The starting Z coordinate. (Motions are independent of X and
90  	 *            Y.) Must be in range {@code 0..2}.
91  	 * @param direction
92  	 *            The direction to move in.
93  	 * @return How to move.
94  	 */
95  	public static DirInfo get(int z, Direction direction) {
96  		return MAP.get(z).get(direction);
97  	}
98  
99  	@Override
100 	public boolean equals(Object o) {
101 		if (!(o instanceof DirInfo)) {
102 			return false;
103 		}
104 		var di = (DirInfo) o;
105 		return z == di.z && dir == di.dir;
106 	}
107 
108 	@Override
109 	public int hashCode() {
110 		return z ^ dir.hashCode();
111 	}
112 
113 	static void load(Connection conn) {
114 		if (MAP.isEmpty()) {
115 			conn.transaction(false, () -> {
116 				try (var di = conn.query(LOAD_DIR_INFO)) {
117 					di.call()
118 							.forEach(row -> new DirInfo(row.getInt("z"),
119 									row.getEnum("direction", Direction.class),
120 									row.getInt("dx"), row.getInt("dy"),
121 									row.getInt("dz")));
122 				}
123 			});
124 			log.debug("created {} DirInfo instances",
125 					MAP.values().stream().mapToInt(Map::size).sum());
126 		}
127 	}
128 }
