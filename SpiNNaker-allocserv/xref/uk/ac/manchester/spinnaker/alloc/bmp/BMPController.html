153 	@PostConstruct
154 	private void init() {
155 		// Set up scheduler
156 		var sched = new ThreadPoolTaskScheduler();
157 		scheduler = sched;
158 		sched.setThreadGroupName("BMP");
159 		sched.initialize();
160 
161 		controllerFactory = controllerFactoryBean::getObject;
162 		allocator.setBMPController(this);
163 
164 		// We do the making of workers later in tests
165 		if (!serviceControl.isUseDummyBMP()) {
166 			makeWorkers();
167 		}
168 
169 		// Set the pool size to match the number of workers
170 		if (workers.size() > 1) {
171 			sched.setPoolSize(workers.size());
172 		}
173 
174 		// Launch the scheduler now it is all set up
175 		sched.initialize();
176 	}

178 	private void makeWorkers() {
179 		// Make workers
180 		try (var c = getConnection();
181 				var getBmps = c.query(GET_ALL_BMPS);
182 				var getBoards = c.query(GET_ALL_BMP_BOARDS)) {
183 			var madeWorkers = c.transaction(false, () -> getBmps.call(row -> {
184 				var m = spallocCore.getMachine(row.getString("machine_name"),
185 						true);
186 				var coords = new BMPCoords(row.getInt("cabinet"),
187 						row.getInt("frame"));
188 				var boards = new HashMap<BMPBoard, String>();
189 				var bmpId = row.getInt("bmp_id");
190 				getBoards.call(r -> {
191 					boards.put(new BMPBoard(r.getInt("board_num")),
192 							r.getString("address"));
193 					return null;
194 				}, bmpId);
195 				var control = controllerFactory.create(m.get(), coords, boards);
196 				var worker = new Worker(control, bmpId);
197 				workers.put(row.getInt("bmp_id"), worker);
198 				return worker;
199 			}));
200 
201 			// Schedule outside the transaction for safety
202 			for (var worker : madeWorkers) {
203 				scheduler.scheduleAtFixedRate(worker, allocProps.getPeriod());
204 			}
205 		}
206 	}

208 	/**
209 	 * Trigger the execution of the workers for the given BMPs now.
210 	 *
211 	 * @param bmps
212 	 *            A list of BMPs that have changed.
213 	 */
214 	public void triggerSearch(Collection<Integer> bmps) {
215 		for (var b : bmps) {
216 			var worker = workers.get(b);
217 			if (worker != null) {
218 				scheduler.schedule(() -> worker.run(), Instant.now());
219 			} else {
220 				log.error("Could not find worker for BMP {}", b);
221 			}
222 		}

225 	/** An action that may throw any of a range of exceptions. */
226 	private interface ThrowingAction {
227 		void act() throws ProcessException, IOException, InterruptedException;
228 	}

230 	private abstract class Request {
231 		final int bmpId;
232 
233 		private int numTries = 0;
234 
235 		Request(int bmpId) {
236 			this.bmpId = bmpId;
237 		}
238 
239 		/**
240 		 * @return Whether this request may be repeated.
241 		 */
242 		boolean isRepeat() {
243 			return numTries < props.getPowerAttempts();
244 		}

246 		/**
247 		 * Basic machinery for handling exceptions that arise while performing a
248 		 * BMP action. Runs on a thread that may touch a BMP directly, but which
249 		 * may not touch the database.
250 		 * <p>
251 		 * Only subclasses should use this!
252 		 *
253 		 * @param body
254 		 *            What to attempt.
255 		 * @param onFailure
256 		 *            What to do on failure.
257 		 * @param onServiceRemove
258 		 *            If the exception looks serious, call this to trigger a
259 		 *            board being taken out of service.
260 		 * @return Whether to stop the retry loop.
261 		 * @throws InterruptedException
262 		 *             If interrupted.
263 		 */
264 		final boolean bmpAction(ThrowingAction body,
265 				Consumer<Exception> onFailure,
266 				Consumer<PermanentProcessException> onServiceRemove)
267 				throws InterruptedException {
268 			boolean isLastTry = numTries++ >= props.getPowerAttempts();
269 			Exception exn;
270 			try {
271 				body.act();
272 				// Exit the retry loop (up the stack); the requests all worked
273 				return true;
274 			} catch (InterruptedException e) {
275 				/*
276 				 * We were interrupted! This happens when we're shutting down.
277 				 * Log (because we're in an inconsistent state) and rethrow so
278 				 * that the outside gets to clean up.
279 				 */
280 				log.error("Requests failed on BMP {} because of "
281 						+ "interruption", bmpId, e);
282 				currentThread().interrupt();
283 				throw e;
284 			} catch (TransientProcessException e) {
285 				if (!isLastTry) {
286 					// Log somewhat gently; we *might* be able to recover...
287 					log.warn("Retrying requests on BMP {} after {}: {}",
288 							bmpId, props.getProbeInterval(),
289 							e.getMessage());
290 					// Ask for a retry
291 					return false;
292 				}
293 				exn = e;
294 				log.error("Requests failed on BMP {}", bmpId, e);
295 			} catch (PermanentProcessException e) {
296 				log.error("BMP {} on {} is unreachable", e.source, bmpId, e);
297 				onServiceRemove.accept(e);
298 				exn = e;
299 			} catch (CallerProcessException e) {
300 				// This is probably a software bug
301 				log.error("SW bug talking to BMP {}", bmpId, e);
302 				exn = e;
303 			} catch (ProcessException | IOException | RuntimeException e) {
304 				log.error("Requests failed on BMP {}", bmpId, e);
305 				exn = e;
306 			}
307 			/*
308 			 * Common permanent failure handling case; arrange for taking a
309 			 * board out of service, mark a request as failed, and stop the
310 			 * retry loop.
311 			 */
312 			onFailure.accept(exn);
313 			return true;
314 		}

373 		boolean processRequest(SpiNNakerControl control) {
374 			while (isRepeat()) {
375 				try {
376 					if (tryProcessRequest(control)) {
377 						return true;
378 					}
379 					sleep(props.getProbeInterval().toMillis());
380 				} catch (InterruptedException e) {
381 					// If this happens, just cancel the transaction;
382 					// when we come back, all things will be redone.
383 					throw new RuntimeException(e);
384 				}
385 			}
386 			return false;
387 		}

393 	/**
394 	 * Describes a request to modify the power status of a collection of boards.
395 	 * The boards must be on a single machine and must all be assigned to a
396 	 * single job.
397 	 * <p>
398 	 * This is the message that is sent from the main thread to the per-BMP
399 	 * worker threads.
400 	 *
401 	 * @author Donal Fellows
402 	 */
403 	private final class PowerRequest extends Request {
404 		private final List<BMPBoard> powerOnBoards = new ArrayList<>();
405 
406 		private final List<BMPBoard> powerOffBoards = new ArrayList<>();
407 
408 		private final List<Link> linkRequests = new ArrayList<>();
409 
410 		private final int jobId;
411 
412 		private final JobState from;
413 
414 		private final JobState to;
415 
416 		private final List<Integer> changeIds = new ArrayList<>();
417 
418 		private final Map<Integer, Integer> boardToId = new HashMap<>();
419 
420 		/**
421 		 * Create a request.
422 		 *
423 		 * @param sql
424 		 *            How to access the database.
425 		 * @param machine
426 		 *            What machine are the boards on? <em>Must not</em> be
427 		 *            {@code null}.
428 		 * @param powerOn
429 		 *            What boards (by DB ID) are to be powered on? May be
430 		 *            {@code null}; that's equivalent to the empty list.
431 		 * @param powerOff
432 		 *            What boards (by DB ID) are to be powered off? May be
433 		 *            {@code null}; that's equivalent to the empty list.
434 		 * @param links
435 		 *            Any link power control requests. By default, links are on
436 		 *            if their board is on and they are connected; it is
437 		 *            <em>useful and relevant</em> to modify the power state of
438 		 *            links on the periphery of an allocation. May be
439 		 *            {@code null}; that's equivalent to the empty list.
440 		 * @param jobId
441 		 *            For what job is this?
442 		 * @param from
443 		 *            What state is the job moving from?
444 		 * @param to
445 		 *            What state is the job moving to?
446 		 * @param changeIds
447 		 *            The DB ids that describe the change, so we can update
448 		 *            those records.
449 		 * @param idToBoard
450 		 *            How to get the physical ID of a board from its database ID
451 		 */
452 		PowerRequest(int bmpId, int jobId, JobState from, JobState to,
453 				List<PowerChange> powerChanges) {
454 			super(bmpId);
455 			for (var change : powerChanges) {
456 				if (change.power) {
457 					powerOnBoards.add(new BMPBoard(change.boardNum));
458 				} else {
459 					powerOffBoards.add(new BMPBoard(change.boardNum));
460 				}
461 				change.offLinks.stream().forEach(link ->
462 						linkRequests.add(new Link(change.boardNum, link)));
463 				changeIds.add(change.changeId);
464 				boardToId.put(change.boardNum, change.boardId);
465 			}
466 			this.jobId = jobId;
467 			this.from = from;
468 			this.to = to;
469 		}
470 
471 		/**
472 		 * Change the power state of boards in this request.
473 		 *
474 		 * @param controllers
475 		 *            How to actually communicate with the machine
476 		 * @throws ProcessException
477 		 *             If the transceiver chokes
478 		 * @throws InterruptedException
479 		 *             If interrupted
480 		 * @throws IOException
481 		 *             If network I/O fails
482 		 */
483 		void changeBoardPowerState(SpiNNakerControl controller)
484 				throws ProcessException, InterruptedException, IOException {
485 
486 			// Send any power on commands
487 			if (!powerOnBoards.isEmpty()) {
488 				controller.powerOnAndCheck(powerOnBoards);
489 			}
490 
491 			// Process perimeter link requests next
492 			for (var linkReq : linkRequests) {
493 				// Set the link state, as required
494 				controller.setLinkOff(linkReq);
495 			}
496 
497 			// Finally send any power off commands
498 			if (!powerOffBoards.isEmpty()) {
499 				controller.powerOff(powerOffBoards);
500 			}
501 		}
502 
503 		/**
504 		 * Handles the database changes after a set of changes to a BMP complete
505 		 * successfully. We will move the job to the state it supposed to be in.
506 		 *
507 		 * @param sql
508 		 *            How to access the DB
509 		 * @return Whether the state of boards or jobs has changed.
510 		 */
511 		private void done() {
512 			try (var c = getConnection();
513 					var deallocateBoards = c.update(DEALLOCATE_BMP_BOARDS_JOB);
514 					var deleteChange = c.update(FINISHED_PENDING);
515 					var setBoardPowerOn = c.update(SET_BOARD_POWER_ON);
516 					var setBoardPowerOff = c.update(SET_BOARD_POWER_OFF)) {
517 				c.transaction(() -> {
518 					int turnedOn = powerOnBoards.stream().map(this::getBoardId)
519 							.mapToInt(setBoardPowerOn::call).sum();
520 					int turnedOff =
521 							powerOffBoards.stream().map(this::getBoardId)
522 									.mapToInt(setBoardPowerOff::call).sum();
523 
524 					if (to == DESTROYED || to == QUEUED) {
525 						/*
526 						 * Need to mark the boards as not allocated; can't do
527 						 * that until they've been switched off.
528 						 */
529 						deallocateBoards.call(jobId, bmpId);
530 					}
531 					int completed = changeIds.stream().mapToInt(
532 							deleteChange::call).sum();
533 
534 					log.debug("BMP ACTION SUCCEEDED ({}:{}->{}): on:{} off:{} "
535 							+ "completed: {}",
536 							jobId, from, to, turnedOn, turnedOff, completed);
537 				});
538 			}
539 
540 			// Tell the allocator something has happened
541 			allocator.updateJob(jobId, from, to);
542 		}
543 
544 		/**
545 		 * Handles the database changes after a set of changes to a BMP complete
546 		 * with a failure. We will roll back the job state to what it was
547 		 * before.
548 		 *
549 		 * @param sql
550 		 *            How to access the DB
551 		 * @return Whether the state of boards or jobs has changed.
552 		 */
553 		private void failed() {
554 			var resetJobAlloc = false;
555 			try (var c = getConnection();
556 					var deallocateBoards = c.update(DEALLOCATE_BMP_BOARDS_JOB);
557 					var deleteChange = c.update(FINISHED_PENDING);
558 					var setBoardPowerOff = c.update(SET_BOARD_POWER_OFF)) {
559 				resetJobAlloc = c.transaction(() -> {
560 					// We should mark the boards as off
561 					int turnedOff =
562 							powerOffBoards.stream().map(this::getBoardId)
563 									.mapToInt(setBoardPowerOff::call).sum();
564 
565 					// Deallocate the boards on this bmp from the job;
566 					// other boards can be deallocated elsewhere.
567 					deallocateBoards.call(jobId, bmpId);
568 
569 					// Delete change ids as they are done even if failed.
570 					var completed = changeIds.stream().mapToInt(
571 							deleteChange::call).sum();
572 
573 					log.debug(
574 							"BMP ACTION FAILED on {} ({}:{}->{}) off:{} "
575 							+ "completed:{}",
576 							bmpId, jobId, from, to, turnedOff, completed);
577 
578 					// If we were meant to be powering up, reset the allocation
579 					// once done here.
580 					return (to == READY && powerOffBoards.isEmpty());
581 				});
582 			}
583 			if (resetJobAlloc) {
584 				allocator.resetPowerOnFailure(jobId);
585 			}
586 		}

588 		/**
589 		 * Process an action to power on or off a set of boards. Runs on a
590 		 * thread that may touch a BMP directly, but which may not touch the
591 		 * database.
592 		 *
593 		 * @param controller
594 		 *            How to actually reach the BMPs.
595 		 * @return Whether this action has "succeeded" and shouldn't be retried.
596 		 * @throws InterruptedException
597 		 *             If interrupted.
598 		 */
599 		@Override
600 		boolean tryProcessRequest(SpiNNakerControl controller)
601 				throws InterruptedException {
602 			boolean ok = bmpAction(() -> {
603 				changeBoardPowerState(controller);
604 				// We want to ensure the lead board is alive
605 				if (!serviceControl.isUseDummyBMP()) {
606 					// Don't bother with pings when the dummy is enabled
607 					controller.ping(powerOnBoards);
608 				}
609 				done();
610 			}, e -> {
611 				failed();
612 				synchronized (BMPController.this) {
613 					bmpProcessingException = e;
614 				}
615 			}, ppe -> {
616 				badBoard(ppe);
617 			});
618 			return ok;
619 		}

621 		@Override
622 		public String toString() {
623 			var sb = new StringBuilder("PowerRequest(for=")
624 					.append(bmpId);
625 			sb.append(";on=").append(powerOnBoards);
626 			sb.append(",off=").append(powerOffBoards);
627 			sb.append(",links=").append(linkRequests);
628 			return sb.append(")").toString();
629 		}

636 		 * somewhere else, and we should mark the board as out of service too;
637 		 * it's never going to work so taking it out right away is the only sane

641 		 *            The failure message.

675 	 * A request to read or write information on a BMP. Includes blacklists,
676 	 * serial numbers, temperature data, etc.
677 	 *
678 	 * @author Donal Fellows
679 	 */
680 	private final class BoardRequest extends Request {
681 		private final NonBootOperation op;
682 
683 		private final int opId;
684 
685 		private final int boardId;
686 
687 		private final BMPCoords bmp;
688 
689 		private final BMPBoard board;
690 
691 		private final String bmpSerialId;
692 
693 		private final Blacklist blacklist;
694 
695 		private final int machineId;
696 
697 		private BoardRequest(int bmpId, NonBootOperation op, Row row) {
698 			super(bmpId);
699 			this.op = op;
700 			opId = row.getInt("op_id");
701 			boardId = row.getInt("board_id");
702 			bmp = new BMPCoords(row.getInt("cabinet"), row.getInt("frame"));
703 			board = new BMPBoard(row.getInt("board_num"));
704 			if (op == WRITE_BL) {
705 				blacklist = row.getSerial("data", Blacklist.class);
706 			} else {
707 				blacklist = null;
708 			}
709 			bmpSerialId = row.getString("bmp_serial_id");
710 			machineId = row.getInt("machine_id");
711 		}
712 
713 		/** The serial number actually read from the board. */
714 		private String readSerial;
715 
716 		/**
717 		 * Access the DB to store the serial number information that we
718 		 * retrieved. A transaction should already be held.
719 		 *
720 		 * @param c
721 		 *            How to access the DB
722 		 * @return Whether we've changed anything
723 		 */
724 		private void recordSerialIds(Connection c) {
725 			try (var setBoardSerialIds = c.update(SET_BOARD_SERIAL_IDS)) {
726 				setBoardSerialIds.call(boardId, readSerial,
727 						phySerMap.getPhysicalId(readSerial));
728 			}
729 		}
730 
731 		/**
732 		 * Access the DB to mark the read request as successful and store the
733 		 * blacklist that was read. A transaction should already be held.
734 		 *
735 		 * @param c
736 		 *            How to access the DB
737 		 * @param readBlacklist
738 		 *            The blacklist that was read
739 		 * @return Whether we've changed anything
740 		 */
741 		private void doneReadBlacklist(Connection c, Blacklist readBlacklist) {
742 			try (var completed = c.update(COMPLETED_BOARD_INFO_READ)) {
743 				log.debug("Completing blacklist read opId {}", opId);
744 				completed.call(readBlacklist, opId);
745 			}
746 		}
747 
748 		/**
749 		 * Access the DB to mark the write request as successful. A transaction
750 		 * should already be held.
751 		 *
752 		 * @param c
753 		 *            How to access the DB
754 		 * @return Whether we've changed anything
755 		 */
756 		private void doneWriteBlacklist(Connection c) {
757 			try (var completed = c.update(COMPLETED_BLACKLIST_WRITE)) {
758 				completed.call(opId);
759 			}
760 		}
761 
762 		/**
763 		 * Access the DB to mark the read request as successful; the actual
764 		 * store of the serial data is elsewhere
765 		 * ({@link #recordSerialIds(Connection)}). A transaction should already
766 		 * be held.
767 		 *
768 		 * @param c
769 		 *            How to access the DB
770 		 * @return Whether we've changed anything
771 		 */
772 		private void doneReadSerial(Connection c) {
773 			try (var completed = c.update(COMPLETED_GET_SERIAL_REQ)) {
774 				completed.call(opId);
775 			}
776 		}
777 
778 		/**
779 		 * Access the DB to mark the read request as successful and store the
780 		 * ADC info that was read. A transaction should be held.
781 		 *
782 		 * @param c
783 		 *            The database connection.
784 		 */
785 		private void doneReadTemps(Connection c, ADCInfo adcInfo) {
786 			try (var completed = c.update(COMPLETED_BOARD_INFO_READ)) {
787 				log.debug("Completing temperature read opId {}", opId);
788 				completed.call(adcInfo, opId);
789 			}
790 		}
791 
792 		/**
793 		 * Access the DB to mark the request as failed and store the exception.
794 		 *
795 		 * @param exn
796 		 *            The exception that caused the failure.
797 		 * @return Whether we've changed anything
798 		 */
799 		private void failed(Exception exn) {
800 			try (var c = getConnection();
801 					var failed = c.update(FAILED_BLACKLIST_OP)) {
802 				c.transaction(() -> failed.call(exn, opId));
803 			}
804 		}

806 		private static final String REPORT_MSG =
807 				"board was not reachable when trying to access its blacklist: ";
808 
809 		/**
810 		 * Access the DB to mark a board as out of service.
811 		 *
812 		 * @param exn
813 		 *            The exception that caused the failure.
814 		 * @return Whether we've changed anything
815 		 */
816 		void takeOutOfService(Exception exn) {
817 			try (var c = getConnection()) {
818 				c.transaction(() -> {
819 					addBoardReport(c, boardId, null, REPORT_MSG + exn);
820 					markBoardAsDead(c, boardId, REPORT_MSG + exn);
821 				});
822 			}
823 		}
824 
825 		/**
826 		 * Process an action to work with a blacklist or serial number. Runs on
827 		 * a thread that may touch a BMP directly, but which may not touch the
828 		 * database.
829 		 *
830 		 * @param controller
831 		 *            How to actually reach the BMP.
832 		 * @return Whether this action has "succeeded" and shouldn't be retried.
833 		 * @throws InterruptedException
834 		 *             If interrupted.
835 		 */
836 		@Override
837 		boolean tryProcessRequest(SpiNNakerControl controller)
838 				throws InterruptedException {
839 			return bmpAction(() -> {
840 				switch (op) {
841 				case WRITE_BL:
842 					writeBlacklist(controller);
843 					break;
844 				case READ_BL:
845 					readBlacklist(controller);
846 					break;
847 				case GET_SERIAL:
848 					readSerial(controller);
849 					break;
850 				case READ_TEMP:
851 					readTemps(controller);
852 					break;
853 				default:
854 					throw new IllegalArgumentException();
855 				}
856 				epochs.blacklistChanged(boardId);
857 				epochs.machineChanged(machineId);
858 			}, e -> {
859 				failed(e);
860 			}, ppe -> {
861 				takeOutOfService(ppe);
862 			});
863 		}
864 
865 		/**
866 		 * Process an action to read a blacklist.
867 		 *
868 		 * @param controller
869 		 *            How to actually reach the BMP.
870 		 * @throws InterruptedException
871 		 *             If interrupted.
872 		 * @throws IOException
873 		 *             If the network is unhappy.
874 		 * @throws ProcessException
875 		 *             If the BMP rejects a message.
876 		 */
877 		private void readBlacklist(SpiNNakerControl controller)
878 				throws InterruptedException, ProcessException, IOException {
879 			readSerial = controller.readSerial(board);
880 			if (bmpSerialId != null && !bmpSerialId.equals(readSerial)) {
881 				/*
882 				 * Doesn't match; WARN but keep going; hardware may just be
883 				 * remapped behind our back.
884 				 */
885 				log.warn(
886 						"blacklist read mismatch: expected serial ID '{}' "
887 								+ "not equal to actual serial ID '{}'",
888 						bmpSerialId, readSerial);
889 			}
890 			var readBlacklist = controller.readBlacklist(board);
891 			try (var c = getConnection()) {
892 				c.transaction(() -> {
893 					recordSerialIds(c);
894 					doneReadBlacklist(c, readBlacklist);
895 				});
896 			}
897 		}
898 
899 		/**
900 		 * Process an action to write a blacklist.
901 		 *
902 		 * @param controller
903 		 *            How to actually reach the BMP.
904 		 * @throws InterruptedException
905 		 *             If interrupted.
906 		 * @throws IOException
907 		 *             If the network is unhappy.
908 		 * @throws ProcessException
909 		 *             If the BMP rejects a message.
910 		 * @throws IllegalStateException
911 		 *             If the operation is applied to a board other than the one
912 		 *             that it is expected to apply to.
913 		 */
914 		private void writeBlacklist(SpiNNakerControl controller)
915 				throws InterruptedException, ProcessException, IOException {
916 			readSerial = controller.readSerial(board);
917 			if (bmpSerialId != null && !bmpSerialId.equals(readSerial)) {
918 				// Doesn't match, so REALLY unsafe to keep going!
919 				throw new IllegalStateException(format(
920 						"aborting blacklist write: expected serial ID '%s' "
921 								+ "not equal to actual serial ID '%s'",
922 						bmpSerialId, readSerial));
923 			}
924 			controller.writeBlacklist(board, requireNonNull(blacklist));
925 			try (var c = getConnection()) {
926 				c.transaction(() -> doneWriteBlacklist(c));
927 			}
928 		}
929 
930 		/**
931 		 * Process an action to read the serial number from a BMP.
932 		 *
933 		 * @param controller
934 		 *            How to actually reach the BMP.
935 		 * @throws InterruptedException
936 		 *             If interrupted.
937 		 * @throws IOException
938 		 *             If the network is unhappy
939 		 * @throws ProcessException
940 		 *             If the BMP rejects a message.
941 		 */
942 		private void readSerial(SpiNNakerControl controller)
943 				throws InterruptedException, ProcessException, IOException {
944 			readSerial = controller.readSerial(board);
945 			try (var c = getConnection()) {
946 				c.transaction(() -> {
947 					recordSerialIds(c);
948 					doneReadSerial(c);
949 				});
950 			}
951 		}
952 
953 		/**
954 		 * Process an action to read some temperature data.
955 		 *
956 		 * @param controller
957 		 *            How to actually reach the BMP.
958 		 * @throws InterruptedException
959 		 *             If interrupted.
960 		 * @throws IOException
961 		 *             If the network is unhappy.
962 		 * @throws ProcessException
963 		 *             If the BMP rejects a message.
964 		 */
965 		private void readTemps(SpiNNakerControl controller)
966 				throws InterruptedException, ProcessException, IOException {
967 			var adcInfo = controller.readTemp(board);
968 			try (var c = getConnection()) {
969 				c.transaction(() -> doneReadTemps(c, adcInfo));
970 			}
971 		}

1021 	/** A worker of a given BMP. */
1022 	private final class Worker implements Runnable {
1023 		/** What are we controlling? */
1024 		private final SpiNNakerControl control;
1025 
1026 		/** Which boards are we looking at? */
1027 		private final int bmpId;
1028 
1029 		Worker(SpiNNakerControl control, int bmpId) {
1030 			this.control = control;
1031 			this.bmpId = bmpId;
1032 
1033 			log.debug("Created worker for boards {}", bmpId);
1034 		}
1035 
1036 		/**
1037 		 * Periodically call to update, or trigger externally.
1038 		 */
1039 		@Override
1040 		public synchronized void run() {
1041 			log.trace("Searching for changes on BMP {}", bmpId);
1042 
1043 			try {
1044 				var changes = getRequestedOperations();
1045 				for (var change : changes) {
1046 					change.processRequest(control);
1047 				}
1048 			} catch (Exception e) {
1049 				log.error("unhandled exception for BMP '{}'", bmpId, e);
1050 			}
1051 		}
1052 
1053 		/**
1054 		 * Get the things that we want the worker to do. <em>Be very
1055 		 * careful!</em> Because this necessarily involves the database, this
1056 		 * must not touch the BMP handle as those operations take a long time
1057 		 * and we absolutely must not have a transaction open at the same time.
1058 		 *
1059 		 * @return List of operations to perform.
1060 		 */
1061 		private List<Request> getRequestedOperations() {
1062 			var requests = new ArrayList<Request>();
1063 			try (var c = getConnection();
1064 					var getPowerRequests = c.query(GET_CHANGES);
1065 					var getBlacklistReads = c.query(GET_BLACKLIST_READS);
1066 					var getBlacklistWrites = c.query(GET_BLACKLIST_WRITES);
1067 					var getReadSerialInfos = c.query(GET_SERIAL_INFO_REQS);
1068 					var getReadTemps = c.query(GET_TEMP_INFO_REQS)) {
1069 				c.transaction(false, () -> {
1070 					// Batch power requests by job
1071 					var powerChanges = new LinkedList<>(
1072 							getPowerRequests.call(PowerChange::new, bmpId));
1073 					while (!powerChanges.isEmpty()) {
1074 						var change = powerChanges.poll();
1075 						var jobChanges = new ArrayList<>(List.of(change));
1076 						while (!powerChanges.isEmpty()
1077 								&& change.isSameJob(powerChanges.peek())) {
1078 							jobChanges.add(powerChanges.poll());
1079 						}
1080 						if (!jobChanges.isEmpty()) {
1081 							log.debug("Running job changes {}", jobChanges);
1082 							requests.add(new PowerRequest(bmpId, change.jobId,
1083 									change.from, change.to, jobChanges));
1084 						}
1085 					}
1086 
1087 					// Leave these until quiet
1088 					if (requests.isEmpty()) {
1089 						requests.addAll(getBlacklistReads.call(
1090 								row -> new BoardRequest(bmpId, READ_BL, row),
1091 								bmpId));
1092 					}
1093 					if (requests.isEmpty()) {
1094 						requests.addAll(getBlacklistWrites.call(
1095 								row -> new BoardRequest(bmpId, WRITE_BL, row),
1096 								bmpId));
1097 						requests.addAll(getReadSerialInfos.call(
1098 								row -> new BoardRequest(bmpId, GET_SERIAL, row),
1099 								bmpId));
1100 						requests.addAll(getReadTemps.call(
1101 								row -> new BoardRequest(bmpId, READ_TEMP, row),
1102 								bmpId));
1103 					}
1104 				});
1105 			} catch (Exception e) {
1106 				log.error("unhandled exception for BMP '{}'", bmpId, e);
1107 			}
1108 			return requests;
1109 		}
1110 	}
