17  package uk.ac.manchester.spinnaker.alloc.bmp;
18  
19  import java.io.IOException;
20  
21  import uk.ac.manchester.spinnaker.alloc.allocator.SpallocAPI.Machine;
22  import uk.ac.manchester.spinnaker.machine.board.BMPCoords;
23  import uk.ac.manchester.spinnaker.transceiver.BMPTransceiverInterface;
24  import uk.ac.manchester.spinnaker.transceiver.SpinnmanException;
25  
26  /**
27   * Creates transceivers for talking to the BMPs of machines. Note that each
28   * machine only has the one BMP that is talked to, and only ever one transceiver
29   * that is used to do it.
30   *
31   * @param <T>
32   *            The actual type of transceiver produced.
33   * @author Donal Fellows
34   */
35  public interface TransceiverFactoryAPI<T extends BMPTransceiverInterface> {
36  	/**
37  	 * Get the transceiver for talking to a given machine's BMPs.
38  	 *
39  	 * @param machineDescription
40  	 *            The machine we're talking about.
41  	 * @param bmp
42  	 *            Which BMP on the machine are we talking to.
43  	 * @return The transceiver. Only operations relating to BMPs are required to
44  	 *         be supported.
45  	 * @throws IOException
46  	 *             If low-level things go wrong.
47  	 * @throws SpinnmanException
48  	 *             If the transceiver can't be built.
49  	 * @throws InterruptedException
50  	 *             If the communications were interrupted.
51  	 */
52  	T getTransciever(Machine machineDescription, BMPCoords bmp)
53  			throws IOException, SpinnmanException, InterruptedException;
54  }
