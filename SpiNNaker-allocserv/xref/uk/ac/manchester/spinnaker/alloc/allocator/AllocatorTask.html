167 	private void updateJobNow(int jobId, JobState sourceState,
168 			JobState targetState) {
169 		try {
170 			var updated = execute(
171 					conn -> update(jobId, sourceState, targetState, conn));
172 			if (updated) {
173 				log.debug("advancing job {} epoch", jobId);
174 				epochs.jobChanged(jobId);
175 			}
176 		} catch (DataAccessException e) {
177 			if (isBusy(e)) {
178 				log.info("database is busy; "
179 						+ "will try allocation processing later");
180 				return;
181 			}
182 			throw e;
183 		}
184 	}

186 	private boolean update(int jobId, JobState sourceState,
187 			JobState targetState, Connection c) {
188 		try (var getChangeStatus = c.query(COUNT_CHANGES_FOR_JOB);
189 				var setJobState = c.update(SET_STATE_PENDING);
190 				var setJobDestroyed = c.update(SET_STATE_DESTROYED);
191 				var deleteChanges = c.update(DELETE_PENDING)) {
192 			// Count pending changes for this state change
193 			var status = getChangeStatus.call1(ChangeStatus::new,
194 					jobId, sourceState, targetState).orElseThrow(
195 							() -> new RuntimeException(
196 									"Error counting job tasks"));
197 
198 			log.debug("Job {} has {} changes remaining", jobId,
199 					status.nChanges);
200 
201 			// If the remaining things are errors, react (if there are errors,
202 			// eventually non-errors will be deleted)
203 			if (status.nErrors > 0 && (status.nErrors == status.nChanges)) {
204 				log.info("Job {} changes resulted in errors.", jobId);
205 
206 				// We can delete the changes now as we know the issues
207 				deleteChanges.call(jobId, sourceState, targetState);
208 
209 				// If we are going to destroyed, we can mostly ignore errors,
210 				// and similarly if we are going to queue it again anyway
211 				if (targetState == DESTROYED || targetState == QUEUED) {
212 					return true;
213 				}
214 
215 				// If the job was ready before we tried to do this, we have to
216 				// destroy the job with an error!
217 				if (sourceState == READY) {
218 					destroyJob(c, jobId, DESTROY_ON_POWER_ERROR);
219 					return true;
220 				}
221 
222 				// If the job was not ready, we need to re-queue and reallocate
223 				// boards
224 				scheduler.schedule(() -> setPower(jobId, OFF, QUEUED),
225 						Instant.now());
226 				return false;
227 			} else if (status.nChanges > 0) {
228 				// There are still changes happening - let them finish first
229 				// even if there are errors as safer to do once everything
230 				// is done.
231 				return false;
232 			}
233 
234 			// If there are no more pending changes and no errors,
235 			// set the job state to the target state
236 			log.debug("Job {} moving to state {}", jobId, targetState);
237 			if (targetState == DESTROYED) {
238 				int rows = setJobDestroyed.call(jobId);
239 				if (rows != 1) {
240 					log.warn("unexpected number of rows affected by "
241 							+ "destroy in state update: {}", rows);
242 				}
243 				return rows > 0;
244 			}
245 
246 			return setJobState.call(targetState, jobId) > 0;
247 		}
248 	}

291 	/**
292 	 * Allocate all current requests for resources.
293 	 */
294 	public synchronized void allocate() {
295 		if (serviceControl.isPaused()) {
296 			return;
297 		}
298 
299 		try {
300 			var allocated = execute(this::allocate);
301 			allocated.updateEpochs();
302 			allocated.updateBMPs();
303 		} catch (DataAccessException e) {
304 			if (isBusy(e)) {
305 				log.info("database is busy; "
306 						+ "will try allocation processing later");
307 				return;
308 			}
309 			throw e;
310 		}
311 	}

498 	private class AllocTask {
499 		final int id;
500 
501 		final int importance;
502 
503 		final int jobId;
504 
505 		final int machineId;
506 
507 		final Rectangle max;
508 
509 		final int maxDeadBoards;
510 
511 		final Integer numBoards;
512 
513 		final Integer width;
514 
515 		final Integer height;
516 
517 		final Integer root;
518 
519 		AllocTask(Row row) {
520 			id = row.getInt("req_id");
521 			importance = row.getInt("importance");
522 			jobId = row.getInt("job_id");
523 			machineId = row.getInt("machine_id");
524 			max = new Rectangle(row);
525 			maxDeadBoards = row.getInt("max_dead_boards");
526 			numBoards = row.getInteger("num_boards");
527 			width = row.getInteger("width");
528 			height = row.getInteger("height");
529 			root = row.getInteger("board_id");
530 		}
531 
532 		Collection<BMPAndMachine> allocate(AllocSQL sql) {
533 			if (nonNull(numBoards) && numBoards > 0) {
534 				// Single-board case gets its own allocator that's better at
535 				// that
536 				if (numBoards == 1) {
537 					log.debug("Allocate one board");
538 					return allocateOneBoard(sql, jobId, machineId);
539 				}
540 				var estimate = new DimensionEstimate(numBoards, max);
541 				return allocateDimensions(sql, jobId, machineId, estimate,
542 						maxDeadBoards);
543 			}
544 
545 			if (nonNull(width) && nonNull(height) && nonNull(root)) {
546 				return allocateTriadsAt(sql, jobId, machineId, root, width,
547 						height,	maxDeadBoards);
548 			}
549 
550 			if (nonNull(width) && nonNull(height) && width > 0 && height > 0) {
551 				// Special case; user is really just asking for one board
552 				if (height == 1 && width == 1 && nonNull(maxDeadBoards)
553 						&& maxDeadBoards == 2) {
554 					return allocateOneBoard(sql, jobId, machineId);
555 				}
556 				var estimate = new DimensionEstimate(width, height, max);
557 				log.debug(
558 						"resolved request for {}x{} boards to {}x{} triads "
559 								+ "with tolerance {}",
560 						width, height, estimate.width, estimate.height,
561 						estimate.tolerance);
562 				return allocateDimensions(sql, jobId, machineId, estimate,
563 						maxDeadBoards);
564 			}
565 
566 			if (nonNull(root)) {
567 				// Ignores maxDeadBoards; is a single-board allocate
568 				return allocateBoard(sql, jobId, machineId, root);
569 			}
570 
571 			log.warn("job {} could not be allocated; "
572 					+ "bad request will be cleared from queue", jobId);
573 			return List.of();
574 		}
575 	}

577 	/**
578 	 * A set of information about the allocations that have been made.
579 	 */
580 	class Allocations {
581 
582 		/** The BMPs that have been affected by the allocations. **/
583 		final Set<Integer> bmps = new HashSet<>();
584 
585 		/** The Machines that have been affected by the allocations. **/
586 		final Set<Integer> machines = new HashSet<>();
587 
588 		/** The jobs that have been affected by the allocations. **/
589 		final List<Integer> jobIds = new ArrayList<>();
590 
591 		Allocations() {
592 			// Does nothing
593 		}
594 
595 		Allocations(int jobId, Collection<BMPAndMachine> bmps) {
596 			addAll(jobId, bmps);
597 		}
598 
599 		void addAll(int jobId, Collection<BMPAndMachine> bmps) {
600 			if (bmps.size() > 0) {
601 				jobIds.add(jobId);
602 				for (var bm : bmps) {
603 					this.bmps.add(bm.bmpId);
604 					this.machines.add(bm.machineId);
605 				}
606 			}
607 		}
608 
609 		void updateEpochs() {
610 			log.debug("Updating jobs {}", jobIds);
611 			for (var job : jobIds) {
612 				epochs.jobChanged(job);
613 			}
614 			log.debug("Updating machines {}", machines);
615 			for (var m : machines) {
616 				epochs.machineChanged(m);
617 			}
618 		}
619 
620 		void updateBMPs() {
621 			if (!bmps.isEmpty() && bmpController != null) {
622 				// Poke the BMP controller to start looking!
623 				log.debug("Triggering BMPs {}", bmps);
624 				bmpController.triggerSearch(bmps);
625 			}
626 		}
627 
628 		boolean notEmpty() {
629 			return !jobIds.isEmpty();
630 		}
631 	}

633 	/**
634 	 * Allocate all current requests for resources.
635 	 *
636 	 * @param conn
637 	 *            The DB connection
638 	 * @return Whether any changes have been done
639 	 */
640 	private Allocations allocate(Connection conn) {
641 		try (var sql = new AllocSQL(conn)) {
642 			int maxImportance = -1;
643 			log.trace("Allocate running");
644 			var allocations = new Allocations();
645 			for (AllocTask task : sql.getTasks.call(AllocTask::new, QUEUED)) {
646 				if (task.importance > maxImportance) {
647 					maxImportance = task.importance;
648 				} else if (task.importance < maxImportance
649 						- allocProps.getImportanceSpan()) {
650 					// Too much of a span
651 					continue;
652 				}
653 				var handled = task.allocate(sql);
654 				// If we handled it, delete the request
655 				if (handled.size() > 0) {
656 					sql.delete.call(task.id);
657 				}
658 				allocations.addAll(task.jobId, handled);
659 				log.debug("allocate for {} (job {}): {}", task.id,
660 						task.jobId, handled);
661 			}
662 			/*
663 			 * Those tasks which weren't allocated get their importance bumped
664 			 * so they get considered with higher priority when the allocator
665 			 * runs next time.
666 			 */
667 			sql.bumpImportance.call();
668 			return allocations;
669 		}

880 	/**
881 	 * Implementation of {@link #tombstone()}. This is done as two transactions
882 	 * to help manage the amount of locking (especially multi-DB locking);
883 	 * nothing else ought to be updating any of these jobs at the time this task
884 	 * usually runs, but we'll still try to keep things minimally locked.
885 	 *
886 	 * @param conn
887 	 *            The DB connection
888 	 * @return Description of the tombstoned IDs
889 	 */
890 	private Copied tombstone(Connection conn, Connection histConn) {
891 		// No tombstoning without the target DB!
892 		if (!isHistoricalDBAvailable()) {
893 			return new Copied(List.of(), List.of());
894 		}
895 
896 		try (var readJobs = conn.query(READ_HISTORICAL_JOBS);
897 				var readAllocs = conn.query(READ_HISTORICAL_ALLOCS);
898 				var deleteJobs = conn.update(DELETE_JOB_RECORD);
899 				var deleteAllocs = conn.update(DELETE_ALLOC_RECORD);
900 				var writeJobs = histConn.update(WRITE_HISTORICAL_JOBS);
901 				var writeAllocs = histConn.update(WRITE_HISTORICAL_ALLOCS)) {
902 			var grace = historyProps.getGracePeriod();
903 			var copied = conn.transaction(
904 					() -> new Copied(readJobs.call(HistoricalJob::new, grace),
905 							readAllocs.call(HistoricalAlloc::new, grace)));
906 			histConn.transaction(() -> {
907 				copied.allocs().forEach((a) -> writeAllocs.call(a.args()));
908 				copied.jobs().forEach((j) -> writeJobs.call(j.args()));
909 			});
910 			conn.transaction(() -> {
911 				copied.allocs().forEach((a) -> deleteAllocs.call(a.allocId));
912 				copied.jobs().forEach((j) -> deleteJobs.call(j.jobId));
913 			});
914 			return copied;
915 		}
916 	}

940 		try (var sql = new DestroySQL(conn)) {

942 					.orElse(DESTROYED) == DESTROYED) {

967 	/**
968 	 * Computes the estimate of what sort of allocation will be required.
969 	 * Converts a number of boards into a close-to-square size to search for.
970 	 * <p>
971 	 * With the big machine's level of resources, that's good enough. For now.
972 	 *
973 	 * @author Donal Fellows
974 	 */
975 	private static final class DimensionEstimate {
976 		/** The estimated width, in triads. */
977 		final int width;
978 
979 		/** The estimated height, in triads. */
980 		final int height;
981 
982 		/**
983 		 * The number of boards in the rectangle of triads that we can tolerate
984 		 * being down due to overallocation (due to the use of rectangles and
985 		 * triads).
986 		 */
987 		final int tolerance;
988 
989 		/**
990 		 * Create an estimate of what to allocate. The old spalloc would take
991 		 * hints at this point on the aspect ratio, but we don't bother; we
992 		 * strongly prefer allocations "nearly square", going for making them
993 		 * slightly taller than wide if necessary.
994 		 *
995 		 * @param numBoards
996 		 *            The number of boards wanted.
997 		 * @param max
998 		 *            The size of the machine.
999 		 */
1000 		DimensionEstimate(int numBoards, Rectangle max) {
1001 			if (numBoards < 1) {
1002 				throw new IllegalArgumentException(
1003 						"number of boards must be greater than zero");
1004 			}
1005 			int numTriads = ceildiv(numBoards, TRIAD_DEPTH);
1006 			width = min((int) ceil(sqrt(numTriads)), max.width);
1007 			height = min(ceildiv(numTriads, width), max.height);
1008 			tolerance = (width * height * TRIAD_DEPTH) - numBoards;
1009 			if (width < 1 || height < 1) {
1010 				throw new IllegalArgumentException(
1011 						"computed dimensions must be greater than zero");
1012 			}
1013 			if (tolerance < 0) {
1014 				throw new IllegalArgumentException(
1015 						"that job cannot possibly fit on this machine");
1016 			}
1017 		}
1018 
1019 		/**
1020 		 * Create an estimate of what to allocate. This does not need to be
1021 		 * "near square".
1022 		 *
1023 		 * @param w
1024 		 *            The width of the allocation requested, in triads.
1025 		 * @param h
1026 		 *            The height of the allocation requested, in triads.
1027 		 * @param max
1028 		 *            The size of the machine.
1029 		 */
1030 		DimensionEstimate(int w, int h, Rectangle max) {
1031 			if (w < 1 || h < 1) {
1032 				throw new IllegalArgumentException(
1033 						"dimensions must be greater than zero");
1034 			}
1035 			int numBoards = w * h * TRIAD_DEPTH;
1036 			width = max(1, min(w, max.width));
1037 			height = max(1, min(h, max.height));
1038 			tolerance = (width * height * TRIAD_DEPTH) - numBoards;
1039 			if (tolerance < 0) {
1040 				throw new IllegalArgumentException(
1041 						"that job cannot possibly fit on this machine");
1042 			}
1043 		}
1044 
1045 		/** @return The estimated dimensions as a rectangle. */
1046 		Rectangle getRect() {
1047 			return new Rectangle(width, height, TRIAD_DEPTH);
1048 		}
1049 	}

1067 	private Collection<BMPAndMachine> allocateDimensions(AllocSQL sql,
1068 			int jobId, int machineId, DimensionEstimate estimate,
1069 			int userMaxDead) {
1070 		int tolerance = userMaxDead + estimate.tolerance;
1071 		int minArea =
1072 				estimate.width * estimate.height * TRIAD_DEPTH - tolerance;
1073 		for (var root : sql.getRectangles
1074 				.call(AllocatorTask::coords, estimate.width, estimate.height,
1075 						machineId, tolerance)) {
1076 			if (minArea > 1) {
1077 				/*
1078 				 * Check that a minimum number of boards are reachable from the
1079 				 * proposed root board. If the root board is isolated, we don't
1080 				 * care if the rest of the allocation works because the rest of
1081 				 * the toolchain won't cope.
1082 				 */
1083 				int size = connectedSize(sql, machineId, root, estimate);
1084 				if (size < minArea) {
1085 					continue;
1086 				}
1087 			}
1088 			return setAllocation(sql, jobId, estimate.getRect(), machineId,
1089 					root);
1090 		}
1091 		log.debug("Could not allocate min area {}", minArea);
1092 		return List.of();
1093 	}

1178 	/**
1179 	 * Does the actual allocation.
1180 	 * <p>
1181 	 * At this point, we've checked that there's enough boards <em>and</em> we
1182 	 * are running in a transaction. We take particular care that we only
1183 	 * actually allocate boards that are reachable from the root board; this is
1184 	 * assumed by the SpiNNaker tools, so we'd better conform to that
1185 	 * expectation. Fortunately, the bigger the allocation, the more likely that
1186 	 * is true (and it is trivially true for single-board allocations.)
1187 	 * <p>
1188 	 * If you want a multi-board allocation, you'd better be allocating a full
1189 	 * triad's-worth of depth or you'll get nothing.
1190 	 *
1191 	 * @param sql
1192 	 *            How to talk to the DB
1193 	 * @param jobId
1194 	 *            What job are we allocating for
1195 	 * @param rect
1196 	 *            Proposed rectangle size
1197 	 * @param machineId
1198 	 *            What machine are we allocating on
1199 	 * @param root
1200 	 *            Proposed root coordinates
1201 	 * @return The BMPs that have been used to make the allocation.
1202 	 */
1203 	private Collection<BMPAndMachine> setAllocation(AllocSQL sql, int jobId,
1204 			Rectangle rect,	int machineId, TriadCoords root) {
1205 		log.debug("performing allocation for {}: {}x{}x{} at {}:{}:{}", jobId,
1206 				rect.width, rect.height, rect.depth, root.x, root.y, root.z);
1207 		var boardsToAllocate = sql.getConnectedBoardIDs
1208 				.call(integer("board_id"), machineId, root.x, root.y, root.z,
1209 						rect.width, rect.height, rect.depth);
1210 		if (boardsToAllocate.isEmpty()) {
1211 			log.debug("No boards to allocate");
1212 			return List.of();
1213 		}
1214 		for (var boardId : boardsToAllocate) {
1215 			sql.allocBoard.call(jobId, boardId);
1216 		}
1217 
1218 		var board = boardsToAllocate.get(0);
1219 		sql.allocJob.call(rect.width, rect.height, rect.depth,
1220 				board, boardsToAllocate.size(), board, jobId);
1221 		log.info("allocated {} boards to {}; issuing power up commands",
1222 				boardsToAllocate.size(), jobId);
1223 		// Any proxies that existed are now defunct; user must make anew
1224 		rememberer.killProxies(jobId);
1225 		return setPower(sql, jobId, ON, READY);
1226 	}

1244 	/**
1245 	 * Issue a request to change the power for the boards of a job.
1246 	 *
1247 	 * @param sql
1248 	 *            How to talk to the DB
1249 	 * @param jobId
1250 	 *            The job in question
1251 	 * @param power
1252 	 *            The power state to switch to
1253 	 * @param targetState
1254 	 *            The state to put the job in afterwards
1255 	 * @return The ids of the BMPs that have been changed
1256 	 */
1257 	private Collection<BMPAndMachine> setPower(PowerSQL sql, int jobId,
1258 			PowerState power, JobState targetState) {
1259 		var sourceState = sql.getJobState.call1(
1260 				enumerate("job_state", JobState.class), jobId).orElseThrow();
1261 		var boards = sql.getJobBoards.call(BoardAndBMP::new, jobId);
1262 		if (boards.isEmpty()) {
1263 			log.debug("No boards for job {}", jobId);
1264 			if (targetState == DESTROYED) {
1265 				log.debug("no boards for {} in destroy", jobId);
1266 			}
1267 			// Make sure we still update the job!
1268 			updateJob(jobId, sourceState, targetState);
1269 			return List.of();
1270 		}
1271 		log.debug("{} boards for job {}", boards.size(), jobId);
1272 
1273 		// Number of changes pending, one per board
1274 		int numPending = 0;
1275 
1276 		var bmps = new HashSet<BMPAndMachine>();
1277 		if (power == ON) {
1278 			/*
1279 			 * This is a bit of a trickier case, as we need to say which links
1280 			 * are to be switched on or, more particularly, which are to be
1281 			 * switched off because they are links to boards that are not
1282 			 * allocated to the job. Off-board links are shut off by default.
1283 			 */
1284 			var perimeterLinks = Row.stream(
1285 					sql.getPerimeter.call(Perimeter::new, jobId))
1286 					.toCollectingMap(Direction.class, (p) -> p.boardId,
1287 							(p) -> p.direction);
1288 
1289 			for (var board : boards) {
1290 				var toChange = perimeterLinks.getOrDefault(board.boardId,
1291 						NO_PERIMETER);
1292 				numPending += sql.issuePowerChange.call(jobId,
1293 						board.boardId,
1294 						sourceState, targetState, true,
1295 						!toChange.contains(Direction.N),
1296 						!toChange.contains(Direction.E),
1297 						!toChange.contains(Direction.SE),
1298 						!toChange.contains(Direction.S),
1299 						!toChange.contains(Direction.W),
1300 						!toChange.contains(Direction.NW));
1301 				bmps.add(board.bmp);
1302 			}
1303 		} else {
1304 			// Powering off; all links switch to off so no perimeter check
1305 			for (var board : boards) {
1306 				numPending += sql.issuePowerChange.call(jobId,
1307 						board.boardId,	sourceState, targetState,
1308 						false, false, false, false,	false, false, false);
1309 				bmps.add(board.bmp);
1310 			}
1311 		}
1312 
1313 		if (targetState == DESTROYED) {
1314 			log.debug("num changes for {} in destroy: {}", jobId, numPending);
1315 			log.info("destroying job {} after power change", jobId);
1316 			int rows = sql.setStateDestroyed.call(jobId);
1317 			if (rows != 1) {
1318 				log.warn("unexpected number of jobs marked destroyed: {}",
1319 						rows);
1320 			}
1321 		} else {
1322 			log.debug("Num changes for target {}: {}", targetState, numPending);
1323 			sql.setStatePending.call(
1324 				numPending > 0 ? POWER : targetState, jobId);
1325 		}
1326 
1327 		return bmps;
1328 	}
