17  package uk.ac.manchester.spinnaker.alloc.allocator;
18  
19  import uk.ac.manchester.spinnaker.alloc.model.JobState;
20  import uk.ac.manchester.spinnaker.alloc.model.PowerState;
21  
22  /**
23   * A service that knows how to initiate the setting of the power state of a job.
24   *
25   * @author Donal Fellows
26   */
27  public interface PowerController {
28  	/**
29  	 * Destroy a job. The power controller has the responsibility because it
30  	 * releases any resources held by the job.
31  	 *
32  	 * @param jobId
33  	 *            The ID of the job.
34  	 * @param reason
35  	 *            Why is the job being destroyed.
36  	 */
37  	void destroyJob(int jobId, String reason);
38  
39  	/**
40  	 * Issue a power change for a job.
41  	 *
42  	 * @param jobId
43  	 *            The ID of the allocated job.
44  	 * @param power
45  	 *            What state to change the job's boards' power to.
46  	 * @param targetState
47  	 *            What state are we aiming to put the job into once the power
48  	 *            has been switched. Should be {@link JobState#READY} or
49  	 *            {@link JobState#DESTROYED}.
50  	 * @return Whether any change has been requested.
51  	 */
52  	boolean setPower(int jobId, PowerState power, JobState targetState);
53  }
