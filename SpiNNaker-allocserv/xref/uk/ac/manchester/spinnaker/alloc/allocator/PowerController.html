16  package uk.ac.manchester.spinnaker.alloc.allocator;
17  
18  import uk.ac.manchester.spinnaker.alloc.model.JobState;
19  import uk.ac.manchester.spinnaker.alloc.model.PowerState;
20  
21  /**
22   * A service that knows how to initiate the setting of the power state of a job.
23   *
24   * @author Donal Fellows
25   */
26  public interface PowerController {
27  	/**
28  	 * Destroy a job. The power controller has the responsibility because it
29  	 * releases any resources held by the job.
30  	 *
31  	 * @param jobId
32  	 *            The ID of the job.
33  	 * @param reason
34  	 *            Why is the job being destroyed.
35  	 */
36  	void destroyJob(int jobId, String reason);
37  
38  	/**
39  	 * Issue a power change for a job.
40  	 *
41  	 * @param jobId
42  	 *            The ID of the allocated job.
43  	 * @param power
44  	 *            What state to change the job's boards' power to.
45  	 * @param targetState
46  	 *            What state are we aiming to put the job into once the power
47  	 *            has been switched. Should be {@link JobState#READY} or
48  	 *            {@link JobState#DESTROYED}.
49  	 * @return Whether any change has been requested.
50  	 */
51  	boolean setPower(int jobId, PowerState power, JobState targetState);
52  }
