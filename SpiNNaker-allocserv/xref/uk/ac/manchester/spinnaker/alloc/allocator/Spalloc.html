432 	@Override
433 	public Job createJobInGroup(String owner, String groupName,
434 			CreateDescriptor descriptor, String machineName, List<String> tags,
435 			Duration keepaliveInterval, byte[] req)
436 					throws IllegalArgumentException {
437 		if (emergencyStop) {
438 			throw new IllegalStateException(
439 					"The Server has been stopped due to an emergency.");
440 		}
441 		return execute(conn -> {
442 			int user = getUser(conn, owner).orElseThrow(
443 					() -> new RuntimeException("no such user: " + owner));
444 			int group = selectGroup(conn, owner, groupName);
445 			if (!quotaManager.mayCreateJob(group)) {
446 				// No quota left
447 				throw new IllegalArgumentException(
448 						"quota exceeded in group " + group);
449 			}
450 
451 			var m = selectMachine(conn, descriptor, machineName, tags);
452 			if (!m.isPresent()) {
453 				throw new IllegalArgumentException(
454 						"no machine available which matches allocation "
455 						+ "request");
456 			}
457 			var machine = m.orElseThrow();
458 
459 			var id = insertJob(conn, machine, user, group, keepaliveInterval,
460 					req);
461 			if (!id.isPresent()) {
462 				throw new RuntimeException("failed to create job");
463 			}
464 			int jobId = id.orElseThrow();
465 
466 			var scale = props.getPriorityScale();
467 
468 			if (machine.getArea() < descriptor.getArea()) {
469 				throw new IllegalArgumentException(
470 						"request cannot fit on machine");
471 			}
472 
473 			// Ask the allocator engine to do the allocation
474 			int numBoards = descriptor.visit(new CreateVisitor<Integer>() {
475 				@Override
476 				public Integer numBoards(CreateNumBoards nb) {
477 					try (var insertReq = conn.update(INSERT_REQ_N_BOARDS)) {
478 						insertReq.call(jobId, nb.numBoards, nb.maxDead,
479 								(int) (nb.getArea() * scale.getSize()));
480 					}
481 					return nb.numBoards;
482 				}
483 
484 				@Override
485 				public Integer dimensions(CreateDimensions d) {
486 					try (var insertReq = conn.update(INSERT_REQ_SIZE)) {
487 						insertReq.call(jobId, d.width, d.height, d.maxDead,
488 								(int) (d.getArea() * scale.getDimensions()));
489 					}
490 					return max(1, d.getArea() - d.maxDead);
491 				}
492 
493 				/*
494 				 * Request by area rooted at specific location; resolve to board
495 				 * ID now, as that doesn't depend on whether the board is
496 				 * currently in use.
497 				 */
498 				@Override
499 				public Integer dimensionsAt(CreateDimensionsAt da) {
500 					try (var insertReq = conn.update(INSERT_REQ_SIZE_BOARD)) {
501 						insertReq.call(jobId,
502 								locateBoard(conn, machine.name, da, true),
503 								da.width, da.height, da.maxDead,
504 								(int) scale.getSpecificBoard());
505 					}
506 					return max(1, da.getArea() - da.maxDead);
507 				}
508 
509 				/*
510 				 * Request by specific location; resolve to board ID now, as
511 				 * that doesn't depend on whether the board is currently in
512 				 * use.
513 				 */
514 				@Override
515 				public Integer board(CreateBoard b) {
516 					try (var insertReq = conn.update(INSERT_REQ_BOARD)) {
517 						/*
518 						 * This doesn't pass along the max dead boards; only
519 						 * after one!
520 						 */
521 						insertReq.call(jobId,
522 								locateBoard(conn, machine.name, b, false),
523 								(int) scale.getSpecificBoard());
524 					}
525 					return 1;
526 				}
527 			});
528 
529 			// DB now changed; can report success
530 			JobLifecycle.log.info(
531 					"created job {} on {} for {} asking for {} board(s)", jobId,
532 					machine.name, owner, numBoards);
533 
534 			allocator.scheduleAllocateNow();
535 			return getJob(jobId, conn).map(ji -> (Job) ji).orElseThrow(
536 					() -> new RuntimeException("Error creating job!"));
537 		});
538 	}
