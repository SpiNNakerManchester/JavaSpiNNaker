140 	// Could be configurable, but no real point
141 	private static final int LIMIT_LIMIT = 200;
142 
143 	/**
144 	 * Adds in the {@code Link:} header with general paging info.
145 	 *
146 	 * @param value The core response.
147 	 * @param ui    Information about URIs
148 	 * @param start The start offset.
149 	 * @param limit The size of chunk.
150 	 * @return Annotated response.
151 	 */
152 	private Response wrapPaging(ListJobsResponse value, UriInfo ui, int start,
153 			int limit) {
154 		var r = ok(value);
155 		var links = new HashMap<String, URI>();
156 		if (start > 0) {
157 			var prev = ui.getRequestUriBuilder()
158 					.replaceQueryParam("wait", false)
159 					.replaceQueryParam("start", max(start - limit, 0)).build();
160 			value.setPrev(prev);
161 			links.put("prev", prev);
162 		}
163 		if (value.jobs.size() == limit) {
164 			var next = ui.getRequestUriBuilder()
165 					.replaceQueryParam("wait", false)
166 					.replaceQueryParam("start", start + limit).build();
167 			value.setNext(next);
168 			links.put("next", next);
169 		}
170 		if (!links.isEmpty()) {
171 			r.header("Link", join(", ", links.entrySet().stream().map(
172 					e -> format("<%s>; rel=\"%s\"", e.getValue(), e.getKey()))
173 					.collect(toList())));
174 		}
175 		return r.build();
176 	}

178 	@Override
179 	public void listJobs(boolean wait, boolean destroyed, int limit, int start,
180 			UriInfo ui, AsyncResponse response) {
181 		if (limit > LIMIT_LIMIT || limit < 1) {
182 			throw new BadArgs("limit must not be bigger than " + LIMIT_LIMIT);
183 		}
184 		if (start < 0) {
185 			throw new BadArgs("start must not be less than 0");
186 		}
187 		var jc = core.getJobs(destroyed, limit, start);
188 		if (wait) {
189 			bgAction(response, () -> {
190 				log.debug("starting wait for change of job list");
191 				jc.waitForChange(properties.getWait());
192 				var newJc = core.getJobs(destroyed, limit, start);
193 				return wrapPaging(new ListJobsResponse(newJc, ui), ui, start,
194 						limit);
195 			});
196 		} else {
197 			fgAction(response, () -> wrapPaging(new ListJobsResponse(jc, ui),
198 					ui, start, limit));
199 		}
200 	}

202 	private static String trim(String str) {
203 		if (isNull(str)) {
204 			return null;
205 		}
206 		return str.strip();
207 	}

209 	@Override
210 	public void createJob(CreateJobRequest req, UriInfo ui,
211 			SecurityContext security, AsyncResponse response) {
212 		var crds = validateAndApplyDefaultsToJobRequest(req, security);
213 
214 		// Async because it involves getting a write lock
215 		bgAction(response, () -> ifElse(
216 				core.createJob(trim(req.owner), trim(req.group), crds,
217 						req.machineName, req.tags, req.keepaliveInterval,
218 						mapper.writeValueAsBytes(req)),
219 				job -> created(ui.getRequestUriBuilder().path("{id}")
220 						.build(job.getId()))
221 								.entity(new CreateJobResponse(job, ui)).build(),
222 				() -> status(BAD_REQUEST).type(TEXT_PLAIN)
223 						// Most likely reason for failure
224 						.entity("out of quota").build()));
225 	}

227 	private CreateDescriptor validateAndApplyDefaultsToJobRequest(
228 			CreateJobRequest req, SecurityContext security) throws BadArgs {
229 		if (isNull(req)) {
230 			throw new BadArgs("request must be supplied");
231 		}
232 
233 		if (!security.isUserInRole("ADMIN") || isNull(req.owner)
234 				|| req.owner.isBlank()) {
235 			req.owner = security.getUserPrincipal().getName();
236 		}
237 		if (isNull(req.owner) || req.owner.isBlank()) {
238 			throw new BadArgs(
239 					"request must be connected to an identified owner");
240 		}
241 		req.owner = req.owner.strip();
242 
243 		var ka = properties.getKeepalive();
244 		if (isNull(req.keepaliveInterval)
245 				|| req.keepaliveInterval.compareTo(ka.getMin()) < 0) {
246 			throw new BadArgs(
247 					"keepalive interval must be at least " + ka.getMin());
248 		}
249 		if (req.keepaliveInterval.compareTo(ka.getMax()) > 0) {
250 			throw new BadArgs(
251 					"keepalive interval must be no more than " + ka.getMax());
252 		}
253 
254 		if (isNull(req.tags)) {
255 			req.tags = new ArrayList<>();
256 			if (isNull(req.machineName)) {
257 				req.tags.add("default");
258 			}
259 		}
260 		if (nonNull(req.machineName) && !req.tags.isEmpty()) {
261 			throw new BadArgs(
262 					"must not specify machine name and tags together");
263 		}
264 
265 		if (isNull(req.maxDeadBoards)) {
266 			req.maxDeadBoards = 0;
267 		} else if (req.maxDeadBoards < 0) {
268 			throw new BadArgs(
269 					"the maximum number of dead boards must not be negative");
270 		}
271 
272 		if (nonNull(req.numBoards)) {
273 			return new CreateNumBoards(req.numBoards, req.maxDeadBoards);
274 		} else if (nonNull(req.dimensions)) {
275 			if (nonNull(req.board)) {
276 				// Both dimensions AND board; rooted rectangle
277 				if (nonNull(req.board.x)) {
278 					return new CreateDimensionsAt(req.dimensions.width,
279 							req.dimensions.height, req.board.x, req.board.y,
280 							req.board.z);
281 				} else if (nonNull(req.board.cabinet)) {
282 					return CreateDimensionsAt.physical(req.dimensions.width,
283 							req.dimensions.height, req.board.cabinet,
284 							req.board.frame, req.board.board,
285 							req.maxDeadBoards);
286 				} else {
287 					return new CreateDimensionsAt(req.dimensions.width,
288 							req.dimensions.height, req.board.address,
289 							req.maxDeadBoards);
290 				}
291 			}
292 			return new CreateDimensions(req.dimensions.width,
293 					req.dimensions.height, req.maxDeadBoards);
294 		} else if (nonNull(req.board)) {
295 			if (nonNull(req.board.x)) {
296 				return triad(req.board.x, req.board.y, req.board.z);
297 			} else if (nonNull(req.board.cabinet)) {
298 				return physical(req.board.cabinet, req.board.frame,
299 						req.board.board);
300 			} else {
301 				return address(req.board.address);
302 			}
303 		} else {
304 			// It's a single board
305 			return new CreateNumBoards(1, 0);
306 		}
307 	}
