70  /**
71   * Manufactures instances of {@link MachineAPI} and {@link JobAPI}. This
72   * indirection allows Spring to insert any necessary security interceptors on
73   * methods.
74   * <p>
75   * Do not call the {@link #machine(Machine,UriInfo) machine()} and
76   * {@link #job(Job,String,Permit,UriInfo) job()} methods directly. Use the
77   * factories.
78   */
79  @Configuration
80  @Role(ROLE_SUPPORT)
81  class SpallocServiceAPIImplBuilder extends BackgroundSupport {
82  	private static final Logger log =
83  			getLogger(SpallocServiceAPIImplBuilder.class);
84  
85  	@Autowired
86  	private SpallocAPI core;
87  
88  	@Autowired
89  	private JsonMapper mapper;
90  
91  	@Autowired
92  	private SpallocProperties props;
93  
94  	@Value("${spring.mvc.servlet.path}")
95  	private String servletPath;
96  
97  	/**
98  	 * Make a machine access interface.
99  	 *
100 	 * @param machine
101 	 *            The machine object to wrap
102 	 * @param ui
103 	 *            How the API was accessed
104 	 * @return A machine access interface. Will be wrapped by Spring with
105 	 *         security support (if needed).
106 	 */
107 	@Bean
108 	@Prototype
109 	@Role(ROLE_APPLICATION)
110 	public MachineAPI machine(Machine machine, UriInfo ui) {
111 		return new MachineAPI() {
112 			@Override
113 			public void describeMachine(boolean wait, AsyncResponse response) {
114 				if (wait) {
115 					bgAction(response, () -> {
116 						log.debug("starting wait for change of machine");
117 						machine.waitForChange(props.getWait());
118 						/*
119 						 * Assume that machines don't change often enough for us
120 						 * to care about whether they vanish; therefore handle
121 						 * still valid after wait
122 						 */
123 						return new MachineResponse(machine, ui);
124 					});
125 				} else {
126 					fgAction(response, () -> new MachineResponse(machine, ui));
127 				}
128 			}
129 
130 			private WhereIsResponse
131 					makeResponse(Optional<BoardLocation> boardLocation) {
132 				return new WhereIsResponse(
133 						boardLocation.orElseThrow(
134 								() -> new NotFound("failed to locate board")),
135 						ui);
136 			}
137 
138 			@Override
139 			public WhereIsResponse whereIsLogicalPosition(int x, int y, int z) {
140 				return makeResponse(machine
141 						.getBoardByLogicalCoords(new TriadCoords(x, y, z)));
142 			}
143 
144 			@Override
145 			public WhereIsResponse whereIsPhysicalPosition(int cabinet,
146 					int frame, int board) {
147 				return makeResponse(machine.getBoardByPhysicalCoords(
148 						new PhysicalCoords(cabinet, frame, board)));
149 			}
150 
151 			@Override
152 			public WhereIsResponse whereIsMachineChipLocation(int x, int y) {
153 				return makeResponse(
154 						machine.getBoardByChip(new ChipLocation(x, y)));
155 			}
156 
157 			@Override
158 			public WhereIsResponse whereIsIPAddress(String address) {
159 				return makeResponse(machine.getBoardByIPAddress(address));
160 			}
161 		};
162 	}

164 	/**
165 	 * Make a job access interface.
166 	 *
167 	 * @param j
168 	 *            The job object to wrap
169 	 * @param caller
170 	 *            What host made the call? Used in keepalive tracking.
171 	 * @param permit
172 	 *            The security permissions pertinent to the job
173 	 * @param ui
174 	 *            How the API was accessed
175 	 * @return A machine access interface. Will be wrapped by Spring with
176 	 *         security support (if needed).
177 	 */
178 	@Bean
179 	@Prototype
180 	@Role(ROLE_APPLICATION)
181 	public JobAPI job(Job j, String caller, Permit permit, UriInfo ui) {
182 		var sp = props.getProxy().isEnable() ? servletPath : null;
183 		return new JobAPIImpl(j, caller, permit, ui, sp);
184 	}
185 
186 	/**
187 	 * Implementation of the JobAPI.
188 	 */
189 	private final class JobAPIImpl implements JobAPI {
190 
191 		private final String caller;
192 
193 		private final Job j;
194 
195 		private final Permit permit;
196 
197 		private final UriInfo ui;
198 
199 		private final String sp;
200 
201 		private JobAPIImpl(Job j, String caller, Permit permit, UriInfo ui,
202 				String sp) {
203 			this.j = j;
204 			this.caller = caller;
205 			this.permit = permit;
206 			this.ui = ui;
207 			this.sp = sp;
208 		}
209 
210 		@Override
211 		public String keepAlive(String reqBody) {
212 			log.debug("keeping job {} alive: {}", j.getId(), caller);
213 			j.access(caller);
214 			return "ok";
215 		}
216 
217 		@Override
218 		public void getState(boolean wait, AsyncResponse response) {
219 			if (wait) {
220 				bgAction(response, permit, () -> {
221 					log.debug("starting wait for change of job");
222 					j.waitForChange(props.getWait());
223 					// Refresh the handle
224 					var nj = core.getJob(permit, j.getId())
225 							.orElseThrow(() -> new ItsGone("no such job"));
226 					log.debug("job state now {}", nj.getState());
227 					return new JobStateResponse(nj, ui, mapper, sp);
228 				});
229 			} else {
230 				fgAction(response, () -> new JobStateResponse(j, ui, mapper,
231 						sp));
232 			}
233 		}
234 
235 		@Override
236 		public Response deleteJob(String reason) {
237 			if (isNull(reason)) {
238 				reason = "unspecified";
239 			}
240 			j.destroy(reason);
241 			return noContent().build();
242 		}
243 
244 		private SubMachine allocation() {
245 			return j.getMachine().orElseThrow(
246 					// No machine allocated yet
247 					EmptyResponse::new);
248 		}
249 
250 		@Override
251 		public SubMachineResponse getMachine() {
252 			j.access(caller);
253 			return new SubMachineResponse(allocation(), ui);
254 		}
255 
256 		@Override
257 		public MachinePower getMachinePower() {
258 			j.access(caller);
259 			return new MachinePower(allocation().getPower());
260 		}
261 
262 		@Override
263 		public void setMachinePower(MachinePower reqBody,
264 				AsyncResponse response) {
265 			// Async because it involves getting a write lock
266 			if (isNull(reqBody)) {
267 				throw new BadArgs("bad power description");
268 			}
269 			bgAction(response, permit, () -> {
270 				j.access(caller);
271 				allocation().setPower(reqBody.getPower());
272 				return accepted().build();
273 			});
274 		}
275 
276 		@Override
277 		public WhereIsResponse getJobChipLocation(int x, int y) {
278 			j.access(caller);
279 			var loc = j.whereIs(x, y).orElseThrow(EmptyResponse::new);
280 			return new WhereIsResponse(loc, ui);
281 		}
282 
283 		@Override
284 		public void reportBoardIssue(IssueReportRequest reqBody,
285 				AsyncResponse response) {
286 			// Async because it involves getting a write lock
287 			if (isNull(reqBody)) {
288 				throw new BadArgs("bad issue description");
289 			}
290 			bgAction(response, () -> new IssueReportResponse(
291 					j.reportIssue(reqBody, permit)));
292 		}
293 
294 		@Override
295 		public void writeDataToJob(int x, int y, long address,
296 				byte[] bytes,
297 				AsyncResponse response) {
298 			bgAction(response, () -> {
299 				var txrx = j.getTransceiver();
300 				txrx.writeMemory(new ChipLocation(x, y),
301 						new MemoryLocation(address), bytes);
302 				return accepted().build();
303 			});
304 		}
305 
306 		@Override
307 		public void readDataFromJob(int x, int y, long address, int size,
308 				AsyncResponse response) {
309 			bgAction(response, () -> {
310 				var txrx = j.getTransceiver();
311 				var arr = new byte[size];
312 				var buffer = txrx.readMemory(new ChipLocation(x, y),
313 						new MemoryLocation(address), size);
314 				buffer.get(arr);
315 				return ok(arr).build();
316 			});
317 
318 		}
319 
320 		@Override
321 		@SuppressWarnings("MustBeClosed")
322 		public void fastDataWrite(@ValidX int gatherX, @ValidY int gatherY,
323 				@Positive int gatherP, @ValidX int ethX,
324 				@ValidY int ethY, @IPAddress String ethAddress, int iptag,
325 				@ValidX int x, @ValidY int y, long address, byte[] bytes,
326 				AsyncResponse response) {
327 			bgAction(response, () -> {
328 				var fdi = j.getFastDataIn(
329 						new CoreLocation(gatherX, gatherY, gatherP),
330 						new IPTag(ethAddress, iptag, ethX, ethY, "localhost",
331 								null, true, "DATA_SPEED_UP"));
332 				fdi.fastWrite(new ChipLocation(x, y),
333 						new MemoryLocation(address), wrap(bytes));
334 				return accepted().build();
335 			});
336 		}
337 
338 		@Override
339 		@SuppressWarnings("MustBeClosed")
340 		public void fastDataRead(@ValidX int gatherX, @ValidY int gatherY,
341 				@ValidX int ethX, @ValidY int ethY,
342 				@IPAddress String ethAddress, int iptag,
343 				@ValidX int x, @ValidY int y, @ValidP int p,
344 				long address, int size,	AsyncResponse response) {
345 			bgAction(response, () -> {
346 				var downloader = j.getDownloader(
347 						new IPTag(ethAddress, iptag, ethX, ethY, "localhost",
348 								null, true, null));
349 				var buffer = downloader.doDownload(new CoreLocation(x, y, p),
350 						new MemoryLocation(address), size);
351 				var arr = new byte[size];
352 				buffer.get(arr);
353 				return ok(arr).build();
354 			});
355 		}
356 
357 		@Override
358 		public void prepareRoutingTables(UriInfo uriInfo,
359 				AsyncResponse response) {
360 			var queryParams = uriInfo.getQueryParameters();
361 			var filters = new HashMap<Integer, DiagnosticFilter>();
362 			for (String param : queryParams.keySet()) {
363 				int index = Integer.parseInt(param);
364 				var values = queryParams.get(param);
365 				if (values.size() != 1) {
366 					throw new RuntimeException(
367 							"Multiple values for index " + param);
368 				}
369 				int value = Integer.parseInt(values.get(0));
370 				filters.put(index, new DiagnosticFilter(value));
371 			}
372 			bgAction(response, () -> {
373 				var txrx = j.getTransceiver();
374 				txrx.resetRouting(filters);
375 				return accepted().build();
376 			});
377 		}
378 	}
