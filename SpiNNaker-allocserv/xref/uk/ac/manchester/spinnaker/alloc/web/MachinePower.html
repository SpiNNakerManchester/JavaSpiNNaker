16  package uk.ac.manchester.spinnaker.alloc.web;
17  
18  import javax.validation.constraints.NotNull;
19  
20  import com.fasterxml.jackson.annotation.JsonProperty;
21  import com.google.errorprone.annotations.Immutable;
22  
23  import uk.ac.manchester.spinnaker.alloc.model.PowerState;
24  
25  /**
26   * Describes the current state of power of the machine (or at least the portion
27   * of it allocated to a job), or a state that the user wants us to switch into.
28   *
29   * @author Donal Fellows
30   */
31  @Immutable
32  public class MachinePower {
33  	@NotNull(message = "power must be specified")
34  	private final PowerState power;
35  
36  	/**
37  	 * Make an instance.
38  	 *
39  	 * @param power
40  	 *            the machine power state
41  	 */
42  	public MachinePower(
43  			@JsonProperty(value = "power", defaultValue = "OFF")
44  			PowerState power) {
45  		this.power = power;
46  	}
47  
48  	/** @return the machine power state */
49  	public PowerState getPower() {
50  		return power;
51  	}
52  }
