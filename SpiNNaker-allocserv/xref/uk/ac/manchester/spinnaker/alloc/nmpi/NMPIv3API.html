16  package uk.ac.manchester.spinnaker.alloc.nmpi;
17  
18  import static com.fasterxml.jackson.databind.PropertyNamingStrategies.SNAKE_CASE;
19  
20  import java.util.List;
21  
22  import javax.ws.rs.Consumes;
23  import javax.ws.rs.GET;
24  import javax.ws.rs.HeaderParam;
25  import javax.ws.rs.POST;
26  import javax.ws.rs.PUT;
27  import javax.ws.rs.Path;
28  import javax.ws.rs.PathParam;
29  import javax.ws.rs.Produces;
30  import javax.ws.rs.QueryParam;
31  
32  import org.apache.cxf.jaxrs.client.JAXRSClientFactory;
33  
34  import com.fasterxml.jackson.databind.ObjectMapper;
35  import com.fasterxml.jackson.jaxrs.json.JacksonJsonProvider;
36  
37  /**
38   * The REST API for the NMPI Service.
39   */
40  @Path("/")
41  public interface NMPIv3API {
42  	/**
43  	 * Get an NMPI Job.
44  	 *
45  	 * @param apiKey
46  	 *            The API key to authenticate with.
47  	 * @param jobId
48  	 *            The job id.
49  	 * @return The job.
50  	 */
51  	@GET
52  	@Path("jobs/{job_id}")
53  	@Produces("application/json")
54  	Job getJob(@HeaderParam("x-api-key") String apiKey,
55  			@PathParam("job_id") int jobId);
56  
57  	/**
58  	 * Update the resources of a Job.
59  	 *
60  	 * @param apiKey
61  	 *            The API key to authenticate with.
62  	 * @param jobId
63  	 *            The job id.
64  	 * @param resources
65  	 *            The resources to set.
66  	 */
67  	@PUT
68  	@Path("jobs/{job_id}")
69  	@Consumes("application/json")
70  	void setJobResources(@HeaderParam("x-api-key") String apiKey,
71  			@PathParam("job_id") int jobId, JobResourceUpdate resources);
72  
73  	/**
74  	 * Get a list of projects that match the parameters.
75  	 *
76  	 * @param apiKey
77  	 *            The API key to authenticate with.
78  	 * @param status
79  	 *            The status of the project to get.
80  	 * @param collab
81  	 *            The name of the collab of the project.
82  	 * @return A list of matching projects.
83  	 */
84  	@GET
85  	@Path("projects/")
86  	@Produces("application/json")
87  	List<Project> getProjects(@HeaderParam("x-api-key") String apiKey,
88  			@QueryParam("status") String status,
89  			@QueryParam("collab") String collab);
90  
91  	/**
92  	 * Create a new session.
93  	 *
94  	 * @param apiKey
95  	 *            The API key to authenticate with.
96  	 * @param session
97  	 *            The session details.
98  	 * @return The created session details.
99  	 */
100 	@POST
101 	@Path("sessions/")
102 	@Consumes("application/json")
103 	@Produces("application/json")
104 	SessionResponse createSession(@HeaderParam("x-api-key") String apiKey,
105 			SessionRequest session);
106 
107 	/**
108 	 * Update a session.
109 	 *
110 	 * @param apiKey
111 	 *            The API key to authenticate with.
112 	 * @param sessionId
113 	 *            The identifier of the session.
114 	 * @param resources
115 	 *            The resources to set.
116 	 */
117 	@PUT
118 	@Path("sessions/{session_id}")
119 	@Consumes("application/json")
120 	void setSessionStatusAndResources(@HeaderParam("x-api-key") String apiKey,
121 			@PathParam("session_id") int sessionId,
122 			SessionResourceUpdate resources);
123 
124 	/**
125 	 * Get a client for the API.
126 	 *
127 	 * @param url
128 	 *            The URL to connect to.
129 	 * @return A proxy of the API.
130 	 */
131 	static NMPIv3API createClient(String url) {
132 		var mapper = new ObjectMapper();
133 		mapper.setPropertyNamingStrategy(SNAKE_CASE);
134 		return JAXRSClientFactory.create(url, NMPIv3API.class,
135 				List.of(new JacksonJsonProvider()));
136 	}
137 }
