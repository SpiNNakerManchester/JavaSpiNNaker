124 	private Optional<UserRecord> makeInitUser(String username) {
125 		if (!authProps.isAddDummyUser()) {
126 			return Optional.empty();
127 		}
128 		var pass = DUMMY_PASSWORD;
129 		boolean poorPassword = true;
130 		if (authProps.isDummyRandomPass()) {
131 			pass = passServices.generatePassword();
132 			poorPassword = false;
133 		}
134 		if (!createUser(username, pass, ADMIN)) {
135 			// User creation failed; probably already exists, which is OK
136 			return Optional.empty();
137 		}
138 		if (authProps.isDummyRandomPass()) {
139 			log.info("admin user {} has password: {}", username, pass);
140 		}
141 		if (poorPassword) {
142 			log.warn("user {} has default password!", username);
143 		}
144 		return Optional.of(userController.getUser(username, null)
145 				.orElseThrow(() -> new SetupException(
146 						"default user was created, yet wasn't found!")));
147 	}

149 	private Optional<GroupRecord> makeInitGroup(String groupname) {
150 		if (groupname.isBlank()) {
151 			// No system group name, so ignore group setup
152 			return Optional.empty();
153 		}
154 		var template = new GroupRecord();
155 		template.setGroupName(groupname);
156 		// What should the default quota be here? Using no-quota-at-all for now
157 		template.setQuota(null);
158 		try {
159 			return userController.createGroup(template, INTERNAL).map(g -> {
160 				log.info("system group '{}' created", groupname);
161 				return g;
162 			});
163 		} catch (DataIntegrityViolationException e) {
164 			if (e.getMessage().contains("A UNIQUE constraint failed")) {
165 				// Already exists; no big deal
166 				return Optional.empty();
167 			} else if (e.getMessage().contains("Duplicate entry")) {
168 				return Optional.empty();
169 			}
170 			throw e;
171 		}
172 	}

200 	@PreAuthorize(IS_ADMIN)
201 	public boolean createUser(String username, String password,
202 			TrustLevel trustLevel) {
203 		var name = username.strip();
204 		if (name.isBlank()) {
205 			// Won't touch the DB if the username is empty
206 			throw new UsernameNotFoundException("empty user name?");
207 		}
208 		var encPass = passServices.encodePassword(password);
209 		try (var sql = new AuthQueries()) {
210 			return sql.transaction(
211 					() -> sql.createUser(username, encPass, trustLevel, null)
212 							.isPresent());
213 		}

216 	/** Marks an authentication that we've already taken a decision about. */
217 	private interface AlreadyDoneMarker extends Authentication {
218 	}
219 
220 	@Override
221 	public Authentication authenticate(Authentication auth)
222 			throws AuthenticationException {
223 		if (isNull(auth)) {
224 			return null;
225 		}
226 		log.debug("requesting auth decision about {}", auth);
227 		if (auth instanceof AlreadyDoneMarker) {
228 			// It's ours; don't repeat the auth
229 			return auth;
230 		}
231 
232 		try {
233 			if (auth instanceof UsernamePasswordAuthenticationToken) {
234 				return authenticateDirect(
235 						(UsernamePasswordAuthenticationToken) auth);
236 			} else if (auth instanceof OAuth2AuthenticationToken) {
237 				/*
238 				 * Technically, at this point we're already authenticated as
239 				 * we've checked that the token from Keycloak is valid. We still
240 				 * have to take an authorization decision though.
241 				 */
242 				var user = ((OAuth2AuthenticationToken) auth).getPrincipal();
243 				return authorizeOpenId(
244 						authProps.getOpenid().getUsernamePrefix()
245 								+ user.getAttribute(PREFERRED_USERNAME),
246 						user.getAttribute(SUB), new OriginatingCredential(user),
247 						auth.getAuthorities());
248 			} else if (auth instanceof BearerTokenAuthentication) {
249 				/*
250 				 * Technically, at this point we're already authenticated as
251 				 * we've checked that the token from Keycloak is valid. We still
252 				 * have to take an authorization decision though.
253 				 */
254 				var bearerAuth = (BearerTokenAuthentication) auth;
255 				var token = bearerAuth.getToken();
256 				return authorizeOpenId(
257 						authProps.getOpenid().getUsernamePrefix()
258 								+ bearerAuth.getTokenAttributes().get(
259 										PREFERRED_USERNAME),
260 						bearerAuth.getName(), new OriginatingCredential(token),
261 						auth.getAuthorities());
262 			} else {
263 				return null;
264 			}
265 		} catch (DataAccessException e) {
266 			throw new InternalAuthenticationServiceException(
267 					"database access problem", e);
268 		}
269 	}
270 
271 	@Override
272 	public Authentication updateAuthentication(HttpServletRequest req,
273 			SecurityContext ctx) {
274 		var current = ctx.getAuthentication();
275 		if (nonNull(current)) {
276 			if (supports(current.getClass())) {
277 				var updated = authenticate(current);
278 				if (nonNull(updated) && updated != current) {
279 					log.debug("filter updated security from {} to {}", current,
280 							updated);
281 					ctx.setAuthentication(updated);
282 					return updated;
283 				}
284 			} else if (!isUnsupportedAuthTokenClass(current.getClass())) {
285 				log.warn("unexpected authentication type {} (token: {})"
286 						+ "on request {}",
287 						current.getClass(), current, req.getRequestURI());
288 			}
289 		}
290 		return null;
291 	}

300 	/** The classes that we <em>know</em> we don't ever want to handle. */
301 	private static final Class<?>[] UNSUPPORTED_AUTH_TOKEN_CLASSES = {
302 		AnonymousAuthenticationToken.class, RememberMeAuthenticationToken.class,
303 		RunAsUserToken.class, TestingAuthenticationToken.class
304 	};
305 
306 	@Override
307 	public boolean supports(Class<?> cls) {
308 		for (var c : SUPPORTED_AUTH_TOKEN_CLASSES) {
309 			if (c.isAssignableFrom(cls)) {
310 				return true;
311 			}
312 		}
313 		if (!isUnsupportedAuthTokenClass(cls)) {
314 			log.warn("asked about supporting {}", cls);
315 		}
316 		return false;
317 	}

319 	/**
320 	 * Check if we're talking about a class that we know we don't ever want to
321 	 * handle.
322 	 *
323 	 * @param cls
324 	 *            The class to check.
325 	 * @return {@code true} if we never want to take a full auth decision about
326 	 *         it.
327 	 */
328 	static boolean isUnsupportedAuthTokenClass(Class<?> cls) {
329 		for (var c : UNSUPPORTED_AUTH_TOKEN_CLASSES) {
330 			if (c.isAssignableFrom(cls)) {
331 				return true;
332 			}
333 		}
334 		return false;
335 	}

349 	 * Do authorization mapping for users coming in with a direct username and
350 	 * password.
351 	 *
352 	 * @param auth
353 	 *            The credentials they presented
354 	 * @return The remapped auth token, or {@code null} if we are throwing
355 	 *         things out at this stage.
356 	 * @throws AuthenticationException
357 	 *             If something is badly wrong.
358 	 */
359 	private Authentication authenticateDirect(
360 			UsernamePasswordAuthenticationToken auth)
361 			throws AuthenticationException {
362 		log.info("authenticating Local Login {}", auth.toString());
363 		// We ALWAYS trim the username; extraneous whitespace is bogus
364 		var name = auth.getName().strip();
365 		if (name.isBlank()) {
366 			// Won't touch the DB if the username is empty
367 			throw new UsernameNotFoundException("empty user name?");
368 		}
369 		var password = auth.getCredentials().toString();
370 		var authorities = new ArrayList<GrantedAuthority>();
371 
372 		try (var queries = new AuthQueries()) {
373 			if (!authLocalAgainstDB(name, password, authorities, queries)) {
374 				return null;
375 			}
376 		}
377 		return new PerformedUsernamePasswordAuthenticationToken(name, password,
378 				authorities);
379 	}

381 	/**
382 	 * Do authorization mapping for users coming in via OpenID Connect from
383 	 * HBP/EBRAINS.
384 	 *
385 	 * @param name
386 	 *            The name of the user <em>with standard namespacing prefix
387 	 *            added</em>.
388 	 * @param subject
389 	 *            The OpenID subject.
390 	 * @param credential
391 	 *            The reason we believe the user should be allowed to use the
392 	 *            service.
393 	 * @param authorities
394 	 *            The claimed authorities derived from the HBP OpenID service.
395 	 *            Interesting because they include claims about organisation and
396 	 *            collabratory membership.
397 	 * @return The remapped auth token, or {@code null} if we are throwing
398 	 *         things out at this stage.
399 	 * @throws AuthenticationException
400 	 *             If something is badly wrong.
401 	 */
402 	private Authentication authorizeOpenId(String name, String subject,
403 			OriginatingCredential credential,
404 			Collection<? extends GrantedAuthority> authorities)
405 			throws AuthenticationException {
406 		log.debug("authenticating OpenID {}", credential);
407 		if (isNull(name)
408 				|| name.equals(authProps.getOpenid().getUsernamePrefix())) {
409 			// No actual name there?
410 			throw new UsernameNotFoundException("empty user name?");
411 		}
412 		try (var queries = new AuthQueries()) {
413 			if (!queries.transaction(() -> authOpenIDAgainstDB(name, subject,
414 					authorities, queries))) {
415 				return null;
416 			}
417 		} catch (RuntimeException e) {
418 			log.warn("serious problem when processing login for OpenID user {}",
419 					name, e);
420 			return null;
421 		}
422 		// Users from OpenID always have the same permissions
423 		return new OpenIDDerivedAuthenticationToken(name, credential);
424 	}

426 	/** Holds either a {@link OAuth2User} or a {@link Jwt}. */
427 	private static final class OriginatingCredential {
428 		private final OAuth2User user;
429 
430 		private final OAuth2AccessToken token;
431 
432 		OriginatingCredential(OAuth2User user) {
433 			this.user = requireNonNull(user);
434 			this.token = null;
435 		}
436 
437 		OriginatingCredential(OAuth2AccessToken token) {
438 			this.user = null;
439 			this.token = requireNonNull(token);
440 		}
441 
442 		@Override
443 		public String toString() {
444 			if (nonNull(user)) {
445 				return user.toString();
446 			} else {
447 				return token.toString();
448 			}
449 		}
450 	}

620 	/**
621 	 * Collection of queries for setting the group memberships of a user. These
622 	 * are queries that would be part of {@link AuthQueries} except for the
623 	 * dance of handling temporary database tables.
624 	 * <p>
625 	 * Use this class by calling {@link #define(String,GroupType)} several
626 	 * times, and then calling {@link #apply(int)} once. Closing this resource
627 	 * removes temporary storage; it does not support nested use.
628 	 *
629 	 * @author Donal Fellows
630 	 */
631 	private final class GroupSynch extends AbstractSQL {
632 		private final Update insert;
633 
634 		private final Update add;
635 
636 		private final Update remove;
637 
638 		GroupSynch(AuthQueries sql) {
639 			super(sql.getConnection());
640 			try (var make = conn.update(GROUP_SYNC_MAKE_TEMP_TABLE)) {
641 				make.call();
642 			} catch (BadSqlGrammarException e) {
643 				if (!e.getMessage().contains("already exists")) {
644 					throw e;
645 				}
646 				log.warn("Group temp table already exists ("
647 						+ e.getMessage() + ")");
648 			}
649 			insert = conn.update(GROUP_SYNC_INSERT_TEMP_ROW);
650 			add = conn.update(GROUP_SYNC_ADD_GROUPS);
651 			remove = conn.update(GROUP_SYNC_REMOVE_GROUPS);
652 		}
653 
654 		@Override
655 		public void close() {
656 			remove.close();
657 			add.close();
658 			insert.close();
659 			/*
660 			 * Seems we can only drop a temporary table if there are no (other)
661 			 * open statements that refer to it *AND* there is no transaction
662 			 * open that used it. Or is that no connection? Either way,
663 			 * SQLITE_LOCKED is not a desired failure state!
664 			 *
665 			 * https://sqlite.org/forum/forumpost/433d2fdb07fc8f13 says some of
666 			 * the constraints, but not all; the need for the transaction to be
667 			 * closed comes from elsewhere in that thread.
668 			 *
669 			 * Fortunately, we can just delete the contents of the temporary
670 			 * table instead, and that's just as good *and can be done in the
671 			 * transaction*.
672 			 */
673 			try (var drop = conn.update(GROUP_SYNC_DROP_TEMP_TABLE)) {
674 				drop.call();
675 			}
676 			super.close();
677 		}

680 		 * List one of the groups that we want a user to be a member of. We're
681 		 * building a set of these.

684 		 *            The name of the group
685 		 * @param type

688 		void define(String name, GroupType type) {

706 		}

742 			Pattern.compile("^collab-(.*)-(administrator|editor|viewer)$");

744 	/**
745 	 * Convert a list of claimed collabs into authorities.
746 	 *
747 	 * @param claim
748 	 *            The (sub-)claim of collabs.
749 	 * @param results
750 	 *            Where to add the generated authorities.
751 	 * @return Whether we processed a claim at all. (The claim could be empty;
752 	 *         that's OK.)
753 	 */
754 	private boolean collabToAuthority(List<String> claim,
755 			Collection<GrantedAuthority> results) {
756 		if (isNull(claim)) {
757 			return false;
758 		}
759 		var seen = new HashSet<>();
760 		for (var collab : claim) {
761 			var reduced = COLLAB_MATCHER.matcher(collab).replaceFirst("$1");
762 			if (!seen.contains(reduced)) {
763 				results.add(new CollabratoryAuthority(reduced));
764 				seen.add(reduced);
765 			}
766 		}
767 		return true;
768 	}

772 	 *

778 	 *         that's OK.)

796 	 * Extract the {@code team/roles} sub-claim. This is <em>awful</em> because
797 	 * the promised types are all inferred and the <em>actual</em> types
798 	 * guaranteed by the way claims are encoded are more that we are taking an
799 	 * {@link Object} because {@link ClaimAccessor#getClaim(String)} is merely
800 	 * pinky-swearing that objects are of the right type, not enforcing it.
801 	 *

803 	 *            Overall claim.

808 			getTeamsFromClaim(Map<String, List<String>> rolesClaim) {
809 		// Messy; all the implicit types and hidden casts!
810 		try {
811 			var teamsClaim = rolesClaim.get("team");
812 			if (!teamsClaim.isEmpty()) {
813 				/*
814 				 * Check invokes a String instance method on the first item of
815 				 * the list; if this works, we're valid. It's pitched into
816 				 * log.trace so things don't complain, but we never expect to
817 				 * enable this.
818 				 */
819 				log.trace("team claim first item length",
820 						teamsClaim.get(0).length());
821 			}
822 			return teamsClaim;
823 		} catch (ClassCastException | NullPointerException
824 				| IndexOutOfBoundsException e) {
825 			log.debug("failed to convert claim", e);
826 		}
827 		return List.of();
828 	}

876 	 * Check if a user can log in, and determine what permissions they have.
877 	 *
878 	 * @param username
879 	 *            The username.
880 	 * @param password
881 	 *            The <em>unencrypted</em> password.
882 	 * @param authorities
883 	 *            Filled out with permissions.
884 	 * @param queries
885 	 *            How to access the database.
886 	 * @return Whether the user is known. If {@code false}, the database cannot
887 	 *         authenticate the user and some other auth method should be tried.
888 	 * @throws UsernameNotFoundException
889 	 *             If no account exists.
890 	 * @throws DisabledException
891 	 *             If the account is administratively disabled.
892 	 * @throws LockedException
893 	 *             If the account is temporarily locked.
894 	 * @throws BadCredentialsException
895 	 *             If the password is invalid.
896 	 */
897 	private boolean authLocalAgainstDB(String username, String password,
898 			List<GrantedAuthority> authorities, AuthQueries queries) {
899 		return ifElse(
900 				queries.transaction(() -> lookUpUserDetails(username, queries)),
901 				details -> {
902 					checkPassword(username, password, details, queries);
903 					// Succeeded; finalize into external form
904 					return queries.transaction(() -> {
905 						queries.noteLoginSuccessForUser(details.userId);
906 						// Convert tiered trust level to grant form
907 						details.trustLevel.getGrants()
908 								.forEach(authorities::add);
909 						return true;
910 					});
911 				}, () -> false);
912 	}

929 	/**
930 	 * Look up a local user. Does all checks <em>except</em> the password check;
931 	 * that's slow (because bcrypt) so we do it outside the transaction. This
932 	 * needs to be done in a writable (exclusive) transaction.
933 	 *
934 	 * @param username
935 	 *            The user name we're looking up.
936 	 * @param queries
937 	 *            How to access the database.
938 	 * @return Whether we know the user, and if so, what do we know about them.
939 	 * @throws AuthenticationException
940 	 *             If the user is known and definitely can't log in for some
941 	 *             reason.
942 	 */
943 	private static Optional<LocalAuthResult> lookUpUserDetails(String username,
944 			AuthQueries queries) {
945 		return queries.getUserBlocked.call1(row -> {
946 			int userId = row.getInt("user_id");
947 			if (row.getBoolean("disabled")) {
948 				log.info("login failure for {}: account is disabled", username);
949 				throw new DisabledException("account is disabled");
950 			}
951 			try {
952 				if (row.getBoolean("locked")) {
953 					// Note that this extends the lock!
954 					throw new LockedException("account is locked");
955 				}
956 				var authInfo = queries.userAuthorities.call1(AuthInfo::new,
957 						userId).get();
958 				var encPass = authInfo.encryptedPassword;
959 				if (isNull(encPass)) {
960 					/*
961 					 * We know this user, but they can't use this authentication
962 					 * method. They'll probably have to use OpenID.
963 					 */
964 					return null;
965 				}
966 				var trust = authInfo.trustLevel;
967 				log.info("User trust level is " + trust);
968 				return new LocalAuthResult(userId, trust, encPass);
969 			} catch (AuthenticationException e) {
970 				queries.noteLoginFailureForUser(userId, username);
971 				log.info("login failure for {}", username, e);
972 				throw e;
973 			}
974 		}, username);
975 	}

977 	/**
978 	 * This is slow (100ms!) so we make sure we aren't holding a transaction
979 	 * open while this step is ongoing.
980 	 *
981 	 * @param username
982 	 *            The username (now validated as existing).
983 	 * @param password
984 	 *            The user-provided password that we're checking.
985 	 * @param queries
986 	 *            How to access the DB.
987 	 * @param details
988 	 *            The results of looking up the user
989 	 */
990 	private void checkPassword(String username, String password,
991 			LocalAuthResult details, AuthQueries queries) {
992 		if (!passServices.matchPassword(password, details.passInfo)) {
993 			queries.transaction(() -> {
994 				queries.noteLoginFailureForUser(details.userId, username);
995 				log.info("login failure for {}: bad password", username);
996 				throw new BadCredentialsException("bad password");
997 			});
998 		}
999 	}

1001 	/**
1002 	 * Check if an OpenID user can use the service. A transaction <em>must</em>
1003 	 * be held open when calling this.
1004 	 *
1005 	 * @param username
1006 	 *            The username, already obtained and verified.
1007 	 * @param subject
1008 	 *            The real OpenID {@code sub} claim, if available/known.
1009 	 * @param authorities
1010 	 *            The claimed authorities derived from the HBP OpenID service.
1011 	 *            Interesting because they include claims about organisation and
1012 	 *            collabratory membership.
1013 	 * @param queries
1014 	 *            How to access the database.
1015 	 * @return Whether the user is allowed to use the service. If {@code false},
1016 	 *         the database cannot authorize the user and some other auth method
1017 	 *         should be tried.
1018 	 * @throws DisabledException
1019 	 *             If the account is administratively disabled.
1020 	 * @throws LockedException
1021 	 *             If the account is temporarily locked.
1022 	 */
1023 	private boolean authOpenIDAgainstDB(String username, String subject,
1024 			Collection<? extends GrantedAuthority> authorities,
1025 			AuthQueries queries) {
1026 		var collabs = new ArrayList<String>();
1027 		var orgs = new ArrayList<String>();
1028 		authorities.forEach(ga -> inflateGroup(ga, collabs, orgs, queries));
1029 		boolean ok = queries.getUserBlocked.call1(userInfo -> {
1030 			log.info("Found user " + username + " in database");
1031 			int userId = userInfo.getInt("user_id");
1032 			synchExternalGroups(username, userId, orgs, collabs, queries);
1033 			if (userInfo.getBoolean("disabled")) {
1034 				log.info("user {} has a disabled account", username);
1035 				throw new DisabledException("account is disabled");
1036 			}
1037 			try {
1038 				if (userInfo.getBoolean("locked")) {
1039 					// Note that this extends the lock!
1040 					throw new LockedException("account is locked");
1041 				}
1042 				var authInfo = queries.userAuthorities.call1(AuthInfo::new,
1043 						userId).get();
1044 				if (nonNull(authInfo.encryptedPassword)) {
1045 					/*
1046 					 * We know this user, but they can't use this authentication
1047 					 * method. They'll probably have to use username+password.
1048 					 */
1049 					return false;
1050 				}
1051 				checkSubject(username, subject,	authInfo.openIdSubject);
1052 				queries.noteLoginSuccessForUser(userId, subject);
1053 				log.info("login success for {}", username);
1054 				return true;
1055 			} catch (AuthenticationException e) {
1056 				queries.noteLoginFailureForUser(userId, username);
1057 				log.info("login failure for {}", username, e);
1058 				throw e;
1059 			}
1060 		}, username).isPresent();
1061 
1062 		if (ok) {
1063 			return true;
1064 		}
1065 
1066 		/*
1067 		 * No such user; need to inflate one now. If we successfully make
1068 		 * the user, they're also immediately authorised as they're
1069 		 * definitely not disabled or locked.
1070 		 */
1071 		return ifElse(queries.createUser(username, null, USER, subject),
1072 				id -> {
1073 					synchExternalGroups(username, id, orgs, collabs,
1074 							queries);
1075 					return true;
1076 				}, () -> {
1077 					// Can't note a failure; no record to note it in!
1078 					log.warn("failed to make user {}", username);
1079 					return false;
1080 				});
1081 	}

1083 	/**
1084 	 * Check if the subject matches and issue a warning if there's a potential
1085 	 * problem.
1086 	 *
1087 	 * @param username
1088 	 *            The username (for logging).
1089 	 * @param subject
1090 	 *            The new user subject ID.
1091 	 * @param oldSubject
1092 	 *            The old user subject ID.
1093 	 */
1094 	private void checkSubject(String username, String subject,
1095 			String oldSubject) {
1096 		if (isNull(subject)) {
1097 			log.warn("null subject for {}", username);
1098 		} else if (subject.isBlank()) {
1099 			log.warn("empty subject for {}", username);
1100 		}
1101 		if (nonNull(oldSubject)) {
1102 			/*
1103 			 * We know this user from before; double check that they're the same
1104 			 * person as before. The {@code sub} claim is a true identifier.
1105 			 */
1106 			if (!requireNonNull(subject).equals(oldSubject)) {
1107 				log.warn("user {} subject changed from {} to {}", username,
1108 						oldSubject, subject);
1109 			}
1110 		}
1111 	}

1113 	private void inflateGroup(GrantedAuthority ga, List<String> collabs,
1114 			List<String> orgs, AuthQueries queries) {
1115 		if (ga instanceof CollabratoryAuthority) {
1116 			var collab = (CollabratoryAuthority) ga;
1117 			var collab1 = collab.getCollabratory();
1118 			if (queries.createGroup(collab1, COLLABRATORY,
1119 					quotaProps.getDefaultCollabQuota())) {
1120 				log.info("created collabratory '{}'", collab1);
1121 			}
1122 			collabs.add(collab.getCollabratory());
1123 		} else if (ga instanceof OrganisationAuthority) {
1124 			var org = (OrganisationAuthority) ga;
1125 			var org1 = org.getOrganisation();
1126 			if (queries.createGroup(org1, ORGANISATION,
1127 					quotaProps.getDefaultOrgQuota())) {
1128 				log.info("created organisation '{}'", org1);
1129 			}
1130 			orgs.add(org.getOrganisation());
1131 		}

1135 	 * Synchronise the lists of different types of groups for a user. Messy
1136 	 * because we want to do minimal inserts and deletes.
1137 	 *
1138 	 * @param username
1139 	 *            Which user (for logging only)
1140 	 * @param userId
1141 	 *            Which user (for DB access only)
1142 	 * @param orgs
1143 	 *            The list of organisation names
1144 	 * @param collabs
1145 	 *            The list of collab names
1146 	 * @param queries
1147 	 *            How to touch the DB
1148 	 */
1149 	private void synchExternalGroups(String username, int userId,
1150 			List<String> orgs, List<String> collabs, AuthQueries queries) {
1151 		if (orgs.isEmpty() && collabs.isEmpty()) {
1152 			log.warn(
1153 					"no organisations or collabratories for user {}; "
1154 							+ "not synching in case this is a glitch "
1155 							+ "(if this is the first time they've logged in, "
1156 							+ "creating a job will fail in this session)",
1157 					username);
1158 			return;
1159 		}
1160 		try (var synch = new GroupSynch(queries)) {
1161 			for (var org : orgs) {
1162 				synch.define(org, ORGANISATION);
1163 			}
1164 			for (var collab : collabs) {
1165 				synch.define(collab, COLLABRATORY);
1166 			}
1167 			synch.apply(userId);
1168 		} catch (RuntimeException e) {
1169 			log.warn("problem when synchronizing group memberships for {}",
1170 					username, e);
1171 			throw e;
1172 		}
1173 	}
