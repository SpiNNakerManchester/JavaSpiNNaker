17  package uk.ac.manchester.spinnaker.alloc.security;
18  
19  import static java.util.Objects.isNull;
20  import static org.springframework.beans.factory.config.BeanDefinition.ROLE_INFRASTRUCTURE;
21  
22  import java.util.ArrayList;
23  import java.util.Collection;
24  import java.util.List;
25  import java.util.Map;
26  import java.util.Optional;
27  import java.util.stream.Stream;
28  
29  import org.springframework.context.annotation.Role;
30  import org.springframework.expression.EvaluationContext;
31  import org.springframework.expression.Expression;
32  import org.springframework.security.access.expression.method.DefaultMethodSecurityExpressionHandler;
33  import org.springframework.security.access.prepost.PostFilter;
34  import org.springframework.stereotype.Component;
35  
36  import uk.ac.manchester.spinnaker.utils.UsedInJavadocOnly;
37  
38  /**
39   * Because Spring otherwise can't apply {@link PostFilter} to {@link Stream}.
40   *
41   * @author Donal Fellows
42   * @see <a href="https://stackoverflow.com/q/66107075/301832">Stack Overflow</a>
43   */
44  @Component
45  @Role(ROLE_INFRASTRUCTURE)
46  @UsedInJavadocOnly(PostFilter.class)
47  class AnyTypeMethodSecurityExpressionHandler
48  		extends DefaultMethodSecurityExpressionHandler {
49  	@Override
50  	public Object filter(Object target, Expression expr,
51  			EvaluationContext ctx) {
52  		if (isNull(target)) {
53  			// We can handle this case here!
54  			return null;
55  		}
56  		if (target instanceof Collection || target.getClass().isArray()
57  				|| target instanceof Map || target instanceof Stream) {
58  			return super.filter(target, expr, ctx);
59  		}
60  		if (target instanceof Optional) {
61  			return filterOptional((Optional<?>) target, expr, ctx);
62  		} else {
63  			return filterOptional(Optional.of(target), expr, ctx).orElse(null);
64  		}
65  	}
66  
67  	@SuppressWarnings("unchecked")
68  	private <T> Optional<T> filterOptional(Optional<T> target, Expression expr,
69  			EvaluationContext ctx) {
70  		// Java 8 language profile makes this a little messy
71  		List<T> a = new ArrayList<T>();
72  		target.ifPresent(a::add);
73  		return ((Stream<T>) super.filter(a.stream(), expr, ctx)).findFirst();
74  	}
75  }
