38  /**
39   * Developer credentials for initial access to client registration using the
40   * HBP/EBRAINS Keycloak service.
41   *
42   * @author Donal Fellows
43   */
44  interface EBRAINSDevCredentials {
45  	/** Where to get the developer access token from. This is non-standard. */
46  	String OIDC_TOKEN_URL =
47  			HBP_OPENID_BASE + "realms/hbp/protocol/openid-connect/token";
48  
49  	/**
50  	 * @return The user that will do the registration. Must not be {@code null}.
51  	 */
52  	String getUser();
53  
54  	/**
55  	 * @return The password for the user that will do the registration. Must not
56  	 *         be {@code null}.
57  	 */
58  	String getPass();
59  
60  	/**
61  	 * Convert the username and password into an access token.
62  	 *
63  	 * @return The access token.
64  	 * @throws IOException
65  	 *             If anything goes wrong. In particular, the user must be known
66  	 *             to the Keycloak service and be authorized to publish service
67  	 *             descriptions.
68  	 */
69  	default String getToken() throws IOException {
70  		var user = requireNonNull(getUser(), "user must be given");
71  		var pass = requireNonNull(getPass(), "password must be given");
72  		var mapper = new ObjectMapper();
73  
74  		var url = new URL(OIDC_TOKEN_URL);
75  		var http = (HttpURLConnection) url.openConnection();
76  		http.setRequestMethod("POST");
77  		http.setDoOutput(true);
78  
79  		// UGLY!
80  		http.setRequestProperty("Authorization", makeBasicAuth("developer:"));
81  
82  		var arguments = new HashMap<String, String>();
83  		arguments.put("grant_type", "password");
84  		arguments.put("username", user);
85  		arguments.put("password", pass);
86  
87  		var out = encodeForm(arguments);
88  		http.setFixedLengthStreamingMode(out.length);
89  		http.setRequestProperty("Content-Type",
90  				"application/x-www-form-urlencoded; charset=UTF-8");
91  		try (var os = http.getOutputStream()) {
92  			os.write(out);
93  		}
94  		try (var is = http.getInputStream()) {
95  			return mapper.readValue(is, AccessTokenResponse.class).getToken();
96  		}
97  	}
98  }

100 abstract class EBRAINSDevCredentialsUtils {
101 	private EBRAINSDevCredentialsUtils() {
102 	}
103 
104 	/**
105 	 * Convert a map to HTML form. Technically the target format supports a
106 	 * multimap, but we don't need that!
107 	 *
108 	 * @param arguments
109 	 *            The map of arguments.
110 	 * @return The encoded form.
111 	 */
112 	static byte[] encodeForm(Map<String, String> arguments) {
113 		var sj = new StringJoiner("&");
114 		for (var entry : arguments.entrySet()) {
115 			sj.add(encode(entry.getKey(), UTF_8) + "="
116 					+ encode(entry.getValue(), UTF_8));
117 		}
118 		return sj.toString().getBytes(UTF_8);
119 	}
120 
121 	/**
122 	 * Convert a username/password pair to the contents of an
123 	 * {@code Authorization} header.
124 	 *
125 	 * @param userPass
126 	 *            The username and password, separated by a colon.
127 	 * @return the header contents
128 	 */
129 	static String makeBasicAuth(String userPass) {
130 		return "Basic " + new String(
131 				getEncoder().encode(userPass.getBytes(UTF_8)), UTF_8);
132 	}
133 }
