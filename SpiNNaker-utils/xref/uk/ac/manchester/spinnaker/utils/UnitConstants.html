17  package uk.ac.manchester.spinnaker.utils;
18  
19  import static java.lang.String.format;
20  
21  import java.time.Duration;
22  
23  /**
24   * Constants relating to physical units.
25   *
26   * @author Christian-B
27   * @author Donal Fellows
28   */
29  public final class UnitConstants {
30  	private UnitConstants() {
31  	}
32  
33  	/** The number of megahertz in each kilohertz. */
34  	public static final int MEGAHERTZ_PER_KILOHERTZ = 1000;
35  
36  	/** The number of megahertz in each hertz. */
37  	public static final int MEGAHERTZ_PER_HERTZ = 1000000;
38  
39  	/** The number of milliseconds per second. */
40  	public static final int MSEC_PER_SEC = 1000;
41  
42  	/** The number of nanoseconds per second. */
43  	public static final int NSEC_PER_SEC = 1000000000;
44  
45  	/** The number of second per minute. */
46  	public static final int SEC_PER_MINUTE = 60;
47  
48  	/** The number of minute per hour. */
49  	public static final int MINUTE_PER_HOUR = 60;
50  
51  	/** The number of nanoseconds per microsecond. */
52  	public static final double NSEC_PER_USEC = 1000.0;
53  
54  	/** The number of nanoseconds per millisecond. */
55  	public static final double NSEC_PER_MSEC = 1000000.0;
56  
57  	/**
58  	 * Formats a duration with hours, minutes seconds and milliseconds as
59  	 * required.
60  	 *
61  	 * @param durationInMillis
62  	 *            A time interval in milliseconds
63  	 * @return A formatted String with only the relative units.
64  	 */
65  	public static String formatDuration(long durationInMillis) {
66  		var d = Duration.ofMillis(durationInMillis);
67  		long hr = d.toHours();
68  		long min = d.toMinutesPart();
69  		long sec = d.toSecondsPart();
70  		long ms = d.toMillisPart();
71  		if (hr > 0) {
72  			return format("%d:%02d:%02d.%03d h", hr, min, sec, ms);
73  		}
74  		if (min > 0) {
75  			return format("%d:%02d.%03d m", min, sec, ms);
76  		}
77  		if (sec > 0) {
78  			return format("%d.%03d s", sec, ms);
79  		}
80  		return ms + " ms";
81  	}
82  }
