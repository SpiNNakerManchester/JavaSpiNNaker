17  package uk.ac.manchester.spinnaker.utils.progress;
18  
19  import java.io.Closeable;
20  import java.io.PrintStream;
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.Iterator;
24  
25  /**
26   * An {@link Iterable} wrapper that will start a {@link ProgressBar} for each
27   * iterator.
28   * <p>
29   * As items are obtained by calling {@link ProgressIterator#next() next()} the
30   * ProgressBar is updated. When {@link ProgressIterator#hasNext() hasNext()}
31   * returns false the ProgressBar is closed resulting in the duration to be
32   * written out. If the Iterable is used in a for loop the duration is started at
33   * the beginning of the for loop and duration finishes and reports as the for
34   * loop ends. (Not at the time the last element is provided.)
35   * <p>
36   * Calling {@link #iterator()} more than once is supported but could result in
37   * mixed output if they are run at the same time.
38   *
39   * @author Christian-B
40   * @param <E>
41   *            Type of elements to be iterated over.
42   */
43  public class ProgressIterable<E> implements Iterable<E>, Closeable {
44  
45  	private final Collection<E> things;
46  	private final String description;
47  	private final ArrayList<ProgressIterator<E>> progressIterables;
48  	private final PrintStream output;
49  
50  	/**
51  	 * Creates an iterable wrapper but not yet a ProgressBar.
52  	 *
53  	 * @param things
54  	 *            A collection to supply a size and and iterable. The type of
55  	 *            elements returned by this iterable is also the type of
56  	 *            elements returned by the ProgressIterable.
57  	 * @param description
58  	 *            A text description to add at the start and when reporting
59  	 *            duration.
60  	 * @param output
61  	 *            The Stream to write output too. For example,
62  	 *            {@link System#out}.
63  	 */
64  	public ProgressIterable(Collection<E> things, String description,
65  			PrintStream output) {
66  		this.things = things;
67  		this.description = description;
68  		this.output = output;
69  		progressIterables = new ArrayList<>();
70  	}
71  
72  	/**
73  	 * Creates an terable wrapper but not yet a ProgressBar. The progress bar
74  	 * will write to {@link System#out}.
75  	 *
76  	 * @param things
77  	 *            A collection to supply a size and and iterable. The type of
78  	 *            elements returned by this iterable is also the type of
79  	 *            elements returned by the ProgressIterable.
80  	 * @param description
81  	 *            A text description to add at the start and when reporting
82  	 *            duration.
83  	 */
84  	public ProgressIterable(Collection<E> things, String description) {
85  		this(things, description, System.out);
86  	}
87  
88  	/**
89  	 * Starts a ProgressBar and returns an iterator over the elements of the
90  	 * iterable contained within this iterable.
91  	 *
92  	 * @return an iterator over the elements of the inner iterable.
93  	 */
94  	@Override
95  	public Iterator<E> iterator() {
96  		ProgressIterator<E> iterator =
97  				new ProgressIterator<>(things, description, output);
98  		progressIterables.add(iterator);
99  		return iterator;
100 	}
101 
102 	/**
103 	 * Closes all created Iterators and their ProgressBar(s).
104 	 * <p>
105 	 * This method allows the Iterable to be used in a
106 	 * {@code try}-with-resources statement, which guarantees the ProgressBar
107 	 * is closed.
108 	 * <p>
109 	 * Note: As {@code hasNext() == false} automatically calls close there is
110 	 * no need to call this method unless you break out of the iterator early.
111 	 * <p>
112 	 * If the bar is already closed, invoking this method has no effect.
113 	 */
114 	@Override
115 	public void close() {
116 		Iterator<ProgressIterator<E>> iter = progressIterables.iterator();
117 		while (iter.hasNext()) {
118 			iter.next().close();
119 			iter.remove();
120 		}
121 	}
122 }
