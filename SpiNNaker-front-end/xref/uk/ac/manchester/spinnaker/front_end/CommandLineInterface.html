100 	static class BuildPropsLoader implements IModelTransformer {
101 		private Properties loadProps() throws IOException {
102 			var prop = new Properties();
103 			try (var clp = getClass().getResourceAsStream(PROPS)) {
104 				prop.load(clp);
105 			}
106 			return prop;
107 		}
108 
109 		@Override
110 		public CommandSpec transform(CommandSpec commandSpec) {
111 			try {
112 				var prop = loadProps();
113 				var jar = prop.getProperty("jar");
114 				var ver = prop.getProperty("version");
115 
116 				commandSpec.name("java -jar " + jar);
117 				commandSpec.version(jar + " version " + ver);
118 			} catch (IOException e) {
119 				e.printStackTrace();
120 				System.exit(USAGE);
121 			}
122 			return commandSpec;
123 		}
124 	}

131 	protected static int run(String... args) {
132 		var cmd = new CommandLine(new CommandLineInterface());
133 		if (args.length == 0) {
134 			cmd.usage(cmd.getErr());
135 			return USAGE;
136 		} else {
137 			return cmd.execute(args);
138 		}
139 	}

205 	/**
206 	 * Run the data specifications in parallel.
207 	 *
208 	 * @param machine
209 	 *            Description of overall machine
210 	 * @param dsFile
211 	 *            Path to the dataspec database
212 	 * @param logfile
213 	 *            The path where the log should write.
214 	 * @param filterSystemCores
215 	 *            If {@code true}, only run the DSE for application vertices. If
216 	 *            {@code false}, only run the DSE for system vertices. If
217 	 *            {@code null}, run the DSE for all vertices.
218 	 * @throws IOException
219 	 *             If the communications fail.
220 	 * @throws SpinnmanException
221 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
222 	 * @throws StorageException
223 	 *             If the database is in an illegal state.
224 	 * @throws ExecutionException
225 	 *             If there was a problem in the parallel queue.
226 	 * @throws InterruptedException
227 	 *             If the wait for everything to complete is interrupted.
228 	 * @throws URISyntaxException
229 	 *             If the proxy URI is provided but not valid.
230 	 */
231 	public void runDSEUploadingViaClassicTransfer(Machine machine,
232 			File dsFile, File logfile, Boolean filterSystemCores)
233 			throws IOException, SpinnmanException, StorageException,
234 			ExecutionException, InterruptedException, URISyntaxException {
235 		setLogfile(logfile);
236 		var db = getDataSpecDB(dsFile);
237 
238 		try (var dseExec = hostFactory.create(machine, db)) {
239 			if (filterSystemCores) {
240 				dseExec.loadCores(false);
241 			} else {
242 				dseExec.loadCores(true);
243 			}
244 		} catch (Exception ex) {
245 			log.error("DSE load failed", ex);
246 			throw ex;
247 		}
248 	}

257 	 * @param dsFile
258 	 *            Path to the dataspec database
259 	 * @param logfile
260 	 *            The path where the log should write.
261 	 * @param reportFolder
262 	 *            Directory containing reports. If {@link Optional#empty()}, no
263 	 *            report will be written.
264 	 * @throws IOException
265 	 *             If the communications fail.
266 	 * @throws SpinnmanException
267 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
268 	 * @throws StorageException
269 	 *             If the database is in an illegal state.
270 	 * @throws ExecutionException
271 	 *             If there was a problem in the parallel queue.
272 	 * @throws InterruptedException
273 	 *             If the wait for everything to complete is interrupted.
274 	 * @throws URISyntaxException
275 	 *             If a proxy URI is provided but invalid.
276 	 */
277 	@Command(name = "dse_app_mon", description = DSE_MON_DESC)
278 	public void runDSEForAppCoresUploadingViaMonitorStreaming(
279 			@Mixin GatherersParam gatherers,
280 			@Mixin MachineParam machine,
281 			@Mixin DsFileParam dsFile,
282 			@Mixin LogfileParam logfile,
283 			@Parameters(description = REPORT, arity = "0..1", index = "3")
284 			Optional<File> reportFolder)
285 			throws IOException, SpinnmanException, StorageException,
286 			ExecutionException, InterruptedException, URISyntaxException {
287 		setLogfile(logfile.get());
288 		var db = getDataSpecDB(dsFile.get());
289 
290 		try (var dseExec = fastFactory.create(machine.get(),
291 						gatherers.get(), reportFolder.orElse(null), db)) {
292 			dseExec.loadCores(gatherers.get());
293 		}
294 	}

298 	 *
299 	 * @param machine
300 	 *            Description of overall machine.
301 	 * @param iobuf
302 	 *            Mapping from APLX executable names (full paths) to what cores
303 	 *            are running those executables, and which we will download
304 	 *            IOBUFs for.
305 	 * @param dbFile
306 	 *            The database that receives the output).
307 	 * @param logfile
308 	 *            The path where the log should write.
309 	 * @param iobufDir
310 	 * 			  The path to the directory where the iobuff would be written
311 	 * @throws IOException
312 	 *             If the communications fail.
313 	 * @throws SpinnmanException
314 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
315 	 * @throws InterruptedException
316 	 *             If interrupted (not expected).
317 	 * @throws URISyntaxException
318 	 *             If the proxy URI is invalid
319 	 * @throws StorageException
320 	 *             If there is an error reading the database
321 	 */
322 	// see https://github.com/SpiNNakerManchester/JavaSpiNNaker/issues/1218
323 	@Command(name = "iobuf", description = IOBUF_DESC)
324 	public void retrieveIOBUFs(
325 			@Mixin MachineParam machine,
326 			@Mixin IobufMapParam iobuf,
327 			@Mixin DbFileParam dbFile,
328 			@Mixin LogfileParam logfile,
329 			@Mixin IobufDirParam iobufDir)
330 			throws IOException, SpinnmanException, InterruptedException,
331 			StorageException, URISyntaxException {
332 		setLogfile(logfile.get());
333 		var db = getBufferManagerDB(dbFile.get());
334 
335 		try (var r = new IobufRetriever(db, machine.get(),
336 						PARALLEL_SIZE)) {
337 			var result = r.retrieveIobufContents(iobuf.get(), iobufDir.get());
338 			MAPPER.writeValue(System.out, result);
339 		}
340 	}
