17  package uk.ac.manchester.spinnaker.front_end.download;
18  
19  import static java.nio.ByteBuffer.allocate;
20  import static java.nio.ByteOrder.LITTLE_ENDIAN;
21  import static uk.ac.manchester.spinnaker.front_end.download.GatherProtocolMessage.ID.CLEAR_TRANSMISSIONS;
22  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
23  import static uk.ac.manchester.spinnaker.messages.sdp.SDPPort.EXTRA_MONITOR_CORE_DATA_SPEED_UP;
24  
25  import java.nio.ByteBuffer;
26  
27  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
28  
29  /**
30   * A message used to request fast data transfer from SpiNNaker to Host.
31   *
32   * @author Alan Stokes
33   */
34  
35  public final class ClearMessage extends GatherProtocolMessage {
36  	private static final int NUM_WORDS = 2;
37  
38  	/**
39  	 * Create a message used to request fast data transfer from SpiNNaker to
40  	 * Host.
41  	 *
42  	 * @param destination
43  	 *            Where to send the message
44  	 * @param transactionId
45  	 *            the transaction id needed
46  	 * @return The created message.
47  	 */
48  	static ClearMessage create(HasCoreLocation destination, int transactionId) {
49  		ByteBuffer payload =
50  				allocate(NUM_WORDS * WORD_SIZE).order(LITTLE_ENDIAN);
51  		payload.putInt(CLEAR_TRANSMISSIONS.value);
52  		payload.putInt(transactionId);
53  		payload.flip();
54  		return new ClearMessage(destination, payload);
55  	}
56  
57  	private ClearMessage(HasCoreLocation destination, ByteBuffer payload) {
58  		super(destination, EXTRA_MONITOR_CORE_DATA_SPEED_UP, payload);
59  	}
60  }
