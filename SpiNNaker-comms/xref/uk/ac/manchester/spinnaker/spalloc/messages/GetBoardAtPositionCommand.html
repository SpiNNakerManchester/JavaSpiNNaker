17  package uk.ac.manchester.spinnaker.spalloc.messages;
18  
19  import javax.validation.Valid;
20  import javax.validation.constraints.NotBlank;
21  
22  import uk.ac.manchester.spinnaker.machine.board.PhysicalCoords;
23  
24  /**
25   * Request the logical location of a board.
26   *
27   * @see <a href=
28   *      "https://spalloc-server.readthedocs.io/en/stable/protocol/#commands.get_board_at_position"
29   *      >Spalloc Server documentation</a>
30   * @see BoardCoordinates The basic result type associated with the request
31   */
32  public class GetBoardAtPositionCommand extends Command<Integer> {
33  	/**
34  	 * Create a request.
35  	 *
36  	 * @param machine
37  	 *            The machine to ask about.
38  	 * @param coords
39  	 *            The logical coordinates of the board.
40  	 */
41  	public GetBoardAtPositionCommand(@NotBlank String machine,
42  			@Valid PhysicalCoords coords) {
43  		super("get_board_at_position");
44  		addKwArg("machine_name", machine);
45  		// The current spalloc server expects the param names x, y, z
46  		addKwArg("x", coords.c);
47  		addKwArg("y", coords.f);
48  		addKwArg("z", coords.b);
49  	}
50  }
