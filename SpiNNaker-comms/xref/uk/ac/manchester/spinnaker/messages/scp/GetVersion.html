17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_VER;
20  
21  import java.nio.ByteBuffer;
22  
23  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
24  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
25  import uk.ac.manchester.spinnaker.messages.model.VersionInfo;
26  
27  /** An SCP request to read the version of software running on a core. */
28  public class GetVersion extends SCPRequest<GetVersion.Response> {
29  	/**
30  	 * @param core
31  	 *            The location of the core to read from.
32  	 */
33  	public GetVersion(HasCoreLocation core) {
34  		super(core, CMD_VER);
35  	}
36  
37  	@Override
38  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
39  		return new Response(buffer);
40  	}
41  
42  	/** An SCP response to a request for the version of software running. */
43  	public static final class Response extends CheckOKResponse {
44  		/** The version information received. */
45  		public final VersionInfo versionInfo;
46  
47  		Response(ByteBuffer buffer) throws UnexpectedResponseCodeException {
48  			super("Version", CMD_VER, buffer);
49  			versionInfo = new VersionInfo(buffer);
50  		}
51  	}
52  }
