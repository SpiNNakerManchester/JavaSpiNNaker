17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static java.nio.ByteOrder.LITTLE_ENDIAN;
20  import static uk.ac.manchester.spinnaker.messages.Constants.UDP_MESSAGE_MAX_SIZE;
21  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_READ;
22  import static uk.ac.manchester.spinnaker.messages.scp.TransferUnit.efficientTransferUnit;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
27  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
28  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
29  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
30  
31  /**
32   * An SCP request to read a region of memory. The response payload is a
33   * read-only little-endian {@link ByteBuffer} intended to be read once.
34   * <p>
35   * Calls {@code sark_cmd_read()} in {@code sark_base.c}.
36   */
37  public class ReadMemory extends SCPRequest<ReadMemory.Response> {
38  	private static int validate(int size) {
39  		if (size < 1 || size > UDP_MESSAGE_MAX_SIZE) {
40  			throw new IllegalArgumentException(
41  					"size must be in range 1 to 256");
42  		}
43  		return size;
44  	}
45  
46  	/**
47  	 * @param core
48  	 *            the core to read via
49  	 * @param address
50  	 *            The positive base address to start the read from
51  	 * @param size
52  	 *            The number of bytes to read, between 1 and 256
53  	 */
54  	public ReadMemory(HasCoreLocation core, MemoryLocation address, int size) {
55  		super(core, CMD_READ, address.address, validate(size),
56  				efficientTransferUnit(address, size).value);
57  	}
58  
59  	/**
60  	 * @param chip
61  	 *            the chip to read via
62  	 * @param address
63  	 *            The positive base address to start the read from
64  	 * @param size
65  	 *            The number of bytes to read, between 1 and 256
66  	 */
67  	public ReadMemory(HasChipLocation chip, MemoryLocation address, int size) {
68  		super(chip.getScampCore(), CMD_READ, address.address, validate(size),
69  				efficientTransferUnit(address, size).value);
70  	}
71  
72  	@Override
73  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
74  		return new Response(buffer);
75  	}
76  
77  	/**
78  	 * An SCP response to a request to read a region of memory on a chip. Note
79  	 * that it is up to the caller to manage the buffer position of the returned
80  	 * response if it is to be read from multiple times.
81  	 */
82  	public static final class Response
83  			extends PayloadedResponse<ByteBuffer, RuntimeException> {
84  		Response(ByteBuffer buffer) throws UnexpectedResponseCodeException {
85  			super("Read", CMD_READ, buffer);
86  		}
87  
88  		/** @return The data read, in a little-endian read-only buffer. */
89  		@Override
90  		protected ByteBuffer parse(ByteBuffer buffer) {
91  			return buffer.asReadOnlyBuffer().order(LITTLE_ENDIAN);
92  		}
93  	}
94  }
