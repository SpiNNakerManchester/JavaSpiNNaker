17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
20  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_RTR;
21  
22  import java.nio.ByteBuffer;
23  
24  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
25  import uk.ac.manchester.spinnaker.messages.model.AppID;
26  
27  /** Sets a fixed route entry. */
28  public final class FixedRouteInitialise extends SCPRequest<CheckOKResponse> {
29  	private static int argument1(AppID appID) {
30  		return appID.appID << BYTE1;
31  	}
32  
33  	/**
34  	 * @param chip
35  	 *            The chip to set the route on.
36  	 * @param entry
37  	 *            the fixed route entry (converted for writing)
38  	 * @param appID
39  	 *            The ID of the application
40  	 */
41  	public FixedRouteInitialise(HasChipLocation chip, int entry, AppID appID) {
42  		super(chip.getScampCore(), CMD_RTR, argument1(appID), entry);
43  	}
44  
45  	@Override
46  	public CheckOKResponse getSCPResponse(ByteBuffer buffer) throws Exception {
47  		return new CheckOKResponse("Fixed Route Initialise", CMD_RTR, buffer);
48  	}
49  }
