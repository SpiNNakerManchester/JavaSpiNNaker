17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.model.AllocFree.FREE_SDRAM_BY_APP_ID;
20  import static uk.ac.manchester.spinnaker.messages.model.AllocFree.FREE_SDRAM_BY_POINTER;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
22  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
23  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_ALLOC;
24  
25  import java.nio.ByteBuffer;
26  
27  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
28  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
29  import uk.ac.manchester.spinnaker.messages.model.AppID;
30  import uk.ac.manchester.spinnaker.messages.model.MemoryAllocationFailedException;
31  
32  /** An SCP Request to free space in the SDRAM. */
33  public class SDRAMDeAlloc extends SCPRequest<SDRAMDeAlloc.Response> {
34  	private final boolean readNumFreedBlocks;
35  
36  	/**
37  	 * Free the memory associated with a given application ID.
38  	 *
39  	 * @param chip
40  	 *            the chip to allocate on
41  	 * @param appID
42  	 *            The ID of the application
43  	 */
44  	public SDRAMDeAlloc(HasChipLocation chip, AppID appID) {
45  		super(chip.getScampCore(), CMD_ALLOC, argument1(appID));
46  		readNumFreedBlocks = true;
47  	}
48  
49  	/**
50  	 * Free a block of memory of known size.
51  	 *
52  	 * @param chip
53  	 *            the chip to allocate on
54  	 * @param baseAddress
55  	 *            The start address in SDRAM to which the block needs to be
56  	 *            deallocated
57  	 */
58  	public SDRAMDeAlloc(HasChipLocation chip, MemoryLocation baseAddress) {
59  		super(chip.getScampCore(), CMD_ALLOC, FREE_SDRAM_BY_POINTER.value,
60  				baseAddress.address);
61  		readNumFreedBlocks = false;
62  	}
63  
64  	private static int argument1(AppID appID) {
65  		return (appID.appID << BYTE1) | (FREE_SDRAM_BY_APP_ID.value << BYTE0);
66  	}
67  
68  	@Override
69  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
70  		return new Response(readNumFreedBlocks, buffer);
71  	}
72  
73  	/** An SCP response to a request to deallocate SDRAM. */
74  	public static class Response extends CheckOKResponse {
75  		/**
76  		 * The number of allocated blocks that have been freed from the appID
77  		 * given, or zero for when the direct amount of space to deallocate was
78  		 * given.
79  		 */
80  		public final int numFreedBlocks;
81  
82  		Response(boolean readFreedBlocks, ByteBuffer buffer) throws Exception {
83  			super("SDRAM Deallocation", CMD_ALLOC, buffer);
84  			if (readFreedBlocks) {
85  				numFreedBlocks = buffer.getInt();
86  				if (numFreedBlocks == 0) {
87  					throw new MemoryAllocationFailedException(
88  							"Could not deallocate SDRAM");
89  				}
90  			} else {
91  				numFreedBlocks = 0;
92  			}
93  		}
94  	}
95  }
