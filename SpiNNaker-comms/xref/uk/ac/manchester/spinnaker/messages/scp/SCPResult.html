17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.utils.CollectionUtils.makeEnumBackingMap;
20  
21  import java.util.Map;
22  
23  /** The SCP Result codes. */
24  public enum SCPResult {
25  	/** SCPCommand completed OK. */
26  	RC_OK(0x80),
27  	/** Bad packet length. */
28  	RC_LEN(0x81),
29  	/** Bad checksum. */
30  	RC_SUM(0x82),
31  	/** Bad/invalid command. */
32  	RC_CMD(0x83),
33  	/** Invalid arguments. */
34  	RC_ARG(0x84),
35  	/** Bad port number. */
36  	RC_PORT(0x85),
37  	/** Timeout. */
38  	RC_TIMEOUT(0x86),
39  	/** No P2P route. */
40  	RC_ROUTE(0x87),
41  	/** Bad CPU number. */
42  	RC_CPU(0x88),
43  	/** SHM destination dead. */
44  	RC_DEAD(0x89),
45  	/** No free shared memory buffers. */
46  	RC_BUF(0x8a),
47  	/** No reply to open. */
48  	RC_P2P_NOREPLY(0x8b),
49  	/** Message was rejected. */
50  	RC_P2P_REJECT(0x8c),
51  	/** Destination busy. */
52  	RC_P2P_BUSY(0x8d),
53  	/** Destination did not respond. */
54  	RC_P2P_TIMEOUT(0x8e),
55  	/** Packet transmission failed. */
56  	RC_PKT_TX(0x8f);
57  
58  	/** The encoded result value. */
59  	public final short value;
60  
61  	private static final Map<Short, SCPResult> MAP =
62  			makeEnumBackingMap(values(), v -> v.value);
63  
64  	SCPResult(int value) {
65  		this.value = (short) value;
66  	}
67  
68  	/**
69  	 * Decode a result.
70  	 *
71  	 * @param value
72  	 *            The value to decode
73  	 * @return The decoded value, or {@code null} if unrecognised.
74  	 */
75  	public static SCPResult get(short value) {
76  		return MAP.get(value);
77  	}
78  }
