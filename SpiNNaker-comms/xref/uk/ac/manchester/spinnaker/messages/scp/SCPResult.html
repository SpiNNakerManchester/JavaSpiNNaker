17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  /** The SCP Result codes. */
23  public enum SCPResult {
24  	/** SCPCommand completed OK. */
25  	RC_OK(0x80),
26  	/** Bad packet length. */
27  	RC_LEN(0x81),
28  	/** Bad checksum. */
29  	RC_SUM(0x82),
30  	/** Bad/invalid command. */
31  	RC_CMD(0x83),
32  	/** Invalid arguments. */
33  	RC_ARG(0x84),
34  	/** Bad port number. */
35  	RC_PORT(0x85),
36  	/** Timeout. */
37  	RC_TIMEOUT(0x86),
38  	/** No P2P route. */
39  	RC_ROUTE(0x87),
40  	/** Bad CPU number. */
41  	RC_CPU(0x88),
42  	/** SHM destination dead. */
43  	RC_DEAD(0x89),
44  	/** No free shared memory buffers. */
45  	RC_BUF(0x8a),
46  	/** No reply to open. */
47  	RC_P2P_NOREPLY(0x8b),
48  	/** Message was rejected. */
49  	RC_P2P_REJECT(0x8c),
50  	/** Destination busy. */
51  	RC_P2P_BUSY(0x8d),
52  	/** Destination did not respond. */
53  	RC_P2P_TIMEOUT(0x8e),
54  	/** Packet transmission failed. */
55  	RC_PKT_TX(0x8f);
56  
57  	/** The encoded result value. */
58  	public final short value;
59  
60  	private static final Map<Short, SCPResult> MAP = new HashMap<>();
61  
62  	SCPResult(int value) {
63  		this.value = (short) value;
64  	}
65  
66  	static {
67  		for (SCPResult r : values()) {
68  			MAP.put(r.value, r);
69  		}
70  	}
71  
72  	/**
73  	 * Decode a result.
74  	 *
75  	 * @param value
76  	 *            The value to decode
77  	 * @return The decoded value, or {@code null} if unrecognised.
78  	 */
79  	public static SCPResult get(short value) {
80  		return MAP.get(value);
81  	}
82  }
