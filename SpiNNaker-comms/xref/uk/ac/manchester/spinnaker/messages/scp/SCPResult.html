16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static uk.ac.manchester.spinnaker.utils.CollectionUtils.makeEnumBackingMap;
19  
20  import java.util.Map;
21  
22  /** The SCP Result codes. */
23  public enum SCPResult {
24  	/** Command completed successfully. */
25  	RC_OK(0x80),
26  	/** Bad packet length. */
27  	RC_LEN(0x81),
28  	/** Bad checksum. */
29  	RC_SUM(0x82),
30  	/** Bad/invalid command. */
31  	RC_CMD(0x83),
32  	/** Invalid arguments. */
33  	RC_ARG(0x84),
34  	/** Bad port number. */
35  	RC_PORT(0x85),
36  	/** Timeout. */
37  	RC_TIMEOUT(0x86),
38  	/** No P2P route. */
39  	RC_ROUTE(0x87),
40  	/** Bad CPU number. */
41  	RC_CPU(0x88),
42  	/** SHM destination dead. */
43  	RC_DEAD(0x89),
44  	/** No free shared memory buffers. */
45  	RC_BUF(0x8a),
46  	/** No reply to open. */
47  	RC_P2P_NOREPLY(0x8b),
48  	/** Message was rejected. */
49  	RC_P2P_REJECT(0x8c),
50  	/** Destination busy. */
51  	RC_P2P_BUSY(0x8d),
52  	/** Destination did not respond. */
53  	RC_P2P_TIMEOUT(0x8e),
54  	/** Packet transmission failed. */
55  	RC_PKT_TX(0x8f);
56  
57  	/** The encoded result value. */
58  	public final short value;
59  
60  	private static final Map<Short, SCPResult> MAP =
61  			makeEnumBackingMap(values(), v -> v.value);
62  
63  	SCPResult(int value) {
64  		this.value = (short) value;
65  	}
66  
67  	/**
68  	 * Decode a result.
69  	 *
70  	 * @param value
71  	 *            The value to decode
72  	 * @return The decoded value, or {@code null} if unrecognised.
73  	 */
74  	public static SCPResult get(short value) {
75  		return MAP.get(value);
76  	}
77  }
