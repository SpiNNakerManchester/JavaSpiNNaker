17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.model.ReinjectionStatus.encodeTimeout;
20  import static uk.ac.manchester.spinnaker.messages.scp.ReinjectorCommand.SET_ROUTER_EMERGENCY_TIMEOUT;
21  
22  import java.nio.ByteBuffer;
23  
24  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
25  
26  /**
27   * An SCP Request to set the router emergency timeout for dropped packet
28   * reinjection. There is no response payload.
29   * <p>
30   * Handled by {@code reinjection_set_emergency_timeout_sdp()} in
31   * {@code extra_monitor_support.c}.
32   */
33  public class SetRouterEmergencyTimeout
34  		extends ReinjectorRequest<CheckOKResponse> {
35  	/**
36  	 * @param core
37  	 *            The coordinates of the monitor core.
38  	 * @param timeoutMantissa
39  	 *            The mantissa of the timeout value, between 0 and 15.
40  	 * @param timeoutExponent
41  	 *            The exponent of the timeout value, between 0 and 15.
42  	 */
43  	public SetRouterEmergencyTimeout(HasCoreLocation core, int timeoutMantissa,
44  			int timeoutExponent) {
45  		super(core, SET_ROUTER_EMERGENCY_TIMEOUT,
46  				encodeTimeout(timeoutMantissa, timeoutExponent));
47  	}
48  
49  	@Override
50  	public CheckOKResponse getSCPResponse(ByteBuffer buffer) throws Exception {
51  		return new CheckOKResponse("Set router emergency timeout",
52  				SET_ROUTER_EMERGENCY_TIMEOUT, buffer);
53  	}
54  }
