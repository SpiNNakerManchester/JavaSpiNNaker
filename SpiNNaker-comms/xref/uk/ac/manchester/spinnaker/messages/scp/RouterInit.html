17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
20  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.HALF1;
22  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_RTR;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
27  import uk.ac.manchester.spinnaker.messages.model.AppID;
28  
29  /** A request to initialise the router on a chip. */
30  public class RouterInit extends SCPRequest<CheckOKResponse> {
31  	/** One reserved for SCAMP. */
32  	private static final int MAX_ENTRIES = 1023;
33  
34  	/**
35  	 * @param chip
36  	 *            The coordinates of the chip to clear the router of
37  	 * @param numEntries
38  	 *            The number of entries in the table (more than 0)
39  	 * @param tableAddress
40  	 *            The allocated table address
41  	 * @param baseAddress
42  	 *            The base address containing the entries
43  	 * @param appID
44  	 *            The ID of the application with which to associate the routes.
45  	 * @throws IllegalArgumentException
46  	 *             If a bad address or entry count is given
47  	 */
48  	public RouterInit(HasChipLocation chip, int numEntries, int tableAddress,
49  			int baseAddress, AppID appID) {
50  		super(chip.getScampCore(), CMD_RTR, argument1(numEntries, appID),
51  				tableAddress, baseAddress);
52  		if (baseAddress < 0) {
53  			throw new IllegalArgumentException(
54  					"baseAddress must not be negative");
55  		}
56  		if (tableAddress < 0) {
57  			throw new IllegalArgumentException(
58  					"tableAddress must not be negative");
59  		}
60  	}
61  
62  	private static int argument1(int numEntries, AppID appID) {
63  		if (numEntries < 1) {
64  			throw new IllegalArgumentException(
65  					"numEntries must be more than 0");
66  		} else if (numEntries > MAX_ENTRIES) {
67  			throw new IllegalArgumentException(
68  					"numEntries must be no more than " + MAX_ENTRIES);
69  		}
70  		return (numEntries << HALF1) | (appID.appID << BYTE1) | (2 << BYTE0);
71  	}
72  
73  	@Override
74  	public CheckOKResponse getSCPResponse(ByteBuffer buffer) throws Exception {
75  		return new CheckOKResponse("Router Init", CMD_RTR, buffer);
76  	}
77  }
