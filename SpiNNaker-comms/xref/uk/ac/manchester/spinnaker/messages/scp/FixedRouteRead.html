16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static uk.ac.manchester.spinnaker.messages.model.RouterCommand.ROUTER_FIXED;
19  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
20  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.TOP_BIT;
22  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_RTR;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
27  import uk.ac.manchester.spinnaker.machine.RoutingEntry;
28  import uk.ac.manchester.spinnaker.messages.model.AppID;
29  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
30  
31  /**
32   * Gets a fixed route entry. The response payload is the
33   * {@linkplain RoutingEntry routing entry}.
34   * <p>
35   * Calls {@code rtr_fr_get()} in {@code sark_hw.c}, via {@code rtr_cmd()} in
36   * {@code scamp-cmd.c}.
37   */
38  public final class FixedRouteRead extends SCPRequest<FixedRouteRead.Response> {
39  	private static int argument1(AppID appID) {
40  		return (appID.appID << BYTE1) | (ROUTER_FIXED.value << BYTE0);
41  	}
42  
43  	private static int argument2() {
44  		return 1 << TOP_BIT;
45  	}
46  
47  	/**
48  	 * @param chip
49  	 *            The chip to get the route from.
50  	 * @param appID
51  	 *            The ID of the application associated with the route
52  	 */
53  	public FixedRouteRead(HasChipLocation chip, AppID appID) {
54  		super(chip.getScampCore(), CMD_RTR, argument1(appID), argument2());
55  	}
56  
57  	@Override
58  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
59  		return new Response(buffer);
60  	}
61  
62  	/** Response for the fixed route read. */
63  	public static final class Response
64  			extends PayloadedResponse<RoutingEntry, RuntimeException> {
65  		private Response(ByteBuffer buffer)
66  				throws UnexpectedResponseCodeException {
67  			super("Read Fixed RoutingEntry route", CMD_RTR, buffer);
68  		}
69  
70  		/** @return the fixed route router route. */
71  		@Override
72  		protected RoutingEntry parse(ByteBuffer buffer) {
73  			return new RoutingEntry(buffer.getInt());
74  		}
75  	}
76  }
