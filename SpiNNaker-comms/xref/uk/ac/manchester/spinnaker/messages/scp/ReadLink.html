16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static java.nio.ByteOrder.LITTLE_ENDIAN;
19  import static uk.ac.manchester.spinnaker.messages.Constants.UDP_MESSAGE_MAX_SIZE;
20  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_LINK_READ;
21  
22  import java.nio.ByteBuffer;
23  
24  import uk.ac.manchester.spinnaker.machine.Direction;
25  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
26  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
27  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
28  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
29  
30  /**
31   * An SCP request to read a region of memory via a link on a chip. The response
32   * payload is a read-only little-endian {@link ByteBuffer} intended to be read
33   * once.
34   * <p>
35   * Calls {@code cmd_link_read()} in {@code scamp-cmd.c}.
36   */
37  public class ReadLink extends SCPRequest<ReadLink.Response> {
38  	private static int validate(int size) {
39  		if (size < 1 || size > UDP_MESSAGE_MAX_SIZE) {
40  			throw new IllegalArgumentException(
41  					"size must be in range 1 to 256");
42  		}
43  		return size;
44  	}
45  
46  	/**
47  	 * @param core
48  	 *            the core to read via
49  	 * @param link
50  	 *            The direction of the link down which to send the query
51  	 * @param baseAddress
52  	 *            The positive base address to start the read from
53  	 * @param size
54  	 *            The number of bytes to read, between 1 and 256
55  	 */
56  	public ReadLink(HasCoreLocation core, Direction link,
57  			MemoryLocation baseAddress, int size) {
58  		super(core, CMD_LINK_READ, baseAddress.address, validate(size),
59  				link.id);
60  	}
61  
62  	/**
63  	 * @param chip
64  	 *            the chip to read via
65  	 * @param link
66  	 *            The direction of the link down which to send the query
67  	 * @param baseAddress
68  	 *            The positive base address to start the read from
69  	 * @param size
70  	 *            The number of bytes to read, between 1 and 256
71  	 */
72  	public ReadLink(HasChipLocation chip, Direction link,
73  			MemoryLocation baseAddress, int size) {
74  		super(chip.getScampCore(), CMD_LINK_READ, baseAddress.address,
75  				validate(size), link.id);
76  	}
77  
78  	@Override
79  	public Response getSCPResponse(ByteBuffer buffer)
80  			throws UnexpectedResponseCodeException {
81  		return new Response(buffer);
82  	}
83  
84  	/**
85  	 * An SCP response to a request to read a region of memory via a link on a
86  	 * chip. Note that it is up to the caller to manage the buffer position of
87  	 * the returned response if it is to be read from multiple times.
88  	 */
89  	protected final class Response
90  			extends PayloadedResponse<ByteBuffer, RuntimeException> {
91  		Response(ByteBuffer buffer) throws UnexpectedResponseCodeException {
92  			super("Read Link", CMD_LINK_READ, buffer);
93  		}
94  
95  		/** @return The data read, as a read-only little-endian buffer. */
96  		@Override
97  		protected ByteBuffer parse(ByteBuffer buffer) {
98  			return buffer.asReadOnlyBuffer().order(LITTLE_ENDIAN);
99  		}
100 	}
101 }
