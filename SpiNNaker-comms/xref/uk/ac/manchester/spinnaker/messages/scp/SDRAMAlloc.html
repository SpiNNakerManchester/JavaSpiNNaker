17  package uk.ac.manchester.spinnaker.messages.scp;
18  
19  import static java.lang.String.format;
20  import static uk.ac.manchester.spinnaker.messages.model.AllocFree.ALLOC_SDRAM;
21  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE0;
22  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE1;
23  import static uk.ac.manchester.spinnaker.messages.scp.Bits.BYTE2;
24  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_ALLOC;
25  
26  import java.nio.ByteBuffer;
27  
28  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
29  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
30  import uk.ac.manchester.spinnaker.messages.model.AppID;
31  import uk.ac.manchester.spinnaker.messages.model.MemoryAllocationFailedException;
32  
33  /** An SCP Request to allocate space in the SDRAM space. */
34  public class SDRAMAlloc extends SCPRequest<SDRAMAlloc.Response> {
35  	private static final int MAX_SDRAM_TAG = 255;
36  
37  	private static final int FLAG_TAG_RETRY = 4;
38  
39  	private final int size;
40  
41  	/**
42  	 * @param chip
43  	 *            the chip to allocate on
44  	 * @param appID
45  	 *            The ID of the application
46  	 * @param size
47  	 *            The size in bytes of memory to be allocated
48  	 */
49  	public SDRAMAlloc(HasChipLocation chip, AppID appID, int size) {
50  		this(chip, appID, size, 0);
51  	}
52  
53  	/**
54  	 * @param chip
55  	 *            the chip to allocate on
56  	 * @param appID
57  	 *            The ID of the application
58  	 * @param size
59  	 *            The size in bytes of memory to be allocated
60  	 * @param tag
61  	 *            the tag for the SDRAM, a 8-bit (chip-wide) tag that can be
62  	 *            looked up by a SpiNNaker application to discover the address
63  	 *            of the allocated block
64  	 * @throws IllegalArgumentException
65  	 *             If a bad tag is given.
66  	 */
67  	public SDRAMAlloc(HasChipLocation chip, AppID appID, int size, int tag) {
68  		super(chip.getScampCore(), CMD_ALLOC, argument1(appID), size, tag);
69  		this.size = size;
70  		if (tag < 0 || tag > MAX_SDRAM_TAG) {
71  			throw new IllegalArgumentException(
72  					"The tag parameter needs to be between 0 and "
73  							+ MAX_SDRAM_TAG);
74  		}
75  	}
76  
77  	private static int argument1(AppID appID) {
78  		return (FLAG_TAG_RETRY << BYTE2) | (appID.appID << BYTE1)
79  				| (ALLOC_SDRAM.value << BYTE0);
80  	}
81  
82  	@Override
83  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
84  		return new Response(size, buffer);
85  	}
86  
87  	/** An SCP response to a request to allocate space in SDRAM. */
88  	public static class Response extends CheckOKResponse {
89  		/** The base address allocated. */
90  		public final MemoryLocation baseAddress;
91  
92  		Response(int size, ByteBuffer buffer) throws Exception {
93  			super("SDRAM Allocation", CMD_ALLOC, buffer);
94  			baseAddress = new MemoryLocation(buffer.getInt());
95  			if (baseAddress.isNull()) {
96  				throw new MemoryAllocationFailedException(
97  						format("Could not allocate %d bytes of SDRAM", size));
98  			}
99  		}
100 	}
101 }
