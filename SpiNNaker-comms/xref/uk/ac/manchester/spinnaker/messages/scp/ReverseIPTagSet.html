18  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.COMMAND_FIELD;
19  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.DEST_P_FIELD;
20  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.DEST_X_FIELD;
21  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.DEST_Y_FIELD;
22  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.PORT_MASK;

25  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.STRIP_FIELD_BIT;
26  import static uk.ac.manchester.spinnaker.messages.model.IPTagFieldDefinitions.THREE_BITS_MASK;
27  import static uk.ac.manchester.spinnaker.messages.scp.IPTagCommand.SET;
28  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_IPTAG;

41  /**
42   * A request to set a Reverse IP Tag. Reverse IP tags are tags that funnel
43   * packets from the outside world to a particular SpiNNaker core. There is no
44   * response payload.
45   * <p>
46   * Handled by {@code cmd_iptag()} in {@code scamp-cmd.c} (or {@code bmp_cmd.c},
47   * if sent to a BMP).
48   *
49   * @see IPTagSet
50   */
51  public class ReverseIPTagSet extends SCPRequest<EmptyResponse> {
52  	/**
53  	 * @param chip
54  	 *            The chip to set the tag on.
55  	 * @param destination
56  	 *            The coordinates of the destination processor.
57  	 * @param port
58  	 *            The port, between 0 and 65535.
59  	 * @param tag
60  	 *            The tag, between 0 and 7.
61  	 * @param sdpPort
62  	 *            The SDP port associated with this tag.
63  	 */
64  	public ReverseIPTagSet(@Valid @NotNull HasChipLocation chip,
65  			@Valid @NotNull HasCoreLocation destination, @UDPPort int port,
66  			@TagID int tag, int sdpPort) {
67  		super(chip.getScampCore(), CMD_IPTAG,
68  				argument1(sdpPort, destination, tag),
69  				argument2(destination, port), 0);
70  	}
71  
72  	// arg1 = flags[11:8] : timeout : command : dest_port : tag
73  	private static int argument1(int sdpPort, HasCoreLocation destination,
74  			int tag) {
75  		final int strip = 1;
76  		final int reverse = 1;
77  		return (reverse << REVERSE_FIELD_BIT) | (strip << STRIP_FIELD_BIT)
78  				| (SET.value << COMMAND_FIELD)
79  				| ((sdpPort & THREE_BITS_MASK) << SDP_PORT_FIELD)
80  				| (destination.getP() << DEST_P_FIELD)
81  				| (tag & THREE_BITS_MASK);
82  	}
83  
84  	// arg2 = dest_addr : port
85  	private static int argument2(HasCoreLocation destination, int port) {
86  		return (destination.getX() << DEST_X_FIELD)
87  				| (destination.getY() << DEST_Y_FIELD) | (port & PORT_MASK);
88  	}
89  
90  	@Override
91  	public EmptyResponse getSCPResponse(ByteBuffer buffer)
92  			throws UnexpectedResponseCodeException {
93  		return new EmptyResponse("Set Reverse IP Tag", CMD_IPTAG, buffer);
94  	}
95  }
