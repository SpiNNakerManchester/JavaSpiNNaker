16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_FILL;
19  
20  import java.nio.ByteBuffer;
21  
22  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
23  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
24  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
25  
26  /**
27   * An SCP request to fill a region of memory on a chip with repeated words of
28   * data. There is no response payload.
29   * <p>
30   * Calls {@code sark_cmd_fill()} in {@code sark_base.c}, or {@code cmd_fill()}
31   * in {@code bmp_cmd.c} if sent to a BMP.
32   */
33  public final class FillRequest extends SCPRequest<EmptyResponse> {
34  	/**
35  	 * @param chip
36  	 *            The chip to read from
37  	 * @param baseAddress
38  	 *            The positive base address to start the fill from. <em>Must be
39  	 *            word-aligned.</em>
40  	 * @param data
41  	 *            The <em>word</em> of data to fill in the space with.
42  	 * @param size
43  	 *            The number of <em>bytes</em> to fill in. <em>Must be a
44  	 *            multiple of 4.</em>
45  	 */
46  	public FillRequest(HasChipLocation chip, MemoryLocation baseAddress,
47  			int data, int size) {
48  		super(chip.getScampCore(), CMD_FILL, baseAddress.address, data, size);
49  	}
50  
51  	@Override
52  	public EmptyResponse getSCPResponse(ByteBuffer buffer)
53  			throws UnexpectedResponseCodeException {
54  		return new EmptyResponse("Fill", CMD_FILL, buffer);
55  	}
56  }
