16  package uk.ac.manchester.spinnaker.messages.scp;
17  
18  import static uk.ac.manchester.spinnaker.messages.scp.ReinjectorCommand.CLEAR;
19  
20  import java.nio.ByteBuffer;
21  
22  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
23  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
24  
25  /**
26   * A request to set the dropped packet reinjected packet types. There is no
27   * response payload.
28   * <p>
29   * Handled by {@code reinjection_clear()} in {@code extra_monitor_support.c}.
30   */
31  public class ClearReinjectionQueue extends ReinjectorRequest<EmptyResponse> {
32  	/**
33  	 * @param core
34  	 *            The coordinates of the monitor core.
35  	 */
36  	public ClearReinjectionQueue(HasCoreLocation core) {
37  		super(core, CLEAR);
38  	}
39  
40  	@Override
41  	public EmptyResponse getSCPResponse(ByteBuffer buffer)
42  			throws UnexpectedResponseCodeException {
43  		return new EmptyResponse("Clear reinjection queue", CLEAR, buffer);
44  	}
45  }
