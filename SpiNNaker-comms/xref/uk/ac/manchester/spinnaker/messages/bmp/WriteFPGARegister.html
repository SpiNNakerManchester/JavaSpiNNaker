17  package uk.ac.manchester.spinnaker.messages.bmp;
18  
19  import static java.nio.ByteBuffer.allocate;
20  import static java.nio.ByteOrder.LITTLE_ENDIAN;
21  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
22  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_LINK_WRITE;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
27  import uk.ac.manchester.spinnaker.messages.model.FPGA;
28  
29  /**
30   * A request for writing data to a FPGA register.
31   *
32   * @see <a href=
33   *      "https://github.com/SpiNNakerManchester/spio/blob/master/designs/spinnaker_fpgas/README.md#spi-interface">
34   *      spinnaker_fpga design README listing of FPGA registers</a>
35   * @see <a href="https://github.com/SpiNNakerManchester/spio/">The SpI/O project
36   *      on GitHub</a>
37   */
38  public class WriteFPGARegister extends BMPRequest<BMPRequest.BMPResponse> {
39  	/**
40  	 * @param fpga
41  	 *            FPGA (0, 1 or 2 on SpiNN-5 board) to communicate with.
42  	 * @param register
43  	 *            Register address to read to. Must be aligned.
44  	 * @param value
45  	 *            A 32-bit value to write to the register.
46  	 * @param board
47  	 *            which board to write the ADC register on
48  	 * @throws IllegalArgumentException
49  	 *             If {@link FPGA#FPGA_ALL} is used or the register address is
50  	 *             not aligned.
51  	 */
52  	public WriteFPGARegister(FPGA fpga, MemoryLocation register, int value,
53  			BMPBoard board) {
54  		super(board, CMD_LINK_WRITE, register.address, WORD_SIZE, fpga.value,
55  				data(value));
56  		if (!register.isAligned()) {
57  			throw new IllegalArgumentException(
58  					"FPGA register addresses must be aligned");
59  		}
60  		if (!fpga.isSingleFPGA()) {
61  			throw new IllegalArgumentException(
62  					"cannot write multiple FPGAs at once with this message");
63  		}
64  	}
65  
66  	private static ByteBuffer data(int value) {
67  		var buffer = allocate(WORD_SIZE).order(LITTLE_ENDIAN);
68  		buffer.putInt(value);
69  		buffer.flip();
70  		return buffer;
71  	}
72  
73  	@Override
74  	public BMPResponse getSCPResponse(ByteBuffer buffer) throws Exception {
75  		return new BMPResponse("Send FPGA register write", CMD_LINK_WRITE,
76  				buffer);
77  	}
78  }
