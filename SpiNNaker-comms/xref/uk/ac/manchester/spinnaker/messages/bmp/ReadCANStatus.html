17  package uk.ac.manchester.spinnaker.messages.bmp;
18  
19  import static uk.ac.manchester.spinnaker.messages.bmp.BMPInfo.CAN_STATUS;
20  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_BMP_INFO;
21  
22  import java.nio.ByteBuffer;
23  import java.util.ArrayList;
24  import java.util.List;
25  
26  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
27  import uk.ac.manchester.spinnaker.utils.MappableIterable;
28  
29  /**
30   * SCP Request for the CAN bus status data from the BMP.
31   */
32  public class ReadCANStatus extends BMPRequest<ReadCANStatus.Response> {
33  	private static final int MAX_BOARDS_PER_FRAME = 24;
34  
35  	private static final BMPBoard FRAME_ROOT = new BMPBoard(0);
36  
37  	public ReadCANStatus() {
38  		// The CAN status is shared between all BMPs; it's how they communicate
39  		super(FRAME_ROOT, CMD_BMP_INFO, CAN_STATUS.value);
40  	}
41  
42  	@Override
43  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
44  		return new Response(buffer);
45  	}
46  
47  	/** An SCP response to a request for the CAN status. */
48  	public final class Response extends BMPRequest.BMPResponse {
49  		/**
50  		 * The status data. The byte at {@code x} is zero if the BMP with that
51  		 * index is disabled.
52  		 */
53  		public final byte[] statusData;
54  
55  		/**
56  		 * What boards are available to be managed by the BMP?
57  		 *
58  		 * @return Ordered sequence of board numbers.
59  		 */
60  		public MappableIterable<Integer> availableBoards() {
61  			List<Integer> boards = new ArrayList<>();
62  			for (int i = 0; i < MAX_BOARDS_PER_FRAME; i++) {
63  				if (statusData[i] != 0) {
64  					boards.add(i);
65  				}
66  			}
67  			return boards::iterator;
68  		}
69  
70  		private Response(ByteBuffer buffer)
71  				throws UnexpectedResponseCodeException {
72  			super("Read CAN Status", CMD_BMP_INFO, buffer);
73  			statusData = new byte[buffer.remaining()];
74  			buffer.get(statusData);
75  		}
76  	}
77  }
