17  package uk.ac.manchester.spinnaker.messages.bmp;
18  
19  import static java.util.stream.Collectors.toUnmodifiableList;
20  import static java.util.stream.IntStream.range;
21  import static uk.ac.manchester.spinnaker.messages.bmp.BMPInfo.CAN_STATUS;
22  import static uk.ac.manchester.spinnaker.messages.scp.SCPCommand.CMD_BMP_INFO;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
27  import uk.ac.manchester.spinnaker.messages.model.UnexpectedResponseCodeException;
28  import uk.ac.manchester.spinnaker.utils.MappableIterable;
29  
30  /**
31   * SCP Request for the CAN bus status data from the BMP. The response payload is
32   * an {@linkplain MappableIterable iterable} of enabled {@linkplain BMPBoard
33   * board numbers}.
34   * <p>
35   * Handled in {@code cmd_bmp_info()} (in {@code bmp_cmd.c}) by reading from
36   * {@code can_status}.
37   */
38  public class ReadCANStatus extends BMPRequest<ReadCANStatus.Response> {
39  	private static final int MAX_BOARDS_PER_FRAME = 24;
40  
41  	private static final BMPBoard FRAME_ROOT = new BMPBoard(0);
42  
43  	/** Create a request. */
44  	public ReadCANStatus() {
45  		// The CAN status is shared between all BMPs; it's how they communicate
46  		super(FRAME_ROOT, CMD_BMP_INFO, CAN_STATUS.value);
47  	}
48  
49  	@Override
50  	public Response getSCPResponse(ByteBuffer buffer) throws Exception {
51  		return new Response(buffer);
52  	}
53  
54  	/** An SCP response to a request for the CAN status. */
55  	public static final class Response
56  			extends BMPRequest.PayloadedResponse<MappableIterable<BMPBoard>> {
57  		private Response(ByteBuffer buffer)
58  				throws UnexpectedResponseCodeException {
59  			super("Read CAN Status", CMD_BMP_INFO, buffer);
60  		}
61  
62  		/**
63  		 * @return Ordered sequence of board numbers available to be managed by
64  		 *         the BMP.
65  		 */
66  		@Override
67  		protected MappableIterable<BMPBoard> parse(ByteBuffer buffer) {
68  			/*
69  			 * The status data. The byte at {@code x} is zero if the BMP with
70  			 * that index is disabled.
71  			 */
72  			var statusData = new byte[buffer.remaining()];
73  			buffer.get(statusData);
74  			var boards = range(0, MAX_BOARDS_PER_FRAME)
75  					.filter(i -> statusData[i] != 0).mapToObj(BMPBoard::new)
76  					.collect(toUnmodifiableList());
77  			return boards::iterator;
78  		}
79  	}
80  }
