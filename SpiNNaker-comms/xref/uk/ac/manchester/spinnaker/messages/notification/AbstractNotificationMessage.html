17  package uk.ac.manchester.spinnaker.messages.notification;
18  
19  import java.nio.ByteBuffer;
20  
21  /**
22   * An notification message.
23   *
24   * @author Donal Fellows
25   */
26  public abstract class AbstractNotificationMessage
27  		implements NotificationMessage {
28  	// Must be power of 2 (minus 1)
29  	private static final int MAX_COMMAND = 0x3FFF;
30  
31  	private static final int FLAG1_BIT = 15;
32  
33  	private static final int FLAG2_BIT = 14;
34  
35  	/** The command code of the message. */
36  	private final NotificationMessageCode command;
37  
38  	protected AbstractNotificationMessage(ByteBuffer buffer) {
39  		command = NotificationMessageCode.get(buffer.getShort() & MAX_COMMAND);
40  	}
41  
42  	public AbstractNotificationMessage(NotificationMessageCode command) {
43  		this.command = command;
44  	}
45  
46  	@Override
47  	public void addToBuffer(ByteBuffer buffer) {
48  		short value = (short) (0 << FLAG1_BIT | 1 << FLAG2_BIT
49  				| command.getValue());
50  		buffer.putShort(value);
51  	}
52  
53  	public static NotificationMessage build(ByteBuffer buffer) {
54  		switch (NotificationMessageCode.get(buffer.getShort(0) & MAX_COMMAND)) {
55  		case DATABASE_CONFIRMATION:
56  			return new DatabaseConfirmation(buffer);
57  		case START_RESUME_NOTIFICATION:
58  			return new StartResume(buffer);
59  		case STOP_PAUSE_NOTIFICATION:
60  			return new PauseStop(buffer);
61  		default:
62  			// TODO Auto-generated method stub
63  			return null;
64  		}
65  	}
66  }
