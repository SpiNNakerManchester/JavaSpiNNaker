16  package uk.ac.manchester.spinnaker.messages.model;
17  
18  import static java.lang.String.format;
19  import static java.net.InetAddress.getByAddress;
20  import static java.util.stream.Collectors.toUnmodifiableList;
21  import static java.util.stream.IntStream.range;
22  import static uk.ac.manchester.spinnaker.messages.Constants.SCP_SCAMP_PORT;
23  
24  import java.net.InetAddress;
25  import java.net.UnknownHostException;
26  import java.util.Collection;
27  import java.util.List;
28  
29  import javax.validation.Valid;
30  
31  import uk.ac.manchester.spinnaker.machine.board.BMPBoard;
32  import uk.ac.manchester.spinnaker.machine.board.BMPCoords;
33  import uk.ac.manchester.spinnaker.utils.validation.IPAddress;
34  import uk.ac.manchester.spinnaker.utils.validation.UDPPort;
35  
36  /**
37   * Contains the details of a connection to a SpiNNaker Board Management
38   * Processor (BMP).
39   */
40  public class BMPConnectionData {
41  	/** The boards to be addressed. Unmodifiable. */
42  	public final Collection<@Valid BMPBoard> boards;
43  
44  	/** The coordinates of the BMP that manages a set of boards. */
45  	@Valid
46  	public final BMPCoords bmp;
47  
48  	/** The IP address of the BMP. */
49  	@IPAddress
50  	public final InetAddress ipAddress;
51  
52  	/**
53  	 * The port number associated with the BMP connection, or {@code null} for
54  	 * the default.
55  	 */
56  	@UDPPort
57  	public final Integer portNumber;
58  
59  	/**
60  	 * @param cabinet
61  	 *            The number of the cabinet containing the frame.
62  	 * @param frame
63  	 *            The number of the frame containing the boards.
64  	 * @param ipAddress
65  	 *            The address of the BMP.
66  	 * @param boards
67  	 *            The boards controlled by the BMP.
68  	 * @param portNumber
69  	 *            The BMP's port.
70  	 */
71  	public BMPConnectionData(int cabinet, int frame, InetAddress ipAddress,
72  			Collection<Integer> boards, Integer portNumber) {
73  		this.bmp = new BMPCoords(cabinet, frame);
74  		this.ipAddress = ipAddress;
75  		this.boards = boards.stream().map(BMPBoard::new)
76  				.collect(toUnmodifiableList());
77  		this.portNumber = portNumber;
78  	}
79  
80  	/**
81  	 * @param coords
82  	 *            The coordinates of the BMP.
83  	 * @param ipAddress
84  	 *            The address of the BMP.
85  	 * @param boards
86  	 *            The boards controlled by the BMP.
87  	 * @param portNumber
88  	 *            The BMP's port.
89  	 */
90  	public BMPConnectionData(BMPCoords coords, InetAddress ipAddress,
91  			Collection<Integer> boards, Integer portNumber) {
92  		this.bmp = coords;
93  		this.ipAddress = ipAddress;
94  		this.boards = boards.stream().map(BMPBoard::new)
95  				.collect(toUnmodifiableList());
96  		this.portNumber = portNumber;
97  	}
98  
99  	private static final int MIN_BYTE_FIELD = 3;
100 
101 	/**
102 	 * Work out the BMP connection IP address given the machine details. This is
103 	 * assumed to be the IP address of the machine, with 1 subtracted from the
104 	 * final part e.g. if the machine IP address is {@code 192.168.0.5}, the BMP
105 	 * IP address is assumed to be {@code 192.168.0.4}.
106 	 * <p>
107 	 * This algorithm is rather hokey. Far better for the user to simply know
108 	 * where the BMP actually is (this is necessary in any large deployment
109 	 * anyway).
110 	 *
111 	 * @param host
112 	 *            the SpiNNaker machine main host
113 	 * @param numBoards
114 	 *            the number of boards in the machine
115 	 * @throws UnknownHostException
116 	 *             If the IP address computations fail
117 	 * @throws IllegalArgumentException
118 	 *             If a host with an address that can't be related to a BMP is
119 	 *             given. Specifically, the given IP address can't end in
120 	 *             {@code .0} or {@code .1} or things will not work.
121 	 */
122 	public BMPConnectionData(InetAddress host, int numBoards)
123 			throws UnknownHostException {
124 		// take the IP address, split by dots, and subtract 1 off last bit
125 		var ipBits = host.getAddress();
126 		if (ipBits[MIN_BYTE_FIELD] == 0 || ipBits[MIN_BYTE_FIELD] == 1) {
127 			// Last digit of valid IP address can't really be 0 or 255
128 			throw new IllegalArgumentException(
129 					"BMP address would have illegal IP address");
130 		}
131 		ipBits[MIN_BYTE_FIELD]--;
132 		ipAddress = getByAddress(ipBits);
133 		portNumber = SCP_SCAMP_PORT;
134 
135 		// Assume a single board with no cabinet or frame specified
136 		bmp = new BMPCoords(0, 0);
137 
138 		// add board scope for each split
139 		// if null, the end user didn't enter anything, so assume one board
140 		// starting at position 0
141 		if (numBoards == 0) {
142 			boards = List.of(new BMPBoard(0));
143 		} else {
144 			boards = range(0, numBoards).mapToObj(BMPBoard::new)
145 					.collect(toUnmodifiableList());
146 		}
147 	}
148 
149 	@Override
150 	public String toString() {
151 		return format(
152 				"(c:%d,f:%d,b:%s; %s)", bmp.cabinet, bmp.frame,
153 				boards.stream().findFirst()
154 						.map(board -> board.board + "...").orElse("<empty>"),
155 				ipAddress);
156 	}
157 }
