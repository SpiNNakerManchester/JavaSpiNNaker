17  package uk.ac.manchester.spinnaker.messages.model;
18  
19  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
20  
21  /**
22   * FPGA registers within a register bank.
23   *
24   * @see <a href=
25   *      "https://github.com/SpiNNakerManchester/spio/tree/master/designs/spinnaker_fpgas">
26   *      Spio documentation</a>
27   * @author Donal Fellows
28   */
29  @SARKStruct("spio")
30  public enum FPGALinkRegisters {
31  	/** Version. {top 24: Module, bottom 8: Protocol} */
32  	VERS(0x00, 32),
33  	/** CRC error counter. */
34  	CRCE(0x04, 32),
35  	/** Frame error counter. */
36  	FRME(0x08, 32),
37  	/** Packet dispatcher busy counter. */
38  	BUSY(0x0C, 32),
39  	/** Local nack'd frame counter. */
40  	LNAK(0x10, 32),
41  	/** Remote nack counter. */
42  	RNAK(0x14, 32),
43  	/** Local ack'd frame counter. */
44  	LACK(0x18, 32),
45  	/** Remote ack counter. */
46  	RACK(0x1C, 32),
47  	/** Local out-of-credit counter. */
48  	LOOC(0x20, 32),
49  	/** Remote out-of-credit counter. */
50  	ROOC(0x24, 32),
51  	/** Credit. */
52  	CRDT(0x28, 3),
53  	/** Frame assembler valid sent frame counter. */
54  	SFRM(0x2C, 32),
55  	/** Frame transmitter frame counter. */
56  	TFRM(0x30, 32),
57  	/** Frame disassembler valid frame counter. */
58  	DFRM(0x34, 32),
59  	/** Packet dispatcher valid received frame counter. */
60  	RFRM(0x38, 32),
61  	/** Empty frame assembler queues. */
62  	EMPT(0x3C, 8),
63  	/** Full frame assembler queues. */
64  	FULL(0x40, 8),
65  	/** Local channel flow control status. */
66  	CFCL(0x44, 8),
67  	/** Remote channel flow control status. */
68  	CFCR(0x48, 8),
69  	/** IDle Sentinel Output value. (Sent in idle frames.) */
70  	IDSO(0x4C, true, 16),
71  	/** IDle Sentinel Input value. (Latest received) */
72  	IDSI(0x50, 16),
73  	/** Handshake. {bit 1: version err, bit 0: complete} */
74  	HAND(0x54, 2),
75  	/** Link reconnection (re-handshake) counter. */
76  	RECO(0x58, 32),
77  	/** 1 = Stop sending data frames. (NB: Will still receive them) */
78  	STOP(0x5C, true, 1);
79  
80  	/** The base addresses of register banks are all multiples of this. */
81  	public static final int BANK_OFFSET_MULTIPLIER = 0x00010000;
82  
83  	/** The offset of the register within the bank of registers. */
84  	public final int offset;
85  
86  	/** Whether this is a writable register. */
87  	public final boolean writable;
88  
89  	/** Size of register, in bits. */
90  	public final int size;
91  
92  	FPGALinkRegisters(int offset, int size) {
93  		this(offset, false, size);
94  	}
95  
96  	FPGALinkRegisters(int offset, boolean writable, int size) {
97  		this.offset = offset;
98  		this.writable = writable;
99  		this.size = size;
100 	}
101 
102 	/**
103 	 * Compute the address of the register in a particular register bank.
104 	 *
105 	 * @param registerBank
106 	 *            Which register bank. Must be 0, 1, or 2.
107 	 * @return The address in the FPGA's address space.
108 	 * @throws IllegalArgumentException
109 	 *             If a bad register bank is given.
110 	 */
111 	public MemoryLocation address(int registerBank) {
112 		if (registerBank < 0 || registerBank > 2) {
113 			throw new IllegalArgumentException(
114 					"registerBank must be 0, 1, or 2");
115 		}
116 		return new MemoryLocation(
117 				BANK_OFFSET_MULTIPLIER * registerBank + offset);
118 	}
119 }
