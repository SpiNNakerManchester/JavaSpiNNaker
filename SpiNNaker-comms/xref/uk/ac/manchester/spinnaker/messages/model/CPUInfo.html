102 	private static final int NUM_REGISTERS = 8;
103 
104 	private static final int APP_NAME_WIDTH = 16;
105 
106 	private static final int SKIP_BYTES = 16;
107 
108 	private static final int NUM_USER_VALUES = 4;
109 
110 	private static int[] getInts(ByteBuffer buffer, int fieldLength) {
111 		var data = new int[fieldLength];
112 		buffer.asIntBuffer().get(data);
113 		buffer.position(buffer.position() + fieldLength * WORD_SIZE);
114 		return data;
115 	}
116 
117 	private static String getStr(ByteBuffer buffer, int fieldLength) {
118 		var data = new byte[fieldLength];
119 		buffer.get(data);
120 		int len = 0;
121 		// Trim from first trailing NUL
122 		for (byte b : data) {
123 			if (b == 0) {
124 				break;
125 			}
126 			len++;
127 		}
128 		return new String(data, 0, len, US_ASCII);
129 	}
130 
131 	/**
132 	 * @param location
133 	 *            Which core was queried for the information.
134 	 * @param buffer
135 	 *            The data received from SDRAM on the board, in a little-endian
136 	 *            buffer.
137 	 */
138 	public CPUInfo(HasCoreLocation location, ByteBuffer buffer) {
139 		core = location.asCoreLocation();
140 		registers = getInts(buffer, NUM_REGISTERS);
141 		processorStateRegister = buffer.getInt();
142 		stackPointer = buffer.getInt();
143 		linkRegister = buffer.getInt();
144 		runTimeError = RunTimeError.get(toUnsignedInt(buffer.get()));
145 		physicalCPUID = buffer.get();
146 		state = CPUState.get(buffer.get());
147 		applicationID = toUnsignedInt(buffer.get());
148 		applicationMailboxDataAddress = buffer.getInt();
149 		monitorMailboxDataAddress = buffer.getInt();
150 		applicationMailboxCommand = MailboxCommand.get(buffer.get());
151 		monitorMailboxCommand = MailboxCommand.get(buffer.get());
152 		softwareErrorCount = buffer.getShort();
153 		softwareSourceFilenameAddress = buffer.getInt();
154 		softwareSourceLineNumber = buffer.getInt();
155 		time = buffer.getInt();
156 		applicationName = getStr(buffer, APP_NAME_WIDTH);
157 		iobufAddress = buffer.getInt();
158 		softwareVersion = buffer.getInt();
159 		buffer.position(buffer.position() + SKIP_BYTES);
160 		user = getInts(buffer, NUM_USER_VALUES);
161 	}

308 	/**
309 	 * @return A description of the state.
310 	 */
311 	public String getStatusDescription() {
312 		if (state != RUN_TIME_EXCEPTION) {
313 			return format("    %d:%d:%d in state %s", core.getX(), core.getY(),
314 					core.getP(), state);
315 		}
316 		var sb = new StringBuilder();
317 		sb.append(format("    %d:%d:%d in state %s:%s\n", core.getX(),
318 				core.getY(), core.getP(), state, runTimeError));
319 		sb.append(format("        r0=%08x, r1=%08x, r2=%08x, r3=%08x\n",
320 				r0.get(registers), r1.get(registers), r2.get(registers),
321 				r3.get(registers)));
322 		sb.append(format("        r4=%08x, r5=%08x, r6=%08x, r7=%08x\n",
323 				r4.get(registers), r5.get(registers), r6.get(registers),
324 				r7.get(registers)));
325 		sb.append(format("        PSR=%08x, SP=%08x, LR=%08x",
326 				processorStateRegister, stackPointer, linkRegister));
327 		return sb.toString();
328 	}
