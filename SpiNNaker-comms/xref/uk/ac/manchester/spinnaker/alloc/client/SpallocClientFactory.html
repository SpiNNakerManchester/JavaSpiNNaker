169 	/**
170 	 * Outputs a form to a connection in
171 	 * {@code application/x-www-form-urlencoded} format.
172 	 *
173 	 * @param connection
174 	 *            The connection. Must have the right verb set.
175 	 * @param map
176 	 *            The contents of the form.
177 	 * @throws IOException
178 	 *             If I/O fails.
179 	 */
180 	static void writeForm(HttpURLConnection connection, Map<String, String> map)
181 			throws IOException {
182 		var form = map.entrySet().stream()
183 				.map(e -> e.getKey() + "=" + encode(e.getValue(), UTF_8))
184 				.collect(joining("&"));
185 
186 		connection.setDoOutput(true);
187 		connection.setRequestProperty(CONTENT_TYPE, FORM_ENCODED);
188 		try (var w =
189 				new OutputStreamWriter(connection.getOutputStream(), UTF_8)) {
190 			w.write(form);
191 		}
192 	}

233 	/**
234 	 * Checks for errors in the response.
235 	 *
236 	 * @param conn
237 	 *            The HTTP connection
238 	 * @param errorMessage
239 	 *            The message to use on error (describes what did not work at a
240 	 *            higher level)
241 	 * @return The input stream so any non-error response content can be
242 	 *         obtained.
243 	 * @throws IOException
244 	 *             If things go wrong with comms.
245 	 * @throws FileNotFoundException
246 	 *             on a {@link HttpURLConnection#HTTP_NOT_FOUND}
247 	 * @throws SpallocException
248 	 *             on other server errors
249 	 */
250 	@MustBeClosed
251 	static InputStream checkForError(HttpURLConnection conn,
252 			String errorMessage) throws IOException {
253 		if (conn.getResponseCode() == HTTP_NOT_FOUND) {
254 			// Special case
255 			throw new FileNotFoundException(errorMessage);
256 		}
257 		if (conn.getResponseCode() >= HTTP_BAD_REQUEST) {
258 			throw new SpallocException(conn.getErrorStream(),
259 					conn.getResponseCode());
260 		}
261 		return conn.getInputStream();
262 	}

314 		final Machine getMachine(String name) throws IOException {
315 			Machine m = MACHINE_MAP.get(name);
316 			if (m == null) {
317 				client.listMachines();
318 				m = MACHINE_MAP.get(name);
319 			}
320 			if (m == null) {
321 				throw new IOException("Machine " + name + " not found");
322 			}
323 			return m;
324 		}

326 		private WhereIs whereis(HttpURLConnection conn) throws IOException {
327 			try (var is = checkForError(conn,
328 					"couldn't get board information")) {
329 				if (conn.getResponseCode() == HTTP_NO_CONTENT) {
330 					throw new FileNotFoundException("machine not allocated");
331 				}
332 				return readJson(is, WhereIs.class);
333 			} finally {
334 				s.trackCookie(conn);
335 			}
336 		}

338 		final WhereIs whereis(URI uri) throws IOException {
339 			return s.withRenewal(() -> {
340 				var conn = s.connection(uri);
341 				var w = whereis(conn);
342 				w.setMachineHandle(getMachine(w.getMachineName()));
343 				w.clearMachineRef();
344 				return w;
345 			});
346 		}

369 		/**
370 		 * Slightly convoluted class to fetch jobs. The complication means we
371 		 * get the initial failure exception nice and early, while we're ready
372 		 * for it. This code would be quite a lot simpler if we didn't want to
373 		 * get the exception during construction.
374 		 */
375 		private class JobLister extends ListFetchingIter<URI> {
376 			private URI next;
377 
378 			private List<URI> first;
379 
380 			JobLister(URI initial) throws IOException {
381 				var first = getJobList(s.connection(initial));
382 				next = first.next;
383 				this.first = first.jobs;
384 			}
385 
386 			private Jobs getJobList(HttpURLConnection conn) throws IOException {
387 				try (var is = checkForError(conn, "couldn't list jobs")) {
388 					return readJson(is, Jobs.class);
389 				} finally {
390 					s.trackCookie(conn);
391 				}
392 			}
393 
394 			@Override
395 			List<URI> fetchNext() throws IOException {
396 				if (nonNull(first)) {
397 					try {
398 						return first;
399 					} finally {
400 						first = null;
401 					}
402 				}
403 				var j = getJobList(s.connection(next));
404 				next = j.next;
405 				return j.jobs;
406 			}
407 
408 			@Override
409 			boolean canFetchMore() {
410 				if (nonNull(first)) {
411 					return true;
412 				}
413 				return nonNull(next);
414 			}
415 		}
416 
417 		private Stream<Job> listJobs(URI flags) throws IOException {
418 			var basicData = new JobLister(
419 					nonNull(flags) ? jobs.resolve(flags) : jobs);
420 			return basicData.stream().flatMap(Collection::stream)
421 					.map(this::job);
422 		}

439 		@Override
440 		public Job createJob(CreateJob createInstructions) throws IOException {
441 			var uri = s.withRenewal(() -> {
442 				var conn = s.connection(jobs, true);
443 				writeObject(conn, createInstructions);
444 				// Get the response entity... and discard it
445 				try (var is = checkForError(conn, "job create failed")) {
446 					readLines(is, UTF_8);
447 					// But we do want the Location header
448 					return URI.create(conn.getHeaderField("Location"));
449 				} finally {
450 					s.trackCookie(conn);
451 				}
452 			});
453 			return job(uri);
454 		}
455 
456 		Job job(URI uri) {
457 			return new JobImpl(this, s, asDir(uri));
458 		}
459 
460 		@Override
461 		public List<Machine> listMachines() throws IOException {
462 			return s.withRenewal(() -> {
463 				var conn = s.connection(machines);
464 				try (var is = checkForError(conn, "list machines failed")) {
465 					var ms = readJson(is, Machines.class);
466 					// Assume we can cache this
467 					for (var bmd : ms.machines) {
468 						log.debug("Machine {} found", bmd.name);
469 						MACHINE_MAP.put(bmd.name,
470 								new MachineImpl(this, s, bmd));
471 					}
472 					return new ArrayList<Machine>(MACHINE_MAP.values());
473 				} finally {
474 					s.trackCookie(conn);
475 				}
476 			});
477 		}

480 	private static final class JobImpl extends Common implements Job {
481 		private final URI uri;
482 
483 		private volatile boolean dead;
484 
485 		@GuardedBy("lock")
486 		private ProxyProtocolClient proxy;
487 
488 		private final Object lock = new Object();
489 
490 		JobImpl(SpallocClient client, Session session, URI uri) {
491 			super(client, session);
492 			this.uri = uri;
493 			this.dead = false;
494 			startKeepalive();
495 		}
496 
497 		@Override
498 		public JobDescription describe(boolean wait) throws IOException {
499 			return s.withRenewal(() -> {
500 				var conn = wait ? s.connection(uri, WAIT_FLAG)
501 						: s.connection(uri);
502 				try (var is = checkForError(conn, "couldn't get job state")) {
503 					return readJson(is, JobDescription.class);
504 				} finally {
505 					s.trackCookie(conn);
506 				}
507 			});
508 		}
509 
510 		@Override
511 		public void keepalive() throws IOException {
512 			s.withRenewal(() -> {
513 				var conn = s.connection(uri, KEEPALIVE, true);
514 				conn.setRequestMethod("PUT");
515 				writeString(conn, "alive");
516 				try (var is = checkForError(conn, "couldn't keep job alive")) {
517 					return readLines(is, UTF_8);
518 					// Ignore the output
519 				} finally {
520 					s.trackCookie(conn);
521 				}
522 			});
523 		}
524 
525 		private static final int DELAY = 20 * MSEC_PER_SEC;
526 
527 		private void startKeepalive() {
528 			if (dead) {
529 				throw new IllegalStateException("job is already deleted");
530 			}
531 			var t = new Daemon(() -> {
532 				try {
533 					while (true) {
534 						sleep(DELAY);
535 						if (dead) {
536 							break;
537 						}
538 						keepalive();
539 					}
540 				} catch (IOException e) {
541 					log.warn("failed to keep job alive for {}", this, e);
542 				} catch (InterruptedException e) {
543 					// If interrupted, we're simply done
544 				}
545 			});
546 			t.setName("keepalive for " + uri);
547 			t.setUncaughtExceptionHandler((th, e) -> {
548 				log.warn("unexpected exception in {}", th, e);
549 			});
550 			t.start();
551 		}
552 
553 		@Override
554 		public void delete(String reason) throws IOException {
555 			dead = true;
556 			s.withRenewal(() -> {
557 				var conn = s.connection(uri, "?reason=" + encode(reason, UTF_8),
558 						true);
559 				conn.setRequestMethod("DELETE");
560 				try (var is = checkForError(conn, "couldn't delete job")) {
561 					readLines(is, UTF_8);
562 					// Ignore the output
563 				} finally {
564 					s.trackCookie(conn);
565 				}
566 				return this;
567 			});
568 			synchronized (lock) {
569 				if (haveProxy()) {
570 					proxy.close();
571 					proxy = null;
572 				}
573 			}
574 		}

576 		@Override
577 		public AllocatedMachine machine() throws IOException {
578 			var am = s.withRenewal(() -> {
579 				var conn = s.connection(uri, MACHINE);
580 				try (var is = checkForError(conn,
581 						"couldn't get allocation description")) {
582 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
583 						throw new IOException("machine not allocated");
584 					}
585 					return readJson(is, AllocatedMachine.class);
586 				} finally {
587 					s.trackCookie(conn);
588 				}
589 			});
590 			am.setMachine(getMachine(am.getMachineName()));
591 			return am;
592 		}

594 		@Override
595 		public boolean getPower() throws IOException {
596 			return s.withRenewal(() -> {
597 				var conn = s.connection(uri, POWER);
598 				try (var is = checkForError(conn, "couldn't get power state")) {
599 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
600 						throw new IOException("machine not allocated");
601 					}
602 					return "ON".equals(readJson(is, Power.class).power);
603 				} finally {
604 					s.trackCookie(conn);
605 				}
606 			});
607 		}
608 
609 		@Override
610 		public boolean setPower(boolean switchOn) throws IOException {
611 			var power = new Power();
612 			power.power = (switchOn ? "ON" : "OFF");
613 			boolean powered = s.withRenewal(() -> {
614 				var conn = s.connection(uri, POWER, true);
615 				conn.setRequestMethod("PUT");
616 				writeObject(conn, power);
617 				try (var is = checkForError(conn, "couldn't set power state")) {
618 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
619 						throw new IOException("machine not allocated");
620 					}
621 					return "ON".equals(readJson(is, Power.class).power);
622 				} finally {
623 					s.trackCookie(conn);
624 				}
625 			});
626 			if (!powered) {
627 				// If someone turns the power off, close the proxy
628 				synchronized (lock) {
629 					if (haveProxy()) {
630 						proxy.close();
631 						proxy = null;
632 					}
633 				}
634 			}
635 			return powered;
636 		}

644 		@GuardedBy("lock")
645 		private boolean haveProxy() {
646 			return nonNull(proxy) && proxy.isOpen();
647 		}
648 
649 		/**
650 		 * @return The websocket-based proxy.
651 		 * @throws IOException
652 		 *             if we can't connect
653 		 * @throws InterruptedException
654 		 *             if we're interrupted while connecting
655 		 */
656 		private ProxyProtocolClient getProxy()
657 				throws IOException, InterruptedException {
658 			synchronized (lock) {
659 				if (haveProxy()) {
660 					return proxy;
661 				}
662 			}
663 			var wssAddr = describe(false).getProxyAddress();
664 			if (isNull(wssAddr)) {
665 				throw new IOException("machine not allocated");
666 			}
667 			synchronized (lock) {
668 				if (!haveProxy()) {
669 					proxy = s.withRenewal(() -> s.websocket(wssAddr));
670 				}
671 				return proxy;
672 			}
673 		}
674 
675 		@MustBeClosed
676 		@Override
677 		public TransceiverInterface getTransceiver()
678 				throws IOException, InterruptedException, SpinnmanException {
679 			var ws = getProxy();
680 			var am = machine();
681 			// TODO: We should know if the machine wraps around or not here...
682 			var version = MachineVersion.TRIAD_NO_WRAPAROUND;
683 			var conns = new ArrayList<Connection>();
684 			var hostToChip = new HashMap<Inet4Address, ChipLocation>();
685 			InetAddress bootChipAddress = null;
686 			for (var bc : am.getConnections()) {
687 				var chipAddr = getByNameQuietly(bc.getHostname());
688 				var chipLoc = bc.getChip().asChipLocation();
689 				conns.add(new ProxiedSCPConnection(chipLoc, ws, chipAddr));
690 				hostToChip.put(chipAddr, bc.getChip());
691 				if (chipLoc.equals(ZERO_ZERO)) {
692 					bootChipAddress = chipAddr;
693 				}
694 			}
695 			if (bootChipAddress != null) {
696 				conns.add(new ProxiedBootConnection(ws, bootChipAddress));
697 			}
698 			return new ProxiedTransceiver(version, conns, hostToChip, ws);
699 		}
