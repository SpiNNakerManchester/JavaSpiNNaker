155 	/**
156 	 * Read an object from a stream.
157 	 *
158 	 * @param <T>
159 	 *            The type of the object to read.
160 	 * @param is
161 	 *            The stream
162 	 * @param cls
163 	 *            The class of object to read.
164 	 * @return The object
165 	 * @throws IOException
166 	 *             If an I/O error happens or the content on the stream can't be
167 	 *             made into an instance of the given class.
168 	 */
169 	static <T> T readJson(InputStream is, Class<T> cls) throws IOException {
170 		BufferedReader streamReader = new BufferedReader(
171 				new InputStreamReader(is, "UTF-8"));
172 		StringBuilder responseStrBuilder = new StringBuilder();
173 
174 		String inputStr;
175 		while ((inputStr = streamReader.readLine()) != null) {
176 			responseStrBuilder.append(inputStr);
177 		}
178 		String json = responseStrBuilder.toString();
179 
180 		try {
181 			return JSON_MAPPER.readValue(json, cls);
182 		} catch (IOException e) {
183 			log.error("Error while reading json {}", json);
184 			throw e;
185 		}
186 	}

188 	/**
189 	 * Outputs a form to a connection in
190 	 * {@code application/x-www-form-urlencoded} format.
191 	 *
192 	 * @param connection
193 	 *            The connection. Must have the right verb set.
194 	 * @param map
195 	 *            The contents of the form.
196 	 * @throws IOException
197 	 *             If I/O fails.
198 	 */
199 	static void writeForm(HttpURLConnection connection, Map<String, String> map)
200 			throws IOException {
201 		var form = map.entrySet().stream()
202 				.map(e -> e.getKey() + "=" + encode(e.getValue(), UTF_8))
203 				.collect(joining("&"));
204 
205 		connection.setDoOutput(true);
206 		connection.setRequestProperty(CONTENT_TYPE, FORM_ENCODED);
207 		try (var w =
208 				new OutputStreamWriter(connection.getOutputStream(), UTF_8)) {
209 			w.write(form);
210 		}
211 	}

252 	/**
253 	 * Checks for errors in the response.
254 	 *
255 	 * @param conn
256 	 *            The HTTP connection
257 	 * @param errorMessage
258 	 *            The message to use on error (describes what did not work at a
259 	 *            higher level)
260 	 * @return The input stream so any non-error response content can be
261 	 *         obtained.
262 	 * @throws IOException
263 	 *             If things go wrong with comms.
264 	 * @throws FileNotFoundException
265 	 *             on a {@link HttpURLConnection#HTTP_NOT_FOUND}
266 	 * @throws SpallocException
267 	 *             on other server errors
268 	 */
269 	@MustBeClosed
270 	static InputStream checkForError(HttpURLConnection conn,
271 			String errorMessage) throws IOException {
272 		if (conn.getResponseCode() == HTTP_NOT_FOUND) {
273 			// Special case
274 			throw new FileNotFoundException(errorMessage);
275 		}
276 		if (conn.getResponseCode() >= HTTP_BAD_REQUEST) {
277 			throw new SpallocException(conn.getErrorStream(),
278 					conn.getResponseCode());
279 		}
280 		return conn.getInputStream();
281 	}

328 		Common(SpallocClient client, Session s) {
329 			this.client = client != null ? client : (SpallocClient) this;

333 		final Machine getMachine(String name) throws IOException {
334 			Machine m = MACHINE_MAP.get(name);
335 			if (m == null) {
336 				client.listMachines();
337 				m = MACHINE_MAP.get(name);
338 			}
339 			if (m == null) {
340 				throw new IOException("Machine " + name + " not found");
341 			}
342 			return m;
343 		}

345 		private WhereIs whereis(HttpURLConnection conn) throws IOException {
346 			try (var is = checkForError(conn,
347 					"couldn't get board information")) {
348 				if (conn.getResponseCode() == HTTP_NO_CONTENT) {
349 					throw new FileNotFoundException("machine not allocated");
350 				}
351 				return readJson(is, WhereIs.class);
352 			} finally {
353 				s.trackCookie(conn);
354 			}
355 		}
356 
357 		final WhereIs whereis(URI uri) throws IOException {
358 			return s.withRenewal(() -> {
359 				var conn = s.connection(uri);
360 				var w = whereis(conn);
361 				w.setMachineHandle(getMachine(w.getMachineName()));
362 				w.clearMachineRef();
363 				return w;
364 			});
365 		}

388 		/**
389 		 * Slightly convoluted class to fetch jobs. The complication means we
390 		 * get the initial failure exception nice and early, while we're ready
391 		 * for it. This code would be quite a lot simpler if we didn't want to
392 		 * get the exception during construction.
393 		 */
394 		private class JobLister extends ListFetchingIter<URI> {
395 			private URI next;
396 
397 			private List<URI> first;
398 
399 			JobLister(URI initial) throws IOException {
400 				var first = getJobList(s.connection(initial));
401 				next = first.next;
402 				this.first = first.jobs;
403 			}
404 
405 			private Jobs getJobList(HttpURLConnection conn) throws IOException {
406 				try (var is = checkForError(conn, "couldn't list jobs")) {
407 					return readJson(is, Jobs.class);
408 				} finally {
409 					s.trackCookie(conn);
410 				}
411 			}
412 
413 			@Override
414 			List<URI> fetchNext() throws IOException {
415 				if (nonNull(first)) {
416 					try {
417 						return first;
418 					} finally {
419 						first = null;
420 					}
421 				}
422 				var j = getJobList(s.connection(next));
423 				next = j.next;
424 				return j.jobs;
425 			}
426 
427 			@Override
428 			boolean canFetchMore() {
429 				if (nonNull(first)) {
430 					return true;
431 				}
432 				return nonNull(next);
433 			}
434 		}
435 
436 		private Stream<Job> listJobs(URI flags) throws IOException {
437 			var basicData = new JobLister(
438 					nonNull(flags) ? jobs.resolve(flags) : jobs);
439 			return basicData.stream().flatMap(Collection::stream)
440 					.map(this::job);
441 		}

446 		}

452 			if (wait) {

458 		@Override
459 		public Job createJob(CreateJob createInstructions) throws IOException {
460 			var uri = s.withRenewal(() -> {
461 				var conn = s.connection(jobs, true);
462 				writeObject(conn, createInstructions);
463 				// Get the response entity... and discard it
464 				try (var is = checkForError(conn, "job create failed")) {
465 					readLines(is, UTF_8);
466 					// But we do want the Location header
467 					return URI.create(conn.getHeaderField("Location"));
468 				} finally {
469 					s.trackCookie(conn);
470 				}
471 			});
472 			var job = job(uri);
473 			job.startKeepalive(
474 					createInstructions.getKeepaliveInterval().toMillis()
475 					/ KEEPALIVE_DIVIDER);
476 			return job;
477 		}
478 
479 		JobImpl job(URI uri) {
480 			return new JobImpl(this, s, asDir(uri));
481 		}
482 
483 		@Override
484 		public List<Machine> listMachines() throws IOException {
485 			return s.withRenewal(() -> {
486 				var conn = s.connection(machines);
487 				try (var is = checkForError(conn, "list machines failed")) {
488 					var ms = readJson(is, Machines.class);
489 					// Assume we can cache this
490 					for (var bmd : ms.machines) {
491 						log.debug("Machine {} found", bmd.name);
492 						MACHINE_MAP.put(bmd.name,
493 								new MachineImpl(this, s, bmd));
494 					}
495 					return new ArrayList<Machine>(MACHINE_MAP.values());
496 				} finally {
497 					s.trackCookie(conn);
498 				}
499 			});
500 		}
501 	}

503 	private static final class JobImpl extends Common implements Job {
504 		private final URI uri;
505 
506 		private volatile boolean dead;
507 
508 		@GuardedBy("lock")
509 		private ProxyProtocolClient proxy;
510 
511 		private final Object lock = new Object();
512 
513 		JobImpl(SpallocClient client, Session session, URI uri) {
514 			super(client, session);
515 			this.uri = uri;
516 			this.dead = false;
517 		}
518 
519 		@Override
520 		public JobDescription describe(boolean wait) throws IOException {
521 			return s.withRenewal(() -> {
522 				var conn = wait ? s.connection(uri, WAIT_FLAG)
523 						: s.connection(uri);
524 				try (var is = checkForError(conn, "couldn't get job state")) {
525 					return readJson(is, JobDescription.class);
526 				} finally {
527 					s.trackCookie(conn);
528 				}
529 			});
530 		}
531 
532 		@Override
533 		public void keepalive() throws IOException {
534 			s.withRenewal(() -> {
535 				var conn = s.connection(uri, KEEPALIVE, true);
536 				conn.setRequestMethod("PUT");
537 				writeString(conn, "alive");
538 				try (var is = checkForError(conn, "couldn't keep job alive")) {
539 					return readLines(is, UTF_8);
540 					// Ignore the output
541 				} finally {
542 					s.trackCookie(conn);
543 				}
544 			});
545 		}
546 
547 		public void startKeepalive(long delayMs) {
548 			if (dead) {
549 				throw new IllegalStateException("job is already deleted");
550 			}
551 			var t = new Daemon(() -> {
552 				try {
553 					while (true) {
554 						sleep(delayMs);
555 						if (dead) {
556 							break;
557 						}
558 						keepalive();
559 					}
560 				} catch (IOException e) {
561 					log.warn("failed to keep job alive for {}", this, e);
562 				} catch (InterruptedException e) {
563 					// If interrupted, we're simply done
564 				}
565 			});
566 			t.setName("keepalive for " + uri);
567 			t.setUncaughtExceptionHandler((th, e) -> {
568 				log.warn("unexpected exception in {}", th, e);
569 			});
570 			t.start();
571 		}
572 
573 		@Override
574 		public void delete(String reason) throws IOException {
575 			dead = true;
576 			s.withRenewal(() -> {
577 				var conn = s.connection(uri, "?reason=" + encode(reason, UTF_8),
578 						true);
579 				conn.setRequestMethod("DELETE");
580 				try (var is = checkForError(conn, "couldn't delete job")) {
581 					readLines(is, UTF_8);
582 					// Ignore the output
583 				} finally {
584 					s.trackCookie(conn);
585 				}
586 				return this;
587 			});
588 			synchronized (lock) {
589 				if (haveProxy()) {
590 					proxy.close();
591 					proxy = null;
592 				}
593 			}
594 		}

596 		@Override
597 		public AllocatedMachine machine() throws IOException {
598 			var am = s.withRenewal(() -> {
599 				var conn = s.connection(uri, MACHINE);
600 				try (var is = checkForError(conn,
601 						"couldn't get allocation description")) {
602 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
603 						throw new IOException("machine not allocated");
604 					}
605 					return readJson(is, AllocatedMachine.class);
606 				} finally {
607 					s.trackCookie(conn);
608 				}
609 			});
610 			am.setMachine(getMachine(am.getMachineName()));
611 			return am;
612 		}
613 
614 		@Override
615 		public boolean getPower() throws IOException {
616 			return s.withRenewal(() -> {
617 				var conn = s.connection(uri, POWER);
618 				try (var is = checkForError(conn, "couldn't get power state")) {
619 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
620 						throw new IOException("machine not allocated");
621 					}
622 					return "ON".equals(readJson(is, Power.class).power);
623 				} finally {
624 					s.trackCookie(conn);
625 				}
626 			});
627 		}
628 
629 		@Override
630 		public boolean setPower(boolean switchOn) throws IOException {
631 			var power = new Power();
632 			power.power = (switchOn ? "ON" : "OFF");
633 			boolean powered = s.withRenewal(() -> {
634 				var conn = s.connection(uri, POWER, true);
635 				conn.setRequestMethod("PUT");
636 				writeObject(conn, power);
637 				try (var is = checkForError(conn, "couldn't set power state")) {
638 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
639 						throw new IOException("machine not allocated");
640 					}
641 					return "ON".equals(readJson(is, Power.class).power);
642 				} finally {
643 					s.trackCookie(conn);
644 				}
645 			});
646 			if (!powered) {
647 				// If someone turns the power off, close the proxy
648 				synchronized (lock) {
649 					if (haveProxy()) {
650 						proxy.close();
651 						proxy = null;
652 					}
653 				}
654 			}
655 			return powered;
656 		}

669 		/**
670 		 * @return The websocket-based proxy.
671 		 * @throws IOException
672 		 *             if we can't connect
673 		 * @throws InterruptedException
674 		 *             if we're interrupted while connecting
675 		 */
676 		private ProxyProtocolClient getProxy()
677 				throws IOException, InterruptedException {
678 			synchronized (lock) {
679 				if (haveProxy()) {
680 					return proxy;
681 				}
682 			}
683 			var wssAddr = describe(false).getProxyAddress();
684 			if (isNull(wssAddr)) {
685 				throw new IOException("machine not allocated");
686 			}
687 			synchronized (lock) {
688 				if (!haveProxy()) {
689 					proxy = s.withRenewal(() -> s.websocket(wssAddr));
690 				}
691 				return proxy;
692 			}
693 		}
694 
695 		@MustBeClosed
696 		@Override
697 		public TransceiverInterface getTransceiver()
698 				throws IOException, InterruptedException, SpinnmanException {
699 			var ws = getProxy();
700 			var am = machine();
701 			// TODO: We should know if the machine wraps around or not here...
702 			var version = MachineVersion.TRIAD_NO_WRAPAROUND;
703 			var conns = new ArrayList<Connection>();
704 			var hostToChip = new HashMap<Inet4Address, ChipLocation>();
705 			InetAddress bootChipAddress = null;
706 			for (var bc : am.getConnections()) {
707 				var chipAddr = getByNameQuietly(bc.getHostname());
708 				var chipLoc = bc.getChip().asChipLocation();
709 				conns.add(new ProxiedSCPConnection(chipLoc, ws, chipAddr));
710 				hostToChip.put(chipAddr, bc.getChip());
711 				if (chipLoc.equals(ZERO_ZERO)) {
712 					bootChipAddress = chipAddr;
713 				}
714 			}
715 			if (bootChipAddress != null) {
716 				conns.add(new ProxiedBootConnection(ws, bootChipAddress));
717 			}
718 			return new ProxiedTransceiver(version, conns, hostToChip, ws);
719 		}

723 			return "Job(" + uri + ")";
724 		}
