16  package uk.ac.manchester.spinnaker.protocols.download;
17  
18  import static java.nio.ByteBuffer.allocate;
19  import static java.nio.ByteOrder.LITTLE_ENDIAN;
20  import static uk.ac.manchester.spinnaker.protocols.download.GatherProtocolMessage.ID.START_SENDING_DATA;
21  import static uk.ac.manchester.spinnaker.messages.Constants.WORD_SIZE;
22  import static uk.ac.manchester.spinnaker.messages.sdp.SDPPort.EXTRA_MONITOR_CORE_DATA_SPEED_UP;
23  
24  import java.nio.ByteBuffer;
25  
26  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
27  import uk.ac.manchester.spinnaker.machine.MemoryLocation;
28  
29  /**
30   * A message used to request fast data transfer from SpiNNaker to Host.
31   */
32  public final class StartSendingMessage extends GatherProtocolMessage {
33  	private static final int NUM_WORDS = 4;
34  
35  	/**
36  	 * Create a message used to request fast data transfer from SpiNNaker to
37  	 * Host.
38  	 *
39  	 * @param destination
40  	 *            Where to send the message
41  	 * @param address
42  	 *            Where to start reading from
43  	 * @param length
44  	 *            How many bytes to read
45  	 * @param transactionId
46  	 *            the transaction ID needed
47  	 * @return The created message.
48  	 */
49  	static StartSendingMessage create(HasCoreLocation destination,
50  			MemoryLocation address, int length, int transactionId) {
51  		var payload = allocate(NUM_WORDS * WORD_SIZE).order(LITTLE_ENDIAN);
52  		payload.putInt(START_SENDING_DATA.value);
53  		payload.putInt(transactionId);
54  		payload.putInt(address.address);
55  		payload.putInt(length);
56  		payload.flip();
57  		return new StartSendingMessage(destination, payload);
58  	}
59  
60  	private StartSendingMessage(HasCoreLocation destination,
61  			ByteBuffer payload) {
62  		super(destination, EXTRA_MONITOR_CORE_DATA_SPEED_UP, payload);
63  	}
64  }
