29  // TODO refactor this to have the functionality exposed higher up
30  /**
31   * A simple wrapper round the basic underlying connection process system.
32   *
33   * @author Donal Fellows
34   */
35  class SendSingleSCPCommandProcess
36  		extends MultiConnectionProcess<SCPConnection> {
37  	/**
38  	 * @param connectionSelector
39  	 *            How to select which connection to use for communication.
40  	 * @param retryTracker
41  	 *            Object used to track how many retries were used in an
42  	 *            operation. May be {@code null} if no suck tracking is
43  	 *            required.
44  	 */
45  	SendSingleSCPCommandProcess(
46  			ConnectionSelector<SCPConnection> connectionSelector,
47  			RetryTracker retryTracker) {
48  		super(connectionSelector, SCP_RETRIES, SCP_TIMEOUT,
49  				DEFAULT_NUM_CHANNELS, DEFAULT_INTERMEDIATE_CHANNEL_WAITS,
50  				retryTracker);
51  	}
52  
53  	/**
54  	 * Execute a call of a request and get a response from it.
55  	 *
56  	 * @param <T>
57  	 *            The type of the response.
58  	 * @param request
59  	 *            The request to make
60  	 * @return The response to the request
61  	 * @throws IOException
62  	 *             If communications fail.
63  	 * @throws ProcessException
64  	 *             If SCAMP on SpiNNaker reports a failure.
65  	 */
66  	public <T extends CheckOKResponse> T execute(SCPRequest<T> request)
67  			throws IOException, ProcessException {
68  		return synchronousCall(request);
69  	}
70  }
