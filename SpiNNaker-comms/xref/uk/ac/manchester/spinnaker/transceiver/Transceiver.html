171 /**
172  * An encapsulation of various communications with the SpiNNaker board. Acts as
173  * a Fa&ccedil;ade for most of the rest of this package.
174  * <p>
175  * The methods of this class are designed to be thread-safe; thus you can make
176  * multiple calls to the same (or different) methods from multiple threads and
177  * expect each call to work as if it had been called sequentially, although the
178  * order of returns is not guaranteed. Note also that with multiple connections
179  * to the board, using multiple threads in this way may result in an increase in
180  * the overall speed of operation, since the multiple calls may be made
181  * separately over the set of given connections.
182  * <p>
183  * For details of thread safety, see the methods annotated with
184  * {@link ParallelSafe}, {@link ParallelSafeWithCare} and {@link ParallelUnsafe}
185  * in {@link TransceiverInterface}. <em>Note that operations on an individual
186  * BMP are <strong>always</strong> parallel-unsafe, other documentation in this
187  * class notwithstanding; BMPs must only ever have one outstanding call made to
188  * them as they do not handle asynchronous calls at all well due to known
189  * firmware bugs.</em>
190  */

346 	/**
347 	 * Create a Transceiver by creating a UDPConnection to the given hostname on
348 	 * port 17893 (the default SCAMP port), and a BootConnection on port 54321
349 	 * (the default boot port), optionally discovering any additional links
350 	 * using the UDPConnection, and then returning the transceiver created with
351 	 * the conjunction of the created UDPConnection and the discovered
352 	 * connections.
353 	 *
354 	 * @param host
355 	 *            The host IP address of the board
356 	 * @param version
357 	 *            The type of SpiNNaker board used within the SpiNNaker machine
358 	 *            being used. May be {@code null} if the board in question can
359 	 *            be assumed to be always already booted.
360 	 * @param numberOfBoards
361 	 *            a number of boards expected to be supported, or {@code null},
362 	 *            which defaults to a single board
363 	 * @param ignoredChips
364 	 *            An optional set of chips to ignore in the machine. Requests
365 	 *            for a "machine" will have these chips excluded, as if they
366 	 *            never existed. The processors of the specified chips are
367 	 *            ignored.
368 	 * @param ignoredCores
369 	 *            An optional map of cores to ignore in the machine. Requests
370 	 *            for a "machine" will have these cores excluded, as if they
371 	 *            never existed.
372 	 * @param ignoredLinks
373 	 *            An optional set of links to ignore in the machine. Requests
374 	 *            for a "machine" will have these links excluded, as if they
375 	 *            never existed.
376 	 * @param bmpConnectionData
377 	 *            the details of the BMP connections used to boot multi-board
378 	 *            systems
379 	 * @param autodetectBMP
380 	 *            True if the BMP of version 4 or 5 boards should be
381 	 *            automatically determined from the board IP address
382 	 * @param bootPortNumber
383 	 *            the port number used to boot the machine
384 	 * @param scampConnections
385 	 *            the list of connections used for SCAMP communications
386 	 * @param maxSDRAMSize
387 	 *            the max size each chip can say it has for SDRAM (mainly used
388 	 *            in debugging purposes)
389 	 * @throws IOException
390 	 *             if networking fails
391 	 * @throws SpinnmanException
392 	 *             If a BMP is uncontactable or SpiNNaker rejects a message.
393 	 * @throws InterruptedException
394 	 *             If the communications were interrupted.
395 	 */
396 	@MustBeClosed
397 	@SuppressWarnings("MustBeClosed")
398 	public Transceiver(InetAddress host, MachineVersion version,
399 			Collection<BMPConnectionData> bmpConnectionData,
400 			Integer numberOfBoards, Set<ChipLocation> ignoredChips,
401 			Map<ChipLocation, Set<Integer>> ignoredCores,
402 			Map<ChipLocation, EnumSet<Direction>> ignoredLinks,
403 			boolean autodetectBMP, List<ConnectionDescriptor> scampConnections,
404 			Integer bootPortNumber, Integer maxSDRAMSize)
405 			throws IOException, SpinnmanException, InterruptedException {
406 		log.info("Creating transceiver for {}", requireNonNull(host,
407 				"SpiNNaker machine host name must be not null"));
408 		var connections = new ArrayList<Connection>();
409 
410 		/*
411 		 * if no BMP has been supplied, but the board is a spinn4 or a spinn5
412 		 * machine, then an assumption can be made that the BMP is at -1 on the
413 		 * final value of the IP address
414 		 */
415 		if (version != null && !version.isFourChip && autodetectBMP
416 				&& (bmpConnectionData == null || bmpConnectionData.isEmpty())) {
417 			bmpConnectionData =
418 					List.of(defaultBMPforMachine(host, numberOfBoards));
419 		}
420 
421 		// handle BMP connections
422 		if (bmpConnectionData != null) {
423 			var bmpIPs = new ArrayList<>();
424 			for (var connData : bmpConnectionData) {
425 				var connection = new BMPConnection(connData);
426 				connections.add(connection);
427 				bmpIPs.add(connection.getRemoteIPAddress());
428 			}
429 			log.info("Transceiver using BMPs: {}", bmpIPs);
430 		}
431 
432 		// handle the SpiNNaker connection
433 		if (scampConnections == null) {
434 			scampConnections = List.of();
435 		}
436 		if (scampConnections.isEmpty()) {
437 			connections.add(createScpConnection(BOOT_CHIP, host));
438 		}
439 
440 		// handle the boot connection
441 		connections.add(new BootConnection(null, null, host, bootPortNumber));
442 
443 		this.version = version;
444 		if (ignoredChips != null) {
445 			ignoreChips.addAll(ignoredChips);
446 		}
447 		if (ignoredCores != null) {
448 			ignoreCores.putAll(ignoredCores);
449 		}
450 		if (ignoredLinks != null) {
451 			ignoreLinks.putAll(ignoredLinks);
452 		}
453 		this.maxSDRAMSize = maxSDRAMSize;
454 
455 		originalConnections.addAll(connections);
456 		allConnections.addAll(connections);
457 		// if there has been SCAMP connections given, build them
458 		for (var desc : scampConnections) {
459 			if (desc.portNumber != null
460 					&& desc.portNumber != SCP_SCAMP_PORT) {
461 				log.warn("ignoring unexpected SCAMP port: {}",
462 						desc.portNumber);
463 			}
464 			connections.add(createScpConnection(desc.chip, desc.hostname));
465 		}
466 		for (Connection conn : connections) {
467 			identifyConnection(conn);
468 		}
469 		scpSelector = makeConnectionSelector();
470 		checkBMPConnections();
471 	}

713 			SystemVariableDefinition dataItem)
714 			throws IOException, ProcessException, InterruptedException {
715 		var buffer = readMemory(chip, SYS_VARS.add(dataItem.offset),
716 				dataItem.type.value);
717 		switch (dataItem.type) {
718 		case BYTE:
719 			return Byte.toUnsignedInt(buffer.get());

723 			return buffer.getInt();

731 			return new MemoryLocation(buffer.getInt());
732 		default:
733 			// Unreachable
734 			throw new IllegalStateException();
735 		}

782 	/** Check that the BMP connections are actually connected to valid BMPs. */
783 	private void checkBMPConnections()
784 			throws IOException, SpinnmanException, InterruptedException {
785 		/*
786 		 * Check that the UDP BMP conn is actually connected to a BMP via the
787 		 * SVER command
788 		 */
789 		for (var conn : bmpConnections) {
790 			// try to send a BMP SVER to check if it responds as expected
791 			try {
792 				var versionInfo = readBMPVersion(conn.getCoords(), conn.boards);
793 				if (!BMP_NAME.equals(versionInfo.name) || !BMP_MAJOR_VERSIONS
794 						.contains(versionInfo.versionNumber.majorVersion)) {
795 					throw new IOException(format(
796 							"The BMP at %s is running %s %s which is "
797 									+ "incompatible with this transceiver, "
798 									+ "required version is %s %s",
799 							conn.getRemoteIPAddress(), versionInfo.name,
800 							versionInfo.versionString, BMP_NAME,
801 							BMP_MAJOR_VERSIONS));
802 				}
803 
804 				log.info("Using BMP at {} with version {} {}",
805 						conn.getRemoteIPAddress(), versionInfo.name,
806 						versionInfo.versionString);
807 			} catch (SocketTimeoutException e) {
808 				/*
809 				 * If it fails to respond due to timeout, maybe that the
810 				 * connection isn't valid.
811 				 */
812 				throw new SpinnmanException(
813 						format("BMP connection to %s is not responding",
814 								conn.getRemoteIPAddress()),
815 						e);
816 			} catch (ProcessException e) {
817 				log.error("Failed to speak to BMP at {}",
818 						conn.getRemoteIPAddress(), e);
819 				throw e;
820 			}
821 		}
822 	}

932 	/**
933 	 * Find connections to the board and store these for future use. Note that
934 	 * connections can be empty, in which case another local discovery mechanism
935 	 * will be used. Note that an exception will be thrown if no initial
936 	 * connections can be found to the board.
937 	 *
938 	 * @return An iterable of discovered connections, not including the
939 	 *         initially given connections in the constructor
940 	 * @throws IOException
941 	 *             if networking fails
942 	 * @throws ProcessException
943 	 *             If SpiNNaker rejects a message.
944 	 * @throws InterruptedException
945 	 *             If the communications were interrupted.
946 	 */
947 	@CheckReturnValue
948 	@SuppressWarnings("MustBeClosed")
949 	public List<SCPConnection> discoverScampConnections()
950 			throws IOException, ProcessException, InterruptedException {
951 		/*
952 		 * Currently, this only finds other UDP connections given a connection
953 		 * that supports SCP - this is done via the machine
954 		 */
955 		if (scpConnections.isEmpty()) {
956 			return List.of();
957 		}
958 
959 		// Get the machine dimensions
960 		var dims = getMachineDimensions();
961 
962 		// Find all the new connections via the machine Ethernet-connected chips
963 		var newConnections = new ArrayList<SCPConnection>();
964 		for (var chip : getSpinn5Geometry().getPotentialRootChips(dims)) {
965 			var ipAddress = getByAddress(
966 					(byte[]) getSystemVariable(chip, ethernet_ip_address));
967 			if (udpScpConnections.containsKey(ipAddress)) {
968 				continue;
969 			}
970 			var conn = searchForProxies(chip);
971 
972 			// if no data, no proxy
973 			if (conn == null) {
974 				conn = createScpConnection(chip, ipAddress);
975 			} else {
976 				// proxy, needs an adjustment
977 				udpScpConnections.remove(conn.getRemoteIPAddress());
978 			}
979 
980 			// check if it works
981 			if (checkConnection(conn, chip)) {
982 				allConnections.add(conn);
983 				udpScpConnections.put(ipAddress, conn);
984 				scpConnections.add(conn);
985 				newConnections.add(conn);
986 			} else {
987 				log.warn(
988 						"Additional Ethernet connection on {} at "
989 								+ "chip {},{} cannot be contacted",
990 						ipAddress, chip.getX(), chip.getY());
991 			}
992 		}
993 
994 		// Update the connection queues after finding new connections
995 		return newConnections;
996 	}

1283 	/**
1284 	 * Try to detect if SCAMP is running, and if not, boot the machine.
1285 	 *
1286 	 * @param numAttempts
1287 	 *            how many attempts should be supported
1288 	 * @param extraBootValues
1289 	 *            Any additional values to set during boot
1290 	 * @return version info for SCAMP on the booted system
1291 	 * @throws IOException
1292 	 *             if networking fails
1293 	 * @throws ProcessException
1294 	 *             If SpiNNaker rejects a message.
1295 	 */
1296 	@CheckReturnValue
1297 	private VersionInfo findScampAndBoot(int numAttempts,
1298 			Map<SystemVariableDefinition, Object> extraBootValues)
1299 			throws InterruptedException, IOException, ProcessException {
1300 		VersionInfo versionInfo = null;
1301 		int triesLeft = numAttempts;
1302 		while (versionInfo == null && triesLeft > 0) {
1303 			try {
1304 				versionInfo = getScampVersion();
1305 				if (versionInfo.core.asChipLocation().equals(BOOT_CHIP)) {
1306 					versionInfo = null;
1307 					sleep(CONNECTION_CHECK_DELAY);
1308 				}
1309 			} catch (ProcessException e) {
1310 				if (e.getCause() instanceof SocketTimeoutException) {
1311 					log.info("Attempting to boot machine");
1312 					bootBoard(extraBootValues);
1313 					triesLeft--;
1314 				} else if (e.getCause() instanceof IOException) {
1315 					throw new IOException(
1316 							"Failed to communicate with the machine", e);
1317 				} else {
1318 					throw e;
1319 				}
1320 			} catch (SocketTimeoutException e) {
1321 				log.info("Attempting to boot machine");
1322 				bootBoard(extraBootValues);
1323 				triesLeft--;
1324 			} catch (IOException e) {
1325 				throw new IOException("Failed to communicate with the machine",
1326 						e);
1327 			}
1328 		}
1329 
1330 		// The last thing we tried was booting, so try again to get the version
1331 		if (versionInfo == null) {
1332 			versionInfo = getScampVersion();
1333 			if (versionInfo.core.asChipLocation().equals(BOOT_CHIP)) {
1334 				versionInfo = null;
1335 			}
1336 		}
1337 		if (versionInfo != null) {
1338 			log.info("Found board with hardware {} firmware {} version {}",
1339 					versionInfo.hardware, versionInfo.name,
1340 					versionInfo.versionNumber);
1341 		}
1342 		return versionInfo;
1343 	}

2149 	@ParallelSafeWithCare
2150 	public void waitForCoresToBeInState(CoreSubsets allCoreSubsets, AppID appID,
2151 			EnumSet<CPUState> cpuStates, Integer timeout, int timeBetweenPolls,
2152 			EnumSet<CPUState> errorStates, int countBetweenFullChecks)
2153 			throws IOException, InterruptedException, SpinnmanException {
2154 		// check that the right number of processors are in the states
2155 		int processorsReady = 0;
2156 		long timeoutTime =
2157 				currentTimeMillis() + (timeout == null ? 0 : timeout);
2158 		int tries = 0;
2159 		while (processorsReady < allCoreSubsets.size()
2160 				&& (timeout == null || currentTimeMillis() < timeoutTime)) {
2161 			// Get the number of processors in the ready states
2162 			processorsReady = 0;
2163 			for (var state : cpuStates) {
2164 				processorsReady += getCoreStateCount(appID, state);
2165 			}
2166 
2167 			// If the count is too small, check for error states
2168 			if (processorsReady < allCoreSubsets.size()) {
2169 				if (inErrorStates(appID, errorStates)) {
2170 					// Small chance that inErrorStates() is wrong
2171 					var errorCores =
2172 							getCoresInState(allCoreSubsets, errorStates);
2173 					if (!errorCores.isEmpty()) {
2174 						throw new CoresNotInStateException(timeout, cpuStates,
2175 								errorCores);
2176 					}
2177 				}
2178 
2179 				/*
2180 				 * If we haven't seen an error, increase the tries, and do a
2181 				 * full check if required
2182 				 */
2183 				if (++tries >= countBetweenFullChecks) {
2184 					var coresInState =
2185 							getCoresInState(allCoreSubsets, cpuStates);
2186 					processorsReady = coresInState.size();
2187 					tries = 0;
2188 				}
2189 
2190 				// If we're still not in the correct state, wait a bit
2191 				if (processorsReady < allCoreSubsets.size()) {
2192 					sleep(timeBetweenPolls);
2193 				}
2194 			}
2195 		}
2196 
2197 		// If we haven't reached the final state, do a final full check
2198 		if (processorsReady < allCoreSubsets.size()) {
2199 			var coresInState = getCoresInState(allCoreSubsets, cpuStates);
2200 
2201 			/*
2202 			 * If we are sure we haven't reached the final state, report a
2203 			 * timeout error
2204 			 */
2205 			if (coresInState.size() != allCoreSubsets.size()) {
2206 				throw new SocketTimeoutException(
2207 						format("waiting for cores %s to reach one of %s",
