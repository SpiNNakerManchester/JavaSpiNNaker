17  package uk.ac.manchester.spinnaker.transceiver;
18  
19  import static java.nio.ByteBuffer.allocate;
20  import static java.nio.ByteOrder.LITTLE_ENDIAN;
21  import static uk.ac.manchester.spinnaker.messages.Constants.CPU_INFO_BYTES;
22  import static uk.ac.manchester.spinnaker.messages.Constants.CPU_INFO_OFFSET;
23  
24  import java.io.IOException;
25  import java.net.InetAddress;
26  import java.net.UnknownHostException;
27  import java.nio.ByteBuffer;
28  
29  import uk.ac.manchester.spinnaker.connections.UDPConnection;
30  import uk.ac.manchester.spinnaker.machine.HasCoreLocation;
31  import uk.ac.manchester.spinnaker.messages.model.BMPConnectionData;
32  
33  /**
34   * Support utilities.
35   */
36  public abstract class Utils {
37  	/**
38  	 * The size of buffer to allocate for SpiNNaker messages.
39  	 */
40  	private static final int SPINNAKER_MESSAGE_BUFFER_SIZE = 300;
41  
42  	private Utils() {
43  	}
44  
45  	/**
46  	 * Work out the BMP connection IP address given the machine details. This is
47  	 * assumed to be the IP address of the machine, with 1 subtracted from the
48  	 * final part e.g. if the machine IP address is 192.168.0.5, the BMP IP
49  	 * address is assumed to be 192.168.0.4
50  	 *
51  	 * @param host
52  	 *            the SpiNNaker machine main host
53  	 * @param numberOfBoards
54  	 *            the number of boards in the machine
55  	 * @return The BMP connection data
56  	 * @throws UnknownHostException
57  	 *             If the IP address computations fail.
58  	 */
59  	public static BMPConnectionData defaultBMPforMachine(InetAddress host,
60  			Integer numberOfBoards) throws UnknownHostException {
61  		return new BMPConnectionData(host,
62  				numberOfBoards == null ? 0 : numberOfBoards);
63  	}
64  
65  	/**
66  	 * Get the address of the {@code vcpu_t} structure for the given core.
67  	 *
68  	 * @param p
69  	 *            The core number
70  	 * @return the address
71  	 */
72  	public static int getVcpuAddress(int p) {
73  		return CPU_INFO_OFFSET + (CPU_INFO_BYTES * p);
74  	}
75  
76  	/**
77  	 * Get the address of the {@code vcpu_t} structure for the given core.
78  	 *
79  	 * @param core
80  	 *            The core
81  	 * @return the address
82  	 */
83  	public static int getVcpuAddress(HasCoreLocation core) {
84  		return CPU_INFO_OFFSET + (CPU_INFO_BYTES * core.getP());
85  	}
86  
87  	/**
88  	 * Sends a port trigger message using a connection to (hopefully) open a
89  	 * port in a NAT and/or firewall to allow incoming packets to be received.
90  	 *
91  	 * @param connection
92  	 *            The UDP connection down which the trigger message should be
93  	 *            sent
94  	 * @param host
95  	 *            The address of the SpiNNaker board to which the message should
96  	 *            be sent
97  	 * @deprecated Call
98       *             {@link UDPConnection#sendPortTriggerMessage(InetAddress)}
99  	 *             directly instead.
100 	 * @throws IOException
101 	 *             If anything goes wrong
102 	 */
103 	@Deprecated
104 	public static void sendPortTriggerMessage(UDPConnection<?> connection,
105 			InetAddress host) throws IOException {
106 		connection.sendPortTriggerMessage(host);
107 	}
108 
109 	/**
110 	 * @return Get a new little-endian buffer sized suitably for SpiNNaker
111 	 *         messages.
112 	 */
113 	public static ByteBuffer newMessageBuffer() {
114 		// TODO How big should this buffer be? 256 or (256 + header size)?
115 		return allocate(SPINNAKER_MESSAGE_BUFFER_SIZE).order(LITTLE_ENDIAN);
116 	}
117 }
