32  
33  /**
34   * Get the CPU information structure for a set of processors.
35   */
36  class GetCPUInfoProcess extends TxrxProcess {
37  	/**
38  	 * @param connectionSelector
39  	 *            How to select how to communicate.
40  	 * @param retryTracker
41  	 *            Object used to track how many retries were used in an
42  	 *            operation. May be {@code null} if no suck tracking is
43  	 *            required.
44  	 */
45  	GetCPUInfoProcess(
46  			ConnectionSelector<? extends SCPConnection> connectionSelector,
47  			RetryTracker retryTracker) {
48  		super(connectionSelector, retryTracker);
49  	}
50  
51  	/**
52  	 * Get CPU information.
53  	 *
54  	 * @param coreSubsets
55  	 *            What processors to get the information from
56  	 * @return The CPU information, in undetermined order.
57  	 * @throws IOException
58  	 *             If anything goes wrong with networking.
59  	 * @throws ProcessException
60  	 *             If SpiNNaker rejects a message.
61  	 * @throws InterruptedException
62  	 *             If the communications were interrupted.
63  	 */
64  	MappableIterable<CPUInfo> getCPUInfo(CoreSubsets coreSubsets)
65  			throws IOException, ProcessException, InterruptedException {
66  		var cpuInfo = new ArrayList<CPUInfo>();
67  		for (var core : requireNonNull(coreSubsets,
68  				"must have actual core subset to iterate over")) {
69  			sendRequest(
70  					new ReadMemory(core.getScampCore(), getVcpuAddress(core),
71  							CPU_INFO_BYTES),
72  					response -> cpuInfo.add(new CPUInfo(core, response.data)));
73  		}
74  		finishBatch();
75  		return cpuInfo::iterator;
76  	}
77  }
