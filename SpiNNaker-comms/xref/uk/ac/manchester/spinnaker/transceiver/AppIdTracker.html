16  package uk.ac.manchester.spinnaker.transceiver;
17  
18  import static java.util.stream.Collectors.toSet;
19  import static java.util.stream.IntStream.rangeClosed;
20  
21  import java.util.Set;
22  
23  import uk.ac.manchester.spinnaker.messages.model.AppID;
24  
25  /** A tracker of application IDs to make it easier to allocate new IDs. */
26  public class AppIdTracker {
27  	private static final int MIN_APP_ID = 17;
28  
29  	private static final int MAX_APP_ID = 254;
30  
31  	private final Set<AppID> freeIDs;
32  
33  	private final int maxID;
34  
35  	private final int minID;
36  
37  	/**
38  	 * Allocate an application ID tracker.
39  	 */
40  	public AppIdTracker() {
41  		this(null, MIN_APP_ID, MAX_APP_ID);
42  	}
43  
44  	/**
45  	 * Allocate an application ID tracker.
46  	 *
47  	 * @param minAppID
48  	 *            The smallest application ID to use
49  	 * @param maxAppID
50  	 *            The largest application ID to use
51  	 */
52  	public AppIdTracker(int minAppID, int maxAppID) {
53  		this(null, minAppID, maxAppID);
54  	}
55  
56  	/**
57  	 * Allocate an application ID tracker.
58  	 *
59  	 * @param appIDsInUse
60  	 *            The IDs that are already in use
61  	 */
62  	public AppIdTracker(Set<AppID> appIDsInUse) {
63  		this(appIDsInUse, MIN_APP_ID, MAX_APP_ID);
64  	}
65  
66  	/**
67  	 * Allocate an application ID tracker.
68  	 *
69  	 * @param appIDsInUse
70  	 *            The IDs that are already in use
71  	 * @param minAppID
72  	 *            The smallest application ID to use
73  	 * @param maxAppID
74  	 *            The largest application ID to use
75  	 */
76  	public AppIdTracker(Set<AppID> appIDsInUse, int minAppID,
77  			int maxAppID) {
78  		freeIDs = rangeClosed(minAppID, maxAppID).mapToObj(AppID::new)
79  				.collect(toSet());
80  		if (appIDsInUse != null) {
81  			freeIDs.removeAll(appIDsInUse);
82  		}
83  		this.minID = minAppID;
84  		this.maxID = maxAppID;
85  	}
86  
87  	/**
88  	 * Get a new unallocated ID.
89  	 *
90  	 * @return The new ID, now allocated.
91  	 * @throws RuntimeException
92  	 *             if there are no IDs available
93  	 */
94  	public AppID allocateNewID() {
95  		var it = freeIDs.iterator();
96  		if (!it.hasNext()) {
97  			throw new RuntimeException("no remaining free IDs");
98  		}
99  		var val = it.next();
100 		it.remove();
101 		return val;
102 	}
103 
104 	/**
105 	 * Allocate a given ID.
106 	 *
107 	 * @param id
108 	 *            The ID to allocate.
109 	 * @throws IllegalArgumentException
110 	 *             if the ID is not present
111 	 */
112 	public void allocateID(AppID id) {
113 		if (!freeIDs.remove(id)) {
114 			throw new IllegalArgumentException(
115 					"id " + id + " was not available for allocation");
116 		}
117 	}
118 
119 	/**
120 	 * Free a given ID.
121 	 *
122 	 * @param id
123 	 *            The ID to free
124 	 * @throws IllegalArgumentException
125 	 *             if the ID is out of range
126 	 */
127 	public void freeID(AppID id) {
128 		if (id.appID < minID || id.appID > maxID) {
129 			throw new IllegalArgumentException(
130 					"ID " + id + " out of allowed range of " + minID
131 							+ " to " + maxID);
132 		}
133 		freeIDs.add(id);
134 	}
135 }
