17  package uk.ac.manchester.spinnaker.connections;
18  
19  import static java.lang.Thread.sleep;
20  import static java.nio.ByteBuffer.allocate;
21  import static java.nio.ByteOrder.BIG_ENDIAN;
22  import static uk.ac.manchester.spinnaker.connections.UDPConnection.TrafficClass.IPTOS_RELIABILITY;
23  import static uk.ac.manchester.spinnaker.messages.Constants.UDP_BOOT_CONNECTION_DEFAULT_PORT;
24  
25  import java.io.IOException;
26  import java.net.InetAddress;
27  import java.nio.ByteBuffer;
28  
29  import uk.ac.manchester.spinnaker.connections.model.BootReceiver;
30  import uk.ac.manchester.spinnaker.connections.model.BootSender;
31  import uk.ac.manchester.spinnaker.messages.boot.BootMessage;
32  
33  /** A connection to the SpiNNaker board that uses UDP to for booting. */
34  public class BootConnection extends UDPConnection<BootMessage>
35  		implements BootSender, BootReceiver {
36  	// Determined by Ethernet MTU, not by SDP buffer size
37  	private static final int BOOT_MESSAGE_SIZE = 1500;
38  
39  	private static final int ANTI_FLOOD_DELAY = 100;
40  
41  	/**
42  	 * Creates a boot connection.
43  	 *
44  	 * @param localHost
45  	 *            The local host to bind to. If {@code null} defaults to bind to
46  	 *            all interfaces, unless remoteHost is specified, in which case
47  	 *            binding is done to the IP address that will be used to send
48  	 *            packets.
49  	 * @param localPort
50  	 *            The local port to bind to, between 1025 and 32767. If
51  	 *            {@code null}, defaults to a random unused local port
52  	 * @param remoteHost
53  	 *            The remote host to send packets to. If {@code null}, the
54  	 *            socket will be available for listening only, and will throw
55  	 *            and exception if used for sending
56  	 * @param remotePort
57  	 *            The remote port to send packets to. If {@code null}, a default
58  	 *            value is used.
59  	 * @throws IOException
60  	 *             If there is an error setting up the communication channel
61  	 */
62  	public BootConnection(InetAddress localHost, Integer localPort,
63  			InetAddress remoteHost, Integer remotePort) throws IOException {
64  		super(localHost, localPort, remoteHost,
65  				remotePort == null ? UDP_BOOT_CONNECTION_DEFAULT_PORT
66  						: remotePort, IPTOS_RELIABILITY);
67  	}
68  
69  	/**
70  	 * Creates a boot connection that binds to all local interfaces on an
71  	 * arbitrary port from the range 1025 to 32767.
72  	 *
73  	 * @param remoteHost
74  	 *            The remote host to send packets to. If {@code null}, the
75  	 *            socket will be available for listening only, and will throw
76  	 *            and exception if used for sending
77  	 * @param remotePort
78  	 *            The remote port to send packets to. If {@code null}, a
79  	 *            default value is used.
80  	 * @throws IOException
81  	 *             If there is an error setting up the communication channel
82  	 */
83  	public BootConnection(InetAddress remoteHost, Integer remotePort)
84  			throws IOException {
85  		super(null, null, remoteHost,
86  				remotePort == null ? UDP_BOOT_CONNECTION_DEFAULT_PORT
87  						: remotePort, IPTOS_RELIABILITY);
88  	}
89  
90  	@Override
91  	public BootMessage receiveMessage(int timeout) throws IOException {
92  		return new BootMessage(receive(timeout));
93  	}
94  
95  	@Override
96  	public void sendBootMessage(BootMessage bootMessage) throws IOException {
97  		ByteBuffer b = allocate(BOOT_MESSAGE_SIZE).order(BIG_ENDIAN);
98  		bootMessage.addToBuffer(b);
99  		b.flip();
100 		send(b);
101 		// Sleep between messages to avoid flooding the machine
102 		try {
103 			sleep(ANTI_FLOOD_DELAY);
104 		} catch (InterruptedException e) {
105 			throw new IOException("interrupted during anti-flood delay", e);
106 		}
107 	}
108 }
