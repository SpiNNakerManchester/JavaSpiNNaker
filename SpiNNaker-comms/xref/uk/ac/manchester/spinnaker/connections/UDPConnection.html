104 	/**
105 	 * Main constructor, any argument of which could {@code null}.
106 	 * <p>
107 	 * No default constructors are provided as it would not be possible to
108 	 * disambiguate between ones with only a local host/port like
109 	 * {@link IPAddressConnection} and ones with only remote host/port like
110 	 * {@link BMPConnection}.
111 	 *
112 	 * @param localHost
113 	 *            The local host to bind to. If {@code null}, it defaults to
114 	 *            binding to all interfaces, unless {@code remoteHost} is
115 	 *            specified, in which case binding is done to the IP address
116 	 *            that will be used to send packets.
117 	 * @param localPort
118 	 *            The local port to bind to, 0 (or {@code null}) or between 1025
119 	 *            and 65535.
120 	 * @param remoteHost
121 	 *            The remote host name or IP address to send packets to. If
122 	 *            {@code null}, the socket will be available for listening only,
123 	 *            and will throw an exception if used for sending.
124 	 * @param remotePort
125 	 *            The remote port to send packets to. If {@code remoteHost} is
126 	 *            {@code null}, this is ignored. If {@code remoteHost} is
127 	 *            specified, this must also be specified as non-zero for the
128 	 *            connection to allow sending.
129 	 * @param trafficClass
130 	 *            What sort of traffic is this socket going to send. If
131 	 *            {@code null}, no traffic class will be used. Receive-only
132 	 *            sockets should leave this as {@code null}.
133 	 * @throws IOException
134 	 *             If there is an error setting up the communication channel
135 	 */
136 	public UDPConnection(InetAddress localHost, Integer localPort,
137 			InetAddress remoteHost, Integer remotePort,
138 			TrafficClass trafficClass) throws IOException {
139 		canSend = (remoteHost != null && remotePort != null && remotePort > 0);
140 		socket = initialiseSocket(localHost, localPort, remoteHost, remotePort,
141 				trafficClass);
142 		if (log.isDebugEnabled()) {
143 			logInitialCreation();
144 		}
145 	}

159 	private void logInitialCreation() {
160 		InetSocketAddress us = null;
161 		try {
162 			us = getLocalAddress();
163 		} catch (Exception ignore) {
164 		}
165 		if (us == null) {
166 			us = new InetSocketAddress((InetAddress) null, 0);
167 		}
168 		InetSocketAddress them = null;
169 		try {
170 			them = getRemoteAddress();
171 		} catch (Exception ignore) {
172 		}
173 		if (them == null) {
174 			them = new InetSocketAddress((InetAddress) null, 0);
175 		}
176 		log.debug("{} socket created ({} <--> {})", getClass().getName(), us,
177 				them);
178 	}

180 	/**
181 	 * Set up a UDP/IPv4 socket.
182 	 *
183 	 * @param localHost
184 	 *            Local side address.
185 	 * @param localPort
186 	 *            Local side port.
187 	 * @param remoteHost
188 	 *            Remote side address.
189 	 * @param remotePort
190 	 *            Remote side port.
191 	 * @param trafficClass
192 	 *            Traffic class, at least for sending. If {@code null}, no
193 	 *            traffic class will be set.
194 	 * @return The configured, connected socket.
195 	 * @throws IOException
196 	 *             If anything fails.
197 	 */
198 	DatagramSocket initialiseSocket(InetAddress localHost, Integer localPort,
199 			InetAddress remoteHost, Integer remotePort,
200 			TrafficClass trafficClass) throws IOException {
201 		// SpiNNaker only speaks IPv4
202 		var sock = new DatagramSocket(createLocalAddress(localHost, localPort));
203 		if (trafficClass != null) {
204 			sock.setTrafficClass(trafficClass.value);
205 		}
206 		sock.setReceiveBufferSize(RECEIVE_BUFFER_SIZE);
207 		if (canSend) {
208 			remoteIPAddress = (Inet4Address) remoteHost;
209 			remoteAddress = new InetSocketAddress(remoteIPAddress, remotePort);
210 			sock.connect(remoteAddress);
211 		}
212 		return sock;
213 	}

215 	private static SocketAddress createLocalAddress(InetAddress localHost,
216 			Integer localPort) throws UnknownHostException {
217 		// Convert null into wildcard
218 		if (localPort == null) {
219 			localPort = 0;
220 		}
221 		Inet4Address localAddr;
222 		try {
223 			if (localHost == null) {
224 				localAddr = (Inet4Address) getByAddress(new byte[IPV4_SIZE]);
225 			} else {
226 				localAddr = (Inet4Address) localHost;
227 			}
228 		} catch (ClassCastException e) {
229 			throw new UnknownHostException("SpiNNaker only talks IPv4");
230 		}
231 		return new InetSocketAddress(localAddr, localPort);
232 	}

311 	 * Receive data from the connection.
312 	 *
313 	 * @param timeout
314 	 *            The timeout in milliseconds, or {@code null} to wait forever
315 	 * @return The data received, in a little-endian buffer
316 	 * @throws SocketTimeoutException
317 	 *             If a timeout occurs before any data is received
318 	 * @throws EOFException
319 	 *             If the connection is closed
320 	 * @throws IOException
321 	 *             If an error occurs receiving the data
322 	 */
323 	public final ByteBuffer receive(Integer timeout)
324 			throws SocketTimeoutException, IOException {
325 		if (timeout != null) {
326 			return receive(convertTimeout(timeout));
327 		}
328 		// Want to wait forever but the underlying engine won't...
329 		while (true) {
330 			try {
331 				return receive(convertTimeout(timeout));
332 			} catch (SocketTimeoutException e) {
333 				continue;
334 			}
335 		}
336 	}

339 	 * Receive data from the connection.
340 	 *
341 	 * @param timeout
342 	 *            The timeout in milliseconds
343 	 * @return The data received, in a little-endian buffer
344 	 * @throws SocketTimeoutException
345 	 *             If a timeout occurs before any data is received
346 	 * @throws EOFException
347 	 *             If the connection is closed
348 	 * @throws IOException
349 	 *             If an error occurs receiving the data
350 	 */
351 	public final ByteBuffer receive(int timeout)
352 			throws SocketTimeoutException, IOException {
353 		if (isClosed()) {
354 			throw new EOFException();
355 		}
356 		return doReceive(timeout);
357 	}

359 	/**
360 	 * Receive data from the connection.
361 	 * <p>
362 	 * This operation is <em>delegatable</em>; see
363 	 * {@link DelegatingSCPConnection}.
364 	 *
365 	 * @param timeout
366 	 *            The timeout in milliseconds
367 	 * @return The data received, in a little-endian buffer
368 	 * @throws SocketTimeoutException
369 	 *             If a timeout occurs before any data is received
370 	 * @throws IOException
371 	 *             If an error occurs receiving the data
372 	 */
373 	ByteBuffer doReceive(int timeout)
374 			throws SocketTimeoutException, IOException {
375 		socket.setSoTimeout(timeout);
376 		var buffer = allocate(receivePacketSize);
377 		var pkt = new DatagramPacket(buffer.array(), receivePacketSize);
378 		socket.receive(pkt);
379 		buffer.position(pkt.getLength()).flip();
380 		logRecv(buffer, pkt.getSocketAddress());
381 		return buffer.order(LITTLE_ENDIAN);
382 	}

434 	/**
435 	 * Create the actual message to send.
436 	 *
437 	 * @param data
438 	 *            The content of the message
439 	 * @param remoteAddress
440 	 *            Where to send it to
441 	 * @return The full packet to send.
442 	 */
443 	private static DatagramPacket formSendPacket(
444 			ByteBuffer data, InetSocketAddress remoteAddress) {
445 		if (!data.hasArray()) {
446 			// Yuck; must copy because can't touch the backing array
447 			var buffer = new byte[data.remaining()];
448 			data.duplicate().get(buffer);
449 			return new DatagramPacket(buffer, 0, buffer.length, remoteAddress);
450 		} else {
451 			// Unsafe, but we can get away with it as we send immediately
452 			// and never actually write to the array
453 			return new DatagramPacket(
454 					data.array(), data.arrayOffset() + data.position(),
455 					data.remaining(), remoteAddress);
456 		}
457 	}

496 	 * Send data down this connection.
497 	 *
498 	 * @param data
499 	 *            The data to be sent

512 		}

514 			throw new EOFException();
515 		}
516 		if (!data.hasRemaining()) {
517 			throw new IllegalStateException(
518 					"data buffer must have bytes to send");
519 		}
520 		doSend(data);
521 	}

625 	private void logSend(ByteBuffer data, SocketAddress addr) {
626 		log.debug("sending data of length {} to {}", data.remaining(), addr);
627 		log.debug("message data: {}", describe(data));
628 	}

630 	private void logRecv(ByteBuffer data, SocketAddress addr) {
631 		log.debug("received data of length {} from {}", data.remaining(), addr);
632 		log.debug("message data: {}", describe(data));
633 	}

635 	private List<String> describe(ByteBuffer data) {
636 		int pos = data.position();
637 		IntFunction<String> foo = i -> hexbyte(data.get(pos + i));
638 		return range(0, data.remaining()).mapToObj(foo).collect(toList());
639 	}

641 	@Override
642 	public boolean isConnected() {
643 		if (!canSend) {
644 			return false;
645 		}
646 		for (int i = 0; i < PING_COUNT; i++) {
647 			if (ping(remoteIPAddress) == 0) {
648 				return true;
649 			}
650 		}
651 		return false;
652 	}

662 	}

669 	/**
670 	 * Sends a port trigger message using a connection to (hopefully) open a
671 	 * port in a NAT and/or firewall to allow incoming packets to be received.
672 	 *
673 	 * @param host
674 	 *            The address of the SpiNNaker board to which the message should
675 	 *            be sent
676 	 * @throws IOException
677 	 *             If anything goes wrong
678 	 */
679 	public final void sendPortTriggerMessage(InetAddress host)
680 			throws IOException {
681 		/*
682 		 * Set up the message so that no reply is expected and it is sent to an
683 		 * invalid port for SCAMP. The current version of SCAMP will reject this
684 		 * message, but then fail to send a response since the
685 		 * REPLY_NOT_EXPECTED flag is set (see scamp-3.c line 728 and 625-644)
686 		 */
687 		var triggerMessage = new SDPMessage(new SDPHeader(REPLY_NOT_EXPECTED,
688 				new CoreLocation(0, 0, 0), RUNNING_COMMAND_SDP_PORT));
689 		sendTo(triggerMessage.getMessageData(null), host, SCP_SCAMP_PORT);
690 	}

692 	@Override
693 	public String toString() {
694 		InetSocketAddress la = null, ra = null;
695 		try {
696 			la = getLocalAddress();
697 		} catch (IOException ignore) {
698 		}
699 		try {
700 			ra = getRemoteAddress();
701 		} catch (IOException ignore) {
702 		}
703 		return format("%s(%s <-%s-> %s)",
704 				getClass().getSimpleName().replaceAll("^.*\\.", ""), la,
705 				isClosed() ? "|" : "", ra);
