17  package uk.ac.manchester.spinnaker.connections;
18  
19  import static java.lang.Runtime.getRuntime;
20  
21  import java.io.IOException;
22  import java.net.InetAddress;
23  import java.util.Calendar;
24  import java.util.HashSet;
25  import java.util.function.BiPredicate;
26  
27  /**
28   * Locate any SpiNNaker machines IP addresses from the auto-transmitted packets
29   * from non-booted SpiNNaker machines.
30   */
31  public abstract class LocateConnectedMachineIPAddress {
32  	private LocateConnectedMachineIPAddress() {
33  	}
34  
35  	/**
36  	 * Locates any SpiNNaker machines IP addresses from the auto-transmitted
37  	 * packets from non-booted SpiNNaker machines.
38  	 *
39  	 * @param handler
40  	 *            A predicate that decides whether to stop searching (we stop if
41  	 *            the predicate returns true). The predicate is given two
42  	 *            arguments: the IP address found and the current time. Note
43  	 *            that each board is only reported once.
44  	 * @throws IOException
45  	 *             If anything goes wrong
46  	 */
47  	public static void locateConnectedMachine(
48  			BiPredicate<InetAddress, Calendar> handler) throws IOException {
49  		try (var connection = new IPAddressConnection()) {
50  			var seenBoards = new HashSet<>();
51  			while (true) {
52  				var ipAddress = connection.receiveMessage();
53  				var now = Calendar.getInstance();
54  				if (ipAddress != null && !seenBoards.contains(ipAddress)) {
55  					seenBoards.add(ipAddress);
56  					if (handler.test(ipAddress, now)) {
57  						break;
58  					}
59  				}
60  			}
61  		}
62  	}
63  
64  	/**
65  	 * A little program that listens for, and prints, the pre-boot messages
66  	 * published by SpiNNaker boards.
67  	 *
68  	 * @param args
69  	 *            ignored
70  	 * @throws IOException
71  	 *             if anything goes wrong.
72  	 */
73  	public static void main(String... args) throws IOException {
74  		System.out.format("The following addresses might be SpiNNaker boards"
75  				+ " (press Ctrl-C to quit):%n");
76  		getRuntime().addShutdownHook(
77  				new Thread(LocateConnectedMachineIPAddress::goodbye));
78  		locateConnectedMachine(LocateConnectedMachineIPAddress::printHost);
79  	}
80  
81  	private static boolean printHost(InetAddress addr, Calendar time) {
82  		System.out.format("%s (%s) at %s%n", addr, addr.getCanonicalHostName(),
83  				time);
84  		return false;
85  	}
86  
87  	private static void goodbye() {
88  		System.out.format("Exiting%n");
89  	}
90  }
