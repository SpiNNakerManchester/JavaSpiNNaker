17  package uk.ac.manchester.spinnaker.connections.model;
18  
19  import static uk.ac.manchester.spinnaker.utils.UnitConstants.MSEC_PER_SEC;
20  
21  import java.io.IOException;
22  
23  /**
24   * How to listen for a message on a connection.
25   *
26   * @param <MessageType>
27   *            The type of message being listened for (typically fixed for a
28   *            particular connection).
29   */
30  public interface Listenable<MessageType> extends MessageReceiver<MessageType> {
31  	/**
32  	 * Do a non-blocking poll of whether there is a message ready to be received
33  	 * without blocking.
34  	 *
35  	 * @return true when there is a packet waiting to be received
36  	 * @throws IOException
37  	 *             If anything goes wrong, e.g., if the socket is closed under
38  	 *             our feet.
39  	 */
40  	default boolean isReadyToReceive() throws IOException {
41  		return isReadyToReceive(null);
42  	}
43  
44  	/**
45  	 * Do a blocking poll of whether there is a message ready to be received
46  	 * without blocking. <i>This method</i> may block until the timeout given.
47  	 *
48  	 * @param timeout
49  	 *            How many seconds to wait for a message to be receivable.
50  	 * @return true when there is a packet waiting to be received
51  	 * @throws IOException
52  	 *             If anything goes wrong, e.g., if the socket is closed under
53  	 *             our feet.
54  	 */
55  	default boolean isReadyToReceive(double timeout) throws IOException {
56  		return isReadyToReceive((int) (timeout * MSEC_PER_SEC));
57  	}
58  
59  	/**
60  	 * Determines if there is a message available to be received without
61  	 * blocking. <i>This method</i> may block until the timeout given.
62  	 *
63  	 * @param timeout
64  	 *            How long to wait, in milliseconds; if zero or {@code null}, a
65  	 *            non-blocking poll is performed.
66  	 * @return true when there is a message waiting to be received
67  	 * @throws IOException
68  	 *             If anything goes wrong, e.g., if the socket is closed under
69  	 *             our feet.
70  	 */
71  	boolean isReadyToReceive(Integer timeout) throws IOException;
72  }
