17  package uk.ac.manchester.spinnaker.connections.model;
18  
19  import java.io.IOException;
20  import java.nio.ByteBuffer;
21  
22  import uk.ac.manchester.spinnaker.machine.ChipLocation;
23  import uk.ac.manchester.spinnaker.messages.scp.SCPRequest;
24  
25  /** A sender of SCP messages. */
26  public interface SCPSender extends Connection {
27  	/**
28  	 * Returns the data of an SCP request as it would be sent down this
29  	 * connection.
30  	 *
31  	 * @param scpRequest
32  	 *            message packet to serialise
33  	 * @return The buffer holding the data. The data should be written into the
34  	 *         start of the buffer and should end at the <i>position</i>.
35  	 */
36  	default ByteBuffer getSCPData(SCPRequest<?> scpRequest) {
37  		return scpRequest.getMessageData(getChip());
38  	}
39  
40  	/**
41  	 * Sends an SCP request down this connection.
42  	 * <p>
43  	 * Messages must have the following properties:
44  	 * <ul>
45  	 * <li><i>sourcePort</i> is {@code null} or 7
46  	 * <li><i>sourceCpu</i> is {@code null} or 31
47  	 * <li><i>sourceChipX</i> is {@code null} or 0
48  	 * <li><i>sourceChipY</i> is {@code null} or 0
49  	 * </ul>
50  	 * <i>tag</i> in the message is optional; if not set, the default set in the
51  	 * constructor will be used.
52  	 * <p>
53  	 * <i>sequence</i> in the message is optional; if not set, <i>(sequence
54  	 * number\ last assigned + 1) % 65536</i> will be used
55  	 *
56  	 * @param scpRequest
57  	 *            message packet to send
58  	 * @throws IOException
59  	 *             If there is an error sending the message
60  	 */
61  	void send(SCPRequest<?> scpRequest) throws IOException;
62  
63  	/**
64  	 * @return The chip at which messages sent down this connection will arrive
65  	 *         at first.
66  	 */
67  	ChipLocation getChip();
68  }
