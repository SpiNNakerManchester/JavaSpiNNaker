30  import javax.validation.Valid;
31  import javax.validation.constraints.AssertTrue;

34  import javax.validation.constraints.NotNull;

36  import org.python.core.PyObject;

38  import com.fasterxml.jackson.annotation.JsonProperty;
39  import com.google.errorprone.annotations.Keep;

41  import uk.ac.manchester.spinnaker.machine.board.BMPCoords;

48  /** A machine description. JSON-serializable. */
49  public final class Machine {
50  	/** The name of the machine. */
51  	@NotBlank
52  	public final String name;
53  
54  	/** The tags of the machine. */
55  	@NotEmpty
56  	public final Set<@NotBlank String> tags;
57  
58  	/** The width of the machine, in triads. */
59  	@ValidTriadWidth
60  	public final int width;
61  
62  	/** The height of the machine, in triads. */
63  	@ValidTriadHeight
64  	public final int height;
65  
66  	/** The dead boards of the machine. */
67  	public final Set<@Valid TriadCoords> deadBoards;
68  
69  	/**
70  	 * The extra dead links of the machine. Doesn't include links to dead
71  	 * boards.
72  	 */
73  	@NotNull
74  	public final Map<@Valid TriadCoords, @NotEmpty EnumSet<Link>> deadLinks;
75  
76  	/** The logical-to-physical board location map. */
77  	@NotNull
78  	public final Map<@Valid TriadCoords, @Valid PhysicalCoords> boardLocations;
79  
80  	/** The IP addresses of the BMPs. */
81  	@JsonProperty("bmp-ips")
82  	@NotNull
83  	public final Map<@Valid BMPCoords, @IPAddress String> bmpIPs;
84  
85  	/** The IP addresses of the boards. */
86  	@JsonProperty("spinnaker-ips")
87  	@NotNull
88  	public final Map<@Valid TriadCoords, @IPAddress String> spinnakerIPs;
89  
90  	private static final int IDX = 3;
91  
92  	Machine(PyObject machine) {
93  		name = getattr(machine, "name").asString();
94  		tags = toSet(getattr(machine, "tags"), PyObject::asString);
95  		width = getattr(machine, "width").asInt();
96  		height = getattr(machine, "height").asInt();
97  		deadBoards = toSet(getattr(machine, "dead_boards"), Machine::xyz);
98  		deadLinks = toCollectingMap(getattr(machine, "dead_links"),
99  				Machine::xyz, () -> noneOf(Link.class),
100 				key -> Link.values()[item(key, IDX).asInt()]);
101 		boardLocations = toMap(getattr(machine, "board_locations"),
102 				Machine::xyz, Machine::cfb);
103 		bmpIPs = toMap(getattr(machine, "bmp_ips"), Machine::cf,
104 				PyObject::asString);
105 		spinnakerIPs = toMap(getattr(machine, "spinnaker_ips"), Machine::xyz,
106 				PyObject::asString);
107 	}
108 
109 	private static TriadCoords xyz(PyObject tuple) {
110 		int index = 0;
111 		int x = item(tuple, index++).asInt();
112 		int y = item(tuple, index++).asInt();
113 		int z = item(tuple, index++).asInt();
114 		return new TriadCoords(x, y, z);
115 	}
116 
117 	private static PhysicalCoords cfb(PyObject tuple) {
118 		int index = 0;
119 		int c = item(tuple, index++).asInt();
120 		int f = item(tuple, index++).asInt();
121 		int b = item(tuple, index++).asInt();
122 		return new PhysicalCoords(c, f, b);
123 	}
124 
125 	private static BMPCoords cf(PyObject tuple) {
126 		int index = 0;
127 		int c = item(tuple, index++).asInt();
128 		int f = item(tuple, index++).asInt();
129 		return new BMPCoords(c, f);
130 	}
131 
132 	@Override
133 	public String toString() {
134 		return new StringBuilder("Machine(").append("name=").append(name)
135 				.append(",").append("tags=").append(tags).append(",")
136 				.append("width=").append(width).append(",").append("height=")
137 				.append(height).append(",").append("deadBoards=")
138 				.append(deadBoards).append(",").append("deadLinks=")
139 				.append(deadLinks).append(",").append("boardLocations=")
140 				.append(boardLocations).append(",").append("bmpIPs=")
141 				.append(bmpIPs).append(",").append("spinnakerIPs=")
142 				.append(spinnakerIPs).append(")").toString();
143 	}
144 
145 	@Keep
146 	@AssertTrue(message = "all boards must have a location and an IP address")
147 	private boolean isEveryBoardWithBMPAndIPaddress() {
148 		return boardLocations.keySet().equals(spinnakerIPs.keySet());
149 	}
150 
151 	@Keep
152 	@AssertTrue(message = "every board's IP address must be unique")
153 	private boolean isEveryBoardIPAddressUnique() {
154 		return spinnakerIPs.size() == Set.copyOf(spinnakerIPs.values()).size();
155 	}
156 
157 	@Keep
158 	@AssertTrue(message = "every BMP's IP address must be unique")
159 	private boolean isEveryBmpIPAddressUnique() {
160 		return bmpIPs.size() == Set.copyOf(bmpIPs.values()).size();
161 	}
162 
163 	@Keep
164 	@AssertTrue(message = "IP addresses may not be assigned to "
165 			+ "both boards and BMPs")
166 	private boolean isBoardAddressSetDisjointFromBmpAddressSet() {
167 		return disjoint(spinnakerIPs.values(), bmpIPs.values());
168 	}
169 
170 	@Keep
171 	@AssertTrue(message = "every board's BMP must be addressable")
172 	private boolean isEveryBoardManaged() {
173 		return boardLocations.values().stream()
174 				.map(PhysicalCoords::getBmpCoords)
175 				.allMatch(bmpIPs::containsKey);
176 	}
177 }
