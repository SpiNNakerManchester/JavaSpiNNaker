17  package uk.ac.manchester.spinnaker.data_spec;
18  
19  import static java.lang.String.format;
20  
21  /**
22   * An exception which occurs when trying to execute an unimplemented command.
23   */
24  public class UnimplementedDSECommandException
25  		extends UnsupportedOperationException {
26  	private static final long serialVersionUID = -2215740111501788733L;
27  
28  	/**
29  	 * Create an instance.
30  	 *
31  	 * @param index
32  	 *            Where the command was located.
33  	 * @param command
34  	 *            The command that was unimplemented.
35  	 */
36  	UnimplementedDSECommandException(int index, Commands command) {
37  		super(format("Command %s (at index %d) in the data specification "
38              + "executor has not yet been implemented", command, index));
39  	}
40  
41  	/**
42  	 * Create an instance.
43  	 *
44  	 * @param index
45  	 *            Where the opcode was located.
46  	 * @param opcode
47  	 *            The opcode that couldn't be converted into a command.
48  	 */
49  	UnimplementedDSECommandException(int index, int opcode) {
50  		super(format("unknown opcocode (%d) at index %d", opcode, index));
51  	}
52  }
