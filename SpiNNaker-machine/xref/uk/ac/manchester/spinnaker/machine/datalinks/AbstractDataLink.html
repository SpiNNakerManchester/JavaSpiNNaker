17  package uk.ac.manchester.spinnaker.machine.datalinks;
18  
19  import java.net.InetAddress;
20  import java.util.Objects;
21  import uk.ac.manchester.spinnaker.machine.ChipLocation;
22  import uk.ac.manchester.spinnaker.machine.Direction;
23  import uk.ac.manchester.spinnaker.machine.HasChipLocation;
24  import uk.ac.manchester.spinnaker.machine.MachineDefaults;
25  
26  /**
27   * Abstract Root of all Data Links.
28   *
29   * @author Christian-B
30   */
31  public class AbstractDataLink implements HasChipLocation {
32  	/** IP address of the Datalink on the board. */
33  	public final InetAddress boardAddress;
34  
35  	/** Coordinates of the location/Chip being linked to. */
36  	public final ChipLocation location;
37  
38  	/** link Direction/id for this link. */
39  	public final Direction direction;
40  
41  	/**
42  	 * Main Constructor of a DataLink.
43  	 *
44  	 * @param location
45  	 *            The location/Chip being linked to
46  	 * @param linkId
47  	 *            The ID/Direction coming out of the Chip
48  	 * @param boardAddress
49  	 *            IP address of the Datalink on the board.
50  	 */
51  	AbstractDataLink(HasChipLocation location, Direction linkId,
52  			InetAddress boardAddress) {
53  		if (location == null) {
54  			throw new IllegalArgumentException("location was null");
55  		}
56  		if (linkId == null) {
57  			throw new IllegalArgumentException("linkId was null");
58  		}
59  		this.location = location.asChipLocation();
60  		this.direction = linkId;
61  		this.boardAddress = boardAddress;
62  	}
63  
64  	@Override
65  	public int getX() {
66  		return location.getX();
67  	}
68  
69  	@Override
70  	public int getY() {
71  		return location.getY();
72  	}
73  
74  	@Override
75  	public ChipLocation asChipLocation() {
76  		return location;
77  	}
78  
79  	@Override
80  	public int hashCode() {
81  		int hash = 41 * (getX() << MachineDefaults.COORD_SHIFT) ^ getY();
82  		hash = 41 * hash + Objects.hashCode(this.boardAddress);
83  		hash = 41 * hash + this.direction.hashCode();
84  		return hash;
85  	}
86  
87  	@Override
88  	public boolean equals(Object obj) {
89  		if (this == obj) {
90  			return true;
91  		}
92  		if (obj == null) {
93  			return false;
94  		}
95  		if (getClass() != obj.getClass()) {
96  			return false;
97  		}
98  		final AbstractDataLink other = (AbstractDataLink) obj;
99  		return sameAs(other);
100 	}
101 
102 	/**
103 	 * Determines if the Objects can be considered the same.
104 	 * <p>
105 	 * Two links where their locations return onSameChipAs can be considered the
106 	 * same even if the location is a different class. This is consistent with
107 	 * hashCode that only considers the location's X and Y
108 	 *
109 	 * @param other
110 	 *            Another DataLink
111 	 * @return True if and only if the IDs match, the boardAddress match and the
112 	 *         links are on the same chip
113 	 */
114 	boolean sameAs(AbstractDataLink other) {
115 		if (!this.direction.equals(other.direction)) {
116 			return false;
117 		}
118 		if (!Objects.equals(this.boardAddress, other.boardAddress)) {
119 			return false;
120 		}
121 		return location.onSameChipAs(other.location);
122 	}
123 }
