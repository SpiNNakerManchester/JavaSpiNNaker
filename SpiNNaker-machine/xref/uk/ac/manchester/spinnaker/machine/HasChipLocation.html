16  package uk.ac.manchester.spinnaker.machine;
17  
18  import com.fasterxml.jackson.annotation.JsonIgnore;
19  
20  /**
21   * The interface supported by any object that is associated with a chip.
22   *
23   * @author Donal Fellows
24   */
25  public interface HasChipLocation {
26  	/**
27  	 * @return The X coordinate of the chip.
28  	 */
29  	@ValidX
30  	int getX();
31  
32  	/**
33  	 * @return The X coordinate of the chip.
34  	 */
35  	@ValidY
36  	int getY();
37  
38  	/**
39  	 * Check if two locations are colocated at the chip level. This does
40  	 * <i>not</i> imply that the two are equal.
41  	 *
42  	 * @param other
43  	 *            The other location to compare to.
44  	 * @return If the two locations have the same X and Y coordinates.
45  	 */
46  	default boolean onSameChipAs(HasChipLocation other) {
47  		return (getX() == other.getX()) && (getY() == other.getY());
48  	}
49  
50  	/**
51  	 * Get the core of the chip that will be running SC&amp;MP. This is always
52  	 * core 0 of the chip, as the core that runs SC&amp;MP always maps itself to
53  	 * be virtual core ID 0.
54  	 *
55  	 * @return The location of the SC&amp;MP core.
56  	 */
57  	@JsonIgnore
58  	default HasCoreLocation getScampCore() {
59  		// SCAMP always runs on core 0 or we can't talk to the chip at all
60  		return new CoreLocation(getX(), getY(), 0);
61  	}
62  
63  	/**
64  	 * Converts (if required) this to a simple X, Y tuple.
65  	 *
66  	 * @return A ChipLocation representation of the X and Y tuple
67  	 */
68  	default ChipLocation asChipLocation() {
69  		return new ChipLocation(getX(), getY());
70  	}
71  }
