17  package uk.ac.manchester.spinnaker.machine;
18  
19  /** A multicast packet routing table entry. */
20  public class MulticastRoutingEntry extends RoutingEntry {
21      private final int key;
22      private final int mask;
23      private final boolean defaultable;
24  
25      /**
26       * Create a multicast routing table entry.
27       *
28       * @param key
29       *            The key of the entry.
30       * @param mask
31       *            The mask of the entry.
32       * @param route
33       *            The route descriptor.
34       * @param defaultable
35       *            Whether this entry is default routable.
36       */
37      public MulticastRoutingEntry(int key, int mask, int route,
38              boolean defaultable) {
39          super(route);
40          this.key = key;
41          this.mask = mask;
42          this.defaultable = defaultable;
43      }
44  
45      /**
46       * Create a multicast routing table entry.
47       *
48       * @param key
49       *            The key of the entry.
50       * @param mask
51       *            The mask of the entry.
52       * @param processorIDs
53       *            The IDs of the processors that this entry routes to.
54       *            The Duplicate IDs are ignored.
55       * @param linkIDs
56       *            The IDs of the links that this entry routes to.
57       *            The Duplicate IDs are ignored.
58       * @param defaultable
59       *            Whether this entry is default routable.
60       */
61      public MulticastRoutingEntry(
62              int key, int mask, Iterable<Integer> processorIDs,
63              Iterable<Direction> linkIDs, boolean defaultable) {
64          super(processorIDs, linkIDs);
65          this.key = key;
66          this.mask = mask;
67          this.defaultable = defaultable;
68      }
69  
70      /** @return the key of this entry */
71      public int getKey() {
72          return key;
73      }
74  
75      /** @return the mask of this entry */
76      public int getMask() {
77          return mask;
78      }
79  
80      /** @return whether this entry is default routable */
81      public boolean isDefaultable() {
82          return defaultable;
83      }
84  }
