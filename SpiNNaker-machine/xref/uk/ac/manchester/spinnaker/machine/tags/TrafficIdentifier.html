17  package uk.ac.manchester.spinnaker.machine.tags;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  /**
23   * The types of traffic an IpTag can handle.
24   *
25   * @author Christian-B
26   */
27  public final class TrafficIdentifier {
28  
29      private static final Map<String, TrafficIdentifier> MAP = new HashMap<>();
30  
31      /** Default if not provided.
32       *
33       * @see <a href=
34       * "https://github.com/SpiNNakerManchester/SpiNNMachine/blob/master/spinn_machine/tags/iptag.py">
35       * Python IPTag</a>
36       */
37      public static final TrafficIdentifier/tags/TrafficIdentifier.html#TrafficIdentifier">TrafficIdentifier DEFAULT = new TrafficIdentifier(
38              "DEFAULT");
39      /** Used to identify buffered traffic.
40       *
41       * @see <a href=
42       * "https://github.com/SpiNNakerManchester/SpiNNFrontEndCommon/blob/master/spinn_front_end_common/interface/buffer_management/recording_utilities.py">
43       * Python Recording Utilities</a>
44       */
45      public static final TrafficIdentifiertags/TrafficIdentifier.html#TrafficIdentifier">TrafficIdentifier BUFFERED = new TrafficIdentifier(
46              "BufferTraffic");
47  
48      /** Label used in Python. */
49      public final String label;
50  
51      /**
52       * Finds a TrafficIdentifier based on the label.
53       *
54       * @param label
55       *            The label to check for. Currently case sensitive.
56       * @return The TrafficIdentifier with this label
57       * @throws IllegalArgumentException
58       *             If this is an unexpected label.
59       */
60      public static TrafficIdentifier getInstance(String label) {
61          if (!MAP.containsKey(label)) {
62              new TrafficIdentifier(label);
63          }
64          return MAP.get(label);
65      }
66  
67      /**
68       * Main Constructor.
69       *
70       * @param label
71       *            Python label.
72       */
73      private TrafficIdentifier(String label) {
74          this.label = label;
75          MAP.put(label, this);
76      }
77  }
