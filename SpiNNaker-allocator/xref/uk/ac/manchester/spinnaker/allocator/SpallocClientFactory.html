140 	/**
141 	 * Outputs a form to a connection in
142 	 * {@code application/x-www-form-urlencoded} format.
143 	 *
144 	 * @param connection
145 	 *            The connection. Must have the right verb set.
146 	 * @param map
147 	 *            The contents of the form.
148 	 * @throws IOException
149 	 *             If I/O fails.
150 	 */
151 	static void writeForm(HttpURLConnection connection, Map<String, String> map)
152 			throws IOException {
153 		var form = map.entrySet().stream()
154 				.map(e -> e.getKey() + "=" + encode(e.getValue(), UTF_8))
155 				.collect(joining("&"));
156 
157 		connection.setDoOutput(true);
158 		connection.setRequestProperty(CONTENT_TYPE, FORM_ENCODED);
159 		try (var w =
160 				new OutputStreamWriter(connection.getOutputStream(), UTF_8)) {
161 			w.write(form);
162 		}
163 	}

204 	/**
205 	 * Checks for errors in the response.
206 	 *
207 	 * @param conn
208 	 *            The HTTP connection
209 	 * @param errorMessage
210 	 *            The message to use on error (describes what did not work at a
211 	 *            higher level)
212 	 * @return The input stream so any non-error response content can be
213 	 *         obtained.
214 	 * @throws IOException
215 	 *             If things go wrong with comms.
216 	 * @throws FileNotFoundException
217 	 *             on a {@link HttpURLConnection#HTTP_NOT_FOUND}
218 	 * @throws SpallocException
219 	 *             on other server errors
220 	 */
221 	@MustBeClosed
222 	static InputStream checkForError(HttpURLConnection conn,
223 			String errorMessage) throws IOException {
224 		if (conn.getResponseCode() == HTTP_NOT_FOUND) {
225 			// Special case
226 			throw new FileNotFoundException(errorMessage);
227 		}
228 		if (conn.getResponseCode() >= HTTP_BAD_REQUEST) {
229 			throw new SpallocException(conn.getErrorStream(),
230 					conn.getResponseCode());
231 		}
232 		return conn.getInputStream();
233 	}

256 	private abstract static class Common {
257 		private final SpallocClient client;
258 
259 		final Session s;

261 		/**
262 		 * Cache of machines, which don't expire.
263 		 */
264 		final Map<String, Machine> machineMap =
265 				synchronizedMap(new HashMap<>());
266 
267 		Common(SpallocClient client, Session s) {
268 			this.client = client != null ? client : (SpallocClient) this;
269 			this.s = s;
270 		}
271 
272 		final Machine getMachine(String name) throws IOException {
273 			Machine m;
274 			do {
275 				m = machineMap.get(name);
276 				if (m == null) {
277 					client.listMachines();
278 				}
279 			} while (m == null);
280 			return m;
281 		}

283 		private WhereIs whereis(HttpURLConnection conn) throws IOException {
284 			try (var is = checkForError(conn,
285 					"couldn't get board information")) {
286 				if (conn.getResponseCode() == HTTP_NO_CONTENT) {
287 					throw new FileNotFoundException("machine not allocated");
288 				}
289 				return readJson(is, WhereIs.class);
290 			} finally {
291 				s.trackCookie(conn);
292 			}
293 		}
294 
295 		final WhereIs whereis(URI uri) throws IOException {
296 			return s.withRenewal(() -> {
297 				var conn = s.connection(uri);
298 				var w = whereis(conn);
299 				w.setMachineHandle(getMachine(w.getMachineName()));
300 				w.clearMachineRef();
301 				return w;
302 			});
303 		}
304 	}

325 		/**
326 		 * Slightly convoluted class to fetch jobs. The complication means we
327 		 * get the initial failure exception nice and early, while we're ready
328 		 * for it. This code would be quite a lot simpler if we didn't want to
329 		 * get the exception during construction.
330 		 */
331 		private class JobLister extends ListFetchingIter<URI> {
332 			private URI next;
333 
334 			private List<URI> first;
335 
336 			JobLister(URI initial) throws IOException {
337 				var first = getJobList(s.connection(initial));
338 				next = first.next;
339 				this.first = first.jobs;
340 			}
341 
342 			private Jobs getJobList(HttpURLConnection conn) throws IOException {
343 				try (var is = checkForError(conn, "couldn't list jobs")) {
344 					return readJson(is, Jobs.class);
345 				} finally {
346 					s.trackCookie(conn);
347 				}
348 			}
349 
350 			@Override
351 			List<URI> fetchNext() throws IOException {
352 				if (nonNull(first)) {
353 					try {
354 						return first;
355 					} finally {
356 						first = null;
357 					}
358 				}
359 				var j = getJobList(s.connection(next));
360 				next = j.next;
361 				return j.jobs;
362 			}
363 
364 			@Override
365 			boolean canFetchMore() {
366 				if (nonNull(first)) {
367 					return true;
368 				}
369 				return nonNull(next);
370 			}
371 		}
372 
373 		private Stream<Job> listJobs(URI flags) throws IOException {
374 			var basicData = new JobLister(
375 					nonNull(flags) ? jobs.resolve(flags) : jobs);
376 			return basicData.stream().flatMap(Collection::stream)
377 					.map(this::job);
378 		}
379

383 		}

389 			if (wait) {

395 		@Override
396 		public Job createJob(CreateJob createInstructions) throws IOException {
397 			var uri = s.withRenewal(() -> {
398 				var conn = s.connection(jobs, true);
399 				writeObject(conn, createInstructions);
400 				// Get the response entity... and discard it
401 				try (var is = checkForError(conn, "job create failed")) {
402 					readLines(is, UTF_8);
403 					// But we do want the Location header
404 					return URI.create(conn.getHeaderField("Location"));
405 				} finally {
406 					s.trackCookie(conn);
407 				}
408 			});
409 			return job(uri);
410 		}
411 
412 		Job job(URI uri) {
413 			return new JobImpl(this, s, asDir(uri));
414 		}
415 
416 		@Override
417 		public List<Machine> listMachines() throws IOException {
418 			return s.withRenewal(() -> {
419 				var conn = s.connection(machines);
420 				try (var is = checkForError(conn, "list machines failed")) {
421 					var ms = readJson(is, Machines.class);
422 					// Assume we can cache this
423 					for (var bmd : ms.machines) {
424 						machineMap.computeIfAbsent(bmd.name,
425 								__ -> new MachineImpl(this, s, bmd));
426 					}
427 					return ms.machines.stream()
428 							.map(bmd -> machineMap.get(bmd.name))
429 							.collect(toList());
430 				} finally {
431 					s.trackCookie(conn);
432 				}
433 			});
434 		}

437 	private static final class JobImpl extends Common implements Job {
438 		private final URI uri;
439 
440 		private volatile boolean dead;
441 
442 		@GuardedBy("lock")
443 		private ProxyProtocolClient proxy;
444 
445 		private final Object lock = new Object();
446 
447 		JobImpl(SpallocClient client, Session session, URI uri) {
448 			super(client, session);
449 			this.uri = uri;
450 			this.dead = false;
451 			startKeepalive();
452 		}
453 
454 		@Override
455 		public JobDescription describe(boolean wait) throws IOException {
456 			return s.withRenewal(() -> {
457 				var conn = wait ? s.connection(uri, WAIT_FLAG)
458 						: s.connection(uri);
459 				try (var is = checkForError(conn, "couldn't get job state")) {
460 					return readJson(is, JobDescription.class);
461 				} finally {
462 					s.trackCookie(conn);
463 				}
464 			});
465 		}
466 
467 		@Override
468 		public void keepalive() throws IOException {
469 			s.withRenewal(() -> {
470 				var conn = s.connection(uri, KEEPALIVE, true);
471 				conn.setRequestMethod("PUT");
472 				writeString(conn, "alive");
473 				try (var is = checkForError(conn, "couldn't keep job alive")) {
474 					return readLines(is, UTF_8);
475 					// Ignore the output
476 				} finally {
477 					s.trackCookie(conn);
478 				}
479 			});
480 		}
481 
482 		private static final int DELAY = 20 * MSEC_PER_SEC;
483 
484 		private void startKeepalive() {
485 			if (dead) {
486 				throw new IllegalStateException("job is already deleted");
487 			}
488 			Thread t = new Daemon(() -> {
489 				try {
490 					while (true) {
491 						sleep(DELAY);
492 						if (dead) {
493 							break;
494 						}
495 						keepalive();
496 					}
497 				} catch (IOException e) {
498 					log.warn("failed to keep job alive for {}", this, e);
499 				} catch (InterruptedException e) {
500 					// If interrupted, we're simply done
501 				}
502 			});
503 			t.setName("keepalive for " + this);
504 			t.setUncaughtExceptionHandler((th, e) -> {
505 				log.warn("unexpected exception in {}", th, e);
506 			});
507 			t.start();
508 		}
509 
510 		@Override
511 		public void delete(String reason) throws IOException {
512 			dead = true;
513 			s.withRenewal(() -> {
514 				var conn = s.connection(uri, "?reason=" + encode(reason, UTF_8),
515 						true);
516 				conn.setRequestMethod("DELETE");
517 				try (var is = checkForError(conn, "couldn't delete job")) {
518 					readLines(is, UTF_8);
519 					// Ignore the output
520 				} finally {
521 					s.trackCookie(conn);
522 				}
523 				return this;
524 			});
525 			synchronized (lock) {
526 				if (haveProxy()) {
527 					proxy.close();
528 					proxy = null;
529 				}
530 			}
531 		}

533 		@Override
534 		public AllocatedMachine machine() throws IOException {
535 			var am = s.withRenewal(() -> {
536 				var conn = s.connection(uri, MACHINE);
537 				try (var is = checkForError(conn,
538 						"couldn't get allocation description")) {
539 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
540 						throw new IOException("machine not allocated");
541 					}
542 					return readJson(is, AllocatedMachine.class);
543 				} finally {
544 					s.trackCookie(conn);
545 				}
546 			});
547 			am.setMachine(getMachine(am.getMachineName()));
548 			return am;
549 		}
550 
551 		@Override
552 		public boolean getPower() throws IOException {
553 			return s.withRenewal(() -> {
554 				var conn = s.connection(uri, POWER);
555 				try (var is = checkForError(conn, "couldn't get power state")) {
556 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
557 						throw new IOException("machine not allocated");
558 					}
559 					return "ON".equals(readJson(is, Power.class).power);
560 				} finally {
561 					s.trackCookie(conn);
562 				}
563 			});
564 		}
565 
566 		@Override
567 		public boolean setPower(boolean switchOn) throws IOException {
568 			var power = new Power();
569 			power.power = (switchOn ? "ON" : "OFF");
570 			boolean powered = s.withRenewal(() -> {
571 				var conn = s.connection(uri, POWER, true);
572 				conn.setRequestMethod("PUT");
573 				writeObject(conn, power);
574 				try (var is = checkForError(conn, "couldn't set power state")) {
575 					if (conn.getResponseCode() == HTTP_NO_CONTENT) {
576 						throw new IOException("machine not allocated");
577 					}
578 					return "ON".equals(readJson(is, Power.class).power);
579 				} finally {
580 					s.trackCookie(conn);
581 				}
582 			});
583 			if (!powered) {
584 				// If someone turns the power off, close the proxy
585 				synchronized (lock) {
586 					if (haveProxy()) {
587 						proxy.close();
588 						proxy = null;
589 					}
590 				}
591 			}
592 			return powered;
593 		}

596 		public WhereIs whereIs(HasChipLocation chip) throws IOException {

598 					format("chip?x=%d&y=%d", chip.getX(), chip.getY())));
599 		}

601 		@GuardedBy("lock")
602 		private boolean haveProxy() {
603 			return nonNull(proxy) && proxy.isOpen();
604 		}
605 
606 		/**
607 		 * @return The websocket-based proxy.
608 		 * @throws IOException
609 		 *             if we can't connect
610 		 * @throws InterruptedException
611 		 *             if we're interrupted while connecting
612 		 */
613 		private ProxyProtocolClient getProxy()
614 				throws IOException, InterruptedException {
615 			synchronized (lock) {
616 				if (haveProxy()) {
617 					return proxy;
618 				}
619 			}
620 			var wssAddr = describe(false).getProxyAddress();
621 			if (isNull(wssAddr)) {
622 				throw new IOException("machine not allocated");
623 			}
624 			synchronized (lock) {
625 				if (!haveProxy()) {
626 					proxy = s.withRenewal(() -> s.websocket(wssAddr));
627 				}
628 				return proxy;
629 			}
630 		}

632 		@Override
633 		public SCPConnection getConnection(HasChipLocation chip)
634 				throws IOException, InterruptedException {
635 			return new ProxiedSCPConnection(chip.asChipLocation(), getProxy());
636 		}

638 		@Override
639 		public EIEIOConnection getEIEIOConnection()
640 				throws IOException, InterruptedException {
641 			var hostToChip = machine().getConnections().stream()
642 					.collect(toMap(c -> getByNameQuietly(c.getHostname()),
643 							ConnectionInfo::getChip));
644 			return new ProxiedEIEIOListenerConnection(hostToChip, getProxy());
645 		}
646 
647 		@Override
648 		public TransceiverInterface getTransceiver()
649 				throws IOException, InterruptedException, SpinnmanException {
650 			var ws = getProxy();
651 			var am = machine();
652 			var conns = new ArrayList<Connection>();
653 			for (var bc : am.getConnections()) {
654 				conns.add(new ProxiedSCPConnection(
655 						bc.getChip().asChipLocation(), ws));
656 			}
657 			conns.add(new ProxiedBootConnection(ws));
658 			return new ProxiedTransceiver(conns, ws);
659 		}

671 		MachineImpl(SpallocClient client, Session session,

714 		@Override
715 		public void waitForChange() throws IOException {
716 			var nbmd = s.withRenewal(() -> {
717 				var conn = s.connection(bmd.uri, WAIT_FLAG);
718 				try (var is = checkForError(conn,
719 						"couldn't wait for state change")) {
720 					return readJson(is, BriefMachineDescription.class);
721 				} finally {
722 					s.trackCookie(conn);
723 				}
724 			});
725 			this.deadBoards = List.copyOf(nbmd.deadBoards);
726 			this.deadLinks = List.copyOf(nbmd.deadLinks);
727 		}
