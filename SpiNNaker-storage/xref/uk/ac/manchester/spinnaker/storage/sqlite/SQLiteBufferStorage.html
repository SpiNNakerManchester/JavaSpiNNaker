73  	private static int getRecordingCore(Connection conn, CoreLocation core)
74  			throws SQLException {
75  		try (var s = conn.prepareStatement(GET_LOCATION)) {
76  			// x, y, processor
77  			setArguments(s, core.getX(), core.getY(), core.getP());
78  			try (var rs = s.executeQuery()) {
79  				while (rs.next()) {
80  					return rs.getInt("core_id");
81  				}
82  			}
83  		}
84  		try (var s = conn.prepareStatement(INSERT_LOCATION,
85  				RETURN_GENERATED_KEYS)) {
86  			// x, y, processor
87  			setArguments(s, core.getX(), core.getY(), core.getP());
88  			s.executeUpdate();
89  			return getLastKey(s).orElseThrow(() -> new IllegalStateException(
90  					"could not make or find recording region core record"));
91  		}
92  	}

94  	private static int getRecordingRegion(Connection conn, int coreID,
95  			Region region) throws SQLException {
96  		try (var s = conn.prepareStatement(GET_REGION)) {
97  			// core_id, local_region_index
98  			setArguments(s, coreID, region.regionIndex);
99  			try (var rs = s.executeQuery()) {
100 				while (rs.next()) {
101 					return rs.getInt("region_id");
102 				}
103 			}
104 		}
105 		try (var s = conn.prepareStatement(INSERT_REGION,
106 				RETURN_GENERATED_KEYS)) {
107 			// core_id, local_region_index, address
108 			setArguments(s, coreID, region.regionIndex,
109 					region.startAddress.address);
110 			s.executeUpdate();
111 			return getLastKey(s).orElseThrow(() -> new IllegalStateException(
112 					"could not make or find recording region record"));
113 		}
114 	}

116 	private void appendRecordingContents(Connection conn, int regionID,
117 			byte[] content) throws SQLException {
118 		int chunkLen = content.length;
119 		var chunk = new ByteArrayInputStream(content);
120 		long timestamp = currentTimeMillis();
121 		if (useMainTable(conn, regionID)) {
122 			log.debug("adding chunk of {} bytes to region table for region {}",
123 					chunkLen, regionID);
124 			addContentToMainRow(conn, regionID, chunkLen, chunk, timestamp);
125 		} else {
126 			log.debug("adding chunk of {} bytes to extra table for region {}",
127 					chunkLen, regionID);
128 			prepareExtraContent(conn, regionID, timestamp);
129 			addExtraContentRow(conn, regionID, chunkLen, chunk);
130 		}
131 	}
132 
133 	private static byte[] read(ByteArrayInputStream chunk, int chunkLen)
134 			throws SQLException {
135 		byte[] buffer = new byte[chunkLen];
136 		int len;
137 		try {
138 			len = IOUtils.read(chunk, buffer, 0, chunkLen);
139 		} catch (IOException e) {
140 			throw new SQLException("failed to read buffer", e);
141 		}
142 		if (len == chunkLen) {
143 			return buffer;
144 		} else if (len <= 0) {
145 			return new byte[0];
146 		}
147 		byte[] nb = new byte[len];
148 		arraycopy(buffer, 0, nb, 0, len);
149 		return nb;
150 	}
151 
152 	private void addContentToMainRow(Connection conn, int regionID,
153 			int chunkLen, ByteArrayInputStream chunk, long timestamp)
154 			throws SQLException {
155 		try (var s = conn.prepareStatement(ADD_CONTENT)) {
156 			// content, append_time, region_id
157 			setArguments(s, read(chunk, chunkLen), chunkLen, timestamp,
158 					regionID);
159 			s.executeUpdate();
160 		}
161 	}
162 
163 	private void prepareExtraContent(Connection conn, int regionID,
164 			long timestamp) throws SQLException {
165 		try (var s = conn.prepareStatement(PREP_EXTRA_CONTENT)) {
166 			// append_time, region_id
167 			setArguments(s, timestamp, regionID);
168 			s.executeUpdate();
169 		}
170 	}
171 
172 	private void addExtraContentRow(Connection conn, int regionID, int chunkLen,
173 			ByteArrayInputStream chunk) throws SQLException {
174 		try (var s = conn.prepareStatement(ADD_EXTRA_CONTENT)) {
175 			// region_id, content
176 			setArguments(s, regionID, read(chunk, chunkLen), chunkLen);
177 			s.executeUpdate();
178 		}
179 	}

181 	private boolean useMainTable(Connection conn, int regionID)
182 			throws SQLException {
183 		try (var s = conn.prepareStatement(GET_MAIN_CONTENT_AVAILABLE)) {
184 			setArguments(s, regionID);
185 			try (var rs = s.executeQuery()) {
186 				while (rs.next()) {
187 					int existing = rs.getInt("existing");
188 					return existing == 1;
189 				}
190 			}
191 		}
192 		return false;
193 	}

195 	@Override
196 	public void appendRecordingContents(Region region, byte[] contents)
197 			throws StorageException {
198 		// Strip off any prefix and suffix added to make the read aligned
199 		byte[] tmp;
200 		if (region.isAligned()) {
201 			tmp = contents;
202 		} else {
203 			tmp = new byte[region.realSize];
204 			arraycopy(contents, region.initialIgnore, tmp, 0, region.realSize);
205 		}
206 		callV(conn -> appendRecordContents(conn, region, tmp),
207 				"creating or adding to a recorded region");
208 	}

210 	/**
211 	 * The core of how to append content to a recording.
212 	 *
213 	 * @param conn
214 	 *            The connection, with a transaction open.
215 	 * @param region
216 	 *            The recording region owning the recording.
217 	 * @param contents
218 	 *            The bytes to append.
219 	 * @throws SQLException
220 	 *             If anything goes wrong.
221 	 * @see #appendRecordingContents(Region,int,byte[])
222 	 */
223 	private void appendRecordContents(Connection conn, Region region,
224 			byte[] contents) throws SQLException {
225 		int coreID = getRecordingCore(conn, region.core);
226 		int regionID = getRecordingRegion(conn, coreID, region);
227 		appendRecordingContents(conn, regionID, contents);
228 	}

233 		return callR(conn -> getRecordingRegionContents(conn, region),

237 	private static byte[] getRecordingRegionContents(Connection conn,
238 			Region region) throws SQLException {
239 		var accum = new ByteArrayOutputStream();
240 		try {
241 			int regionID = -1;
242 			try (var s = conn.prepareStatement(FETCH_RECORDING)) {
243 				// x, y, processor, local_region_index
244 				setArguments(s, region.core.getX(), region.core.getY(),
245 						region.core.getP(), region.regionIndex);
246 				try (var rs = s.executeQuery()) {
247 					while (rs.next()) {
248 						accum.write(rs.getBytes("content"));
249 						regionID = rs.getInt("region_id");
250 					}
251 				}
252 			}
253 			if (regionID < 0) {
254 				throw new IllegalArgumentException("core " + region.core
255 						+ " has no data for region " + region.regionIndex);
256 			}
257 			try (var s = conn.prepareStatement(FETCH_EXTRA_RECORDING)) {
258 				// region_id
259 				setArguments(s, regionID);
260 				try (var rs = s.executeQuery()) {
261 					while (rs.next()) {
262 						accum.write(rs.getBytes("content"));
263 					}
264 				}
265 			}
266 		} catch (IOException | OutOfMemoryError e) {
267 			throw new RuntimeException("BLOB sequence too large for Java", e);
268 		}
269 		return accum.toByteArray();
270 	}

272 	@Override
273 	public List<CoreLocation> getCoresWithStorage() throws StorageException {
274 		return callR(conn -> {
275 			try (var s = conn.prepareStatement(GET_CORES_WITH_STORAGE);
276 					var rs = s.executeQuery()) {
277 				var result = new ArrayList<CoreLocation>();
278 				while (rs.next()) {
279 					int x = rs.getInt("x");
280 					int y = rs.getInt("y");
281 					int p = rs.getInt("processor");
282 					result.add(new CoreLocation(x, y, p));
283 				}
284 				return result;
285 			}
286 		}, "listing cores");
287 	}

289 	@Override
290 	public List<Integer> getRegionsWithStorage(HasCoreLocation core)
291 			throws StorageException {
292 		return callR(conn -> {
293 			try (var s = conn.prepareStatement(GET_REGIONS_WITH_STORAGE)) {
294 				setArguments(s, core.getX(), core.getY(), core.getP());
295 				try (var rs = s.executeQuery()) {
296 					var result = new ArrayList<Integer>();
297 					while (rs.next()) {
298 						result.add(rs.getInt("local_region_index"));
299 					}
300 					return result;
301 				}
302 			}
303 		}, "listing regions for a core");
304 	}
